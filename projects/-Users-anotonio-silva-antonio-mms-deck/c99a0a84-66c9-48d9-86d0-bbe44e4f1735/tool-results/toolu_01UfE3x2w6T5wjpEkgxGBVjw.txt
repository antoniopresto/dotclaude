    64→/**
    65→ * LOD Transition Configuration
    66→ *
    67→ * Defines blend zones around LOD thresholds for smooth transitions.
    68→ * During blend zones, both adjacent layers render with interpolated opacity.
    69→ */
    70→const LOD_THRESHOLDS = {
    71→  macroToMeso: 0.5,  // Below 0.5 = MACRO
    72→  mesoToMicro: 2.0,  // Above 2.0 = MICRO
    73→};
    74→
    75→// Size of blend zone around each threshold (±this value)
    76→const BLEND_ZONE_SIZE = 0.15;
    77→
    78→/**
    79→ * Calculate layer opacities for smooth LOD transitions
    80→ *
    81→ * Returns opacity values for each layer (0-1).
    82→ * During blend zones, adjacent layers overlap with interpolated opacity.
    83→ */
    84→function calculateLayerOpacities(zoom: number): {
    85→  macro: number;
    86→  meso: number;
    87→  micro: number;
    88→} {
    89→  const { macroToMeso, mesoToMicro } = LOD_THRESHOLDS;
    90→
    91→  // Define blend zone boundaries
    92→  const macroEnd = macroToMeso - BLEND_ZONE_SIZE; // 0.35
    93→  const mesoStart = macroToMeso + BLEND_ZONE_SIZE; // 0.65
    94→  const mesoEnd = mesoToMicro - BLEND_ZONE_SIZE; // 1.85
    95→  const microStart = mesoToMicro + BLEND_ZONE_SIZE; // 2.15
    96→
    97→  let macro = 0;
    98→  let meso = 0;
    99→  let micro = 0;
   100→
   101→  // Calculate MACRO opacity
   102→  if (zoom <= macroEnd) {
   103→    macro = 1;
   104→  } else if (zoom < mesoStart) {
   105→    // Blend zone MACRO → MESO
   106→    const t = (zoom - macroEnd) / (mesoStart - macroEnd);
   107→    macro = 1 - t;
   108→  }
   109→
   110→  // Calculate MESO opacity
   111→  if (zoom < macroEnd) {
   112→    meso = 0;
   113→  } else if (zoom < mesoStart) {
   114→    // Blend zone MACRO → MESO
   115→    const t = (zoom - macroEnd) / (mesoStart - macroEnd);
   116→    meso = t;
   117→  } else if (zoom <= mesoEnd) {
   118→    meso = 1;
   119→  } else if (zoom < microStart) {
   120→    // Blend zone MESO → MICRO
   121→    const t = (zoom - mesoEnd) / (microStart - mesoEnd);
   122→    meso = 1 - t;
   123→  }
   124→
   125→  // Calculate MICRO opacity
   126→  if (zoom <= mesoEnd) {
   127→    micro = 0;
   128→  } else if (zoom < microStart) {
   129→    // Blend zone MESO → MICRO
   130→    const t = (zoom - mesoEnd) / (microStart - mesoEnd);
   131→    micro = t;
   132→  } else {
   133→    micro = 1;
   134→  }
   135→
   136→  return { macro, meso, micro };
   137→}

   139→/**
   140→ * LODLayers - Orchestrates layer rendering with smooth transitions
   141→ *
   142→ * Renders multiple layers simultaneously during blend zones,
   143→ * with opacity interpolation for smooth visual transitions.
   144→ */
   145→function LODLayers({
   146→  tasks,
   147→  trainCount,
   148→  zoom,
   149→  viewportBounds,
   150→  spatialIndex,
   151→  hoveredTaskId,
   152→  selectedTaskId,
   153→  focusedTaskId,
   154→}: {
   155→  tasks: ReturnType<typeof getRenderData>['tasks'];
   156→  trainCount: number;
   157→  zoom: number;
   158→  viewportBounds: ViewportBounds;
   159→  spatialIndex: ISpatialIndex;
   160→  hoveredTaskId: string | null;
   161→  selectedTaskId: string | null;
   162→  focusedTaskId: string | null;
   163→}) {
   164→  const opacities = useMemo(() => calculateLayerOpacities(zoom), [zoom]);
   165→
   166→  return (
   167→    <group>
   168→      {/* Render each layer if opacity > 0 */}
   169→      {opacities.macro > 0 && (
   170→        <MacroLayer
   171→          tasks={tasks}
   172→          trainCount={trainCount}
   173→          opacity={opacities.macro}
   174→        />
   175→      )}
   176→      {opacities.meso > 0 && (
   177→        <MesoLayer
   178→          tasks={tasks}
   179→          trainCount={trainCount}
   180→          opacity={opacities.meso}
   181→          viewportBounds={viewportBounds}
   182→          spatialIndex={spatialIndex}
   183→          hoveredTaskId={hoveredTaskId}
   184→          selectedTaskId={selectedTaskId}
   185→          focusedTaskId={focusedTaskId}
   186→        />
   187→      )}
   188→      {opacities.micro > 0 && (
   189→        <MicroLayer
   190→          tasks={tasks}
   191→          trainCount={trainCount}
   192→          opacity={opacities.micro}
   193→          viewportBounds={viewportBounds}
   194→          hoveredTaskId={hoveredTaskId}
   195→          selectedTaskId={selectedTaskId}
   196→          focusedTaskId={focusedTaskId}
   197→        />
   198→      )}
   199→    </group>
   200→  );
   201→}

   347→/**
   348→ * Get tasks visible within the current viewport
   349→ * Sorted by train index (Y) then start day (X) for logical tab order
   350→ */
   351→function getVisibleTasks(
   352→  tasks: ReturnType<typeof getRenderData>['tasks'],
   353→  viewportBounds: ViewportBounds
   354→): ReturnType<typeof getRenderData>['tasks'] {
   355→  const visibleTasks = tasks.filter((task) => {
   356→    const taskStartX = task.startDay * pxPerDay;
   357→    const taskEndX = taskStartX + task.duration * pxPerDay;
   358→    const taskY = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
   359→
   360→    // Check if task overlaps with viewport
   361→    const xOverlap = taskEndX >= viewportBounds.minX && taskStartX <= viewportBounds.maxX;
   362→    const yOverlap = taskY >= viewportBounds.minY - pxPerTrain && taskY <= viewportBounds.maxY + pxPerTrain;
   363→
   364→    return xOverlap && yOverlap;
   365→  });
   366→
   367→  // Sort by train index (row), then by start day (column) for natural reading order
   368→  return visibleTasks.sort((a, b) => {
   369→    if (a.trainIndex !== b.trainIndex) {
   370→      return a.trainIndex - b.trainIndex;
   371→    }
   372→    return a.startDay - b.startDay;
   373→  });
   374→}

   376→/**
   377→ * Find next task in a specific direction from the current focused task
   378→ * Direction: 'left' | 'right' | 'up' | 'down'
   379→ */
   380→function findNextTaskInDirection(
   381→  tasks: ReturnType<typeof getRenderData>['tasks'],
   382→  currentTaskId: string,
   383→  direction: 'left' | 'right' | 'up' | 'down'
   384→): string | null {
   385→  const currentTask = tasks.find((t) => t.id === currentTaskId);
   386→  if (!currentTask) return null;
   387→
   388→  const currentX = currentTask.startDay * pxPerDay + (currentTask.duration * pxPerDay) / 2;
   389→  const currentY = -currentTask.trainIndex;
   390→
   391→  let bestMatch: typeof currentTask | null = null;
   392→  let bestScore = Infinity;
   393→
   394→  for (const task of tasks) {
   395→    if (task.id === currentTaskId) continue;
   396→
   397→    const taskX = task.startDay * pxPerDay + (task.duration * pxPerDay) / 2;
   398→    const taskY = -task.trainIndex;
   399→
   400→    let isInDirection = false;
   401→    let score = 0;
   402→
   403→    switch (direction) {
   404→      case 'left':
   405→        isInDirection = taskX < currentX;
   406→        if (isInDirection) {
   407→          // Prefer same row, then closest to the left
   408→          const rowPenalty = Math.abs(taskY - currentY) * 10000;
   409→          score = (currentX - taskX) + rowPenalty;
   410→        }
   411→        break;
   412→      case 'right':
   413→        isInDirection = taskX > currentX;
   414→        if (isInDirection) {
   415→          const rowPenalty = Math.abs(taskY - currentY) * 10000;
   416→          score = (taskX - currentX) + rowPenalty;
   417→        }
   418→        break;
   419→      case 'up':
   420→        isInDirection = taskY > currentY;
   421→        if (isInDirection) {
   422→          // Prefer closest row above with similar X
   423→          const xPenalty = Math.abs(taskX - currentX) / 100;
   424→          score = (taskY - currentY) + xPenalty;
   425→        }
   426→        break;
   427→      case 'down':
   428→        isInDirection = taskY < currentY;
   429→        if (isInDirection) {
   430→          const xPenalty = Math.abs(taskX - currentX) / 100;
   431→          score = (currentY - taskY) + xPenalty;
   432→        }
   433→        break;
   434→    }
   435→
   436→    if (isInDirection && score < bestScore) {
   437→      bestScore = score;
   438→      bestMatch = task;
   439→    }
   440→  }
   441→
   442→  return bestMatch?.id || null;
   443→}

   445→/**
   446→ * Find task at screen position (hit testing for hover/click)
   447→ *
   448→ * Converts screen coordinates to world coordinates and finds
   449→ * which task (if any) is under the cursor.
   450→ *
   451→ * @param screenX - Screen X coordinate (e.g., clientX)
   452→ * @param screenY - Screen Y coordinate (e.g., clientY)
   453→ * @param containerRect - Container element bounding rect
   454→ * @param cameraX - Camera X position in world coordinates
   455→ * @param cameraY - Camera Y position in world coordinates
   456→ * @param zoom - Current zoom level
   457→ * @param canvasWidth - Canvas width in pixels
   458→ * @param canvasHeight - Canvas height in pixels
   459→ * @param tasks - Array of tasks to test against
   460→ * @returns Task ID if found, null otherwise
   461→ */
   462→function findTaskAtPosition(
   463→  screenX: number,
   464→  screenY: number,
   465→  containerRect: DOMRect,
   466→  cameraX: number,
   467→  cameraY: number,
   468→  zoom: number,
   469→  canvasWidth: number,
   470→  canvasHeight: number,
   471→  tasks: ReturnType<typeof getRenderData>['tasks']
   472→): string | null {
   473→  // Convert screen coords to canvas coords (relative to container)
   474→  const canvasX = screenX - containerRect.left;
   475→  const canvasY = screenY - containerRect.top;
   476→
   477→  // Calculate half viewport in world units
   478→  const halfViewX = canvasWidth / 2 / zoom;
   479→  const halfViewY = canvasHeight / 2 / zoom;
   480→
   481→  // Convert canvas coords to world coords
   482→  // canvasX=0 is left edge of viewport, which is cameraX - halfViewX in world
   483→  // canvasY=0 is top edge of viewport, which is cameraY + halfViewY in world
   484→  const worldX = cameraX - halfViewX + (canvasX / zoom);
   485→  const worldY = cameraY + halfViewY - (canvasY / zoom);
   486→
   487→  // Find task at world position
   488→  for (const task of tasks) {
   489→    const taskStartX = task.startDay * pxPerDay;
   490→    const taskEndX = taskStartX + task.duration * pxPerDay;
   491→    const taskCenterY = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
   492→    const taskTopY = taskCenterY + TASK_HEIGHT / 2;
   493→    const taskBottomY = taskCenterY - TASK_HEIGHT / 2;
   494→
   495→    if (
   496→      worldX >= taskStartX &&
   497→      worldX <= taskEndX &&
   498→      worldY <= taskTopY &&
   499→      worldY >= taskBottomY
   500→    ) {
   501→      return task.id;
   502→    }
   503→  }
   504→
   505→  return null;
   506→}

   508→/**
   509→ * Camera controller component (inside R3F context)
   510→ * Handles initial positioning and syncs camera with external state
   511→ *
   512→ * IMPORTANT: Uses R3F's `size` from useThree() which is the actual canvas size,
   513→ * not the container div size (which can differ due to flexbox/layout).
   514→ */
   515→function CameraController({
   516→  cameraX,
   517→  cameraY,
   518→  zoom,
   519→  worldBounds,
   520→  isInitialized,
   521→  onInitialized,
   522→  onViewportChange,
   523→}: {
   524→  cameraX: number;
   525→  cameraY: number;
   526→  zoom: number;
   527→  worldBounds: WorldBounds;
   528→  isInitialized: boolean;
   529→  onInitialized: (pos: { x: number; y: number }, size: { width: number; height: number }) => void;
   530→  onViewportChange?: (viewport: ViewportInfo) => void;
   531→}) {
   532→  const { camera, size } = useThree();
   533→
   534→  // Initialize camera position on first render using actual R3F size
   535→  useEffect(() => {
   536→    if (isInitialized || size.width === 0 || size.height === 0) return;
   537→
   538→    const halfViewX = size.width / 2 / zoom;
   539→    const halfViewY = size.height / 2 / zoom;
   540→
   541→    // Position camera to show top-left corner (X=0, Y=0)
   542→    const initialCameraX = halfViewX;
   543→    const initialCameraY = -halfViewY;
   544→
   545→    // Clamp to world bounds
   546→    const minCameraX = worldBounds.minX + halfViewX;
   547→    const maxCameraX = worldBounds.maxX - halfViewX;
   548→    const minCameraY = worldBounds.minY + halfViewY;
   549→    const maxCameraY = worldBounds.maxY - halfViewY;
   550→
   551→    const clampedX = Math.max(minCameraX, Math.min(maxCameraX, initialCameraX));
   552→    const clampedY = Math.max(minCameraY, Math.min(maxCameraY, initialCameraY));
   553→
   554→    // Update camera
   555→    camera.position.x = clampedX;
   556→    camera.position.y = clampedY;
   557→    (camera as any).zoom = zoom;
   558→    camera.updateProjectionMatrix();
   559→
   560→    // Notify parent of initial position and actual canvas size
   561→    onInitialized({ x: clampedX, y: clampedY }, { width: size.width, height: size.height });
   562→
   563→    // Notify viewport change with full viewport info
   564→    const topY = clampedY + halfViewY;
   565→    const leftX = clampedX - halfViewX;
   566→    const rightX = clampedX + halfViewX;
   567→    onViewportChange?.({ topY, leftX, rightX });
   568→  }, [size, zoom, worldBounds, camera, isInitialized, onInitialized, onViewportChange]);
   569→
   570→  // Sync camera with external state changes (after initialization)
   571→  useEffect(() => {
   572→    if (!isInitialized) return;
   573→
   574→    camera.position.x = cameraX;
   575→    camera.position.y = cameraY;
   576→    (camera as any).zoom = zoom;
   577→    camera.updateProjectionMatrix();
   578→  }, [camera, cameraX, cameraY, zoom, isInitialized]);
   579→
   580→  return null;
   581→}

   832→  /**
   833→   * Clamp camera position to world bounds
   834→   */
   835→  const clampCameraToWorld = useCallback(
   836→    (x: number, y: number, currentZoom: number): { x: number; y: number } => {
   837→      const halfViewX = canvasSize.width / 2 / currentZoom;
   838→      const halfViewY = canvasSize.height / 2 / currentZoom;
   839→
   840→      const minCameraX = worldBounds.minX + halfViewX;
   841→      const maxCameraX = worldBounds.maxX - halfViewX;
   842→      const minCameraY = worldBounds.minY + halfViewY;
   843→      const maxCameraY = worldBounds.maxY - halfViewY;

   846→        maxCameraX < minCameraX
   847→          ? (worldBounds.minX + worldBounds.maxX) / 2
   848→          : Math.max(minCameraX, Math.min(maxCameraX, x));
   849→
   850→      const clampedY =
   851→        maxCameraY < minCameraY
   852→          ? (worldBounds.minY + worldBounds.maxY) / 2
   853→          : Math.max(minCameraY, Math.min(maxCameraY, y));
   854→

   856→    },
   857→    [canvasSize, worldBounds]

   859→
   860→  /**
   861→   * Stop any active momentum animation
   862→   */
   863→  const stopMomentum = useCallback(() => {
   864→    if (momentumAnimationRef.current.animationId) {
   865→      cancelAnimationFrame(momentumAnimationRef.current.animationId);

   870→  }, []);
   871→
   872→  /**
   873→   * Stop any active zoom animation
   874→   */
   875→  const stopZoomAnimation = useCallback(() => {

   877→      cancelAnimationFrame(zoomAnimationRef.current.animationId);
   878→      zoomAnimationRef.current.animationId = null;
   879→    }
   880→    // Clear wheel accumulator
   881→    if (wheelAccumulatorRef.current.timeout) {
   882→      clearTimeout(wheelAccumulatorRef.current.timeout);
   883→      wheelAccumulatorRef.current.timeout = null;
   884→    }
   885→    wheelAccumulatorRef.current.delta = 0;
   886→  }, []);
   887→
   888→  /**
   889→   * Animate zoom smoothly from current to target
   890→   * Uses quadratic easing for Google Maps-like feel
   891→   *
   892→   * Now uses canvasStore for animation state - no stale closures in the animation loop
   893→   */
   894→  const animateZoomTo = useCallback(
   895→    (
   896→      targetZoom: number,
   897→      targetCameraX: number,
   898→      targetCameraY: number,
   899→      duration: number = ZOOM_ANIMATION_DURATION
   900→    ) => {
   901→      // Stop any existing animation
   902→      stopZoomAnimation();
   903→
   904→      // Get current state from store (always fresh)
   905→      const currentState = canvasStore.getState();
   906→      const currentCamera = currentState.viewport.camera;
   907→
   908→      // Start animation in store
   909→      canvasStore.dispatch(
   910→        canvasActions.viewport.startAnimation({
   911→          startX: currentCamera.x,
   912→          startY: currentCamera.y,
   913→          startZoom: zoom,
   914→          targetX: targetCameraX,
   915→          targetY: targetCameraY,
   916→          targetZoom: targetZoom,
   917→          duration: duration,
   918→        })
   919→      );
   920→
   921→      const animate = () => {
   922→        // Read animation state from store (always current, no stale closures)
   923→        const state = canvasStore.getState();
   924→        const anim = state.viewport.animation;
   925→
   926→        if (!anim.isAnimating) {
   927→          zoomAnimationRef.current.animationId = null;
   928→          return;
   929→        }
   930→
   931→        const elapsed = Date.now() - anim.startTime;
   932→        const progress = Math.min(elapsed / anim.duration, 1);
   933→        const easedProgress = Easing.quadOut(progress);
   934→
   935→        const currentZoom = anim.startZoom + (anim.targetZoom - anim.startZoom) * easedProgress;
   936→        const currentCameraX = anim.startX + (anim.targetX - anim.startX) * easedProgress;
   937→        const currentCameraY = anim.startY + (anim.targetY - anim.startY) * easedProgress;
   938→
   939→        // Clamp camera position
   940→        const clamped = clampCameraToWorld(currentCameraX, currentCameraY, currentZoom);
   941→
   942→        // Update store with new camera position
   943→        canvasStore.dispatch(
   944→          canvasActions.viewport.updateAnimation({ camera: clamped })
   945→        );
   946→        onZoomChange(currentZoom);
   947→        notifyViewportChange(clamped.x, clamped.y, currentZoom);
   948→
   949→        if (progress < 1) {
   950→          zoomAnimationRef.current.animationId = requestAnimationFrame(animate);
   951→        } else {
   952→          canvasStore.dispatch(canvasActions.viewport.stopAnimation());
   953→          zoomAnimationRef.current.animationId = null;
   954→        }
   955→      };
   956→
   957→      zoomAnimationRef.current.animationId = requestAnimationFrame(animate);
   958→    },
   959→    [zoom, clampCameraToWorld, onZoomChange, notifyViewportChange, stopZoomAnimation]
   960→  );
   961→
   962→  /**
   963→   * Calculate average velocity from samples
   964→   * Samples contain individual deltas (movement per event), not absolute positions
   965→   */
   966→  const calculateAverageVelocity = useCallback(() => {
   967→    const samples = velocitySamplesRef.current;
   968→    if (samples.length < 2) return { vx: 0, vy: 0 };
   969→
   970→    // Use only recent samples (last N samples)
   971→    const recentSamples = samples.slice(-MOMENTUM_VELOCITY_SAMPLES);
   972→    if (recentSamples.length < 2) return { vx: 0, vy: 0 };
   973→
   974→    // Calculate total displacement and time
   975→    const first = recentSamples[0];
   976→    const last = recentSamples[recentSamples.length - 1];
   977→    const dt = last.time - first.time;
   978→
   979→    if (dt <= 0) return { vx: 0, vy: 0 };
   980→
   981→    // Sum all deltas to get total displacement
   982→    let totalDx = 0;
   983→    let totalDy = 0;
   984→    for (const sample of recentSamples) {
   985→      totalDx += sample.x;
   986→      totalDy += sample.y;
   987→    }
   988→
   989→    // Velocity in world units per millisecond
   990→    const vx = totalDx / dt;
   991→    const vy = totalDy / dt;
   992→
   993→    return { vx, vy };
   994→  }, []);
   995→
   996→  /**
   997→   * Start momentum animation after drag release
   998→   *
   999→   * Now uses canvasStore for momentum state - no stale closures in the animation loop
  1000→   */
  1001→  const startMomentum = useCallback(
  1002→    (vx: number, vy: number) => {
  1003→      // Convert velocity from world/ms to world/frame (at 60fps, ~16.67ms per frame)
  1004→      const frameTime = 16.67;
  1005→      const velocityX = vx * frameTime;
  1006→      const velocityY = vy * frameTime;
  1007→
  1008→      // Start momentum in store
  1009→      canvasStore.dispatch(
  1010→        canvasActions.viewport.startMomentum({ velocityX, velocityY })
  1011→      );
  1012→
  1013→      // Clear samples
  1014→      velocitySamplesRef.current = [];
  1015→
  1016→      const animate = () => {
  1017→        // Read momentum state from store (always current, no stale closures)
  1018→        const state = canvasStore.getState();
  1019→        const mom = state.viewport.momentum;
  1020→
  1021→        if (!mom.isActive) {
  1022→          momentumAnimationRef.current.animationId = null;
  1023→          return;
  1024→        }
  1025→
  1026→        // Check if velocity is negligible
  1027→        const speed = Math.sqrt(mom.velocityX ** 2 + mom.velocityY ** 2);
  1028→        if (speed < MOMENTUM_MIN_VELOCITY) {
  1029→          canvasStore.dispatch(canvasActions.viewport.stopMomentum());
  1030→          momentumAnimationRef.current.animationId = null;
  1031→          return;
  1032→        }
  1033→
  1034→        // Apply velocity to camera position
  1035→        // Using store.getState() - no stale closures!
  1036→        const prevPos = state.viewport.camera;
  1037→        const newX = prevPos.x + mom.velocityX;
  1038→        const newY = prevPos.y - mom.velocityY; // Inverted for camera Y
  1039→        const clamped = clampCameraToWorld(newX, newY, zoom);
  1040→
  1041→        // Update store with new position and apply friction
  1042→        canvasStore.dispatch(
  1043→          canvasActions.viewport.updateMomentum({
  1044→            velocityX: mom.velocityX * MOMENTUM_FRICTION,
  1045→            velocityY: mom.velocityY * MOMENTUM_FRICTION,
  1046→            camera: clamped,
  1047→          })
  1048→        );
  1049→        notifyViewportChange(clamped.x, clamped.y, zoom);
  1050→
  1051→        // Continue animation
  1052→        momentumAnimationRef.current.animationId = requestAnimationFrame(animate);
  1053→      };
  1054→
  1055→      // Start animation
  1056→      momentumAnimationRef.current.animationId = requestAnimationFrame(animate);
  1057→    },
  1058→    [clampCameraToWorld, notifyViewportChange, zoom]
  1059→  );
  1060→
  1061→  /**
  1062→   * Calculate target camera position for zoom toward a point
  1063→   */
  1064→  const calculateZoomTowardPoint = useCallback(
  1065→    (cursorX: number, cursorY: number, fromZoom: number, toZoom: number) => {
  1066→      // Convert cursor to world coordinates (before zoom)
  1067→      const worldX = cursorX / fromZoom + cameraPos.x;
  1068→      const worldY = -cursorY / fromZoom + cameraPos.y;
  1069→
  1070→      // Calculate new camera position to keep cursor point fixed
  1071→      const newCameraX = worldX - cursorX / toZoom;
  1072→      const newCameraY = worldY + cursorY / toZoom;
  1073→
  1074→      return clampCameraToWorld(newCameraX, newCameraY, toZoom);
  1075→    },
  1076→    [cameraPos, clampCameraToWorld]
  1077→  );
  1078→
  1079→  /**
  1080→   * Execute accumulated zoom with smooth animation
  1081→   * Called after wheel events are accumulated (for trackpad smoothing)
  1082→   * With cross-platform normalized delta (F121)
  1083→   *
  1084→   * Now uses canvasStore to get current animation state
  1085→   */
  1086→  const executeAccumulatedZoom = useCallback(() => {
  1087→    const accumulator = wheelAccumulatorRef.current;
  1088→    if (Math.abs(accumulator.delta) < 0.001) {
  1089→      accumulator.delta = 0;
  1090→      return;
  1091→    }
  1092→
  1093→    const rect = containerRef.current?.getBoundingClientRect();
  1094→    if (!rect) return;
  1095→
  1096→    // Get current state from store (always fresh)
  1097→    const state = canvasStore.getState();
  1098→    const anim = state.viewport.animation;
  1099→
  1100→    // Get the current zoom (may be in animation)
  1101→    const currentZoom = anim.isAnimating ? anim.targetZoom : zoom;
  1102→
  1103→    // Calculate zoom factor from accumulated normalized delta (F121)
  1104→    // Normalized delta is already scaled by sensitivity, use exponential zoom
  1105→    // Negative delta = zoom in, positive delta = zoom out
  1106→    const zoomFactor = Math.exp(-accumulator.delta);
  1107→    const newZoom = Math.max(effectiveMinZoom, Math.min(MAX_ZOOM, currentZoom * zoomFactor));
  1108→
  1109→    if (Math.abs(newZoom - currentZoom) < 0.001) {
  1110→      accumulator.delta = 0;
  1111→      return;
  1112→    }
  1113→
  1114→    // Cursor position relative to canvas center
  1115→    const cursorX = accumulator.cursorX - rect.left - rect.width / 2;
  1116→    const cursorY = accumulator.cursorY - rect.top - rect.height / 2;
  1117→
  1118→    // Get current camera position (may be in animation)
  1119→    const currentCameraX = anim.isAnimating ? anim.targetX : state.viewport.camera.x;
  1120→    const currentCameraY = anim.isAnimating ? anim.targetY : state.viewport.camera.y;
  1121→
  1122→    // Calculate target camera position
  1123→    const worldX = cursorX / currentZoom + currentCameraX;
  1124→    const worldY = -cursorY / currentZoom + currentCameraY;
  1125→    const newCameraX = worldX - cursorX / newZoom;
  1126→    const newCameraY = worldY + cursorY / newZoom;
  1127→    const clamped = clampCameraToWorld(newCameraX, newCameraY, newZoom);
  1128→
  1129→    // Reset accumulator
  1130→    accumulator.delta = 0;
  1131→    accumulator.timeout = null;
  1132→
  1133→    // Animate to new zoom
  1134→    animateZoomTo(newZoom, clamped.x, clamped.y, ZOOM_ANIMATION_DURATION);
  1135→  }, [zoom, effectiveMinZoom, clampCameraToWorld, animateZoomTo]);
  1136→
  1137→  /**
  1138→   * Normalize wheel delta for cross-platform consistency (F121)
  1139→   * Handles deltaMode (LINE/PAGE), detects trackpad vs mouse
  1140→   * Returns normalized pixel delta
  1141→   */
  1142→  const normalizeWheelDelta = useCallback((e: WheelEvent): number => {
  1143→    let delta = e.deltaY;
  1144→
  1145→    // Normalize deltaMode to pixels (F121: cross-platform normalization)
  1146→    // deltaMode 0 = PIXEL, 1 = LINE, 2 = PAGE
  1147→    if (e.deltaMode === 1) {
  1148→      // LINE mode (Firefox on Windows)
  1149→      delta *= LINE_HEIGHT;
  1150→    } else if (e.deltaMode === 2) {
  1151→      // PAGE mode (rare, but handle it)
  1152→      delta *= PAGE_HEIGHT;
  1153→    }
  1154→
  1155→    // Detect trackpad vs mouse by event frequency
  1156→    const accumulator = wheelAccumulatorRef.current;
  1157→    const now = performance.now();
  1158→    const timeSinceLastEvent = now - accumulator.lastEventTime;
  1159→
  1160→    if (timeSinceLastEvent < TRACKPAD_DETECTION_THRESHOLD) {
  1161→      accumulator.rapidEventCount++;
  1162→      if (accumulator.rapidEventCount >= TRACKPAD_EVENT_COUNT_THRESHOLD) {
  1163→        accumulator.isTrackpad = true;
  1164→      }
  1165→    } else {
  1166→      // Reset detection if events are spaced out (mouse behavior)
  1167→      accumulator.rapidEventCount = 1;
  1168→      // Also check delta magnitude - trackpads produce small deltas
  1169→      if (Math.abs(delta) > 50 && e.deltaMode === 0) {
  1170→        accumulator.isTrackpad = false;
  1171→      }
  1172→    }
  1173→    accumulator.lastEventTime = now;
  1174→
  1175→    // Apply sensitivity based on input type
  1176→    const sensitivity = accumulator.isTrackpad
  1177→      ? TRACKPAD_ZOOM_SENSITIVITY
  1178→      : MOUSE_ZOOM_SENSITIVITY;
  1179→
  1180→    // Apply sensitivity and clamp to prevent jumps
  1181→    const normalizedDelta = delta * sensitivity;
  1182→    return Math.max(-MAX_ZOOM_DELTA_PER_EVENT, Math.min(MAX_ZOOM_DELTA_PER_EVENT, normalizedDelta));
  1183→  }, []);

  1185→/**
  1186→   * Handle wheel events - Google Maps-like behavior (BUG-003)
  1187→   *
  1188→   * Behavior:
  1189→   * - No modifier (2-finger scroll): Pan in both directions (deltaX → horizontal, deltaY → vertical)
  1190→   * - Ctrl+scroll (pinch gesture): Zoom toward cursor position
  1191→   *
  1192→   * This follows Google Maps UX pattern where:
  1193→   * - Trackpad 2-finger drag pans the map
  1194→   * - Trackpad pinch (sends ctrlKey=true) zooms
  1195→   * - Mouse Ctrl+scroll zooms
  1196→   *
  1197→   * With cross-platform normalization (F121)
  1198→   */
  1199→  const handleWheel = useCallback(
  1200→    (e: WheelEvent) => {
  1201→      e.preventDefault();
  1202→      if (!isInitialized) return;
  1203→
  1204→      // Stop momentum on any wheel interaction
  1205→      stopMomentum();
  1206→
  1207→      const rect = containerRef.current?.getBoundingClientRect();
  1208→      if (!rect) return;
  1209→
  1210→      // Normalize deltas for cross-platform consistency
  1211→      const getNormalizedDeltas = () => {
  1212→        let deltaX = e.deltaX;
  1213→        let deltaY = e.deltaY;
  1214→        if (e.deltaMode === 1) {
  1215→          deltaX *= LINE_HEIGHT;
  1216→          deltaY *= LINE_HEIGHT;
  1217→        } else if (e.deltaMode === 2) {
  1218→          deltaX *= PAGE_HEIGHT;
  1219→          deltaY *= PAGE_HEIGHT;
  1220→        }
  1221→        return { deltaX, deltaY };
  1222→      };
  1223→
  1224→      // Ctrl+scroll = ZOOM (pinch gesture on trackpad sends ctrlKey=true)
  1225→      // BUG-009: Apply zoom DIRECTLY for trackpad pinch (no animation delay)
  1226→      // This eliminates the 50ms accumulation + 250ms animation lag for responsive feel
  1227→      if (e.ctrlKey) {
  1228→        // Normalize delta for cross-platform consistency (F121)
  1229→        const normalizedDelta = normalizeWheelDelta(e);
  1230→
  1231→        // Skip if delta is too small
  1232→        if (Math.abs(normalizedDelta) < 0.001) return;
  1233→
  1234→        // Calculate zoom factor from normalized delta
  1235→        // Negative delta = zoom in, positive delta = zoom out
  1236→        const zoomFactor = Math.exp(-normalizedDelta);
  1237→        const newZoom = Math.max(effectiveMinZoom, Math.min(MAX_ZOOM, zoom * zoomFactor));
  1238→
  1239→        // Skip if zoom change is negligible
  1240→        if (Math.abs(newZoom - zoom) < 0.001) return;
  1241→
  1242→        // Cursor position relative to canvas center
  1243→        const cursorX = e.clientX - rect.left - rect.width / 2;
  1244→        const cursorY = e.clientY - rect.top - rect.height / 2;
  1245→
  1246→        // Get current camera position
  1247→        const currentCamera = canvasStore.getState().viewport.camera;
  1248→
  1249→        // Calculate new camera position to keep cursor point fixed in world space
  1250→        const worldX = cursorX / zoom + currentCamera.x;
  1251→        const worldY = -cursorY / zoom + currentCamera.y;
  1252→        const newCameraX = worldX - cursorX / newZoom;
  1253→        const newCameraY = worldY + cursorY / newZoom;
  1254→        const clamped = clampCameraToWorld(newCameraX, newCameraY, newZoom);
  1255→
  1256→        // Apply directly (no animation) - same as handlePinchZoom for responsive feel
  1257→        updateCameraPos(clamped);
  1258→        onZoomChange(newZoom);
  1259→        notifyViewportChange(clamped.x, clamped.y, newZoom);
  1260→        return;
  1261→      }
  1262→
  1263→      // Default behavior: PAN in both directions (Google Maps style)
  1264→      // 2-finger trackpad scroll sends both deltaX and deltaY
  1265→      const { deltaX, deltaY } = getNormalizedDeltas();
  1266→
  1267→      // Apply pan sensitivity (scaled by zoom for consistent world movement)
  1268→      const panDeltaX = (deltaX * SCROLL_PAN_SENSITIVITY) / zoom;
  1269→      const panDeltaY = (deltaY * SCROLL_PAN_SENSITIVITY) / zoom;
  1270→
  1271→      const currentCamera = canvasStore.getState().viewport.camera;
  1272→      const newX = currentCamera.x + panDeltaX;
  1273→      const newY = currentCamera.y - panDeltaY; // Negative because Y is inverted in canvas coords
  1274→      const clamped = clampCameraToWorld(newX, newY, zoom);
  1275→
  1276→      updateCameraPos(clamped);
  1277→      notifyViewportChange(clamped.x, clamped.y, zoom);
  1278→    },
  1279→    [isInitialized, stopMomentum, normalizeWheelDelta, effectiveMinZoom, zoom, clampCameraToWorld, updateCameraPos, onZoomChange, notifyViewportChange]
