    11→import Anthropic from '@anthropic-ai/sdk';
    12→import type { ChatMessage, ClaudeResponse, CloudModel } from '../types';
    13→
    14→// Singleton instance
    15→let anthropicInstance: Anthropic | null = null;
    16→
    17→// Model configurations
    18→const MODEL_MAP: Record<CloudModel, string> = {
    19→  'claude-opus': 'claude-opus-4-5-20251101',
    20→  'claude-sonnet': 'claude-sonnet-4-20250514',
    21→};
    22→
    23→const MAX_TOKENS: Record<CloudModel, number> = {
    24→  'claude-opus': 32000,
    25→  'claude-sonnet': 16000,
    26→};
    27→
    28→export function getAnthropic(): Anthropic {
    29→  if (!anthropicInstance) {
    30→    const apiKey = process.env.ANTHROPIC_API_KEY;
    31→    if (!apiKey) {
    32→      throw new Error(
    33→        'ANTHROPIC_API_KEY not set. Export it in your shell or .env file.'
    34→      );
    35→    }
    36→    anthropicInstance = new Anthropic({ apiKey });
    37→  }
    38→  return anthropicInstance;
    39→}

    41→export async function chat(
    42→  model: CloudModel,
    43→  messages: ChatMessage[],
    44→  options?: {
    45→    temperature?: number;
    46→    maxTokens?: number;
    47→    system?: string;
    48→  }
    49→): Promise<ClaudeResponse> {
    50→  const anthropic = getAnthropic();
    51→
    52→  // Separate system message from conversation
    53→  const systemMessage =
    54→    options?.system ||
    55→    messages.find((m) => m.role === 'system')?.content ||
    56→    '';
    57→
    58→  const conversationMessages = messages
    59→    .filter((m) => m.role !== 'system')
    60→    .map((m) => ({
    61→      role: m.role as 'user' | 'assistant',
    62→      content: m.content,
    63→    }));
    64→
    65→  const response = await anthropic.messages.create({
    66→    model: MODEL_MAP[model],
    67→    max_tokens: options?.maxTokens ?? MAX_TOKENS[model],
    68→    temperature: options?.temperature ?? 0.7,
    69→    system: systemMessage,
    70→    messages: conversationMessages,
    71→  });
    72→
    73→  return {
    74→    id: response.id,
    75→    content: response.content as Array<{ type: 'text'; text: string }>,
    76→    model: response.model,
    77→    stopReason: response.stop_reason || 'end_turn',
    78→    usage: {
    79→      inputTokens: response.usage.input_tokens,
    80→      outputTokens: response.usage.output_tokens,
    81→    },
    82→  };
    83→}

    85→export async function streamChat(
    86→  model: CloudModel,
    87→  messages: ChatMessage[],
    88→  onToken: (token: string) => void,
    89→  options?: {
    90→    temperature?: number;
    91→    maxTokens?: number;
    92→    system?: string;
    93→  }
    94→): Promise<void> {
    95→  const anthropic = getAnthropic();
    96→
    97→  const systemMessage =
    98→    options?.system ||
    99→    messages.find((m) => m.role === 'system')?.content ||
   100→    '';
   101→
   102→  const conversationMessages = messages
   103→    .filter((m) => m.role !== 'system')
   104→    .map((m) => ({
   105→      role: m.role as 'user' | 'assistant',
   106→      content: m.content,
   107→    }));
   108→
   109→  const stream = anthropic.messages.stream({
   110→    model: MODEL_MAP[model],
   111→    max_tokens: options?.maxTokens ?? MAX_TOKENS[model],
   112→    temperature: options?.temperature ?? 0.7,
   113→    system: systemMessage,
   114→    messages: conversationMessages,
   115→  });
   116→
   117→  for await (const event of stream) {
   118→    if (
   119→      event.type === 'content_block_delta' &&
   120→      event.delta.type === 'text_delta'
   121→    ) {
   122→      onToken(event.delta.text);
   123→    }
   124→  }
   125→}
