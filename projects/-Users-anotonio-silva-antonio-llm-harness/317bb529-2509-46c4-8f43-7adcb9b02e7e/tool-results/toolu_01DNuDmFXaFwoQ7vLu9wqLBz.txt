     1→#!/usr/bin/env bun
     2→/**
     3→ * Session CLI
     4→ *
     5→ * Interactive harness session for executing tasks.
     6→ *
     7→ * Usage: bun run src/cli/session.ts [project-root]
     8→ */
     9→
    10→import { createHarness } from '../harness';
    11→import { getAllAgents } from '../agents';
    12→
    13→async function main() {
    14→  const projectRoot = process.argv[2] || process.cwd();
    15→
    16→  console.log('=== LLM Harness Session ===\n');
    17→  console.log(`Project: ${projectRoot}\n`);
    18→
    19→  const harness = createHarness(projectRoot);
    20→  await harness.startSession();
    21→
    22→  // Show available agents
    23→  console.log('\nAvailable agents:');
    24→  getAllAgents().forEach((agent) => {
    25→    console.log(`  - ${agent.id}: ${agent.name} (${agent.model})`);
    26→  });
    27→
    28→  // Show feature status
    29→  const stats = harness.getFeatureStats();
    30→  console.log(`\nFeatures: ${stats.complete}/${stats.total} complete`);
    31→  console.log(`  Pending: ${stats.pending}`);
    32→  console.log(`  In Progress: ${stats.inProgress}`);
    33→  console.log(`  Failed: ${stats.failed}`);
    34→
    35→  // Show next feature
    36→  const next = harness.getNextFeature();
    37→  if (next) {
    38→    console.log(`\nNext: ${next.id} - ${next.description}`);
    39→  }
    40→
    41→  console.log('\nCommands:');
    42→  console.log('  /task <description>  - Route and execute a task');
    43→  console.log('  /chat <agent> <msg>  - Chat with specific agent');
    44→  console.log('  /status              - Show feature status');
    45→  console.log('  /progress            - Show recent progress');
    46→  console.log('  /start <id>          - Start a feature');
    47→  console.log('  /complete <id>       - Complete a feature');
    48→  console.log('  /quit                - End session');
    49→  console.log('');
    50→
    51→  const readline = await import('readline');
    52→  const rl = readline.createInterface({
    53→    input: process.stdin,
    54→    output: process.stdout,
    55→  });
    56→
    57→  const askQuestion = () => {
    58→    rl.question('> ', async (input) => {
    59→      const trimmed = input.trim();
    60→
    61→      if (!trimmed) {
    62→        askQuestion();
    63→        return;
    64→      }
    65→
    66→      // Parse command
    67→      if (trimmed.startsWith('/')) {
    68→        const [cmd, ...args] = trimmed.slice(1).split(' ');
    69→
    70→        switch (cmd) {
    71→          case 'quit':
    72→          case 'exit':
    73→            harness.endSession('Session ended by user');
    74→            console.log('Goodbye!');
    75→            rl.close();
    76→            return;
    77→
    78→          case 'task':
    79→            if (!args.length) {
    80→              console.log('Usage: /task <description>');
    81→              break;
    82→            }
    83→            const task = args.join(' ');
    84→            console.log('\nExecuting task...');
    85→            const result = await harness.executeTask(task);
    86→            console.log('\n--- Response ---');
    87→            console.log(result);
    88→            console.log('----------------\n');
    89→            break;
    90→
    91→          case 'chat':
    92→            if (args.length < 2) {
    93→              console.log('Usage: /chat <agent-id> <message>');
    94→              break;
    95→            }
    96→            const [agentId, ...msgParts] = args;
    97→            const message = msgParts.join(' ');
    98→            console.log(`\nChatting with ${agentId}...`);
    99→            try {
   100→              const response = await harness.chatWithAgent(agentId, message);
   101→              console.log('\n--- Response ---');
   102→              console.log(response);
   103→              console.log('----------------\n');
   104→            } catch (e) {
   105→              console.error(`Error: ${e instanceof Error ? e.message : e}`);
   106→            }
   107→            break;
   108→
   109→          case 'status':
   110→            const s = harness.getFeatureStats();
   111→            console.log(`\nFeatures: ${s.complete}/${s.total} complete`);
   112→            console.log(`  Pending: ${s.pending}`);
   113→            console.log(`  In Progress: ${s.inProgress}`);
   114→            console.log(`  Failed: ${s.failed}\n`);
   115→            break;
   116→
   117→          case 'progress':
   118→            const recent = harness.getRecentProgress(10);
   119→            console.log('\nRecent progress:');
   120→            recent.forEach((line) => console.log(`  ${line}`));
   121→            console.log('');
   122→            break;
   123→
   124→          case 'start':
   125→            if (!args[0]) {
   126→              console.log('Usage: /start <feature-id>');
   127→              break;
   128→            }
   129→            harness.startFeature(args[0]);
   130→            console.log(`Started feature ${args[0]}`);
   131→            break;
   132→
   133→          case 'complete':
   134→            if (!args[0]) {
   135→              console.log('Usage: /complete <feature-id>');
   136→              break;
   137→            }
   138→            harness.completeFeature(args[0], args.slice(1).join(' ') || undefined);
   139→            console.log(`Completed feature ${args[0]}`);
   140→            break;
   141→
   142→          default:
   143→            console.log(`Unknown command: ${cmd}`);
   144→        }
   145→      } else {
   146→        // Treat as task by default
   147→        console.log('\nExecuting task...');
   148→        const result = await harness.executeTask(trimmed);
   149→        console.log('\n--- Response ---');
   150→        console.log(result);
   151→        console.log('----------------\n');
   152→      }
   153→
   154→      askQuestion();
   155→    });
   156→  };
   157→
   158→  askQuestion();
   159→}
   160→
   161→main().catch(console.error);
