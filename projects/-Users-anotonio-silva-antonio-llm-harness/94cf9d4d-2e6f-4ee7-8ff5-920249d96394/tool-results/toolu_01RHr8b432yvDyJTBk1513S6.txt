   170→// Train Pool Panel Component
   171→function TrainPoolPanel({
   172→  pool,
   173→  onSelectTrain,
   174→  selectedTrainId,
   175→}: {
   176→  pool: TrainPoolItem[];
   177→  onSelectTrain: (trainId: string) => void;
   178→  selectedTrainId: string | null;
   179→}) {
   180→  const [searchValue, setSearchValue] = useState('');
   181→  const [fleetFilter, setFleetFilter] = useState<
   182→    'ALL' | 'KINKI_SHARYO' | 'ALSTOM'
   183→  >('ALL');
   184→  const [isFleetOpen, setIsFleetOpen] = useState(false);
   185→
   186→  const filteredPool = useMemo(() => {
   187→    let result = pool;
   188→
   189→    if (fleetFilter !== 'ALL') {
   190→      result = result.filter(t => t.fleetType === fleetFilter);
   191→    }
   192→
   193→    if (searchValue.length >= 2) {
   194→      const search = searchValue.toLowerCase();
   195→      result = result.filter(t => t.trainId.toLowerCase().includes(search));
   196→    }
   197→
   198→    return result.filter(t => t.fitnessStatus === 'FIT').slice(0, 15);
   199→  }, [
   200→    pool,
   201→    fleetFilter,
   202→    searchValue,
   203→  ]);
   204→
   205→  return (
   206→    <Card isCompact>
   207→      <CardHeader>
   208→        <CardTitle>Train Pool</CardTitle>
   209→      </CardHeader>
   210→      <CardBody>
   211→        <Flex direction={{ default: 'column' }} gap={{ default: 'gapSm' }}>
   212→          <FlexItem>
   213→            <Flex gap={{ default: 'gapSm' }}>
   214→              <FlexItem grow={{ default: 'grow' }}>
   215→                <SearchInput
   216→                  placeholder="Search..."
   217→                  value={searchValue}
   218→                  onChange={(_e, v) => setSearchValue(v)}
   219→                  onClear={() => setSearchValue('')}
   220→                />
   221→              </FlexItem>
   222→              <FlexItem>
   223→                <Dropdown
   224→                  isOpen={isFleetOpen}
   225→                  onOpenChange={setIsFleetOpen}
   226→                  onSelect={() => setIsFleetOpen(false)}
   227→                  toggle={toggleRef => (
   228→                    <MenuToggle
   229→                      ref={toggleRef}
   230→                      onClick={() => setIsFleetOpen(!isFleetOpen)}
   231→                      isExpanded={isFleetOpen}
   232→                      isCompact
   233→                    >
   234→                      {fleetFilter === 'ALL'
   235→                        ? 'All'
   236→                        : fleetFilter === 'KINKI_SHARYO'
   237→                          ? 'KS'
   238→                          : 'AL'}
   239→                    </MenuToggle>
   240→                  )}
   241→                >
   242→                  <DropdownList>
   243→                    <DropdownItem
   244→                      key="all"
   245→                      onClick={() => setFleetFilter('ALL')}
   246→                    >
   247→                      All
   248→                    </DropdownItem>
   249→                    <DropdownItem
   250→                      key="ks"
   251→                      onClick={() => setFleetFilter('KINKI_SHARYO')}
   252→                    >
   253→                      Kinki Sharyo
   254→                    </DropdownItem>
   255→                    <DropdownItem
   256→                      key="al"
   257→                      onClick={() => setFleetFilter('ALSTOM')}
   258→                    >
   259→                      Alstom
   260→                    </DropdownItem>
   261→                  </DropdownList>
   262→                </Dropdown>
   263→              </FlexItem>
   264→            </Flex>
   265→          </FlexItem>
   266→
   267→          <FlexItem>
   268→            <div className="pf-v6-u-font-size-sm pf-v6-u-color-200 pf-v6-u-mb-sm">
   269→              {filteredPool.length} available (FIT)
   270→            </div>
   271→          </FlexItem>
   272→
   273→          <FlexItem>
   274→            <Flex direction={{ default: 'column' }} gap={{ default: 'gapXs' }}>
   275→              {filteredPool.map(train => (
   276→                <FlexItem key={train.trainId}>
   277→                  <div
   278→                    onClick={() => onSelectTrain(train.trainId)}
   279→                    className={`pf-v6-u-p-sm pf-v6-u-border-radius-sm ${
   280→                      selectedTrainId === train.trainId
   281→                        ? 'pf-v6-u-background-color-primary-100'
   282→                        : 'pf-v6-u-background-color-200'
   283→                    }`}
   284→                    role="button"
   285→                    tabIndex={0}
   286→                    onKeyDown={e => {
   287→                      if (e.key === 'Enter' || e.key === ' ') {
   288→                        onSelectTrain(train.trainId);
   289→                      }
   290→                    }}
   291→                  >
   292→                    <Flex
   293→                      justifyContent={{ default: 'justifyContentSpaceBetween' }}
   294→                    >
   295→                      <FlexItem>
   296→                        <Label
   297→                          color={
   298→                            train.fitnessStatus === 'FIT' ? 'green' : 'orange'
   299→                          }
   300→                          isCompact
   301→                        >
   302→                          {train.trainId}
   303→                        </Label>
   304→                      </FlexItem>
   305→                      <FlexItem>
   306→                        <span className="pf-v6-u-font-size-xs">
   307→                          {formatMileage(train.currentMileage)}
   308→                        </span>
   309→                      </FlexItem>
   310→                    </Flex>
   311→                    <Progress
   312→                      value={train.pmProgress}
   313→                      title="PM Progress"
   314→                      size="sm"
   315→                      variant={
   316→                        train.pmProgress > 90
   317→                          ? 'danger'
   318→                          : train.pmProgress > 75
   319→                            ? 'warning'
   320→                            : undefined
   321→                      }
   322→                      measureLocation={ProgressMeasureLocation.outside}
   323→                      className="pf-v6-u-mt-xs"
   324→                    />
   325→                  </div>
   326→                </FlexItem>
   327→              ))}
   328→            </Flex>
   329→          </FlexItem>
   330→        </Flex>
   331→      </CardBody>
   332→    </Card>
   333→  );
   334→}

   425→              </div>

   429→              <Flex
   430→                justifyContent={{ default: 'justifyContentSpaceBetween' }}

   526→// AI Recommendations Panel Component
   527→function RecommendationsPanel({
   528→  schedule,
   529→  onAcceptAll,
   530→  isAcceptingAll,
   531→}: {
   532→  schedule: DailyScheduleData;
   533→  pool: TrainPoolItem[];
   534→  onAcceptAll: () => void;
   535→  isAcceptingAll: boolean;
   536→}) {
   537→  const pendingSuggestions = schedule.assignments.filter(
   538→    a => a.status === 'AI_SUGGESTED',
   539→  );
   540→
   541→  return (
   542→    <Card isCompact>
   543→      <CardHeader>
   544→        <CardTitle>
   545→          <AutomationIcon className="pf-v6-u-mr-sm" />
   546→          AI Recommendations
   547→        </CardTitle>
   548→      </CardHeader>
   549→      <CardBody>
   550→        {pendingSuggestions.length > 0 ? (
   551→          <Flex direction={{ default: 'column' }} gap={{ default: 'gapMd' }}>
   552→            <FlexItem>
   553→              <Button
   554→                variant="primary"
   555→                onClick={onAcceptAll}
   556→                isLoading={isAcceptingAll}
   557→                isDisabled={isAcceptingAll}
   558→              >
   559→                Accept All ({pendingSuggestions.length})
   560→              </Button>
   561→            </FlexItem>
   562→
   563→            <Divider />
   564→
   565→            {pendingSuggestions.slice(0, 3).map(assignment => {
   566→              const route = schedule.routes.find(
   567→                r => r.id === assignment.routeId,
   568→              );
   569→              if (!route || !assignment.aiRecommendation) return null;
   570→
   571→              return (
   572→                <FlexItem key={assignment.routeId}>
   573→                  <Panel variant="bordered">
   574→                    <PanelHeader>
   575→                      <strong>{route.name}</strong>
   576→                    </PanelHeader>
   577→                    <PanelMain>
   578→                      <PanelMainBody>
   579→                        <div className="pf-v6-u-mb-sm">
   580→                          <Label color="blue" isCompact>
   581→                            {assignment.aiRecommendation.trainId}
   582→                          </Label>
   583→                          <span className="pf-v6-u-ml-sm">
   584→                            Score: {assignment.aiRecommendation.score}
   585→                          </span>
   586→                        </div>
   587→
   588→                        {assignment.aiRecommendation.alternatives.length >
   589→                          0 && (
   590→                          <div className="pf-v6-u-font-size-xs pf-v6-u-color-200">
   591→                            <div className="pf-v6-u-mb-xs">Alternatives:</div>
   592→                            {assignment.aiRecommendation.alternatives.map(
   593→                              alt => (
   594→                                <div key={alt.trainId}>
   595→                                  • {alt.trainId} ({alt.score}) - {alt.reason}
   596→                                </div>
   597→                              ),
   598→                            )}
   599→                          </div>
   600→                        )}
   601→                      </PanelMainBody>
   602→                    </PanelMain>
   603→                  </Panel>
   604→                </FlexItem>
   605→              );
   606→            })}
   607→          </Flex>
   608→        ) : (
   609→          <div className="pf-v6-u-text-align-center pf-v6-u-color-200">
   610→            <CheckCircleIcon className="pf-v6-u-mb-sm" />
   611→            <div>All recommendations accepted</div>
   612→          </div>
   613→        )}
   614→
   615→        {/* Warnings */}
   616→        {schedule.warnings.length > 0 && (
   617→          <>
   618→            <Divider className="pf-v6-u-my-md" />
   619→            <div className="pf-v6-u-font-weight-bold pf-v6-u-mb-sm">
   620→              <ExclamationTriangleIcon
   621→                className="pf-v6-u-mr-sm"
   622→                color="var(--pf-t--global--color--status--warning--default)"
   623→              />
   624→              Warnings ({schedule.warnings.length})
   625→            </div>
   626→            {schedule.warnings.map((warning, i) => (
   627→              <div key={i} className="pf-v6-u-font-size-sm pf-v6-u-mb-xs">
   628→                • {warning.message}
   629→              </div>
   630→            ))}
   631→          </>
   632→        )}
   633→      </CardBody>
   634→    </Card>
   635→  );
   636→}
