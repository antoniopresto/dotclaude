   532→function generateTrainRow(
   533→  trainId: string,
   534→  fleetType: FleetType,
   535→  index: number,
   536→  now: Date
   537→): TrainGridRow {
   538→  // PM1 threshold: 50,000 km
   539→  const pm1Threshold = 50000;
   540→
   541→  // Randomize mileage between 30,000 and 52,000 km for realistic distribution
   542→  const currentMileage = 30000 + Math.floor(Math.random() * 22000);
   543→  const percentComplete = (currentMileage / pm1Threshold) * 100;
   544→
   545→  // Determine status based on percentage
   546→  let pmStatus: 'SAFE' | 'APPROACHING' | 'CRITICAL' | 'OVERDUE';
   547→  if (percentComplete >= 100) pmStatus = 'OVERDUE';
   548→  else if (percentComplete >= 95) pmStatus = 'CRITICAL';
   549→  else if (percentComplete >= 80) pmStatus = 'APPROACHING';
   550→  else pmStatus = 'SAFE';
   551→
   552→  // Some trains have issues
   553→  const isCritical = index % 17 === 0;
   554→  const isRestricted = index % 23 === 0;
   555→  const isInMaintenance = index % 31 === 0;
   556→  const isOOS = index % 41 === 0;
   557→
   558→  // Calculate days remaining (assuming 400 km/day average)
   559→  const kmRemaining = Math.max(0, pm1Threshold - currentMileage);
   560→  const daysRemaining = Math.ceil(kmRemaining / 400);
   561→
   562→  // Fitness status
   563→  let fitnessStatus: FitnessStatus = 'FIT';
   564→  if (isOOS) fitnessStatus = 'UNFIT';
   565→  else if (isRestricted) fitnessStatus = 'RESTRICTED';
   566→
   567→  // Train status
   568→  let trainStatus: TrainStatus = 'IN_SERVICE';
   569→  if (isOOS) trainStatus = 'OUT_OF_SERVICE';
   570→  else if (isInMaintenance) trainStatus = 'IN_MAINTENANCE';
   571→
   572→  // Last service date (random within last 30 days)
   573→  const lastServiceDays = Math.floor(Math.random() * 30);
   574→  const lastService = new Date(now);
   575→  lastService.setDate(lastService.getDate() - lastServiceDays);
   576→
   577→  // Next PM date
   578→  const nextPMDate = new Date(now);
   579→  nextPMDate.setDate(nextPMDate.getDate() + daysRemaining);
   580→
   581→  // Random assignment for some trains
   582→  const hasAssignment = Math.random() > 0.3;
   583→  const routeNumber = Math.floor(Math.random() * 24) + 1;
   584→
   585→  const pmWindow: PMWindowData = {
   586→    level: 'PM1',
   587→    targetMileage: pm1Threshold,
   588→    currentMileage,
   589→    percentComplete: Math.min(100, percentComplete),
   590→    kmRemaining,
   591→    status: pmStatus,
   592→    daysRemaining,
   593→  };
   594→
   595→  return {
   596→    id: trainId,
   597→    trainId,
   598→    name: `Train ${trainId.replace('KS-', '').replace('AL-', '')}`,
   599→    fleetType,
   600→    currentMileage,
   601→    lifetimeMileage: currentMileage + Math.floor(Math.random() * 500000),
   602→    pmWindow,
   603→    fitnessStatus,
   604→    trainStatus,
   605→    nextPM: {
   606→      level: 'PM1',
   607→      daysRemaining,
   608→      estimatedDate: nextPMDate,
   609→    },
   610→    lastService,
   611→    assignment:
   612→      hasAssignment && trainStatus === 'IN_SERVICE' && fitnessStatus !== 'UNFIT'
   613→        ? {
   614→            routeId: `RT-${String(routeNumber).padStart(3, '0')}`,
   615→            routeName: `Route ${routeNumber}`,
   616→            estimatedKm: 300 + Math.floor(Math.random() * 200),
   617→          }
   618→        : undefined,
   619→    baseDepot: fleetType === 'KINKI_SHARYO' ? 'Depot A' : 'Depot B',
   620→    lastUpdated: now,
   621→  };
   622→}

   854→// Calculate train score for a route
   855→function calculateTrainScore(train: TrainPoolItem, route: Route): number {
   856→  let score = 100;
   857→
   858→  // Fitness check (30% weight)
   859→  if (route.fitnessRequirement === 'FIT' && train.fitnessStatus !== 'FIT') {
   860→    return 0; // Cannot assign
   861→  }
   862→  if (train.fitnessStatus === 'FIT') {
   863→    score += 0; // Already at max
   864→  } else {
   865→    score -= 15;
   866→  }
   867→
   868→  // Mileage balance (40% weight) - prefer lower mileage
   869→  const mileageScore = Math.max(0, 100 - (train.currentMileage / 50000) * 100);
   870→  score = score * 0.6 + mileageScore * 0.4;
   871→
   872→  // PM distance (30% weight) - prefer trains far from PM
   873→  const pmScore = Math.min(100, train.pmDaysRemaining * 3);
   874→  score = score * 0.7 + pmScore * 0.3;
   875→
   876→  return Math.round(score);
   877→}

   879→// Generate AI recommendations
   880→function generateAIRecommendations(
   881→  routes: Route[],
   882→  pool: TrainPoolItem[],
   883→): RouteAssignment[] {
   884→  const assignments: RouteAssignment[] = [];
   885→  const usedTrains = new Set<string>();
   886→
   887→  // Pre-calculate scores for all train-route combinations
   888→  for (const route of routes) {
   889→    const candidates = pool
   890→      .filter(t => !usedTrains.has(t.trainId))
   891→      .filter(
   892→        t => route.fitnessRequirement !== 'FIT' || t.fitnessStatus === 'FIT',
   893→      )
   894→      .map(t => ({
   895→        train: t,
   896→        score: calculateTrainScore(t, route),
   897→      }))
   898→      .filter(c => c.score > 0)
   899→      .sort((a, b) => b.score - a.score);
   900→
   901→    if (candidates.length === 0) {
   902→      assignments.push({
   903→        routeId: route.id,
   904→        status: 'UNASSIGNED',
   905→      });
   906→      continue;
   907→    }
   908→
   909→    const best = candidates[0];
   910→    const alternatives = candidates.slice(1, 4).map(c => ({
   911→      trainId: c.train.trainId,
   912→      score: c.score,
   913→      reason:
   914→        c.train.currentMileage > best.train.currentMileage
   915→          ? `Higher mileage (+${(c.train.currentMileage - best.train.currentMileage).toLocaleString()} km)`
   916→          : c.train.fleetType !== best.train.fleetType
   917→            ? 'Different fleet'
   918→            : 'Similar performance',
   919→    }));
   920→
   921→    // Generate reasons for the recommendation
   922→    const reasons: string[] = [];
   923→    if (best.train.currentMileage < 40000) {
   924→      reasons.push('Lowest mileage in available pool');
   925→    }
   926→    if (best.train.pmDaysRemaining > 14) {
   927→      reasons.push(`PM1 not due for ${best.train.pmDaysRemaining} days`);
   928→    }
   929→    if (best.train.fitnessStatus === 'FIT') {
   930→      reasons.push('Matches route fitness requirement');
   931→    }
   932→
   933→    usedTrains.add(best.train.trainId);
   934→
   935→    // Randomly decide if already assigned or AI suggested
   936→    const isAssigned = Math.random() > 0.4;
   937→
   938→    assignments.push({
   939→      routeId: route.id,
   940→      trainId: best.train.trainId,
   941→      trainScore: best.score,
   942→      status: isAssigned ? 'ASSIGNED' : 'AI_SUGGESTED',
   943→      aiRecommendation: {
   944→        trainId: best.train.trainId,
   945→        score: best.score,
   946→        reasons,
   947→        alternatives,
   948→      },
   949→    });
   950→  }
   951→
   952→  return assignments;
   953→}
