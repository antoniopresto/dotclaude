   154→```typescript
   155→import { useCallback } from 'react';
   156→import { useShallow } from 'zustand/shallow';
   157→import { useRootStore } from '~/store';
   158→import {
   159→  getPointsProducts,
   160→  clearPointsProducts,
   161→  getPointsBalance,
   162→  getParameterPointsProgram,
   163→} from '../services';
   164→import {
   165→  PointsProductsStatus,
   166→  PointsBalanceStatus,
   167→  ParameterPointsProgramStatus,
   168→} from '../models';
   169→
   170→export function usePoints() {
   171→  // Products from products store
   172→  const pointsProducts = useRootStore(useShallow((state) => state.pointsProducts));
   173→  const { pointsProductsActions } = useRootStore.getState();
   174→
   175→  // Balance from extract store (pointsBalance)
   176→  const pointsBalance = useRootStore(useShallow((state) => state.pointsBalance));
   177→
   178→  // Parameterization from parameter store
   179→  const parameterPointsProgram = useRootStore(
   180→    useShallow((state) => state.parameterPointsProgram)
   181→  );
   182→
   183→  const requestInitialData = useCallback(async () => {
   184→    await Promise.all([
   185→      getPointsBalance(),
   186→      getPointsProducts(),
   187→      getParameterPointsProgram(),
   188→    ]);
   189→  }, []);
   190→
   191→  const requestPointsBalance = useCallback(async () => {
   192→    await getPointsBalance();
   193→  }, []);
   194→
   195→  const requestPointsProducts = useCallback(async () => {
   196→    await getPointsProducts();
   197→  }, []);
   198→
   199→  const requestParameterization = useCallback(async () => {
   200→    await getParameterPointsProgram();
   201→  }, []);
   202→
   203→  const clear = useCallback(() => {
   204→    clearPointsProducts();
   205→  }, []);
   206→
   207→  return {
   208→    // Balance (from extract store - pointsBalance)
   209→    points: pointsBalance.points,
   210→    isLoadingBalance: pointsBalance.status === PointsBalanceStatus.LOADING,
   211→    isErrorBalance: pointsBalance.status === PointsBalanceStatus.ERROR,
   212→
   213→    // Products (from products store)
   214→    products: pointsProducts.products,
   215→    isLoadingProducts: pointsProducts.status === PointsProductsStatus.LOADING,
   216→    isErrorProducts: pointsProducts.status === PointsProductsStatus.ERROR,
   217→
   218→    // Parameterization (from parameter store)
   219→    pointsQuantity: parameterPointsProgram.pointsQuantity,
   220→    conversionFactor: parameterPointsProgram.conversionFactor,
   221→    isLoadingParameterization:
   222→      parameterPointsProgram.status === ParameterPointsProgramStatus.LOADING,
   223→    isErrorParameterization:
   224→      parameterPointsProgram.status === ParameterPointsProgramStatus.ERROR,
   225→
   226→    // Actions
   227→    requestPointsBalance,
   228→    requestPointsProducts,
   229→    requestParameterization,
   230→    requestInitialData,
   231→    clear,
   232→  };
   233→}
   234→```

   257→
   258→```typescript
   259→import { useCallback } from 'react';
   260→import { useShallow } from 'zustand/shallow';
   261→import { useRootStore } from '~/store';
   262→import { getMarketplace, clearMarketplace } from '../services';
   263→import { MarketplaceStatus } from '../models';
   264→
   265→export function useMarketplace() {
   266→  const marketplace = useRootStore(useShallow((state) => state.marketplace));
   267→
   268→  const requestGetMarketplace = useCallback(
   269→    async (params: Record<string, string>) => {
   270→      await getMarketplace(params);
   271→    },
   272→    []
   273→  );
   274→
   275→  const clear = useCallback(() => {
   276→    clearMarketplace();
   277→  }, []);
   278→
   279→  return {
   280→    data: marketplace.data,
   281→    isLoading: marketplace.status === MarketplaceStatus.LOADING,
   282→    isError: marketplace.status === MarketplaceStatus.ERROR,
   283→    requestGetMarketplace,
   284→    clear,
   285→  };
   286→}
   287→```
