   746→            e.preventDefault();

   751→        function changeZoom(factor, centerX, centerY) {
   752→            const oldZoom = zoom;
   753→            zoom = Math.max(0.1, Math.min(5, zoom * factor));
   754→
   755→            if (centerX !== undefined && centerY !== undefined) {
   756→                const container = document.getElementById('canvasContainer');
   757→                const rect = container.getBoundingClientRect();
   758→                const mouseX = centerX - rect.left;
   759→                const mouseY = centerY - rect.top;
   760→
   761→                panOffset.x = mouseX - (mouseX - panOffset.x) * (zoom / oldZoom);
   762→                panOffset.y = mouseY - (mouseY - panOffset.y) * (zoom / oldZoom);
   763→            }
   764→
   765→            updateCamera();
   766→            updateUI();
   767→            render();
   768→        }

   776→        function onMouseMove(e) {
   777→            if (!isDragging) return;
   778→
   779→            const dx = e.clientX - lastMousePos.x;
   780→            const dy = e.clientY - lastMousePos.y;
   781→
   782→            panOffset.x += dx;
   783→            panOffset.y += dy;
   784→
   785→            lastMousePos = { x: e.clientX, y: e.clientY };
   786→
   787→            updateCamera();
   788→            updateUI();
   789→            render();
   790→        }
