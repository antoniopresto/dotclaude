   414→        // ============================================
   415→        // DATA GENERATION
   416→        // ============================================
   417→        function generateData() {
   418→            // Generate trains
   419→            state.trains = Array.from({ length: CONFIG.TRAINS }, (_, i) => ({
   420→                id: `T${String(i + 1).padStart(3, '0')}`,
   421→                index: i
   422→            }));
   423→
   424→            // Generate tasks matching the image pattern
   425→            state.tasks = [];
   426→            const seededRandom = createSeededRandom(42);
   427→
   428→            state.trains.forEach((train, trainIndex) => {
   429→                // Generate 3-8 tasks per train
   430→                const taskCount = 3 + Math.floor(seededRandom() * 6);
   431→
   432→                for (let t = 0; t < taskCount; t++) {
   433→                    const startDay = Math.floor(seededRandom() * 140);
   434→                    const duration = 3 + Math.floor(seededRandom() * 12);
   435→
   436→                    // Random type distribution matching the image
   437→                    const typeRoll = seededRandom();
   438→                    let type, color;
   439→                    if (typeRoll < 0.35) {
   440→                        type = 'inspection';
   441→                        color = 'gray';
   442→                    } else if (typeRoll < 0.70) {
   443→                        type = 'maintenance';
   444→                        color = 'blue';
   445→                    } else {
   446→                        type = 'corrective';
   447→                        color = 'orange';
   448→                    }
   449→
   450→                    // Some tasks have conflicts (5%)
   451→                    const hasConflict = seededRandom() < 0.05;
   452→
   453→                    state.tasks.push({
   454→                        id: `${train.id}-${t}`,
   455→                        trainId: train.id,
   456→                        trainIndex,
   457→                        startDay,
   458→                        duration,
   459→                        type,
   460→                        color,
   461→                        hasConflict
   462→                    });
   463→                }
   464→            });
   465→        }
   466→
   467→        function createSeededRandom(seed) {
   468→            return function() {
   469→                seed = (seed * 1103515245 + 12345) & 0x7fffffff;
   470→                return seed / 0x7fffffff;
   471→            };
   472→        }

   477→        function initDeck() {

   481→            canvas.width = container.clientWidth;

   521→        // ============================================
   522→        // RENDER LAYERS
   523→        // ============================================
   524→        function renderLayers() {
   525→            const zoom = Math.pow(2, state.viewState.zoom);
   526→            const dayWidth = CONFIG.DAY_WIDTH * zoom;
   527→            const rowHeight = CONFIG.ROW_HEIGHT;
   528→
   529→            // Row backgrounds
   530→            const rowBackgrounds = state.trains.map((train, i) => ({
   531→                polygon: [
   532→                    [0, i * rowHeight],
   533→                    [CONFIG.VISIBLE_DAYS * CONFIG.DAY_WIDTH, i * rowHeight],
   534→                    [CONFIG.VISIBLE_DAYS * CONFIG.DAY_WIDTH, (i + 1) * rowHeight],
   535→                    [0, (i + 1) * rowHeight]
   536→                ],
   537→                color: i % 2 === 0 ? COLORS.rowEven : COLORS.rowOdd
   538→            }));
   539→
   540→            // Task rectangles
   541→            const taskRects = state.tasks.map(task => {
   542→                const x = task.startDay * CONFIG.DAY_WIDTH;
   543→                const y = task.trainIndex * rowHeight + 2;
   544→                const width = task.duration * CONFIG.DAY_WIDTH;
   545→                const height = rowHeight - 4;
   546→
   547→                return {
   548→                    polygon: [
   549→                        [x, y],
   550→                        [x + width, y],
   551→                        [x + width, y + height],
   552→                        [x, y + height]
   553→                    ],
   554→                    color: COLORS[task.color],
   555→                    hasConflict: task.hasConflict
   556→                };
   557→            });
   558→
   559→            // Conflict borders
   560→            const conflictTasks = state.tasks.filter(t => t.hasConflict).map(task => {
   561→                const x = task.startDay * CONFIG.DAY_WIDTH;
   562→                const y = task.trainIndex * rowHeight + 2;
   563→                const width = task.duration * CONFIG.DAY_WIDTH;
   564→                const height = rowHeight - 4;
   565→
   566→                return {
   567→                    polygon: [
   568→                        [x, y],
   569→                        [x + width, y],
   570→                        [x + width, y + height],
   571→                        [x, y + height]
   572→                    ]
   573→                };
   574→            });
   575→
   576→            const layers = [
   577→                // Row backgrounds
   578→                new deck.PolygonLayer({
   579→                    id: 'row-backgrounds',
   580→                    data: rowBackgrounds,
   581→                    getPolygon: d => d.polygon,
   582→                    getFillColor: d => d.color,
   583→                    stroked: false,
   584→                    filled: true
   585→                }),
   586→                // Task blocks
   587→                new deck.PolygonLayer({
   588→                    id: 'task-blocks',
   589→                    data: taskRects,
   590→                    getPolygon: d => d.polygon,
   591→                    getFillColor: d => d.color,
   592→                    stroked: false,
   593→                    filled: true,
   594→                    pickable: true
   595→                }),
   596→                // Conflict borders
   597→                new deck.PolygonLayer({
   598→                    id: 'conflict-borders',
   599→                    data: conflictTasks,
   600→                    getPolygon: d => d.polygon,
   601→                    getFillColor: [0, 0, 0, 0],
   602→                    getLineColor: COLORS.conflict,
   603→                    getLineWidth: 2,
   604→                    stroked: true,
   605→                    filled: false,
   606→                    lineWidthUnits: 'pixels'
   607→                })
   608→            ];
   609→
   610→            state.deckInstance.setProps({ layers });
   611→        }

   643→        function renderTimeline() {
   644→            const container = document.getElementById('timeline-header');
   645→            const canvas = document.getElementById('deck-canvas');
   646→            const zoom = Math.pow(2, state.viewState.zoom);
   647→            const viewportWidth = canvas.width;
   648→
   649→            // Calculate world coordinates visible on canvas
   650→            const targetX = state.viewState.target[0];
   651→            const worldLeft = (targetX - viewportWidth / 2) / zoom;
   652→            const worldRight = (targetX + viewportWidth / 2) / zoom;
   653→
   654→            // Convert world X to days
   655→            const startDay = Math.floor(worldLeft / CONFIG.DAY_WIDTH);
   656→            const endDay = Math.ceil(worldRight / CONFIG.DAY_WIDTH);
   657→
   658→            // Determine interval based on zoom
   659→            let interval = 7; // Weekly by default
   660→            if (zoom < 0.3) interval = 30;
   661→            else if (zoom < 0.5) interval = 14;
   662→            else if (zoom > 2) interval = 1;
   663→
   664→            let html = '';
   665→            const baseDate = new Date(CONFIG.START_DATE);
   666→
   667→            for (let day = Math.max(0, Math.floor(startDay / interval) * interval);
   668→                 day <= Math.min(CONFIG.VISIBLE_DAYS, endDay + interval);
   669→                 day += interval) {
   670→                const date = new Date(baseDate);
   671→                date.setDate(date.getDate() + day);
   672→
   673→                // Calculate screen position
   674→                const worldX = day * CONFIG.DAY_WIDTH;
   675→                const screenX = (worldX - worldLeft) * zoom;
   676→
   677→                if (screenX >= -100 && screenX < viewportWidth + 100) {
