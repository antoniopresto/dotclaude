    70→        // GENERATE TASK DATA
    71→        // ============================================
    72→        function generateTaskData() {
    73→            const tasks = [];
    74→
    75→            // Create diagonal pattern matching the original image exactly
    76→            // The original shows continuous diagonal bands across both years
    77→            for (let trainIndex = 0; trainIndex < CONFIG.TRAINS_COUNT; trainIndex++) {
    78→                const trainId = `T${String(trainIndex + 1).padStart(3, '0')}`;
    79→
    80→                // First set of tasks - starts from T001 at month 0 and progresses diagonally
    81→                // Each train is offset by ~0.48 months (about 2 weeks per train)
    82→                const startMonth1 = trainIndex * 0.48;
    83→
    84→                // Show tasks if they fall within the 24-month view
    85→                if (startMonth1 < 24) {
    86→                    tasks.push({
    87→                        trainId,
    88→                        trainIndex,
    89→                        blueStart: startMonth1,
    90→                        blueDuration: 2.2,  // More compact blocks
    91→                        orangeStart: startMonth1 + 1.8,
    92→                        orangeDuration: 0.8,  // Smaller orange blocks
    93→                        set: 1
    94→                    });
    95→                }
    96→
    97→                // Second set of tasks - repeats 12 months later for each train
    98→                const startMonth2 = startMonth1 + 12;
    99→
   100→                // Show if within view
   101→                if (startMonth2 < 24) {
   102→                    tasks.push({
   103→                        trainId,
   104→                        trainIndex,
   105→                        blueStart: startMonth2,
   106→                        blueDuration: 2.2,
   107→                        orangeStart: startMonth2 + 1.8,
   108→                        orangeDuration: 0.8,
   109→                        set: 2
   110→                    });
   111→                }
   112→            }
   113→
   114→            return tasks;
   115→        }

   118→        // GENERATE MANPOWER DATA
   119→        // ============================================
   120→        function generateManpowerData() {
   121→            const data = [];
   122→            // Use seeded pseudo-random for consistent results matching the original
   123→            let seed = 12345;
   124→            const seededRandom = () => {
   125→                seed = (seed * 1103515245 + 12345) & 0x7fffffff;
   126→                return (seed / 0x7fffffff);
   127→            };
   128→
   129→            for (let month = 0; month < 24; month++) {
   130→                // Relatively uniform distribution with slight variations
   131→                const mechanical = Math.floor(55 + seededRandom() * 15);
   132→                const electrical = Math.floor(35 + seededRandom() * 10);
   133→                data.push({ mechanical, electrical });
   134→            }
   135→            return data;
   136→        }

   364→        function renderMainGrid(container, width, layout, taskData) {
   365→            const { textPrimary, textSecondary, rowEven, rowOdd, borderLight, taskBlue, taskOrange } = CONFIG.COLORS;
   366→            const { gridStartY, gridHeight, sidebarWidth, trainRowHeight, monthColumnWidth, bottomMonthY, monthHeaderHeight } = layout;

   369→

   371→            const trainIdLabel = new PIXI.Text({
   372→                text: 'Train ID',

   374→            });
   375→            trainIdLabel.rotation = -Math.PI / 2;
   376→            trainIdLabel.position.set(12, gridHeight / 2 + 30);
   377→            container.addChild(trainIdLabel);
   378→
   379→            // Draw alternating row backgrounds and train IDs
   380→            for (let i = 0; i < CONFIG.TRAINS_COUNT; i++) {
   381→                const y = i * trainRowHeight;
   382→                const rowColor = i % 2 === 0 ? rowEven : rowOdd;
   383→
   384→                // Row background
   385→                const rowBg = new PIXI.Graphics();
   386→                rowBg.rect(sidebarWidth, y, width - sidebarWidth, trainRowHeight);
   387→                rowBg.fill(rowColor);
   388→                container.addChild(rowBg);
   389→
   390→                // Row border
   391→                const rowBorder = new PIXI.Graphics();
   392→                rowBorder.rect(sidebarWidth, y + trainRowHeight - 1, width - sidebarWidth, 1);
   393→                rowBorder.fill(0xE8E8E8);
   394→                container.addChild(rowBorder);
   395→
   396→                // Train ID
   397→                const trainId = `T${String(i + 1).padStart(3, '0')}`;
   398→                const trainIdText = new PIXI.Text({
   399→                    text: trainId,
   400→                    style: { fontFamily: 'Consolas, monospace', fontSize: 9, fill: textPrimary }
   401→                });
   402→                trainIdText.position.set(20, y + 1);
   403→                container.addChild(trainIdText);
   404→            }
   405→
   406→            // Draw vertical month grid lines
   407→            for (let i = 0; i <= 24; i++) {
   408→                const x = sidebarWidth + (i * monthColumnWidth);
   409→                const lineColor = i === 12 ? 0xC0C0C0 : borderLight; // Thicker line between years
   410→                const lineWidth = i === 12 ? 2 : 1;
   411→
   412→                const gridLine = new PIXI.Graphics();
   413→                gridLine.rect(x, 0, lineWidth, gridHeight);
   414→                gridLine.fill(lineColor);
   415→                container.addChild(gridLine);
   416→            }
   417→
   418→            // Draw task blocks
   419→            taskData.forEach(task => {
   420→                const y = task.trainIndex * trainRowHeight;
   421→
   422→                // Blue task block
   423→                const blueX = sidebarWidth + (task.blueStart * monthColumnWidth);
   424→                const blueWidth = task.blueDuration * monthColumnWidth;
   425→
   426→                if (blueX >= sidebarWidth && blueX < width - 10) {
   427→                    const blueBlock = new PIXI.Graphics();
   428→                    blueBlock.rect(blueX + 2, y + 1, Math.min(blueWidth - 4, width - blueX - 10), trainRowHeight - 2);
   429→                    blueBlock.fill(taskBlue);
   430→                    container.addChild(blueBlock);
   431→                }
   432→
   433→                // Orange task block
   434→                const orangeX = sidebarWidth + (task.orangeStart * monthColumnWidth);
   435→                const orangeWidth = task.orangeDuration * monthColumnWidth;
   436→
   437→                if (orangeX >= sidebarWidth && orangeX < width - 10) {
   438→                    const orangeBlock = new PIXI.Graphics();
   439→                    orangeBlock.rect(orangeX + 2, y + 1, Math.min(orangeWidth - 4, width - orangeX - 10), trainRowHeight - 2);
   440→                    orangeBlock.fill(taskOrange);
   441→                    container.addChild(orangeBlock);
   442→                }
   443→            });
   444→
   445→            // Bottom month labels
   446→            const bottomContainer = new PIXI.Container();
   447→            bottomContainer.position.y = gridHeight;
   448→            container.addChild(bottomContainer);
   449→
   450→            for (let i = 0; i < 24; i++) {
   451→                const monthName = CONFIG.MONTHS[i % 12];
   452→                const x = sidebarWidth + (i * monthColumnWidth) + (monthColumnWidth / 2);
   453→
   454→                const monthText = new PIXI.Text({
   455→                    text: monthName,
   456→                    style: { fontFamily: 'Segoe UI', fontSize: 11, fill: textSecondary }
   457→                });
   458→                monthText.anchor.set(0.5, 0);
   459→                monthText.position.set(x, 5);
   460→                bottomContainer.addChild(monthText);
   461→            }
   462→        }
