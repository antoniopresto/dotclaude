   239→  [Types.SET_CONVERSION_FACTOR](
   240→    state: State,
   241→    { conversionFactor }: { conversionFactor: number },

   246→  [Types.SET_CONVERSION_FACTOR](
   247→    state: State,
   248→    { conversionFactor }: { conversionFactor: number },

   250→    return { ...state, conversionFactor };
   251→  },
   252→  [Types.REQUEST_PARAMETERIZATION](state: State): State {

   254→  },
   255→  [Types.SUCCESS_PARAMETERIZATION](
   256→    state: State,
   257→    {
   258→      parameterization,
   259→      pointsQuantity
   260→    }: {
   261→      parameterization: ParametrizationData[];
   262→      pointsQuantity: number;
   263→    },
   264→  ): State {
   265→
   266→    const pointsRescueItem = parameterization?.find(
   267→      item => item.category === 'pointsRescueSentence'
   268→    );
   269→
   270→    if (pointsRescueItem?.value) {
   271→      try {
   272→        let parsedValue = JSON.parse(pointsRescueItem.value);
   273→
   274→        if (typeof parsedValue === 'string') {
   275→          parsedValue = JSON.parse(parsedValue);
   276→        }
   277→
   278→        if (parsedValue && parsedValue.pointsQuantity !== undefined) {
   279→          const ptsQuantity = Number(parsedValue.pointsQuantity);
   280→
   281→          if (!isNaN(ptsQuantity)) {
   282→            pointsQuantity = ptsQuantity;
   283→          }
   284→        }
   285→      } catch (error) {
   286→        console.error('Error parsing pointsQuantity:', error);
   287→      }
   288→    }
   289→
   290→    return {
   291→      ...state,
   292→      parameterization,
   293→      pointsQuantity,
   294→      isLoadingParameterization: false,
   295→      isErrorParameterization: false,
   296→    };
   297→  },
   298→  [Types.FAILED_PARAMETERIZATION](state: State): State {
   299→    return { ...state, isLoadingParameterization: false, isErrorParameterization: true };
   300→  },
   301→});
