    43→# -----------------------------------------------------------------------------
    44→# File Comparison Strategy
    45→# -----------------------------------------------------------------------------
    46→#
    47→# Pattern matching for TypeScript is unreliable. Instead:
    48→# 1. This script does basic structural checks (file existence, unmapped files)
    49→# 2. Actual semantic comparison is done by Claude agents in PROMPT.md

    55→generate_comparison_manifest() {
    56→    local manifest_file="comparison_manifest.json"
    57→
    58→    echo "{" > "$manifest_file"
    59→    echo '  "generated_at": "'$(date -Iseconds)'",' >> "$manifest_file"
    60→    echo '  "source_dir": "'$SOURCE_DIR'",' >> "$manifest_file"
    61→    echo '  "target_dir": "'$TARGET_DIR'",' >> "$manifest_file"
    62→    echo '  "file_pairs": [' >> "$manifest_file"
    63→
    64→    local first=true
    65→    local i=0
    66→    for source_rel in "${SOURCE_FILES[@]}"; do
    67→        local target_rel="${TARGET_FILES[$i]}"
    68→        local source_file="$SOURCE_DIR/$source_rel"
    69→        local target_file="$TARGET_DIR/$target_rel"
    70→
    71→        if [ -f "$source_file" ] && [ -f "$target_file" ]; then
    72→            if [ "$first" = true ]; then
    73→                first=false
    74→            else
    75→                echo "," >> "$manifest_file"
    76→            fi
    77→            echo -n '    {"source": "'$source_rel'", "target": "'$target_rel'"}' >> "$manifest_file"
    78→        fi
    79→        i=$((i + 1))
    80→    done
    81→
    82→    echo "" >> "$manifest_file"
    83→    echo "  ]," >> "$manifest_file"
    84→    echo '  "unmapped_files": [' >> "$manifest_file"
    85→
    86→    # Add unmapped files
    87→    first=true
    88→    while IFS= read -r target_file; do
    89→        local rel_path="${target_file#$TARGET_DIR/}"
    90→        if ! is_mapped_target "$rel_path" && ! is_approved_new_file "$rel_path"; then
    91→            if [ "$first" = true ]; then
    92→                first=false
    93→            else
    94→                echo "," >> "$manifest_file"
    95→            fi
    96→            echo -n '    "'$rel_path'"' >> "$manifest_file"
    97→        fi
    98→    done < <(find "$TARGET_DIR" -name "*.ts" -o -name "*.tsx" 2>/dev/null)
    99→
   100→    echo "" >> "$manifest_file"
   101→    echo "  ]" >> "$manifest_file"
   102→    echo "}" >> "$manifest_file"
   103→
   104→    log_success "Generated comparison manifest: $manifest_file"
   105→}

   145→is_approved_new_file() {
   146→    local rel_path="$1"

   220→get_target_for_source() {
   221→    local source="$1"
   222→    local i=0
   223→    for src in "${SOURCE_FILES[@]}"; do
   224→        if [ "$src" = "$source" ]; then
   225→            echo "${TARGET_FILES[$i]}"
   226→            return 0
   227→        fi
   228→        i=$((i + 1))
   229→    done
   230→    return 1
   231→}
   232→
   233→is_mapped_target() {
   234→    local target="$1"
   235→    for tgt in "${TARGET_FILES[@]}"; do
   236→        if [ "$tgt" = "$target" ]; then
   237→            return 0
   238→        fi
   239→    done
   240→    return 1
   241→}

   243→# -----------------------------------------------------------------------------
   244→# Validation Functions (Structural Only)
   245→# -----------------------------------------------------------------------------
   246→#
   247→# Semantic comparison (interfaces, enums, functions) is done by Claude agents.
   248→# This script only does structural validation:
   249→# - File existence
   250→# - Unmapped files detection
   251→# - Generate manifest for agent comparison
   252→# -----------------------------------------------------------------------------
   253→
   254→validate_file_pair() {
   255→    local source_rel="$1"
   256→    local target_rel="$2"
   257→
   258→    local source_file="$SOURCE_DIR/$source_rel"
   259→    local target_file="$TARGET_DIR/$target_rel"
   260→
   261→    # Check if source exists
   262→    if [ ! -f "$source_file" ]; then
   263→        log_warning "Source file not found: $source_rel"
   264→        return 0
   265→    fi
   266→
   267→    # Check if target exists
   268→    if [ ! -f "$target_file" ]; then
   269→        log_error "Target file missing: $target_rel (expected from $source_rel)"
   270→        return 1
   271→    fi
   272→
   273→    # Basic check: files exist and are readable
   274→    log_success "File pair exists: $source_rel → $target_rel"
   275→    return 0
   276→}
   277→
   278→# -----------------------------------------------------------------------------
   279→# Check for Unmapped Files in Target (potential UNSAFE additions)
   280→# -----------------------------------------------------------------------------
   281→
   282→check_unmapped_target_files() {
   283→    echo ""
   284→    echo "Checking for unmapped files in target..."
   285→    echo ""
   286→
   287→    local unmapped_count=0
   288→
   289→    # Find all TypeScript files in target
   290→    while IFS= read -r target_file; do
   291→        local rel_path="${target_file#$TARGET_DIR/}"
   292→
   293→        # Check if this file is a mapped target or an approved new file
   294→        if is_mapped_target "$rel_path"; then
   295→            : # File is mapped, OK
   296→        elif is_approved_new_file "$rel_path"; then
   297→            log_info "New file (approved): $rel_path"
   298→        else
   299→            log_error "UNMAPPED FILE in target: $rel_path (no source equivalent, not approved)"
   300→            unmapped_count=$((unmapped_count + 1))
   301→        fi
   302→    done < <(find "$TARGET_DIR" -name "*.ts" -o -name "*.tsx" 2>/dev/null)
   303→
   304→    if [ $unmapped_count -gt 0 ]; then
   305→        echo ""
   306→        echo "Found $unmapped_count unmapped file(s). These may be UNSAFE additions."
   307→        echo "Either add them to APPROVED_NEW_FILES or delete them."
   308→    fi
   309→}
   310→
   311→
   312→# -----------------------------------------------------------------------------
   313→# Main Validation
   314→# -----------------------------------------------------------------------------
   315→
   316→echo "=============================================="
   317→echo "  Migration Validation Script"
   318→echo "=============================================="
   319→echo ""
   320→echo "Source: $SOURCE_DIR"
   321→echo "Target: $TARGET_DIR"
   322→echo ""
   323→
   324→# Check directories exist
   325→if [ ! -d "$SOURCE_DIR" ]; then
   326→    log_error "Source directory not found: $SOURCE_DIR"
   327→    exit 1

   329→
   330→if [ ! -d "$TARGET_DIR" ]; then
   331→    log_error "Target directory not found: $TARGET_DIR"
   332→    exit 1
   333→fi
   334→
   335→echo "Phase 1: Structural validation (file existence)..."
   336→echo ""
   337→
   338→# Validate each mapped file pair exists
   339→i=0
   340→for source_rel in "${SOURCE_FILES[@]}"; do
   341→    target_rel="${TARGET_FILES[$i]}"
   342→    validate_file_pair "$source_rel" "$target_rel"
   343→    i=$((i + 1))
   344→done
   345→
   346→echo ""
   347→echo "Phase 2: Checking for unmapped files..."
   348→
   349→# Check for unmapped files in target (potential UNSAFE additions)
   350→check_unmapped_target_files
   351→
   352→echo ""
   353→echo "Phase 3: Generating comparison manifest for agent validation..."
   354→
   355→# Generate manifest for Claude agent comparison
   356→generate_comparison_manifest

   360→echo "  Structural Validation Summary"

   362→

   366→    echo "Fix structural issues first:"
   367→    echo "1. Missing target files must be created"
   368→    echo "2. Unmapped files must be approved or removed"
   369→    echo ""
   370→    exit 1
   371→fi
   372→
   373→if [ $WARNINGS_FOUND -gt 0 ]; then
   374→    echo -e "${YELLOW}STRUCTURAL OK with $WARNINGS_FOUND warning(s)${NC}"
   375→else
   376→    echo -e "${GREEN}STRUCTURAL OK: All files exist${NC}"
   377→fi
   378→
   379→echo ""
   380→echo "=============================================="
   381→echo "  Next Step: Semantic Validation"
   382→echo "=============================================="
   383→echo ""
   384→echo "Manifest generated: comparison_manifest.json"
   385→echo ""
   386→echo "Claude should now run semantic comparison using:"
   387→echo "  - 3 parallel comparison agents (interfaces, functions, state)"
   388→echo "  - 1 judge agent to evaluate SAFE vs UNSAFE"
   389→echo ""
   390→echo "See PROMPT.md for agent comparison instructions."
   391→echo ""
   392→
   393→exit 0
