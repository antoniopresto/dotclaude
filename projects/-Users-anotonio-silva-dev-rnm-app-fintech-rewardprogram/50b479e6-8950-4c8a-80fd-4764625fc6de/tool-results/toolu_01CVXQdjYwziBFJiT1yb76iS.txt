    26→## Introduction
    27→
    28→The goal of this guide is to ensure **consistency, testability and scalability** across our frontend projects.
    29→
    30→Key decisions:
    31→- **Zustand** is the **exclusive** global/shared state manager.
    32→- UI follows the **Hook Component** pattern (logic hook + dumb view).
    33→- Domain logic lives in **`/services`** with **SOLID** layering.
    34→- Cache is **typed and reusable**.
    35→
    36→---
    37→
    38→## Components & Screens — Hook Component Pattern
    39→
    40→Each screen is split into a logic hook and a render-only component.
    41→
    42→```
    43→screens/Home/
    44→├── index.ts        # logic: export useHomeScreen()
    45→├── view.tsx        # render: export default HomeScreen
    46→└── Models/
    47→    └── index.d.ts    # types (Props, State, Handlers, Return)
    48→```
    49→
    50→**Models/index.d.ts**
    51→```ts
    52→export interface HomeParams {
    53→  welcome?: string;
    54→}
    55→export interface HomeState {
    56→  greeting: string;
    57→  isLoading: boolean;
    58→}
    59→export interface HomeHandlers {
    60→  onPressCTA: () => void;
    61→}
    62→export type UseHomeReturn = HomeState & HomeHandlers;
    63→```
    64→
    65→**index.ts (logic hook)**
    66→```ts
    67→import { useCallback, useMemo, useState, useEffect } from "react";
    68→import type { HomeParams, UseHomeReturn } from "./Models";
    69→import { useUserStore } from "@/services/user/store"; // Zustand slice example
    70→
    71→export function useHomeScreen(params?: HomeParams): UseHomeReturn {
    72→  const name = useUserStore((s) => s.name);
    73→  const [isLoading, setIsLoading] = useState(true);
    74→
    75→  useEffect(() => {
    76→    const t = setTimeout(() => setIsLoading(false), 300);
    77→    return () => clearTimeout(t);
    78→  }, []);
    79→
    80→  const greeting = useMemo(
    81→    () => params?.welcome ?? `Hello, ${name || "guest"}!`,
    82→    [params?.welcome, name]
    83→  );
    84→
    85→  const onPressCTA = useCallback(() => {
    86→    // Call services, analytics, navigation, etc.
    87→  }, []);
    88→
    89→  return { greeting, isLoading, onPressCTA };
    90→}
    91→```
    92→
    93→**view.tsx (render)**
    94→```tsx
    95→import React from "react";
    96→import { View } from "react-native";
    97→import { Text, Button } from "@semparar/design-system-app";
    98→import { useHomeScreen } from "./index";
    99→import type { HomeParams } from "./Models";
   100→
   101→type Props = HomeParams;
   102→
   103→export default function HomeScreen(props: Props) {
   104→  const { greeting, isLoading, onPressCTA } = useHomeScreen(props);
   105→
   106→  if (isLoading) {
   107→    return (
   108→      <View>
   109→        <Text>Loading...</Text>
   110→      </View>
   111→    );
   112→  }
   113→
   114→  return (
   115→    <View>
   116→      <Text>{greeting}</Text>
   117→      <Button onPress={onPressCTA} text="Continue" />
   118→    </View>
   119→  );
   120→}
   121→```

   125→## Atomic Components — Hook Component Pattern
   126→
   127→```
   128→atomic/atoms/Button/
   129→├── index.ts        # logic: export useButton()
   130→├── view.tsx        # render: export default Button
   131→└── Models/
   132→    └── index.d.ts
   133→```
   134→
   135→**Models/index.d.ts**
   136→```ts
   137→export interface ButtonProps {
   138→  text: string;
   139→  disabled?: boolean;
   140→  onPress?: () => void;
   141→}
   142→export interface ButtonState {
   143→  pressed: boolean;
   144→}
   145→export interface ButtonHandlers {
   146→  handlePress: () => void;
   147→}
   148→export type UseButtonReturn = ButtonState & ButtonHandlers;
   149→```
   150→
   151→**index.ts (logic hook)**
   152→```ts
   153→import { useCallback, useState } from "react";
   154→import type { ButtonProps, UseButtonReturn } from "./Models";
   155→
   156→export function useButton(props: ButtonProps): UseButtonReturn {
   157→  const [pressed, setPressed] = useState(false);
   158→
   159→  const handlePress = useCallback(() => {
   160→    setPressed(true);
   161→    props.onPress?.();
   162→    setTimeout(() => setPressed(false), 150);
   163→  }, [props.onPress]);
   164→
   165→  return { pressed, handlePress };
   166→}
   167→```
   168→
   169→**view.tsx (render)**
   170→```tsx
   171→import React from "react";
   172→import { Pressable, View } from "react-native";
   173→import { Text } from "@semparar/design-system-app";
   174→import { useButton } from "./index";
   175→import type { ButtonProps } from "./Models";
   176→
   177→export default function Button(props: ButtonProps) {
   178→  const { pressed, handlePress } = useButton(props);
   179→
   180→  return (
   181→    <Pressable disabled={props.disabled} onPress={handlePress}>
   182→      <View>
   183→        <Text>{props.text}{pressed ? " •" : ""}</Text>
   184→      </View>
   185→    </Pressable>
   186→  );
   187→}
   188→```

   192→## Project Organization
   193→
   194→```
   195→src/
   196→├── @types/            # Global TS types
   197→├── assets/            # Images, fonts, icons
   198→├── atomic/            # UI components (Hook Component pattern)
   199→│   ├── atoms/
   200→│   ├── molecules/
   201→│   └── organisms/
   202→├── modules/           # Microfrontends shared and native modules
   203→├── screens/           # Page-level (Hook Component pattern)
   204→├── services/          # Domain logic (SOLID layout)
   205→│   ├── api/           # HTTP layer (Axios/Fetch)
   206→│   ├── class/         # Class with logical requests
   207→│   ├── hooks/         # Specific service hooks
   208→│   ├── repository/    # Data abstraction & mapping
   209→│   ├── services/      # Business orchestration
   210→│   ├── storage/       # Storage orchestration with cache or not
   211→│   ├── store/         # Zustand slices (only)
   212→│   ├── models/        # Domain types & contracts
   213→│   ├── utils/         # Reusable service helps
   214→│   └── index.ts       # Public exports
   215→├── hooks/             # Cross-cutting React hooks
   216→├── routes/            # Navigation config
   217→├── settings/          # Env & global config (axios, etc.)
   218→└── utils/             # General-purpose helpers
   219→```
   220→
   221→> UI **never** calls API directly. All effects & side effects live in the **logic hooks** (which call `/services`).

   224→
   225→## Services Architecture
   226→
   227→**Flow:** `UI → services → repository → api` and `UI → services → store (Zustand)`
   228→
   229→- **api/**: request clients; no domain logic.
   230→- **class/**: class with logical requests.
   231→- **hooks/**: specific service hooks.
   232→- **repository/**: maps transport to domain, aggregates sources.
   233→- **services/**: orchestration and business rules.
   234→- **storage/**: storage orchestration with cache or not.
   235→- **store/**: Zustand slices (single source of app state).
   236→- **models/**: types, DTOs, contracts.
   237→- **utils/**: reusable service helpers.
   238→
   239→Principles:
   240→- Single responsibility per folder/file.
   241→- No circular dependencies.
   242→- Public APIs exported via `services/index.ts` per domain.
   243→

   245→
   246→## State Management (Zustand-only)
   247→
   248→- All global/shared state is **Zustand**.
   249→- Slices live under `/services/<domain>/store/`.
   250→- Use `persist` (with `createJSONStorage`) only when needed.
   251→- Keep setters pure; side effects go in hooks or services.
   252→
   253→**Slice example**
   254→```ts
   255→import type { StateCreator } from "zustand";
   256→import type { AuthActions, AuthState } from "../models/store";
   257→
   258→export type AuthSlice = {
   259→    auth: AuthState;
   260→    authActions: AuthActions
   261→};
   262→
   263→const initialAuth = {
   264→    user: null,
   265→    loading: false,
   266→    error: undefined,
   267→}
   268→
   269→export const createAuthSlice: StateCreator<AuthSlice, [], [], AuthSlice> = (set) => ({
   270→    auth: initialAuth,
   271→    authActions: {
   272→        setLoading: (v) => set((s) => ({ auth: { ...s.auth, isLoading: v } })),
   273→        setUser: (v) => set((s) => ({ auth: { ...s.auth, user: v } })),
   274→        setError: (v) => set((s) => ({ auth: { ...s.auth, error: v } })),
   275→        reset: () => set({ auth: initialAuth }),
   276→    }
   277→});
   278→```
   279→
   280→**Root Store**
   281→
   282→- Applies to ALL Sem Parar frontend apps — including SuperApp.
   283→
   284→```ts
   285→import { create } from "zustand";
   286→import { devtools, persist, createJSONStorage } from "zustand/middleware";
   287→import { immer } from "zustand/middleware/immer";
   288→
   289→import { createUserSlice, type UserSlice } from "@semparar/mf-app-user/src/newArch/user";
   290→import { createAuthSlice, type AuthSlice } from "@semparar/mf-app-yourmf";
   291→
   292→import storage from "~/redux/store/storage";
   293→
   294→/** Add your slice Type */
   295→export type RootStore = UserSlice & AuthSlice;
   296→
   297→export const useRootStore = create<RootStore>()(
   298→  devtools(
   299→    persist(
   300→      immer((...a) => ({
   301→        ...createUserSlice(...a),
   302→        ...createAuthSlice(...a), // Add your slice here
   303→      })),
   304→      {
   305→        name: "root-storage",
   306→        storage: createJSONStorage(() => storage),
   307→        partialize: (s) => ({
   308→          user: s.user,
   309→          auth: s.auth, // Add your store data
   310→        }),
   311→        merge: (persisted, current) => ({ ...current, ...(persisted as Partial<RootStore>) }),
   312→      }
   313→    ),
   314→    { name: "RootStore" }
   315→  )
   316→);
   317→```
   318→
   319→- For Use on Class.
   320→
   321→```ts
   322→class Auth {
   323→  private readonly setAuth;
   324→
   325→  constructor () {
   326→    this.setUser = useRootStore.getState().authActions.setUser;
   327→  }
   328→
   329→  function doSomething(data) {
   330→    this.setUser({ ...data });
   331→  }
   332→}
   333→```
   334→
   335→- For Use on Component/Screen.
   336→- UI never imports the store directly — uses selectors
   337→
   338→```ts
   339→const loading = useRootStore((s) => s.user.isLoading);
   340→const token = useRootStore((s) => s.auth.token);
   341→```

   362→## Code Standards
   363→
   364→**Language**: English only.  
   365→**Naming**: `camelCase` (vars/functions), `PascalCase` (components/classes), `UPPER_SNAKE_CASE` (constants).
   366→
   367→**Explanatory variables**
   368→```js
   369→const onlyNumbersRegex = /^\d+$/;
   370→const getNumberValidation = (message) => (value) => !onlyNumbersRegex.test(value) && message;
   371→const isNumber = getNumberValidation("error message");
   372→isNumber(1);
   373→```
   374→
   375→**Semantic constants**
   376→```js
   377→const DAY_IN_MILLISECONDS = 86400000;
   378→setTimeout(doSomething, DAY_IN_MILLISECONDS);
   379→```
   380→
   381→**Library workarounds with link**
   382→```js
   383→// Refer to issue https://github.com/ngokevin/react-file-reader-input/issues/11#issuecomment-363484861
   384→// eslint-disable-next-line no-param-reassign
   385→event.target.value = "";
   386→```
   387→
   388→**JSDoc**
   389→```js
   390→/**
   391→ * Truncates an event name to the max length.
   392→ * @param {*} eventName
   393→ * @param {*} maxLength
   394→ */
   395→```
   396→
   397→**Hooks: multi-line deps & IIFE**
   398→```ts
   399→useEffect(() => {
   400→  (async () => {
   401→    await CustomMath.randomSum();
   402→    setSum(42);
   403→  })();
   404→}, [
   405→  setSum,
   406→]);
   407→```
   408→
   409→**Extract conditions**
   410→```ts
   411→const hasSelectedOffer = !!(selectedProduct || selectedLeasingOffer);
   412→if (!hasSelectedOffer && !individualOffer) return;
   413→```
   414→
   415→**Optional chaining**
   416→```ts
   417→const answers = formikForm?.values?.answers;
   418→```
   419→
   420→**JSX indentation**
   421→```tsx
   422→{
   423→  !isEnd && (
   424→    <Button text="CTA" onPress={handleNavigate} />
   425→  )
   426→}
   427→```
   428→
   429→**Early returns**
   430→```ts
   431→if (!data) return null;
   432→return <View />;
   433→```
   434→
   435→**Nullish coalescing**
   436→```ts
   437→return data ?? 0;
   438→```

   448→## Cache Pattern
   449→
   450→Keep cache typed, serializable, and domain.
   451→
   452→```ts
   453→import storage from "~/redux/store/storage";
   454→
   455→export interface CachedData<T> {
   456→  timestamp: number;
   457→  data: T;
   458→}
   459→
   460→export interface StorageContract<T> {
   461→  get(): Promise<CachedData<T> | null>;
   462→  save(data: T): Promise<void>;
   463→  remove(): Promise<void>;
   464→}
   465→
   466→export class GenericStorage<T> implements StorageContract<T> {
   467→  constructor(private readonly key: string) {}
   468→
   469→  async get(): Promise<CachedData<T> | null> {
   470→    const json = await storage.getItem(this.key);
   471→    if (!json) return null;
   472→    return JSON.parse(json) as CachedData<T>;
   473→  }
   474→
   475→  async save(data: T): Promise<void> {
   476→    const payload: CachedData<T> = { timestamp: Date.now(), data };
   477→    await storage.setItem(this.key, JSON.stringify(payload));
   478→  }
   479→
   480→  async remove(): Promise<void> {
   481→    await storage.removeItem(this.key);
   482→  }
   483→}
   484→```
   485→
   486→**Guidelines**
   487→- One storage class per module/feature.
   488→- Types/interfaces live in `Models/` where appropriate.
   489→- Use `timestamp` for freshness/expiry strategies.
   490→- Never persist secrets/PII outside secure storage.
   491→- Keep payloads small and JSON-serializable.
