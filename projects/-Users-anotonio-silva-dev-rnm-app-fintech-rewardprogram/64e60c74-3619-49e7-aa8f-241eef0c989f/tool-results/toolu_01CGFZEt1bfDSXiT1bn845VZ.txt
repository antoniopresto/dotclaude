    27→## Introduction
    28→
    29→The goal of this guide is to ensure **consistency, testability and scalability** across our frontend projects.
    30→
    31→Key decisions:
    32→
    33→- **Zustand** is the **exclusive** global/shared state manager.
    34→- UI follows the **Hook Component** pattern (logic hook + dumb view).
    35→- Domain logic lives in **`/services`** with **SOLID** layering.
    36→- Cache is **typed and reusable**.
    37→
    38→---
    39→
    40→## Components & Screens — Hook Component Pattern
    41→
    42→Each screen is split into a logic hook and a render-only component.
    43→
    44→```
    45→screens/Home/
    46→├── index.ts        # logic: export useHomeScreen()
    47→├── view.tsx        # render: export default HomeScreen
    48→└── Models/
    49→    └── index.d.ts    # types (Props, State, Handlers, Return)
    50→```
    51→
    52→**Models/index.d.ts**
    53→
    54→```ts
    55→export interface HomeParams {
    56→  welcome?: string;
    57→}
    58→export interface HomeState {
    59→  greeting: string;
    60→  isLoading: boolean;
    61→}
    62→export interface HomeHandlers {
    63→  onPressCTA: () => void;
    64→}
    65→export type UseHomeReturn = HomeState & HomeHandlers;
    66→```
    67→
    68→**index.ts (logic hook)**
    69→
    70→```ts
    71→import { useCallback, useMemo, useState, useEffect } from 'react';
    72→import type { HomeParams, UseHomeReturn } from './Models';
    73→import { useUserStore } from '@/services/user/store'; // Zustand slice example
    74→
    75→export function useHomeScreen(params?: HomeParams): UseHomeReturn {
    76→  const name = useUserStore(s => s.name);
    77→  const [isLoading, setIsLoading] = useState(true);
    78→
    79→  useEffect(() => {
    80→    const t = setTimeout(() => setIsLoading(false), 300);
    81→    return () => clearTimeout(t);
    82→  }, []);
    83→
    84→  const greeting = useMemo(
    85→    () => params?.welcome ?? `Hello, ${name || 'guest'}!`,
    86→    [
    87→      params?.welcome,
    88→      name,
    89→    ],
    90→  );
    91→
    92→  const onPressCTA = useCallback(() => {
    93→    // Call services, analytics, navigation, etc.
    94→  }, []);
    95→
    96→  return {
    97→    greeting,
    98→    isLoading,
    99→    onPressCTA,
   100→  };
   101→}
   102→```
   103→
   104→**view.tsx (render)**
   105→
   106→```tsx
   107→import React from 'react';
   108→import { View } from 'react-native';
   109→import { Text, Button } from '@semparar/design-system-app';
   110→import { useHomeScreen } from './index';
   111→import type { HomeParams } from './Models';
   112→
   113→type Props = HomeParams;
   114→
   115→export default function HomeScreen(props: Props) {
   116→  const { greeting, isLoading, onPressCTA } = useHomeScreen(props);
   117→
   118→  if (isLoading) {
   119→    return (
   120→      <View>
   121→        <Text>Loading...</Text>
   122→      </View>
   123→    );
   124→  }
   125→
   126→  return (
   127→    <View>
   128→      <Text>{greeting}</Text>
   129→      <Button onPress={onPressCTA} text="Continue" />
   130→    </View>
   131→  );
   132→}
   133→```

   137→## Atomic Components — Hook Component Pattern
   138→
   139→```
   140→atomic/atoms/Button/
   141→├── index.ts        # logic: export useButton()
   142→├── view.tsx        # render: export default Button
   143→└── Models/
   144→    └── index.d.ts
   145→```
   146→
   147→**Models/index.d.ts**
   148→
   149→```ts
   150→export interface ButtonProps {
   151→  text: string;
   152→  disabled?: boolean;
   153→  onPress?: () => void;
   154→}
   155→export interface ButtonState {
   156→  pressed: boolean;
   157→}
   158→export interface ButtonHandlers {
   159→  handlePress: () => void;
   160→}
   161→export type UseButtonReturn = ButtonState & ButtonHandlers;
   162→```
   163→
   164→**index.ts (logic hook)**
   165→
   166→```ts
   167→import { useCallback, useState } from 'react';
   168→import type { ButtonProps, UseButtonReturn } from './Models';
   169→
   170→export function useButton(props: ButtonProps): UseButtonReturn {
   171→  const [pressed, setPressed] = useState(false);
   172→
   173→  const handlePress = useCallback(() => {
   174→    setPressed(true);
   175→    props.onPress?.();
   176→    setTimeout(() => setPressed(false), 150);
   177→  }, [props.onPress]);
   178→
   179→  return {
   180→    pressed,
   181→    handlePress,
   182→  };
   183→}
   184→```
   185→
   186→**view.tsx (render)**
   187→
   188→```tsx
   189→import React from 'react';
   190→import { Pressable, View } from 'react-native';
   191→import { Text } from '@semparar/design-system-app';
   192→import { useButton } from './index';
   193→import type { ButtonProps } from './Models';
   194→
   195→export default function Button(props: ButtonProps) {
   196→  const { pressed, handlePress } = useButton(props);
   197→
   198→  return (
   199→    <Pressable disabled={props.disabled} onPress={handlePress}>
   200→      <View>
   201→        <Text>
   202→          {props.text}
   203→          {pressed ? ' •' : ''}
   204→        </Text>
   205→      </View>
   206→    </Pressable>
   207→  );
   208→}
   209→```

   213→## Project Organization
   214→
   215→```
   216→src/
   217→├── @types/            # Global TS types
   218→├── assets/            # Images, fonts, icons
   219→├── atomic/            # UI components (Hook Component pattern)
   220→│   ├── atoms/
   221→│   ├── molecules/
   222→│   └── organisms/
   223→├── modules/           # Microfrontends shared and native modules
   224→├── screens/           # Page-level (Hook Component pattern)
   225→├── services/          # Domain logic (SOLID layout)
   226→│   ├── api/           # HTTP layer (Axios/Fetch)
   227→│   ├── class/         # Class with logical requests
   228→│   ├── hooks/         # Specific service hooks
   229→│   ├── repository/    # Data abstraction & mapping
   230→│   ├── services/      # Business orchestration
   231→│   ├── storage/       # Storage orchestration with cache or not
   232→│   ├── store/         # Zustand slices (only)
   233→│   ├── models/        # Domain types & contracts
   234→│   ├── utils/         # Reusable service helps
   235→│   └── index.ts       # Public exports
   236→├── hooks/             # Cross-cutting React hooks
   237→├── routes/            # Navigation config
   238→├── settings/          # Env & global config (axios, etc.)
   239→└── utils/             # General-purpose helpers
   240→```
   241→
   242→> UI **never** calls API directly. All effects & side effects live in the **logic hooks** (which call `/services`).

   245→
   246→## Services Architecture
   247→
   248→**Flow:** `UI → services → repository → api` and `UI → services → store (Zustand)`
   249→
   250→- **api/**: request clients; no domain logic.
   251→- **class/**: class with logical requests.
   252→- **hooks/**: specific service hooks.
   253→- **repository/**: maps transport to domain, aggregates sources.
   254→- **services/**: orchestration and business rules.
   255→- **storage/**: storage orchestration with cache or not.
   256→- **store/**: Zustand slices (single source of app state).
   257→- **models/**: types, DTOs, contracts.
   258→- **utils/**: reusable service helpers.
   259→
   260→Principles:
   261→
   262→- Single responsibility per folder/file.
   263→- No circular dependencies.
   264→- Public APIs exported via `services/index.ts` per domain.
   265→

   268→## State Management (Zustand-only)
   269→
   270→- All global/shared state is **Zustand**.
   271→- Slices live under `/services/<domain>/store/`.
   272→- Use `persist` (with `createJSONStorage`) only when needed.
   273→- Keep setters pure; side effects go in hooks or services.
   274→
   275→**Slice example**
   276→
   277→```ts
   278→import type { StateCreator } from 'zustand';
   279→import type { AuthActions, AuthState } from '../models/store';
   280→
   281→export type AuthSlice = {
   282→  auth: AuthState;
   283→  authActions: AuthActions;
   284→};
   285→
   286→const initialAuth = {
   287→  user: null,
   288→  loading: false,
   289→  error: undefined,
   290→};
   291→
   292→export const createAuthSlice: StateCreator<
   293→  AuthSlice,
   294→  [],
   295→  [],
   296→  AuthSlice
   297→> = set => ({
   298→  auth: initialAuth,
   299→  authActions: {
   300→    setLoading: v =>
   301→      set(s => ({
   302→        auth: {
   303→          ...s.auth,
   304→          isLoading: v,
   305→        },
   306→      })),
   307→    setUser: v =>
   308→      set(s => ({
   309→        auth: {
   310→          ...s.auth,
   311→          user: v,
   312→        },
   313→      })),
   314→    setError: v =>
   315→      set(s => ({
   316→        auth: {
   317→          ...s.auth,
   318→          error: v,
   319→        },
   320→      })),
   321→    reset: () => set({ auth: initialAuth }),
   322→  },
   323→});
   324→```

   326→**Root Store**
   327→
   328→- Applies to ALL Sem Parar frontend apps — including SuperApp.
   329→
   330→```ts
   331→import { create } from 'zustand';
   332→import { devtools, persist, createJSONStorage } from 'zustand/middleware';
   333→import { immer } from 'zustand/middleware/immer';
   334→
   335→import {
   336→  createUserSlice,
   337→  type UserSlice,
   338→} from '@semparar/mf-app-user/src/newArch/user';
   339→import { createAuthSlice, type AuthSlice } from '@semparar/mf-app-yourmf';
   340→
   341→import storage from '~/redux/store/storage';
   342→
   343→/** Add your slice Type */
   344→export type RootStore = UserSlice & AuthSlice;
   345→
   346→export const useRootStore = create<RootStore>()(
   347→  devtools(
   348→    persist(
   349→      immer((...a) => ({
   350→        ...createUserSlice(...a),
   351→        ...createAuthSlice(...a), // Add your slice here
   352→      })),
   353→      {
   354→        name: 'root-storage',
   355→        storage: createJSONStorage(() => storage),
   356→        partialize: s => ({
   357→          user: s.user,
   358→          auth: s.auth, // Add your store data
   359→        }),
   360→        merge: (persisted, current) => ({
   361→          ...current,
   362→          ...(persisted as Partial<RootStore>),
   363→        }),
   364→      },
   365→    ),
   366→    { name: 'RootStore' },
   367→  ),
   368→);
   369→```
   370→
   371→- For Use on Class.
   372→
   373→```ts
   374→class Auth {
   375→  private readonly setAuth;
   376→
   377→  constructor () {
   378→    this.setUser = useRootStore.getState().authActions.setUser;
   379→  }
   380→
   381→  function doSomething(data) {
   382→    this.setUser({ ...data });
   383→  }
   384→}
   385→```
   386→
   387→- For Use on Component/Screen.
   388→- UI never imports the store directly — uses selectors
   389→
   390→```ts
   391→const loading = useRootStore(s => s.user.isLoading);
   392→const token = useRootStore(s => s.auth.token);
   393→```

   415→## Code Standards
   416→
   417→**Language**: English only.  
   418→**Naming**: `camelCase` (vars/functions), `PascalCase` (components/classes), `UPPER_SNAKE_CASE` (constants).
   419→
   420→**Explanatory variables**
   421→
   422→```js
   423→const onlyNumbersRegex = /^\d+$/;
   424→const getNumberValidation = message => value =>
   425→  !onlyNumbersRegex.test(value) && message;
   426→const isNumber = getNumberValidation('error message');
   427→isNumber(1);
   428→```
   429→
   430→**Semantic constants**
   431→
   432→```js
   433→const DAY_IN_MILLISECONDS = 86400000;
   434→setTimeout(doSomething, DAY_IN_MILLISECONDS);
   435→```
   436→
   437→**Library workarounds with link**
   438→
   439→```js
   440→// Refer to issue https://github.com/ngokevin/react-file-reader-input/issues/11#issuecomment-363484861
   441→// eslint-disable-next-line no-param-reassign
   442→event.target.value = '';
   443→```
   444→
   445→**JSDoc**
   446→
   447→```js
   448→/**
   449→ * Truncates an event name to the max length.
   450→ * @param {*} eventName
   451→ * @param {*} maxLength
   452→ */
   453→```
   454→
   455→**Hooks: multi-line deps & IIFE**
   456→
   457→```ts
   458→useEffect(() => {
   459→  (async () => {
   460→    await CustomMath.randomSum();
   461→    setSum(42);
   462→  })();
   463→}, [setSum]);
   464→```
   465→
   466→**Extract conditions**
   467→
   468→```ts
   469→const hasSelectedOffer = !!(selectedProduct || selectedLeasingOffer);
   470→if (!hasSelectedOffer && !individualOffer) return;
   471→```
   472→
   473→**Optional chaining**
   474→
   475→```ts
   476→const answers = formikForm?.values?.answers;
   477→```
   478→
   479→**JSX indentation**
   480→
   481→```tsx
   482→{
   483→  !isEnd && <Button text="CTA" onPress={handleNavigate} />;
   484→}
   485→```
   486→
   487→**Early returns**
   488→
   489→```ts
   490→if (!data) return null;
   491→return <View />;
   492→```
   493→
   494→**Nullish coalescing**
   495→
   496→```ts
   497→return data ?? 0;
   498→```

   509→## Cache Pattern
   510→
   511→Keep cache typed, serializable, and domain.
   512→
   513→```ts
   514→import storage from '~/redux/store/storage';
   515→
   516→export interface CachedData<T> {
   517→  timestamp: number;
   518→  data: T;
   519→}
   520→
   521→export interface StorageContract<T> {
   522→  get(): Promise<CachedData<T> | null>;
   523→  save(data: T): Promise<void>;
   524→  remove(): Promise<void>;
   525→}
   526→
   527→export class GenericStorage<T> implements StorageContract<T> {
   528→  constructor(private readonly key: string) {}
   529→
   530→  async get(): Promise<CachedData<T> | null> {
   531→    const json = await storage.getItem(this.key);
   532→    if (!json) return null;
   533→    return JSON.parse(json) as CachedData<T>;
   534→  }
   535→
   536→  async save(data: T): Promise<void> {
   537→    const payload: CachedData<T> = {
   538→      timestamp: Date.now(),
   539→      data,
   540→    };
   541→    await storage.setItem(this.key, JSON.stringify(payload));
   542→  }
   543→
   544→  async remove(): Promise<void> {
   545→    await storage.removeItem(this.key);
   546→  }
   547→}
   548→```
   549→
   550→**Guidelines**
   551→
   552→- One storage class per module/feature.
   553→- Types/interfaces live in `Models/` where appropriate.
   554→- Use `timestamp` for freshness/expiry strategies.
   555→- Never persist secrets/PII outside secure storage.
   556→- Keep payloads small and JSON-serializable.
