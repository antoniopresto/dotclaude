     1→import React, { FC, useEffect, useState, useLayoutEffect, useMemo, useRef } from 'react';
     2→import { NavigationProp, useNavigation, useIsFocused } from '@react-navigation/native';
     3→import { Dimensions } from 'react-native';
     4→import { connect } from 'react-redux';
     5→
     6→import { useScreenViewTracking } from '../../hooks/tracking';
     7→import { PointsActions } from '../../redux/ducks/points';
     8→import { rescueActions, RescuePayload } from '../../redux/ducks/rescue';
     9→import { useFocusEffect } from '@react-navigation/native';
    10→
    11→import theme from '../../styles/theme';
    12→import { InvoiceCashbackProps } from './models';
    13→import { PropsRootStack } from '../../routes/Models';
    14→import InvoiceCashbackSummaryView from './view';
    15→import { bindActionCreators, Dispatch } from 'redux';
    16→import { usePermission } from '../../services';
    17→import { AppReducers } from '../../redux/reducers/Models';
    18→
    19→const InvoiceCashbackSummary: FC<InvoiceCashbackProps> = ({
    20→  route,
    21→  pointsReducer,
    22→  rescueReducer,
    23→  rescueRequest,
    24→  rescueClear,
    25→}) => {
    26→  useScreenViewTracking({ route, title: 'Cashback na fatura' });
    27→  const navigation = useNavigation<NavigationProp<PropsRootStack>>();
    28→  const isFocused = useIsFocused();
    29→  const { height } = Dimensions.get('window');
    30→  const { hasRedeemed, markAsRedeemed } = usePermission();
    31→  const alreadyMarkedRef = useRef(false);
    32→  const [hasAttemptedRescue, setHasAttemptedRescue] = useState(false);
    33→
    34→  useLayoutEffect(() => {
    35→    if (isFocused) {
    36→      setHasAttemptedRescue(false);
    37→      rescueClear();
    38→    }
    39→  }, [isFocused, rescueClear]);
    40→
    41→  useFocusEffect(
    42→    React.useCallback(() => {
    43→      setHasAttemptedRescue(false);
    44→      rescueClear();
    45→      return undefined;
    46→    }, [rescueClear])
    47→  );
    48→
    49→  const conversionFactor = pointsReducer.conversionFactor;
    50→
    51→  const handleGoBack = () => {
    52→    setHasAttemptedRescue(false);
    53→    rescueClear();
    54→    navigation.goBack();
    55→  };
    56→
    57→  const invoiceList = (pts: number): any[] => {
    58→    const whiteShape = theme.colors.LIGHT_BASE_01;
    59→    const fintechColor = theme.colors.SECONDARY_COLOR_02;
    60→    const finalPrice = pts * conversionFactor;
    61→    const formattedPrice = new Intl.NumberFormat('pt-BR', {
    62→      style: 'currency',
    63→      currency: 'BRL',
    64→    }).format(finalPrice);
    65→
    66→    return [
    67→      {
    68→        type: 'default',
    69→        icon: 'icon-switch-vertical',
    70→        iconColor: fintechColor,
    71→        iconColorShape: whiteShape,
    72→        description: `Cada ${pts.toLocaleString(
    73→          'pt-br'
    74→        )} pontos valem ${formattedPrice} no cashback`,
    75→      },
    76→      {
    77→        type: 'default',
    78→        icon: 'icon-calendar',
    79→        iconColor: fintechColor,
    80→        iconColorShape: whiteShape,
    81→        description: 'Você define quanto cashback quer aplicar na sua fatura aberta',
    82→      },
    83→      {
    84→        type: 'default',
    85→        icon: 'icon-payment-invoice',
    86→        iconColor: fintechColor,
    87→        iconColorShape: whiteShape,
    88→        description: 'Se o valor for maior que o da fatura, a diferença será aplicada nas próximas',
    89→      },
    90→      {
    91→        type: 'default',
    92→        icon: 'icon-clock',
    93→        iconColor: fintechColor,
    94→        iconColorShape: whiteShape,
    95→        description: 'O cashback acontece em até 72h',
    96→      },
    97→    ];
    98→  };
    99→
   100→  const pointsSelected = route?.params?.pointsSelected ?? 0;
   101→  const reaisSelected = route?.params?.reaisSelected ?? 0;
   102→  const productCode = route?.params?.productCode;
   103→  const campaingId = route?.params?.campaingId;
   104→
   105→  const onConfirmRescue = () => {
   106→    setHasAttemptedRescue(true);
   107→    const payload: RescuePayload = {
   108→      value: pointsSelected,
   109→      campaignId: campaingId,
   110→      productCode: productCode,
   111→      description: 'Cashback resgatado',
   112→    };
   113→    rescueRequest({ payload });
   114→  };
   115→
   116→  const onExitResult = () => {
   117→    setHasAttemptedRescue(false);
   118→    rescueClear();
   119→    navigation.navigate('RewardProgram');
   120→  };
   121→
   122→  const isRescueLoading = rescueReducer?.isLoading;
   123→  const isRescueFailed = rescueReducer?.isError;
   124→  const isRescueConcluded = !isRescueFailed && !!rescueReducer?.status;
   125→  const showResult = isFocused && hasAttemptedRescue && (isRescueConcluded || isRescueFailed);
   126→
   127→  const items = invoiceList(pointsSelected, reaisSelected).map((item) => ({
   128→    icon: item.icon,
   129→    iconColor: item.iconColor,
   130→    description: item.description,
   131→    iconColorShape: item.iconColorShape,
   132→    type: item.type,
   133→  }));
   134→
   135→  const modalizeRef1 = useRef(null);
   136→
   137→  const onOpenDefault = () => {
   138→    modalizeRef1.current?.open();
   139→  };
   140→
   141→  useEffect(() => {
   142→    if (showResult && isRescueConcluded && !alreadyMarkedRef.current && hasRedeemed !== true) {
   143→      alreadyMarkedRef.current = true;
   144→      markAsRedeemed().catch(() => {});
   145→    }
   146→  }, [showResult, isRescueConcluded, hasRedeemed, markAsRedeemed]);
   147→
   148→  const nfPtBR = useMemo(() => new Intl.NumberFormat('pt-BR'), []);
   149→  const getFormatPoints = (n: number) => nfPtBR.format(Math.max(0, Math.floor(n || 0)));
   150→  const formatCurrencyBR = (n: number) =>
   151→    new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' })
   152→      .format(Number.isFinite(n) ? n : 0)
   153→      .replace(/\s/g, '');
   154→
   155→  const formattedSelectedPoints = useMemo(
   156→    () => getFormatPoints(pointsSelected || 0),
   157→    [pointsSelected]
   158→  );
   159→
   160→  const formattedSelectedReais = useMemo(
   161→    () => formatCurrencyBR(reaisSelected || 0),
   162→    [reaisSelected]
   163→  );
   164→
   165→  const effectiveShowResult =
   166→    typeof showResult !== 'undefined' ? showResult : !!(isRescueConcluded || isRescueFailed);
   167→  const screenTitle = effectiveShowResult ? 'Conclusão' : 'Cashback na fatura';
   168→
   169→  return (
   170→    <InvoiceCashbackSummaryView
   171→      handleGoBack={handleGoBack}
   172→      onConfirmRescue={onConfirmRescue}
   173→      onExitResult={onExitResult}
   174→      isRescueLoading={isRescueLoading}
   175→      isRescueConcluded={isRescueConcluded}
   176→      isRescueFailed={isRescueFailed}
   177→      items={items}
   178→      effectiveShowResult={showResult}
   179→      onOpenDefault={onOpenDefault}
   180→      modalizeRef1={modalizeRef1}
   181→      formattedSelectedPoints={formattedSelectedPoints}
   182→      formattedSelectedReais={formattedSelectedReais}
   183→      screenTitle={screenTitle}
   184→      height={height}
   185→    />
   186→  );
   187→};
   188→
   189→export const mapStateToProps = ({ pointsReducer, rescueReducer }: AppReducers) => ({
   190→  pointsReducer,
   191→  rescueReducer,
   192→});
   193→
   194→export const mapDispatchToProps = (dispatch: Dispatch) => {
   195→  return bindActionCreators(
   196→    {
   197→      ...PointsActions,
   198→      rescueRequest: rescueActions.rescueRequest,
   199→      rescueClear: rescueActions.rescueClear,
   200→    },
   201→    dispatch
   202→  );
   203→};
   204→
   205→export default connect(mapStateToProps, mapDispatchToProps)(InvoiceCashbackSummary);
