     1→import { Action } from 'redux';
     2→import createReducer from '~/redux/store/createReducer';
     3→
     4→enum RescueTypes {
     5→  REQUEST = '@rescue/REQUEST',
     6→  SUCCESS = '@rescue/SUCCESS',
     7→  FAILED = '@rescue/FAILED',
     8→  CLEAR = '@rescue/CLEAR',
     9→}
    10→export interface ResgateResponse {
    11→  status: {
    12→    message: string;
    13→  };
    14→  data?: {
    15→    status: string;
    16→  };
    17→}
    18→export interface RescuePayload {
    19→  value: number;
    20→  campaignId?: string;
    21→  productCode?: string;
    22→  description: string;
    23→  participantId?: string;
    24→}
    25→
    26→export interface RescueReducerProps {
    27→  isLoading: boolean;
    28→  isError: boolean;
    29→  status: ResgateResponse['status'] | undefined;
    30→}
    31→
    32→type State = RescueReducerProps;
    33→
    34→const initialState: RescueReducerProps = {
    35→  isLoading: false,
    36→  isError: false,
    37→  status: undefined,
    38→};
    39→
    40→const rescueReducer = createReducer(initialState, {
    41→  [RescueTypes.REQUEST](state: State) {
    42→    return {
    43→      ...state,
    44→      isLoading: true,
    45→      isError: false,
    46→    };
    47→  },
    48→  [RescueTypes.SUCCESS](state: State, { status }: { status: ResgateResponse['status'] }) {
    49→    return {
    50→      ...state,
    51→      status,
    52→      isLoading: false,
    53→      isError: false,
    54→    };
    55→  },
    56→  [RescueTypes.FAILED](state: State, { status }: { status?: ResgateResponse['status'] }) {
    57→    return {
    58→      ...state,
    59→      status,
    60→      isLoading: false,
    61→      isError: true,
    62→    };
    63→  },
    64→  [RescueTypes.CLEAR](): State {
    65→    return initialState;
    66→  },
    67→});
    68→
    69→type RescueActions = {
    70→  rescueRequest: (data: {
    71→    payload?: RescuePayload;
    72→    onSuccess?: (status: ResgateResponse['status']) => void;
    73→    onFailed?: (status?: ResgateResponse['status']) => void;
    74→  }) => Action;
    75→  rescueSuccess: (data: { status: ResgateResponse['status'] }) => Action;
    76→  rescueFailed: (data: { status?: ResgateResponse['status'] }) => Action;
    77→  rescueClear: () => Action;
    78→};
    79→
    80→export const rescueActions: RescueActions = {
    81→  rescueRequest: (data: {
    82→    payload?: RescuePayload;
    83→    onSuccess?: (status: ResgateResponse['status']) => void;
    84→    onFailed?: (status?: ResgateResponse['status']) => void;
    85→  }) => ({
    86→    type: RescueTypes.REQUEST,
    87→    data,
    88→  }),
    89→  rescueSuccess: (data: { status: ResgateResponse['status'] }) => ({
    90→    type: RescueTypes.SUCCESS,
    91→    ...data,
    92→  }),
    93→  rescueFailed: (data: { status?: ResgateResponse['status'] }) => ({
    94→    type: RescueTypes.FAILED,
    95→    ...data,
    96→  }),
    97→  rescueClear: () => ({
    98→    type: RescueTypes.CLEAR,
    99→  }),
   100→};
   101→
   102→export default rescueReducer;
