     1→import React, { FC, useCallback, useEffect, useMemo, useRef, useState } from 'react';
     2→import { connect } from 'react-redux';
     3→import { Dispatch, bindActionCreators } from 'redux';
     4→import { useFocusEffect } from '@react-navigation/native';
     5→import { eventSelectContent, eventScreenViewV3, eventModalView } from '~/constants/functions';
     6→import ApiError from '../../atomic/organisms/ApiError';
     7→import BasicScreen from '../../atomic/templates/BasicScreen';
     8→import { PointsActions } from '../../redux/ducks/points';
     9→import { AppReducers } from '../../redux/reducers/Models';
    10→import { PointsProps } from './Models';
    11→import PointsView from './view';
    12→import { useCardBuilderWhereUseMyPoints } from '../../hooks/custom/useCardBuilderWhereUseMyPoints';
    13→import { PropsModalizeRef } from '@semparar/design-system-app/src/@types/modelize';
    14→import { getParameterPointsProgram } from '../../services';
    15→
    16→const Points: FC<PointsProps> = ({
    17→  navigation,
    18→  requestPointsBalance,
    19→  requestPointsProducts,
    20→  requestInitialData,
    21→  clear,
    22→  pointsReducer,
    23→}) => {
    24→  const hasFocused = useRef(false);
    25→  const modalizeRef = useRef<PropsModalizeRef>(null as unknown as PropsModalizeRef);
    26→  const [isMyOrders, setIsMyOrders] = useState(false);
    27→
    28→  const handleMyOrders = () => {
    29→    navigation.navigate('MyOrdersRewardProgram')
    30→  }
    31→
    32→  const handleGoBack = () => {
    33→    navigation.goBack();
    34→  };
    35→  const handlerGoHome = () => {
    36→    navigation.navigate('Home');
    37→  };
    38→
    39→  const handleOpenDefault = () => {
    40→    eventSelectContent({
    41→      content_type: 'cta',
    42→      content_id: 'informacao_pontos',
    43→      location_1: 'monetizacao',
    44→      location_2: 'hub_recompensas_spmais',
    45→      location_3: 'home_hub_recompensas',
    46→    });
    47→
    48→    modalizeRef.current?.open?.();
    49→  };
    50→
    51→  const handleOpenMyOrders = () => {
    52→    setIsMyOrders(true);
    53→
    54→    eventModalView({
    55→      content_id: 'menu_header',
    56→      location_1: 'monetizacao',
    57→      location_2: 'hub_recompensas_spmais',
    58→      location_3: 'home_hub_recompensas',
    59→    });
    60→  };
    61→
    62→  const buildPointsCardItems = useCardBuilderWhereUseMyPoints();
    63→
    64→  const requestPointsCallback = useCallback(() => {
    65→    requestPointsBalance();
    66→    requestPointsProducts();
    67→    getParameterPointsProgram();
    68→  }, [requestPointsBalance, requestPointsProducts]);
    69→
    70→  const handleApiErrorPress = () => {
    71→    if (pointsReducer.isErrorBalance || pointsReducer.isErrorProducts) {
    72→      requestPointsCallback();
    73→    }
    74→  };
    75→
    76→  useFocusEffect(
    77→    useCallback(() => {
    78→      if (!hasFocused.current) {
    79→        eventScreenViewV3('home_hub_recompensas', 'hub_recompensas_spmais', 'monetizacao', {
    80→          location_1: 'monetizacao',
    81→          location_2: 'hub_recompensas_spmais',
    82→          location_3: 'home_hub_recompensas',
    83→        });
    84→        hasFocused.current = true;
    85→      }
    86→      return () => {
    87→        hasFocused.current = false;
    88→      };
    89→    }, [])
    90→  );
    91→
    92→  useEffect(() => {
    93→    requestInitialData();
    94→    getParameterPointsProgram();
    95→
    96→    return () => {
    97→      clear();
    98→    };
    99→  }, []);
   100→
   101→  useFocusEffect(
   102→    useCallback(() => {
   103→      requestPointsCallback();
   104→      return undefined;
   105→    }, [requestPointsCallback])
   106→  );
   107→
   108→  const view = useMemo(() => {
   109→    if (pointsReducer.isErrorProducts || pointsReducer.isErrorBalance) {
   110→      return (
   111→        <BasicScreen
   112→          leftIconOnPress={handleGoBack}
   113→          title="Programa de Pontos"
   114→          errorButtonOnPress={handleApiErrorPress}
   115→        >
   116→          <ApiError onPress={handleApiErrorPress} />
   117→        </BasicScreen>
   118→      );
   119→    }
   120→
   121→    const normalizedItems = buildPointsCardItems(pointsReducer.products);
   122→
   123→    const onOpenExtract = () => {
   124→      eventSelectContent({
   125→        content_type: 'cta',
   126→        content_id: 'exibir_extrato',
   127→        location_1: 'monetizacao',
   128→        location_2: 'hub_recompensas_spmais',
   129→        location_3: 'home_hub_recompensas',
   130→      });
   131→      navigation.navigate('ExtractRewardProgram', {
   132→        analytics: {
   133→          screen_name: 'extrato_hub_recompensas',
   134→          screen_class: 'hub_recompensas_spmais',
   135→          screen_bussines: 'monetizacao',
   136→          screen_info: {
   137→            location_1: 'monetizacao',
   138→            location_2: 'hub_recompensas_spmais',
   139→            location_3: 'extrato_hub_recompensas',
   140→          },
   141→        },
   142→      });
   143→    };
   144→
   145→    return (
   146→      <PointsView
   147→        items={normalizedItems}
   148→        isLoadingBalance={pointsReducer.isLoadingBalance}
   149→        handleGoBack={handleGoBack}
   150→        pointsAvailable={pointsReducer.points}
   151→        pointsOptions={pointsReducer.products}
   152→        onOpenExtract={onOpenExtract}
   153→        onOpenDefault={handleOpenDefault}
   154→        modalizeRef={modalizeRef}
   155→        isMyOrders={isMyOrders}
   156→        setIsMyOrders={setIsMyOrders}
   157→        handlerGoHome={handlerGoHome}
   158→        onMenuPress={handleOpenMyOrders}
   159→        handleMyOrders={handleMyOrders}
   160→      />
   161→    );
   162→  }, [pointsReducer, navigation, buildPointsCardItems, isMyOrders]);
   163→
   164→  return view;
   165→};
   166→
   167→export const mapStateToProps = ({ pointsReducer }: AppReducers) => ({
   168→  pointsReducer,
   169→});
   170→
   171→export const mapDispatchToProps = (dispatch: Dispatch) => {
   172→  return bindActionCreators(
   173→    {
   174→      ...PointsActions,
   175→    },
   176→    dispatch
   177→  );
   178→};
   179→
   180→export default connect(mapStateToProps, mapDispatchToProps)(Points);
