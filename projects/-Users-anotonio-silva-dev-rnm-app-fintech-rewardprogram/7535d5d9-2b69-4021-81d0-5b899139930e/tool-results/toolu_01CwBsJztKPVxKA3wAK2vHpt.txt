     6→export async function performRescue(
     7→  payload: RescuePayload,
     8→  callbacks?: RescueCallbacks
     9→): Promise<void> {
    10→  const { rescueActions } = useRootStore.getState();
    11→
    12→  try {
    13→    rescueActions.setStatus(RescueStatus.LOADING);
    14→
    15→    const participantId = payload.participantId || await getParticipantIdFromToken();
    16→    const rescuePayload: RescuePayload = {
    17→      ...payload,
    18→      participantId,
    19→    };
    20→
    21→    const response = await rescueApi.rescue(rescuePayload);
    22→
    23→    if (response?.data?.status === 'OK' || response?.status?.message) {
    24→      rescueActions.setResponseStatus(response.status);
    25→      rescueActions.setStatus(RescueStatus.SUCCESS);
    26→      callbacks?.onSuccess?.(response.status);
    27→    } else {
    28→      rescueActions.setResponseStatus(response?.status);
    29→      rescueActions.setStatus(RescueStatus.ERROR);
    30→      callbacks?.onFailed?.(response?.status);
    31→    }
    32→  } catch (error) {
    33→    rescueActions.setError('Erro ao realizar resgate');
    34→    rescueActions.setStatus(RescueStatus.ERROR);
    35→    callbacks?.onFailed?.();
    36→  }
    37→}
