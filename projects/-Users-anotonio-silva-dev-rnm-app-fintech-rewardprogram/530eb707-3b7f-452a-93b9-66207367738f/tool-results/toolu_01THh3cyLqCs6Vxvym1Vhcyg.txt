+# Frontend Style Guide — Sem Parar (Zustand-only + Hook Component Pattern)
+
+Welcome to the **Sem Parar Frontend Style Guide**.  
+This version consolidates: **Zustand as the only state manager**, **Hook Component pattern** for Screens and Atomic components, **services architecture (SOLID)**, and an **cache pattern**.

+## Introduction
+
+The goal of this guide is to ensure **consistency, testability and scalability** across our frontend projects.  
+
+Key decisions:
+- **Zustand** is the **exclusive** global/shared state manager.
+- UI follows the **Hook Component** pattern (logic hook + dumb view).
+- Domain logic lives in **`/services`** with **SOLID** layering.
+- Cache is **typed and reusable**.
+
+---
+
+## Components & Screens — Hook Component Pattern
+
+Each screen is split into a logic hook and a render-only component.
+
+```
+screens/Home/
+├── index.ts        # logic: export useHomeScreen()
+├── view.tsx        # render: export default HomeScreen
+└── Models/
+    └── index.d.ts    # types (Props, State, Handlers, Return)
+```
+
+**Models/index.d.ts**
+```ts
+export interface HomeParams {
+  welcome?: string;
+}
+export interface HomeState {
+  greeting: string;
+  isLoading: boolean;
+}
+export interface HomeHandlers {
+  onPressCTA: () => void;
+}
+export type UseHomeReturn = HomeState & HomeHandlers;
+```
+
+**index.ts (logic hook)**
+```ts
+import { useCallback, useMemo, useState, useEffect } from "react";
+import type { HomeParams, UseHomeReturn } from "./Models";
+import { useUserStore } from "@/services/user/store"; // Zustand slice example
+
+export function useHomeScreen(params?: HomeParams): UseHomeReturn {
+  const name = useUserStore((s) => s.name);
+  const [isLoading, setIsLoading] = useState(true);
+
+  useEffect(() => {
+    const t = setTimeout(() => setIsLoading(false), 300);
+    return () => clearTimeout(t);
+  }, []);
+
+  const greeting = useMemo(
+    () => params?.welcome ?? `Hello, ${name || "guest"}!`,
+    [params?.welcome, name]
+  );
+
+  const onPressCTA = useCallback(() => {
+    // Call services, analytics, navigation, etc.
+  }, []);
+
+  return { greeting, isLoading, onPressCTA };
+}
+```
+
+**view.tsx (render)**
+```tsx
+import React from "react";
+import { View } from "react-native";
+import { Text, Button } from "@semparar/design-system-app";
+import { useHomeScreen } from "./index";
+import type { HomeParams } from "./Models";
+
+type Props = HomeParams;
+
+export default function HomeScreen(props: Props) {
+  const { greeting, isLoading, onPressCTA } = useHomeScreen(props);
+
+  if (isLoading) {
+    return (
+      <View>
+        <Text>Loading...</Text>
+      </View>
+    );
+  }
+
+  return (
+    <View>
+      <Text>{greeting}</Text>
+      <Button onPress={onPressCTA} text="Continue" />
+    </View>
+  );
+}
+```

+## Atomic Components — Hook Component Pattern
+
+```
+atomic/atoms/Button/

+export function useButton(props: ButtonProps): UseButtonReturn {

+## Services Architecture
+
+**Flow:** `UI → services → repository → api` and `UI → services → store (Zustand)`
+
+- **api/**: request clients; no domain logic.  
+- **class/**: class with logical requests.  
+- **hooks/**: specific service hooks.
+- **repository/**: maps transport to domain, aggregates sources.  
+- **services/**: orchestration and business rules.  
+- **storage/**: storage orchestration with cache or not.  
+- **store/**: Zustand slices (single source of app state).  
+- **models/**: types, DTOs, contracts.  
+- **utils/**: reusable service helpers.
+
+Principles:
+- Single responsibility per folder/file.
+- No circular dependencies.
+- Public APIs exported via `services/index.ts` per domain.
+

+## State Management (Zustand-only)
+
+- All global/shared state is **Zustand**.  
+- Slices live under `/services/<domain>/store/`.  
+- Use `persist` (with `createJSONStorage`) only when needed.  
+- Keep setters pure; side effects go in hooks or services.
+
+**Slice example**
+```ts
+import type { StateCreator } from "zustand";
+import type { AuthActions, AuthState } from "../models/store";
+
+export type AuthSlice = {
+    auth: AuthState;
+    authActions: AuthActions
+};
+
+const initialAuth = {
+    user: null,
+    loading: false,
+    error: undefined,
+}
+
+export const createAuthSlice: StateCreator<AuthSlice, [], [], AuthSlice> = (set) => ({
+    auth: initialAuth,
+    authActions: {
+        setLoading: (v) => set((s) => ({ auth: { ...s.auth, isLoading: v } })),
+        setUser: (v) => set((s) => ({ auth: { ...s.auth, user: v } })),
+        setError: (v) => set((s) => ({ auth: { ...s.auth, error: v } })),
+        reset: () => set({ auth: initialAuth }),
+    }
+});

+## Cache Pattern
+
+Keep cache typed, serializable, and domain.
+
+```ts
+import storage from "~/redux/store/storage";
+
+export interface CachedData<T> {
+  timestamp: number;
+  data: T;
+}
+
+export interface StorageContract<T> {
+  get(): Promise<CachedData<T> | null>;
+  save(data: T): Promise<void>;
+  remove(): Promise<void>;
+}
+
+export class GenericStorage<T> implements StorageContract<T> {
+  constructor(private readonly key: string) {}
+
+  async get(): Promise<CachedData<T> | null> {
+    const json = await storage.getItem(this.key);
+    if (!json) return null;
+    return JSON.parse(json) as CachedData<T>;
+  }
+
+  async save(data: T): Promise<void> {
+    const payload: CachedData<T> = { timestamp: Date.now(), data };
+    await storage.setItem(this.key, JSON.stringify(payload));
+  }
+
+  async remove(): Promise<void> {
+    await storage.removeItem(this.key);
+  }
+}
+```
+
+**Guidelines**
+- One storage class per module/feature.
+- Types/interfaces live in `Models/` where appropriate.
+- Use `timestamp` for freshness/expiry strategies.
+- Never persist secrets/PII outside secure storage.
+- Keep payloads small and JSON-serializable.

+import type { PointsTransaction, AppliedFilters } from '../../store/rewardProgram/models';
+import { TransactionType, TYPE_LABELS } from './Models';
+
+function getTransactionDescription(type: TransactionType): string {
+  return TYPE_LABELS[type] || '';
+}
+
+function matchesPeriodFilter(
+  transactionDate: Date,
+  periodStart: Date | null,
+  todayEnd: Date,
+): boolean {
+  if (!periodStart) return true;
+  return transactionDate >= periodStart && transactionDate <= todayEnd;
+}
+
+function matchesTypeFilter(
+  transactionType: string,
+  typesSet: Set<string> | null,
+): boolean {
+  if (!typesSet) return true;
+  return typesSet.has(transactionType);
+}
+
+export function getFilteredTransactions(
+  transactions: PointsTransaction[],
+  filters?: AppliedFilters,
+): PointsTransaction[] {
+  if (!filters || (!filters.period && (!filters.types || filters.types.length === 0))) {
+    return transactions;
+  }
+
+  const now = new Date();
+  const todayEnd = new Date(now);
+  todayEnd.setHours(23, 59, 59, 999);
+
+  const periodStart = filters.period
+    ? (() => {
+      const date = new Date(now);
+      date.setDate(date.getDate() - filters.period);
+      date.setHours(0, 0, 0, 0);
+      return date;
+    })()
+    : null;
+
+  const hasTypeFilter = filters.types && filters.types.length > 0;
+  const typesSet = hasTypeFilter ? new Set(filters.types) : null;
+
+  return transactions.filter((transaction) => {
+    const transactionDate = new Date(transaction.date);
+
+    if (!matchesPeriodFilter(transactionDate, periodStart, todayEnd)) {
+      return false;
+    }
+
+    if (!matchesTypeFilter(transaction.type, typesSet)) {
+      return false;
+    }
+
+    return true;
+  });
+}
+
+function matchesSearchText(
+  transaction: PointsTransaction,
+  normalizedSearch: string,
+  isExpiringTab: boolean,
+): boolean {
+  const amountStr = String(transaction.amount);
+
+  if (amountStr.includes(normalizedSearch)) {
+    return true;
+  }
+
+  if (isExpiringTab) {
+    return false;
+  }
+
+  const description = getTransactionDescription(transaction.type).toLowerCase();
+  if (description.includes(normalizedSearch)) {
+    return true;
+  }
+
+  if (transaction.description?.toLowerCase().includes(normalizedSearch)) {
+    return true;
+  }
+
+  return false;
+}
+
+export function getFilteredTransactionsByText(
+  transactions: PointsTransaction[],
+  searchText: string,
+  isExpiringTab = false,
+): PointsTransaction[] {
+  if (!searchText?.trim() || !transactions?.length) {
+    return transactions;
+  }
+
+  const normalizedSearch = searchText.toLowerCase().trim();
+
+  return transactions.filter((transaction) =>
+    matchesSearchText(transaction, normalizedSearch, isExpiringTab),
+  );
+}

+import { TransactionType, MyOrdersDetailProps } from "./Models";
+
+export type TransactionGainType = 'CARD' | 'TAG' | 'CASHBACK';
+
+export const getTransactionGainType = (transaction: MyOrdersDetailProps['transaction']): TransactionGainType => {
+  if (!transaction) return 'CARD';
+  
+  const provider = transaction.details?.provider?.toLowerCase() || '';
+  
+  if (provider.includes('cashback')) {
+    return 'CASHBACK';
+  }
+  
+  const icon = transaction.icon?.toLowerCase() || '';
+  const description = transaction.description?.toLowerCase() || '';
+  
+  const tagIndicators = ['tag', 'pedágio', 'pedagio', 'estacionamento', 'abastecimento', 'abastece', 'drive', 'lava', 'rápido'];
+  
+  const isTag = tagIndicators.some(indicator => 
+    icon.includes(indicator) || 
+    description.includes(indicator) || 
+    provider.includes(indicator)
+  );
+  
+  return isTag ? 'TAG' : 'CARD';
+};
+
+export const getFormattedPointsValue = (amount: number, type: TransactionType): string => {
+  const prefix = (type === 'CREDIT' || type === 'REFUNDED') ? '+ ' : '- ';
+  const label = Math.abs(amount) === 1 ? 'ponto' : 'pontos';
+  return `${prefix}${Math.abs(amount).toLocaleString('pt-BR')} ${label}`;
+};
+
+export const getFormattedHighlightText = (date: string, gainType: TransactionGainType, moment: any): string => {
+  const transactionDate = moment(date);
+  const day = transactionDate.format('DD');
+  const month = transactionDate.format('MMM').toLowerCase();
+  const year = transactionDate.format('YYYY');
+  const time = transactionDate.format('HH:mm');
+  
+  return `Pedido feito em ${day} ${month} ${year} às ${time}`;
+};
+
+export const getFormattedPointsValueForMarketplace = (valuePoints: number): string => {
+  const absValue = Math.abs(valuePoints);
+  const label = absValue === 1 ? 'ponto' : 'pontos';
+  return `${absValue.toLocaleString('pt-BR')} ${label}`;
+};
+
+export const getPaymentTypeLabel = (paymentType: string): string => {
+  const labels: Record<string, string> = {
+    POINTS: 'Pontos',
+    CARD: 'Cartão',
+    PIX: 'PIX',
+  };
+  return labels[paymentType] || paymentType;
+};
+
+export const getMarketplaceTitle = (transaction: MyOrdersDetailProps['transaction']): string | null => {
+  if (!transaction?.marketplace?.orders) return null;
+
+  const allItems: Array<{ name: string }> = [];
+  transaction.marketplace.orders.forEach(order => {
+    if (order.items) {
+      allItems.push(...order.items);
+    }
+  });
+
+  if (allItems.length === 0) return null;
+  if (allItems.length === 1) return allItems[0].name;
+  
+  const firstItemName = allItems[0].name;
+  const secondItemFirstWord = allItems[1].name.split(' ')[0];
+  
+  if (allItems.length === 2) {
+    return `${firstItemName} e ${secondItemFirstWord}`;
+  }
+  
+  const remainingCount = allItems.length - 2;
+  return `${firstItemName}, ${secondItemFirstWord} e outros ${remainingCount}`;
+};
+
+export const getCleanedImageUrl = (url: string | undefined): string | undefined => {
+  if (!url) return undefined;
+  
+  let cleanedUrl = url.trim();
+  
+  if (cleanedUrl.startsWith('imageUrl:')) {
+    cleanedUrl = cleanedUrl.replace(/^imageUrl:/i, '');
+  }
+  
+  if (cleanedUrl.startsWith('http://') || cleanedUrl.startsWith('https://')) {
+    return cleanedUrl;
+  }
+  
+  return undefined;
+};

+import { api } from '@semparar/mf-app-user';
+import type { AxiosResponse } from 'axios';
+import { EXTRACT_CONSTANTS } from '../constants';

+export type ExtractRequestType = 'Credito' | 'Debito' | 'Expirado' | 'Resgate';
+
+export interface ExtractRequestDto {
+  participantId: string;
+  page?: number;
+  pageSize?: number;
+  type?: ExtractRequestType;
+}
+
+export interface TransactionDto {
+  id: string;
+  date: string;
+  description: string;
+  type: string;
+  amount: number;
+  icon: string;
+  status?: string;
+  origination?: string;

+
+export interface PaginationDto {
+  page: number;
+  pageSize: number;
+  total: number;
+  totalPages: number;

+export interface PointsExtractApiResponse {
+  data: {
+    transactions: TransactionDto[];
+  };
+  pagination: PaginationDto;
+  status: {
+    returnCode: number;
+    message: string;
+  };

+interface PointsBalanceApiResponse {
+  data: {
+    data: { points?: number };
+    status: {
+      returnCode?: number;
+      codigoRetorno?: number;
+      message?: string;
+    };
+  };
+}
+
+function buildExtractRequestPayload(payload: ExtractRequestDto): ExtractRequestDto {
+  return {
+    participantId: payload.participantId,
+    page: payload.page ?? EXTRACT_CONSTANTS.DEFAULT_PAGE,
+    pageSize: payload.pageSize ?? EXTRACT_CONSTANTS.DEFAULT_PAGE_SIZE,
+    ...(payload.type ? { type: payload.type } : {}),

+function getGroupedTransactionsByDate(
+  transactions: PointsTransaction[],
+): GroupedTransactions[] {
+  if (transactions.length === 0) {
+    return [];
+  }
+
+  const sortedTransactions = [...transactions].sort((a, b) =>
+    moment(b.date).diff(moment(a.date)),
+  );
+
+  const groupedByDate = sortedTransactions.reduce((acc, transaction) => {
+    const formattedDate = getFormattedDateForHeader(transaction.date);
+
+    if (!acc[formattedDate]) {
+      acc[formattedDate] = [];
+    }
+
+    acc[formattedDate].push(
+      getTransactionCardItem(transaction) as propsItemsListCardTransition,
+    );
+    return acc;
+  }, {} as Record<string, propsItemsListCardTransition[]>);
+
+  const getSortedItemsByDate = (
+    items: propsItemsListCardTransition[],
+    date: string,
+  ) => {
+    return items.sort((a, b) => {
+      const transactionA = sortedTransactions.find(
+        (t) =>
+          getFormattedDateForHeader(t.date) === date &&
+          getTransactionCardItem(t).title === a.title,
+      );
+      const transactionB = sortedTransactions.find(
+        (t) =>
+          getFormattedDateForHeader(t.date) === date &&
+          getTransactionCardItem(t).title === b.title,
+      );
+
+      if (transactionA && transactionB) {
+        return moment(transactionB.date).diff(moment(transactionA.date));
+      }
+      return 0;
+    });
+  };
+
+  const getSortedByDate = (a: GroupedTransactions, b: GroupedTransactions) => {
+    const dateA = moment(a.date.split(' ').slice(1).join(' '), 'DD MMMM');
+    const dateB = moment(b.date.split(' ').slice(1).join(' '), 'DD MMMM');
+    return dateB.diff(dateA);
+  };
+
+  return Object.entries(groupedByDate)
+    .map(([date, items]) => ({
+      date,
+      items: getSortedItemsByDate(items as propsItemsListCardTransition[], date),
+    }))
+    .sort(getSortedByDate);
+}
