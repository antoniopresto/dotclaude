+import { TransactionType, MyOrdersDetailProps } from "./Models";
+
+export type TransactionGainType = 'CARD' | 'TAG' | 'CASHBACK';
+
+export const getTransactionGainType = (transaction: MyOrdersDetailProps['transaction']): TransactionGainType => {
+  if (!transaction) return 'CARD';
+  
+  const provider = transaction.details?.provider?.toLowerCase() || '';
+  
+  if (provider.includes('cashback')) {
+    return 'CASHBACK';
+  }
+  
+  const icon = transaction.icon?.toLowerCase() || '';
+  const description = transaction.description?.toLowerCase() || '';
+  
+  const tagIndicators = ['tag', 'pedágio', 'pedagio', 'estacionamento', 'abastecimento', 'abastece', 'drive', 'lava', 'rápido'];
+  
+  const isTag = tagIndicators.some(indicator => 
+    icon.includes(indicator) || 
+    description.includes(indicator) || 
+    provider.includes(indicator)
+  );
+  
+  return isTag ? 'TAG' : 'CARD';
+};
+
+export const getFormattedPointsValue = (amount: number, type: TransactionType): string => {
+  const prefix = (type === 'CREDIT' || type === 'REFUNDED') ? '+ ' : '- ';
+  const label = Math.abs(amount) === 1 ? 'ponto' : 'pontos';
+  return `${prefix}${Math.abs(amount).toLocaleString('pt-BR')} ${label}`;
+};
+
+export const getFormattedHighlightText = (date: string, gainType: TransactionGainType, moment: any): string => {
+  const transactionDate = moment(date);
+  const day = transactionDate.format('DD');
+  const month = transactionDate.format('MMM').toLowerCase();
+  const year = transactionDate.format('YYYY');
+  const time = transactionDate.format('HH:mm');
+  
+  return `Pedido feito em ${day} ${month} ${year} às ${time}`;
+};
+
+export const getFormattedPointsValueForMarketplace = (valuePoints: number): string => {
+  const absValue = Math.abs(valuePoints);
+  const label = absValue === 1 ? 'ponto' : 'pontos';
+  return `${absValue.toLocaleString('pt-BR')} ${label}`;
+};
+
+export const getPaymentTypeLabel = (paymentType: string): string => {
+  const labels: Record<string, string> = {
+    POINTS: 'Pontos',
+    CARD: 'Cartão',
+    PIX: 'PIX',
+  };
+  return labels[paymentType] || paymentType;
+};
+
+export const getMarketplaceTitle = (transaction: MyOrdersDetailProps['transaction']): string | null => {
+  if (!transaction?.marketplace?.orders) return null;
+
+  const allItems: Array<{ name: string }> = [];
+  transaction.marketplace.orders.forEach(order => {
+    if (order.items) {
+      allItems.push(...order.items);
+    }
+  });
+
+  if (allItems.length === 0) return null;
+  if (allItems.length === 1) return allItems[0].name;
+  
+  const firstItemName = allItems[0].name;
+  const secondItemFirstWord = allItems[1].name.split(' ')[0];
+  
+  if (allItems.length === 2) {
+    return `${firstItemName} e ${secondItemFirstWord}`;
+  }
+  
+  const remainingCount = allItems.length - 2;
+  return `${firstItemName}, ${secondItemFirstWord} e outros ${remainingCount}`;
+};
+
+export const getCleanedImageUrl = (url: string | undefined): string | undefined => {
+  if (!url) return undefined;
+  
+  let cleanedUrl = url.trim();
+  
+  if (cleanedUrl.startsWith('imageUrl:')) {
+    cleanedUrl = cleanedUrl.replace(/^imageUrl:/i, '');
+  }
+  
+  if (cleanedUrl.startsWith('http://') || cleanedUrl.startsWith('https://')) {
+    return cleanedUrl;
+  }
+  
+  return undefined;
+};

+import { PointsTransaction } from '../models/extract.model';
+import type { OrderDetailDataDto, getOrderDetailApi } from '../api/orderDetail.api';
+
+function getTransactionFromOrderDetail(dto: OrderDetailDataDto): PointsTransaction {
+  const items = dto.items || [];
+  const benefits = dto.benefits || [];
+  
+  const totalValue = items.reduce((sum, item) => sum + (item.unitaryValue * item.quantity), 0);
+  
+  const fields: Array<{ label: string; value: string }> = [];
+  
+  if (items.length > 0) {
+    items.forEach((item, index) => {
+      fields.push({
+        label: `Item ${index + 1}`,
+        value: `${item.description} - ${item.quantity}x R$ ${item.unitaryValue.toFixed(2)}`,
+      });
+    });
+  }
+
+  if (benefits.length > 0) {
+    benefits.forEach((benefit, index) => {
+      fields.push({
+        label: `Benefício ${index + 1}`,
+        value: `${benefit.description} - R$ ${Math.abs(benefit.value).toFixed(2)}`,
+      });
+    });
+  }
+
+  return {
+    id: dto.id,
+    date: dto.date,
+    description: dto.description,
+    type: 'RESCUE',
+    amount: dto.amount,
+    currencyValue: totalValue > 0 ? totalValue : Math.abs(dto.amount),
+    icon: 'icon-cashback',
+    details: {
+      provider: 'Cashback',
+      extraInfo: '',
+      fields,
+    },
+    status: '',
+    marketplace: dto.marketplace,
+  };
+}
+
+export function getOrderDetailFromResponse(
+  response: Awaited<ReturnType<typeof getOrderDetailApi>>,
+): PointsTransaction | null {
+  const apiResponse = response?.data;
+
+  if (!apiResponse) {
+    return null;
+  }
+
+  const status = apiResponse.status;
+  const returnCode = status?.returnCode;
+
+  if (returnCode !== 200 && returnCode !== 0) {
+    return null;
+  }
+
+  const orderDetailData = apiResponse.data;
+
+  if (!orderDetailData) {
+    return null;
+  }
+
+  return getTransactionFromOrderDetail(orderDetailData);
+}

+import { getOrderDetailApi, type OrderDetailRequestDto } from '../api/orderDetail.api';
+import { getOrderDetailFromResponse } from '../repository/orderDetail.repository';
+import { useRootStore } from '../..';
+import {
+  OrderDetailStatus,
+} from '../models/orderDetail.model';
+import { eventServerResponse } from '~/constants/functions';
+import { getParticipantIdFromToken } from '../utils';
+
+interface GetOrderDetailPayload {
+  transactionId: string;
+  participantId?: string;
+}
+
+function handleOrderDetailSuccess(
+  orderDetail: ReturnType<typeof getOrderDetailFromResponse>,
+  transactionId: string,
+): void {
+  const { orderDetailActions } = useRootStore.getState();
+
+  if (orderDetail) {
+    orderDetailActions.setOrderDetail(orderDetail, transactionId);
+    eventServerResponse('ORDER_DETAIL_SUCCESS');
+  } else {
+    handleOrderDetailError();
+  }
+}
+
+function handleOrderDetailError(): void {
+  const { orderDetailActions } = useRootStore.getState();
+  orderDetailActions.setOrderDetailStatus(OrderDetailStatus.ERROR);
+  eventServerResponse('ORDER_DETAIL_ERROR');
+}
+
+export async function getOrderDetail(
+  payload: GetOrderDetailPayload,
+): Promise<void> {
+  const { orderDetailActions, orderDetail } = useRootStore.getState();
+
+  const isDifferentTransaction = orderDetail.transactionId !== payload.transactionId;
+  const isAlreadyLoading = orderDetail.status === OrderDetailStatus.LOADING;
+
+  if (isAlreadyLoading && !isDifferentTransaction) {
+    return;
+  }
+
+  if (isDifferentTransaction) {
+    orderDetailActions.reset();
+  }
+
+  orderDetailActions.setOrderDetailStatus(OrderDetailStatus.LOADING);
+
+  try {
+    const participantId =
+      payload?.participantId ?? (await getParticipantIdFromToken());
+
+    if (!participantId) {
+      handleOrderDetailError();
+      return;
+    }
+
+    const requestPayload: OrderDetailRequestDto = {
+      transactionId: payload.transactionId,
+      participantId,
+    };
+
+    const response = await getOrderDetailApi(requestPayload);
+    const orderDetailData = getOrderDetailFromResponse(response);
+
+    if (orderDetailData) {
+      handleOrderDetailSuccess(orderDetailData, payload.transactionId);
+      return;
+    }
+
+    handleOrderDetailError();
+  } catch {
+    handleOrderDetailError();
+  }
+}
