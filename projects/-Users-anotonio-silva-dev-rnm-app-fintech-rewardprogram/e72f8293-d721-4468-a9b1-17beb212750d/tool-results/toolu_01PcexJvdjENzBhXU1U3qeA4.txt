     1→import React, { FC, useCallback, useEffect, useMemo, useRef, useState } from 'react';
     2→import { useFocusEffect } from '@react-navigation/native';
     3→
     4→import { eventSelectContent, eventScreenViewV3, eventModalView } from '~/constants/functions';
     5→import ApiError from '../../atomic/organisms/ApiError';
     6→import BasicScreen from '../../atomic/templates/BasicScreen';
     7→import { PointsProps } from './Models';
     8→import PointsView from './view';
     9→import { useCardBuilderWhereUseMyPoints } from '../../hooks/custom/useCardBuilderWhereUseMyPoints';
    10→import { PropsModalizeRef } from '@semparar/design-system-app/src/@types/modelize';
    11→import { getParameterPointsProgram } from '../../services/services';
    12→import { useRootStore } from '~/store';
    13→import { getPointsProducts, getPointsBalance } from '../../services/services';
    14→import { PointsProductsStatus, PointsBalanceStatus } from '../../services/models';
    15→
    16→const Points: FC<PointsProps> = ({ navigation }) => {
    17→  const hasFocused = useRef(false);
    18→  const modalizeRef = useRef<PropsModalizeRef>(null as unknown as PropsModalizeRef);
    19→  const [isMyOrders, setIsMyOrders] = useState(false);
    20→
    21→  const { products, status: productsStatus } = useRootStore((state) => state.pointsProducts);
    22→  const { points, status: balanceStatus } = useRootStore((state) => state.pointsBalance);
    23→  const { pointsProductsActions } = useRootStore.getState();
    24→  const { pointsExtractActions } = useRootStore.getState();
    25→
    26→  const isLoadingBalance = balanceStatus === PointsBalanceStatus.LOADING || balanceStatus === PointsBalanceStatus.IDLE;
    27→  const isLoadingProducts = productsStatus === PointsProductsStatus.LOADING || productsStatus === PointsProductsStatus.IDLE;
    28→  const isErrorBalance = balanceStatus === PointsBalanceStatus.ERROR;
    29→  const isErrorProducts = productsStatus === PointsProductsStatus.ERROR;
    30→
    31→  const handleGoBack = () => {
    32→    navigation.goBack();
    33→  };
    34→
    35→  const handleOpenDefault = () => {
    36→    eventSelectContent({
    37→      content_type: 'cta',
    38→      content_id: 'informacao_pontos',
    39→      location_1: 'monetizacao',
    40→      location_2: 'hub_recompensas_spmais',
    41→      location_3: 'home_hub_recompensas',
    42→    });
    43→
    44→    modalizeRef.current?.open?.();
    45→  };
    46→
    47→  const handleOpenMyOrders = () => {
    48→    setIsMyOrders(true);
    49→
    50→    eventModalView({
    51→      content_id: 'menu_header',
    52→      location_1: 'monetizacao',
    53→      location_2: 'hub_recompensas_spmais',
    54→      location_3: 'home_hub_recompensas',
    55→    });
    56→  };
    57→
    58→  const buildPointsCardItems = useCardBuilderWhereUseMyPoints();
    59→
    60→  const requestPointsCallback = useCallback(() => {
    61→    getPointsBalance();
    62→    getPointsProducts();
    63→    getParameterPointsProgram();
    64→  }, []);
    65→
    66→  const handleApiErrorPress = () => {
    67→    if (isErrorBalance || isErrorProducts) {
    68→      pointsExtractActions.setBalanceStatus(PointsBalanceStatus.IDLE);
    69→      pointsProductsActions.setStatus(PointsProductsStatus.IDLE);
    70→      requestPointsCallback();
    71→    }
    72→  };
    73→
    74→  useFocusEffect(
    75→    useCallback(() => {
    76→      if (!hasFocused.current) {
    77→        eventScreenViewV3('home_hub_recompensas', 'hub_recompensas_spmais', 'monetizacao', {
    78→          location_1: 'monetizacao',
    79→          location_2: 'hub_recompensas_spmais',
    80→          location_3: 'home_hub_recompensas',
    81→        });
    82→        hasFocused.current = true;
    83→      }
    84→      return () => {
    85→        hasFocused.current = false;
    86→      };
    87→    }, [])
    88→  );
    89→
    90→  useEffect(() => {
    91→    getParameterPointsProgram();
    92→    return () => {
    93→      pointsProductsActions.reset();
    94→      pointsExtractActions.reset();
    95→    };
    96→  }, []);
    97→
    98→  useFocusEffect(
    99→    useCallback(() => {
   100→      requestPointsCallback();
   101→      return undefined;
   102→    }, [requestPointsCallback])
   103→  );
   104→
   105→  const view = useMemo(() => {
   106→    if (isErrorProducts || isErrorBalance) {
   107→      return (
   108→        <BasicScreen
   109→          leftIconOnPress={handleGoBack}
   110→          title="Programa de Pontos"
   111→          errorButtonOnPress={handleApiErrorPress}
   112→        >
   113→          <ApiError onPress={handleApiErrorPress} />
   114→        </BasicScreen>
   115→      );
   116→    }
   117→
   118→    const normalizedItems = buildPointsCardItems(products);
   119→
   120→    const onOpenExtract = () => {
   121→      eventSelectContent({
   122→        content_type: 'cta',
   123→        content_id: 'exibir_extrato',
   124→        location_1: 'monetizacao',
   125→        location_2: 'hub_recompensas_spmais',
   126→        location_3: 'home_hub_recompensas',
   127→      });
   128→      navigation.navigate('PointsExtract', {
   129→        analytics: {
   130→          screen_name: 'extrato_hub_recompensas',
   131→          screen_class: 'hub_recompensas_spmais',
   132→          screen_bussines: 'monetizacao',
   133→          screen_info: {
   134→            location_1: 'monetizacao',
   135→            location_2: 'hub_recompensas_spmais',
   136→            location_3: 'extrato_hub_recompensas',
   137→          },
   138→        },
   139→      });
   140→    };
   141→
   142→    return (
   143→      <PointsView
   144→        items={normalizedItems}
   145→        isLoadingBalance={isLoadingBalance || isLoadingProducts}
   146→        handleGoBack={handleGoBack}
   147→        pointsAvailable={points}
   148→        pointsOptions={products}
   149→        onOpenExtract={onOpenExtract}
   150→        onOpenDefault={handleOpenDefault}
   151→        modalizeRef={modalizeRef}
   152→        isMyOrders={isMyOrders}
   153→        setIsMyOrders={setIsMyOrders}
   154→        onMenuPress={handleOpenMyOrders}
   155→      />
   156→    );
   157→  }, [products, points, isLoadingBalance, isLoadingProducts, isErrorProducts, isErrorBalance, navigation, buildPointsCardItems, isMyOrders]);
   158→
   159→  return view;
   160→};
   161→
   162→export default Points;
