    88→  date: string;
    89→  items: propsItemsListCardTransition[];

    92→function getGroupedTransactionsByDate(
    93→  transactions: PointsTransaction[],
    94→): GroupedTransactions[] {
    95→  if (transactions.length === 0) {
    96→    return [];
    97→  }
    98→
    99→  const sortedTransactions = [...transactions].sort((a, b) =>
   100→    moment(b.date).diff(moment(a.date)),
   101→  );
   102→
   103→  const groupedByDate = sortedTransactions.reduce((acc, transaction) => {
   104→    const formattedDate = getFormattedDateForHeader(transaction.date);
   105→
   106→    if (!acc[formattedDate]) {
   107→      acc[formattedDate] = [];
   108→    }
   109→
   110→    acc[formattedDate].push(
   111→      getTransactionCardItem(transaction) as propsItemsListCardTransition,
   112→    );
   113→    return acc;
   114→  }, {} as Record<string, propsItemsListCardTransition[]>);
   115→
   116→  const getSortedItemsByDate = (
   117→    items: propsItemsListCardTransition[],
   118→    date: string,
   119→  ) => {
   120→    return items.sort((a, b) => {
   121→      const transactionA = sortedTransactions.find(
   122→        (t) =>
   123→          getFormattedDateForHeader(t.date) === date &&
   124→          getTransactionCardItem(t).title === a.title,
   125→      );
   126→      const transactionB = sortedTransactions.find(
   127→        (t) =>
   128→          getFormattedDateForHeader(t.date) === date &&
   129→          getTransactionCardItem(t).title === b.title,
   130→      );
   131→
   132→      if (transactionA && transactionB) {
   133→        return moment(transactionB.date).diff(moment(transactionA.date));
   134→      }
   135→      return 0;
   136→    });
   137→  };
   138→
   139→  const getSortedByDate = (a: GroupedTransactions, b: GroupedTransactions) => {
   140→    const dateA = moment(a.date.split(' ').slice(1).join(' '), 'DD MMMM');
   141→    const dateB = moment(b.date.split(' ').slice(1).join(' '), 'DD MMMM');
   142→    return dateB.diff(dateA);
   143→  };
   144→
   145→  return Object.entries(groupedByDate)
   146→    .map(([date, items]) => ({
   147→      date,
   148→      items: getSortedItemsByDate(items as propsItemsListCardTransition[], date),
   149→    }))
   150→    .sort(getSortedByDate);
   151→}

   190→export function processPointsExtractData(
   191→  extractData: PointsExtractData | null | undefined,
   192→): ProcessedExtractData {
   193→  if (!extractData || typeof extractData !== 'object') {
   194→    return DEFAULT_PROCESSED_EXTRACT_DATA;
   195→  }
   196→
   197→  try {
   198→    const transactions = extractData.transactions || [];
   199→    const groupedTransactions = getGroupedTransactionsByDate(transactions);
   200→    const mappedTransactions = PointsDataGroupedByDayMapper.getMappedData(groupedTransactions);
   201→
   202→    return {
   203→      groupedTransactions,
   204→      mappedTransactions,
   205→    };
   206→  } catch (error) {
   207→    console.warn('Erro ao processar dados do extrato:', error);
   208→    return DEFAULT_PROCESSED_EXTRACT_DATA;
   209→  }
   210→}
