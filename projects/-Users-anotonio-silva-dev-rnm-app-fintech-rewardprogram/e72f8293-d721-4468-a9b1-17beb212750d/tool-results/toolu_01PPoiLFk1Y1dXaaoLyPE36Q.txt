     1→import React, { FC, useEffect, useState, useLayoutEffect, useMemo, useRef } from 'react';
     2→import { NavigationProp, useNavigation, useIsFocused } from '@react-navigation/native';
     3→import { Dimensions } from 'react-native';
     4→import { useFocusEffect } from '@react-navigation/native';
     5→
     6→import { useScreenViewTracking } from '../../hooks/tracking';
     7→
     8→import theme from '../../styles/theme';
     9→import { CashbackInvoiceItem, InvoiceCashbackProps } from './models';
    10→import { PropsRootStack } from '../../routes/Models';
    11→import InvoiceCashbackSummaryView from './view';
    12→import { useInvoiceCashbackPermission } from '../../services/hooks';
    13→import { useRootStore } from '~/store';
    14→import { postRescue, clearRescue } from '../../services/services';
    15→import { RescueStatus, RescuePayload } from '../../services/models';
    16→
    17→const InvoiceCashbackSummary: FC<InvoiceCashbackProps> = ({ route }) => {
    18→  useScreenViewTracking({ route, title: 'Cashback na fatura' });
    19→  const navigation = useNavigation<NavigationProp<PropsRootStack>>();
    20→  const isFocused = useIsFocused();
    21→  const { height } = Dimensions.get('window');
    22→  const { hasRedeemed, markAsRedeemed } = useInvoiceCashbackPermission();
    23→  const alreadyMarkedRef = useRef(false);
    24→  const [hasAttemptedRescue, setHasAttemptedRescue] = useState(false);
    25→
    26→  const { conversionFactor } = useRootStore((state) => state.pointsProducts);
    27→  const { status: rescueStatus, response: rescueResponse } = useRootStore((state) => state.rescue);
    28→
    29→  const isRescueLoading = rescueStatus === RescueStatus.LOADING;
    30→  const isRescueFailed = rescueStatus === RescueStatus.ERROR;
    31→  const isRescueConcluded = rescueStatus === RescueStatus.SUCCESS && !!rescueResponse;
    32→
    33→  useLayoutEffect(() => {
    34→    if (isFocused) {
    35→      setHasAttemptedRescue(false);
    36→      clearRescue();
    37→    }
    38→  }, [isFocused]);
    39→
    40→  useFocusEffect(
    41→    React.useCallback(() => {
    42→      setHasAttemptedRescue(false);
    43→      clearRescue();
    44→      return undefined;
    45→    }, [])
    46→  );
    47→
    48→  const handleGoBack = () => {
    49→    setHasAttemptedRescue(false);
    50→    clearRescue();
    51→    navigation.goBack();
    52→  };
    53→
    54→  const invoiceList = (pts: number): CashbackInvoiceItem[] => {
    55→    const whiteShape = theme.colors.LIGHT_BASE_01;
    56→    const fintechColor = theme.colors.SECONDARY_COLOR_02;
    57→    const finalPrice = pts * conversionFactor;
    58→    const formattedPrice = new Intl.NumberFormat('pt-BR', {
    59→      style: 'currency',
    60→      currency: 'BRL',
    61→    }).format(finalPrice);
    62→
    63→    return [
    64→      {
    65→        type: 'default',
    66→        icon: 'icon-switch-vertical',
    67→        iconColor: fintechColor,
    68→        iconColorShape: whiteShape,
    69→        description: `Cada ${pts.toLocaleString('pt-br')} pontos valem ${formattedPrice} no cashback`,
    70→      },
    71→      {
    72→        type: 'default',
    73→        icon: 'icon-calendar',
    74→        iconColor: fintechColor,
    75→        iconColorShape: whiteShape,
    76→        description: 'Você define quanto cashback quer aplicar na sua fatura aberta',
    77→      },
    78→      {
    79→        type: 'default',
    80→        icon: 'icon-payment-invoice',
    81→        iconColor: fintechColor,
    82→        iconColorShape: whiteShape,
    83→        description: 'Se o valor for maior que o da fatura, a diferença será aplicada nas próximas',
    84→      },
    85→      {
    86→        type: 'default',
    87→        icon: 'icon-clock',
    88→        iconColor: fintechColor,
    89→        iconColorShape: whiteShape,
    90→        description: 'O cashback acontece em até 72h',
    91→      },
    92→    ];
    93→  };
    94→
    95→  const pointsSelected = route?.params?.pointsSelected ?? 0;
    96→  const reaisSelected = route?.params?.reaisSelected ?? 0;
    97→  const productCode = route?.params?.productCode;
    98→  const campaingId = route?.params?.campaingId;
    99→
   100→  const onConfirmRescue = async () => {
   101→    setHasAttemptedRescue(true);
   102→    const payload: RescuePayload = {
   103→      value: pointsSelected,
   104→      campaignId: campaingId,
   105→      productCode: productCode,
   106→      description: 'Cashback resgatado',
   107→    };
   108→    await postRescue(payload);
   109→  };
   110→
   111→  const onExitResult = () => {
   112→    setHasAttemptedRescue(false);
   113→    clearRescue();
   114→    navigation.navigate('RewardProgram');
   115→  };
   116→
   117→  const showResult = isFocused && hasAttemptedRescue && (isRescueConcluded || isRescueFailed);
   118→
   119→  const items = invoiceList(pointsSelected).map((item) => ({
   120→    icon: item.icon,
   121→    iconColor: item.iconColor,
   122→    description: item.description,
   123→    iconColorShape: item.iconColorShape,
   124→    type: item.type,
   125→  }));
   126→
   127→  const modalizeRef1 = useRef(null);
   128→
   129→  const onOpenDefault = () => {
   130→    modalizeRef1.current?.open();
   131→  };
   132→
   133→  useEffect(() => {
   134→    if (showResult && isRescueConcluded && !alreadyMarkedRef.current && hasRedeemed !== true) {
   135→      alreadyMarkedRef.current = true;
   136→      markAsRedeemed().catch(() => {});
   137→    }
   138→  }, [showResult, isRescueConcluded, hasRedeemed, markAsRedeemed]);
   139→
   140→  const nfPtBR = useMemo(() => new Intl.NumberFormat('pt-BR'), []);
   141→  const getFormatPoints = (n: number) => nfPtBR.format(Math.max(0, Math.floor(n || 0)));
   142→  const formatCurrencyBR = (n: number) =>
   143→    new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' })
   144→      .format(Number.isFinite(n) ? n : 0)
   145→      .replace(/\s/g, '');
   146→
   147→  const formattedSelectedPoints = useMemo(
   148→    () => getFormatPoints(pointsSelected || 0),
   149→    [pointsSelected]
   150→  );
   151→
   152→  const formattedSelectedReais = useMemo(
   153→    () => formatCurrencyBR(reaisSelected || 0),
   154→    [reaisSelected]
   155→  );
   156→
   157→  const effectiveShowResult =
   158→    typeof showResult !== 'undefined' ? showResult : !!(isRescueConcluded || isRescueFailed);
   159→  const screenTitle = effectiveShowResult ? 'Conclusão' : 'Cashback na fatura';
   160→
   161→  return (
   162→    <InvoiceCashbackSummaryView
   163→      handleGoBack={handleGoBack}
   164→      onConfirmRescue={onConfirmRescue}
   165→      onExitResult={onExitResult}
   166→      isRescueLoading={isRescueLoading}
   167→      isRescueConcluded={isRescueConcluded}
   168→      isRescueFailed={isRescueFailed}
   169→      items={items}
   170→      effectiveShowResult={showResult}
   171→      onOpenDefault={onOpenDefault}
   172→      modalizeRef1={modalizeRef1}
   173→      formattedSelectedPoints={formattedSelectedPoints}
   174→      formattedSelectedReais={formattedSelectedReais}
   175→      screenTitle={screenTitle}
   176→      height={height}
   177→    />
   178→  );
   179→};
   180→
   181→export default InvoiceCashbackSummary;
