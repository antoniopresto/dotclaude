     1→import React, { FC, useEffect, useMemo, useRef, useState } from 'react';
     2→import { NavigationProp, useFocusEffect, useNavigation } from '@react-navigation/native';
     3→import { Dimensions } from 'react-native';
     4→
     5→import { useScreenViewTracking } from '../../hooks/tracking';
     6→
     7→import theme from '../../styles/theme';
     8→import { CashbackInvoiceItem, InvoiceCashbackProps } from './models';
     9→import { PropsRootStack } from '../../routes/Models';
    10→import InvoiceCashbackPointsView from './view';
    11→import { eventScreenViewV3, eventSelectContent, eventModalView } from '~/constants/functions';
    12→import {
    13→  getFormattedPoints,
    14→  formatCurrencyBR,
    15→  getParsedCurrency,
    16→  getParsePoints,
    17→} from '../../utils/functions';
    18→import { useRootStore } from '~/store';
    19→import { getParameterPointsProgram } from '../../services/services';
    20→import { ParameterPointsProgramStatus } from '../../services/models';
    21→
    22→const InvoiceCashbackPoints: FC<InvoiceCashbackProps> = ({ route }) => {
    23→  useScreenViewTracking({ route, title: 'Cashback na fatura' });
    24→  const { height, width } = Dimensions.get('window');
    25→
    26→  const { points } = useRootStore((state) => state.pointsBalance);
    27→  const { pointsQuantity, status: paramStatus } = useRootStore((state) => state.parameterPointsProgram);
    28→
    29→  const isLoadingParameterization = paramStatus === ParameterPointsProgramStatus.LOADING;
    30→
    31→  useEffect(() => {
    32→    getParameterPointsProgram();
    33→  }, []);
    34→
    35→  const hasFocused = useRef(false);
    36→  useFocusEffect(
    37→    React.useCallback(() => {
    38→      if (!hasFocused.current) {
    39→        eventScreenViewV3('desconto_fatura', 'hub_recompensas_spmais', 'monetizacao', {
    40→          location_1: 'monetizacao',
    41→          location_2: 'hub_recompensas_spmais',
    42→          location_3: 'desconto_fatura',
    43→        });
    44→        hasFocused.current = true;
    45→      }
    46→      return () => {
    47→        hasFocused.current = false;
    48→      };
    49→    }, [])
    50→  );
    51→
    52→  const navigation = useNavigation<NavigationProp<PropsRootStack>>();
    53→
    54→  const conversionFactor = route?.params?.conversionFactor ?? 0;
    55→  const pointsPrice = Number((Math.trunc(points || 0) * conversionFactor).toFixed(2));
    56→
    57→  const finalPrice = pointsPrice || 0;
    58→  const [pointsValue, setPointsValue] = useState('');
    59→  const [reaisValue, setReaisValue] = useState('');
    60→  const [pointsError, setPointsError] = useState(false);
    61→  const [reaisError, setReaisError] = useState(false);
    62→  const convertionPrice = pointsQuantity * conversionFactor;
    63→  const convertionValue = new Intl.NumberFormat('pt-BR', {
    64→    style: 'currency',
    65→    currency: 'BRL',
    66→  }).format(convertionPrice);
    67→
    68→  const handleGoBack = () => {
    69→    navigation.goBack();
    70→  };
    71→
    72→  const handleContinue = () => {
    73→    eventSelectContent({
    74→      content_type: 'cta',
    75→      content_id: 'continuar',
    76→      location_1: 'monetizacao',
    77→      location_2: 'hub_recompensas_spmais',
    78→      location_3: 'desconto_fatura',
    79→    });
    80→
    81→    const reaisVal = conversionFactor * getParsePoints(pointsValue);
    82→
    83→    navigation.navigate('InvoiceCashbackSummary', {
    84→      pointsSelected: getParsePoints(pointsValue),
    85→      reaisSelected: reaisVal,
    86→      conversionFactor,
    87→      productCode: route?.params?.productCode,
    88→      campaingId: route?.params?.campaingId,
    89→    });
    90→  };
    91→
    92→  const invoiceList = (): CashbackInvoiceItem[] => {
    93→    const whiteShape = theme.colors.LIGHT_BASE_01;
    94→    const fintechColor = theme.colors.SECONDARY_COLOR_02;
    95→
    96→    return [
    97→      {
    98→        type: 'default',
    99→        icon: 'icon-switch-vertical',
   100→        iconColor: fintechColor,
   101→        iconColorShape: whiteShape,
   102→        description: `Cada ${pointsQuantity.toLocaleString('pt-br')} pontos valem ${convertionValue} no cashback`,
   103→      },
   104→      {
   105→        type: 'default',
   106→        icon: 'icon-calendar',
   107→        iconColor: fintechColor,
   108→        iconColorShape: whiteShape,
   109→        description: 'Você define quanto cashback quer aplicar na sua fatura aberta',
   110→      },
   111→      {
   112→        type: 'default',
   113→        icon: 'icon-payment-invoice',
   114→        iconColor: fintechColor,
   115→        iconColorShape: whiteShape,
   116→        description: 'Se o valor for maior que o da fatura, a diferença será aplicada nas próximas',
   117→      },
   118→      {
   119→        type: 'default',
   120→        icon: 'icon-clock',
   121→        iconColor: fintechColor,
   122→        iconColorShape: whiteShape,
   123→        description: 'O cashback acontece em até 72h',
   124→      },
   125→    ];
   126→  };
   127→
   128→  const rate = useMemo(() => {
   129→    if (conversionFactor && conversionFactor > 0) {
   130→      return conversionFactor;
   131→    }
   132→    return 0;
   133→  }, [conversionFactor]);
   134→
   135→  const items = invoiceList().map((item) => ({
   136→    icon: item.icon,
   137→    iconColor: item.iconColor,
   138→    description: item.description,
   139→    iconColorShape: item.iconColorShape,
   140→    type: item.type,
   141→  }));
   142→
   143→  const handleChangePoints = (text: string) => {
   144→    const val = getParsePoints(text);
   145→    const formatted = val ? getFormattedPoints(val) : '';
   146→    setPointsValue(formatted);
   147→
   148→    if (rate > 0) {
   149→      setReaisValue(val > 0 ? formatCurrencyBR(val * rate) : '');
   150→    }
   151→    if (val > (Math.trunc(Number(points)) || 0)) {
   152→      setPointsError(true);
   153→    } else {
   154→      setPointsError(false);
   155→    }
   156→  };
   157→
   158→  const handleChangeReais = (text: string) => {
   159→    const val = getParsedCurrency(text);
   160→    const formatted = val > 0 ? formatCurrencyBR(val) : '';
   161→    setReaisValue(formatted);
   162→
   163→    if (rate > 0) {
   164→      const ptsFromReais = Math.round(val / rate);
   165→      setPointsValue(ptsFromReais > 0 ? getFormattedPoints(ptsFromReais) : '');
   166→    }
   167→    if (val > finalPrice) {
   168→      setReaisError(true);
   169→    } else {
   170→      setReaisError(false);
   171→    }
   172→  };
   173→
   174→  const handleUseAll = () => {
   175→    setPointsValue(getFormattedPoints(points || 0));
   176→    setReaisValue(formatCurrencyBR(finalPrice));
   177→    setPointsError(false);
   178→    setReaisError(false);
   179→  };
   180→
   181→  useFocusEffect(
   182→    React.useCallback(() => {
   183→      setPointsValue('');
   184→      setReaisValue('');
   185→      setPointsError(false);
   186→      setReaisError(false);
   187→      return () => {
   188→        setPointsValue('');
   189→        setReaisValue('');
   190→        setPointsError(false);
   191→        setReaisError(false);
   192→      };
   193→    }, [])
   194→  );
   195→
   196→  const modalizeRef1 = useRef(null);
   197→
   198→  const onOpenDefault = () => {
   199→    eventSelectContent({
   200→      content_type: 'cta',
   201→      content_id: 'informacoes',
   202→      location_1: 'monetizacao',
   203→      location_2: 'hub_recompensas_spmais',
   204→      location_3: 'desconto_fatura',
   205→    });
   206→
   207→    modalizeRef1.current?.open();
   208→
   209→    eventModalView({
   210→      content_id: 'info_desconto_fatura',
   211→      location_1: 'monetizacao',
   212→      location_2: 'hub_recompensas_spmais',
   213→      location_3: 'desconto_fatura',
   214→    });
   215→  };
   216→
   217→  const handleModalizeContinue = () => {
   218→    eventSelectContent({
   219→      content_type: 'cta',
   220→      content_id: 'entendi',
   221→      location_1: 'monetizacao',
   222→      location_2: 'hub_recompensas_spmais',
   223→      location_3: 'info_desconto_fatura',
   224→    });
   225→
   226→    modalizeRef1.current?.close();
   227→  };
   228→
   229→  const handleSelectEvent = (contentId: string) => {
   230→    eventSelectContent({
   231→      content_type: 'form',
   232→      content_id: contentId,
   233→      location_1: 'monetizacao',
   234→      location_2: 'hub_recompensas_spmais',
   235→      location_3: 'desconto_fatura',
   236→    });
   237→  };
   238→
   239→  return (
   240→    <InvoiceCashbackPointsView
   241→      handleGoBack={handleGoBack}
   242→      cashbackInvoiceList={invoiceList}
   243→      points={Math.trunc(Number(points)) || 0}
   244→      handleContinue={handleContinue}
   245→      price={pointsPrice}
   246→      conversionFactor={conversionFactor}
   247→      formattedPrice={formatCurrencyBR(finalPrice)}
   248→      handleChangeReais={handleChangeReais}
   249→      pointsValue={pointsValue}
   250→      reaisValue={reaisValue}
   251→      convertionValue={convertionValue}
   252→      pointsQuantity={pointsQuantity}
   253→      pointsError={pointsError}
   254→      reaisError={reaisError}
   255→      handleChangePoints={handleChangePoints}
   256→      handleUseAll={handleUseAll}
   257→      items={items}
   258→      handleModalizeContinue={handleModalizeContinue}
   259→      onOpenDefault={onOpenDefault}
   260→      modalizeRef1={modalizeRef1}
   261→      handleSelectEvent={handleSelectEvent}
   262→      height={height}
   263→      width={width}
   264→      isLoadingParameterization={isLoadingParameterization}
   265→    />
   266→  );
   267→};
   268→
   269→export default InvoiceCashbackPoints;
