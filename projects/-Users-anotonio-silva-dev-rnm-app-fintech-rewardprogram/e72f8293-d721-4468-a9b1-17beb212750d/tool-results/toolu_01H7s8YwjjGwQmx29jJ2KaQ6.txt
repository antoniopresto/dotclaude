    13→  const { hasRedeemed, status } = useRootStore((state) => state.rewardsPermission);

    15→  const isLoading = status === PermissionStatus.LOADING;
    16→  const isError = status === PermissionStatus.ERROR;
    17→
    18→  const checkHasRedeemed = useCallback(
    19→    async ({ forceRefresh = false }: { forceRefresh?: boolean } = {}): Promise<boolean> => {
    20→      if (hasRedeemed !== null && !forceRefresh) {
    21→        return hasRedeemed;
    22→      }
    23→
    24→      if (!accountId) {
    25→        throw new Error('Expected accountId to be defined.');
    26→      }
    27→
    28→      return checkHasRedeemedService(accountId);
    29→    },
    30→    [accountId, hasRedeemed]
    31→  );
    32→
    33→  const markAsRedeemed = useCallback(async (): Promise<boolean> => {
    34→    if (!accountId) {
    35→      return false;
    36→    }
    37→
    38→    return markAsRedeemedService(accountId);
    39→  }, [accountId]);
    40→
    41→  return {
    42→    hasRedeemed,
    43→    isLoading,
    44→    isError,
    45→    checkHasRedeemed,
