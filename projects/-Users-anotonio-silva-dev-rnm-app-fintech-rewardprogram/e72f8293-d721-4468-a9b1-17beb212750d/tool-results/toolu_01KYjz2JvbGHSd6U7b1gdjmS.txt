     1→import React, { useCallback, useMemo, useRef, useState } from "react"
     2→import { NavigationProp, useFocusEffect, useNavigation } from "@react-navigation/native";
     3→import { PropsRootStack } from "../../routes/Models";
     4→import { format, parseISO } from "date-fns";
     5→import { PropsModalizeRef } from "@semparar/design-system-app/src/@types/modelize";
     6→import { eventSelectContent, eventScreenViewV3, eventModalView } from '~/constants/functions';
     7→import { getPointsExtract } from '../../services/services';
     8→import { useShallow } from "zustand/react/shallow";
     9→import { useRootStore } from '~/store';
    10→import { AppliedFilters, PERIOD_DAYS_MAP, PointsExtractStatus, PointsTransaction } from "../../services/models";
    11→import { Dimensions } from "react-native";
    12→import { FilterGroup } from "../Extract/Models";
    13→import { getFilteredTransactions } from "../Extract/utils";
    14→import MyOrdersRewardProgramView from "./view";
    15→
    16→
    17→const MyOrdersRewardProgram = () => {
    18→
    19→    const [isRefreshing, setIsRefreshing] = useState(false);
    20→
    21→    const modalizeRefFilters = useRef<PropsModalizeRef>(null);
    22→    const emptyImage = "https://image.mkt-semparar.com.br/lib/fe2f117170640475721171/m/1/shopping_cart.png"
    23→
    24→    const SKELETON_COUNT = 3;
    25→    const { width } = Dimensions.get('window');
    26→
    27→    const { pointsExtract, pointsExtractFilters, pointsExtractActions } =
    28→        useRootStore(
    29→            useShallow((state) => ({
    30→                pointsExtract: state.pointsExtract,
    31→                pointsExtractFilters: state.pointsExtractFilters,
    32→                pointsExtractActions: state.pointsExtractActions
    33→            })),
    34→        );
    35→
    36→    const appliedTransactionsFilters = pointsExtractFilters.appliedTransactionsFilters;
    37→
    38→    const getFormatDate = (iso?: string) => {
    39→        if (!iso) return '';
    40→        const date = parseISO(iso);
    41→        return format(date, 'dd/MM/yyyy');
    42→    };
    43→
    44→    const periodFilters = [
    45→        { label: '7 dias', value: 0 },
    46→        { label: '15 dias', value: 1 },
    47→        { label: '30 dias', value: 2 },
    48→        { label: '60 dias', value: 3 },
    49→        { label: '90 dias', value: 4 },
    50→    ]
    51→
    52→
    53→    const typeFilters = [
    54→        { label: 'Mais recentes', value: 'ASC' },
    55→        { label: 'Mais antigas', value: 'DESC' },
    56→    ]
    57→
    58→    const getFilterGroups: FilterGroup[] = useMemo(() => {
    59→        return [{
    60→            label: 'Períodos',
    61→            options: periodFilters,
    62→        }, {
    63→            label: 'Relevância',
    64→            options: typeFilters,
    65→        }];
    66→    }, []);
    67→
    68→    const clearTransactionsFilters = useCallback(() => {
    69→        pointsExtractActions.clearTransactionsFilters();
    70→    }, [pointsExtractActions]);
    71→
    72→    const setTransactionsFilters = useCallback(
    73→        (filters: AppliedFilters) => {
    74→            pointsExtractActions.setTransactionsFilters(filters);
    75→        },
    76→        [pointsExtractActions],
    77→    );
    78→
    79→    const handleApplyFilters = useCallback(
    80→        (filters: Record<number, unknown>) => {
    81→            const period = filters[0] as number | undefined;
    82→            const types = filters[1] as string | undefined;
    83→
    84→            const hasPeriod = period !== undefined && period !== null;
    85→            const hasOrdering = types !== undefined && types !== null;
    86→
    87→            const parts: string[] = ['filtrar'];
    88→
    89→            if (hasPeriod) {
    90→                const periodDay = periodFilters[period].label.split(' ')[0];
    91→                parts.push(`${periodDay}_dias`);
    92→            }
    93→
    94→            if (hasOrdering) {
    95→                const label = typeFilters.find(
    96→                    item => item.value === types
    97→                )?.label;
    98→
    99→                if (label) {
   100→                    parts.push(label.toLowerCase());
   101→                }
   102→            }
   103→
   104→            const trackingLabel = parts.join('_');
   105→
   106→            eventSelectContent({
   107→                content_type: 'cta',
   108→                content_id: `${trackingLabel}`,
   109→                location_1: 'monetizacao',
   110→                location_2: 'hub_recompensas_spmais',
   111→                location_3: 'filtro_meus_pedidos',
   112→            });
   113→
   114→            if (!hasPeriod && !hasOrdering) {
   115→                clearTransactionsFilters();
   116→            } else {
   117→                setTransactionsFilters({
   118→                    period: hasPeriod ? period : undefined,
   119→                    order: hasOrdering ? [types] : undefined,
   120→                });
   121→            }
   122→        },
   123→        [setTransactionsFilters, clearTransactionsFilters],
   124→    );
   125→
   126→    const handleOpenFilters = () => {
   127→        eventSelectContent({
   128→            content_type: 'cta',
   129→            content_id: 'filtro_pedidos',
   130→            location_1: 'monetizacao',
   131→            location_2: 'hub_recompensas_spmais',
   132→            location_3: 'meus_pedidos',
   133→        });
   134→
   135→        modalizeRefFilters.current?.open?.();
   136→
   137→        eventModalView({
   138→            content_id: 'filtro_meus_pedidos',
   139→            location_1: 'monetizacao',
   140→            location_2: 'hub_recompensas_spmais',
   141→            location_3: 'meus_pedidos',
   142→        });
   143→    }
   144→
   145→    const handleDetailsOrders = () => {
   146→        eventSelectContent({
   147→            content_type: 'cta',
   148→            content_id: 'detalhe_pedido',
   149→            location_1: 'monetizacao',
   150→            location_2: 'hub_recompensas_spmais',
   151→            location_3: 'meus_pedidos',
   152→        });
   153→    }
   154→
   155→    const getTransactionsWithFilters = useCallback(
   156→        (transactions: PointsTransaction[]): PointsTransaction[] => {
   157→            let filtered = transactions;
   158→            if (appliedTransactionsFilters) {
   159→                const filtersWithDays: AppliedFilters = {
   160→                    ...appliedTransactionsFilters,
   161→                    period: appliedTransactionsFilters.period !== undefined
   162→                        ? PERIOD_DAYS_MAP[appliedTransactionsFilters.period]
   163→                        : undefined,
   164→                };
   165→                filtered = getFilteredTransactions(filtered, filtersWithDays);
   166→            }
   167→
   168→            return filtered;
   169→        },
   170→        [appliedTransactionsFilters],
   171→    );
   172→
   173→    const getActiveFiltersCount = () => {
   174→        let count = 0;
   175→        if (appliedTransactionsFilters?.period !== undefined && appliedTransactionsFilters?.period !== null) count++;
   176→        if (appliedTransactionsFilters?.order?.length) count++;
   177→        return count > 0 ? count.toString() : undefined;
   178→    }
   179→
   180→    const getInitialFilters = useMemo(() => {
   181→        const filters: Record<number, unknown> = {};
   182→        if (appliedTransactionsFilters?.period !== undefined && appliedTransactionsFilters?.period !== null) {
   183→            filters[0] = appliedTransactionsFilters.period;
   184→        }
   185→        if (appliedTransactionsFilters?.order?.length) {
   186→            filters[1] = appliedTransactionsFilters.order[0];
   187→        }
   188→        return filters;
   189→    }, [appliedTransactionsFilters]);
   190→
   191→    const getOrdersTransactions = () => {
   192→        if (!pointsExtract.extractData) return [];
   193→
   194→        const transactions = getTransactionsWithFilters(
   195→            pointsExtract.extractData.transactions,
   196→        );
   197→
   198→        return transactions
   199→    }
   200→    const ordersTransactionsData = getOrdersTransactions()
   201→
   202→    const isLoading = pointsExtract.status === PointsExtractStatus.LOADING
   203→    const isError = pointsExtract.status === PointsExtractStatus.ERROR
   204→
   205→    const navigation = useNavigation<NavigationProp<PropsRootStack>>();
   206→
   207→    const handleGoBack = () => {
   208→        clearTransactionsFilters()
   209→        pointsExtractActions.reset()
   210→        navigation.goBack()
   211→    }
   212→
   213→    const handleRefresh = async () => {
   214→        setIsRefreshing(true);
   215→        await getPointsExtract({ type: 'Resgate' });
   216→        setIsRefreshing(false);
   217→    };
   218→
   219→    const hasFocused = useRef(false);
   220→    useFocusEffect(
   221→        React.useCallback(() => {
   222→            if (!hasFocused.current) {
   223→                eventScreenViewV3(
   224→                    'meus_pedidos',
   225→                    'hub_recompensas_spmais',
   226→                    'monetizacao',
   227→                    {
   228→                        location_1: 'monetizacao',
   229→                        location_2: 'hub_recompensas_spmais',
   230→                        location_3: 'meus_pedidos',
   231→                    }
   232→                );
   233→                pointsExtractActions.reset()
   234→                getPointsExtract({ type: 'Resgate' });
   235→                hasFocused.current = true;
   236→            }
   237→            return () => {
   238→                hasFocused.current = false;
   239→            };
   240→        }, [])
   241→    )
   242→
   243→    return (
   244→        <MyOrdersRewardProgramView
   245→            myOrders={ordersTransactionsData}
   246→            emptyImage={emptyImage}
   247→            isLoading={isLoading}
   248→            isError={isError}
   249→            filterGroups={getFilterGroups}
   250→            modalizeRefFilters={modalizeRefFilters}
   251→            formatDate={getFormatDate}
   252→            handleGoBack={handleGoBack}
   253→            handleOpenFilters={handleOpenFilters}
   254→            handleDetailsOrders={handleDetailsOrders}
   255→            isRefreshing={isRefreshing}
   256→            handleRefresh={handleRefresh}
   257→            skeletonCount={SKELETON_COUNT}
   258→            screenWidth={width}
   259→            handleApplyFilters={handleApplyFilters}
   260→            initialFilters={getInitialFilters}
   261→            activeFiltersCount={getActiveFiltersCount()}
   262→
   263→        />
   264→    )
   265→
   266→}
   267→
   268→export default MyOrdersRewardProgram;
   269→
