     1→//! Harness CLI - Database utility for AI agents and humans.
     2→//!
     3→//! This CLI provides programmatic access to the Harness storage layer,
     4→//! enabling AI agents to manage tasks, progress, and sessions.
     5→//!
     6→//! ## Usage
     7→//!
     8→//! ```bash
     9→//! harness init                              # Create .harness/harness.db
    10→//! harness db status                         # Show DB info
    11→//! harness task add "Title" [-d "desc"]      # Add a task
    12→//! harness task list [--status=TODO]         # List tasks
    13→//! harness session start                     # Start work session
    14→//! ```
    15→

    28→#[command(propagate_version = true)]

   354→
   355→fn main() -> ExitCode {
   356→    let cli = Cli::parse();
   357→
   358→    // Determine project directory
   359→    let project_dir = cli.dir.unwrap_or_else(|| std::env::current_dir().unwrap_or_default());
   360→
   361→    let result = match cli.command {
   362→        Commands::Init => cmd_init(&project_dir, cli.format),
   363→        Commands::Db { command } => match command {
   364→            DbCommands::Status => cmd_db_status(&project_dir, cli.format),
   365→        },
   366→        Commands::Task { command } => cmd_task(&project_dir, command, cli.format),
   367→        Commands::Step { command } => cmd_step(&project_dir, command, cli.format),
   368→        Commands::Session { command } => cmd_session(&project_dir, command, cli.format),
   369→        Commands::Progress { command } => cmd_progress(&project_dir, command, cli.format),
   370→        Commands::Status => cmd_status(&project_dir, cli.format),
   371→    };
   372→
   373→    match result {
   374→        Ok(()) => ExitCode::SUCCESS,
   375→        Err(e) => {
   376→            eprintln!("Error: {}", e);
   377→            match e {
   378→                CliError::NotHarnessProject => ExitCode::from(2),
   379→                CliError::Database(_) => ExitCode::from(3),
   380→                _ => ExitCode::FAILURE,
   381→            }
   382→        }
   383→    }
   384→}
   385→
   386→// =============================================================================
   387→// ERROR HANDLING
   388→// =============================================================================
   389→
   390→#[derive(Debug)]
   391→enum CliError {
   392→    NotHarnessProject,
   393→    Database(db::DbError),
   394→    NotFound(String),
   395→    NoActiveSession,
   396→    Io(std::io::Error),
   397→}
   398→
   399→impl std::fmt::Display for CliError {
   400→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   401→        match self {
   402→            CliError::NotHarnessProject => write!(f, "Not a Harness project (no .harness/ folder)"),
   403→            CliError::Database(e) => write!(f, "Database error: {}", e),
   404→            CliError::NotFound(what) => write!(f, "Not found: {}", what),
   405→            CliError::NoActiveSession => write!(f, "No active session"),
   406→            CliError::Io(e) => write!(f, "IO error: {}", e),
   407→        }
   408→    }
   409→}
   410→
   411→impl From<db::DbError> for CliError {
   412→    fn from(e: db::DbError) -> Self {
   413→        CliError::Database(e)
   414→    }
   415→}
   416→
   417→impl From<std::io::Error> for CliError {
   418→    fn from(e: std::io::Error) -> Self {
   419→        CliError::Io(e)
   420→    }
   421→}
   422→
   423→type CliResult<T> = Result<T, CliError>;

   429→fn cmd_init(project_dir: &PathBuf, format: OutputFormat) -> CliResult<()> {
   430→    let harness_dir = project_dir.join(db::HARNESS_FOLDER);
   431→
   432→    // Create .harness/ if it doesn't exist
   433→    if !harness_dir.exists() {
   434→        std::fs::create_dir_all(&harness_dir)?;
   435→    }
   436→
   437→    // Open or create the database
   438→    let conn = db::open_or_create(project_dir)?;
   439→
   440→    let db_path = db::get_db_path(project_dir);
   441→
   442→    match format {
   443→        OutputFormat::Json => {
   444→            println!(
   445→                r#"{{"success":true,"path":"{}","version":"{}"}}"#,
   446→                db_path.display(),
   447→                db::SCHEMA_VERSION
   448→            );
   449→        }
   450→        OutputFormat::Quiet => {
   451→            println!("{}", db_path.display());
   452→        }
   453→        OutputFormat::Text => {
   454→            println!("Initialized Harness database at {}", db_path.display());
   455→            println!("Schema version: {}", db::SCHEMA_VERSION);
   456→
   457→            // Show stats
   458→            let stats = db::DbStats::collect(&conn, Some(&db_path))?;
   459→            println!("\nDatabase stats:");
   460→            println!("  Tasks:    {}", stats.task_count);
   461→            println!("  Steps:    {}", stats.step_count);
   462→            println!("  Sessions: {}", stats.session_count);
   463→            println!("  Progress: {}", stats.progress_count);
   464→        }
   465→    }
   466→
   467→    Ok(())
   468→}
   469→
   470→fn cmd_db_status(project_dir: &PathBuf, format: OutputFormat) -> CliResult<()> {
   471→    if !db::has_harness_folder(project_dir) {
   472→        return Err(CliError::NotHarnessProject);
   473→    }
   474→
   475→    let conn = db::open(project_dir)?;
   476→    let db_path = db::get_db_path(project_dir);
   477→    let stats = db::DbStats::collect(&conn, Some(&db_path))?;
   478→    let version = db::get_schema_version(&conn)?.unwrap_or_else(|| "unknown".to_string());
   479→
   480→    match format {
   481→        OutputFormat::Json => {
   482→            println!(
   483→                r#"{{"path":"{}","version":"{}","size_bytes":{},"tasks":{},"steps":{},"sessions":{},"progress":{}}}"#,
   484→                db_path.display(),
   485→                version,
   486→                stats.file_size_bytes,
   487→                stats.task_count,
   488→                stats.step_count,
   489→                stats.session_count,
   490→                stats.progress_count
   491→            );
   492→        }
   493→        OutputFormat::Quiet => {
   494→            println!("{}", db_path.display());
   495→        }
   496→        OutputFormat::Text => {
   497→            println!("Database: {}", db_path.display());
   498→            println!("Schema version: {}", version);
   499→            println!("Size: {} bytes", stats.file_size_bytes);
   500→            println!();
   501→            println!("Counts:");
   502→            println!("  Tasks:    {}", stats.task_count);
   503→            println!("  Steps:    {}", stats.step_count);
   504→            println!("  Sessions: {}", stats.session_count);
   505→            println!("  Progress: {}", stats.progress_count);
   506→        }
   507→    }
   508→
   509→    Ok(())
   510→}

   512→fn cmd_task(project_dir: &PathBuf, command: TaskCommands, format: OutputFormat) -> CliResult<()> {
   513→    if !db::has_harness_folder(project_dir) {
   514→        return Err(CliError::NotHarnessProject);
   515→    }
   516→
   517→    let conn = db::open(project_dir)?;
   518→
   519→    match command {
   520→        TaskCommands::Add {
   521→            title,
   522→            description,
   523→            category,
   524→            status,
   525→        } => {
   526→            let mut task = db::ProjectTask::new(&title).with_status(status.into());
   527→
   528→            if let Some(ref desc) = description {
   529→                task = task.with_description(desc);
   530→            }
   531→            if let Some(ref cat) = category {
   532→                task = task.with_category(cat);
   533→            }
   534→
   535→            db::create_task(&conn, &task)?;
   536→
   537→            match format {
   538→                OutputFormat::Json => {
   539→                    println!("{}", serde_json::to_string(&task).unwrap_or_default());
   540→                }
   541→                OutputFormat::Quiet => {
   542→                    println!("{}", task.id);
   543→                }
   544→                OutputFormat::Text => {
   545→                    println!("Created task: {}", task.id);
   546→                    println!("  Title:  {}", task.title);
   547→                    println!("  Status: {}", task.status.as_str());
   548→                }
   549→            }
   550→        }
   551→
   552→        TaskCommands::List { status, limit, cursor } => {
   553→            let result = if let Some(status) = status {
   554→                db::list_tasks_by_status(&conn, status.into(), cursor.as_deref(), Some(limit))?
   555→            } else {
   556→                db::list_tasks(&conn, cursor.as_deref(), Some(limit))?
   557→            };
   558→
   559→            match format {
   560→                OutputFormat::Json => {
   561→                    println!("{}", serde_json::to_string(&result).unwrap_or_default());
   562→                }
   563→                OutputFormat::Quiet => {
   564→                    for task in &result.items {
   565→                        println!("{}", task.id);
   566→                    }
   567→                }
   568→                OutputFormat::Text => {
   569→                    if result.items.is_empty() {
   570→                        println!("No tasks found.");
   571→                    } else {
   572→                        println!("{:<26}  {:<8}  {}", "ID", "STATUS", "TITLE");
   573→                        println!("{}", "-".repeat(70));
   574→                        for task in &result.items {
   575→                            println!(
   576→                                "{:<26}  {:<8}  {}",
   577→                                task.id,
   578→                                task.status.as_str(),
   579→                                truncate(&task.title, 32)
   580→                            );
   581→                        }
   582→                        if result.has_next {
   583→                            println!();
   584→                            println!("More results available. Use --cursor={}", result.next_cursor.as_deref().unwrap_or(""));
   585→                        }
   586→                    }
   587→                }
   588→            }
   589→        }
   590→
   591→        TaskCommands::Search { query, limit } => {
   592→            let tasks = db::search_tasks(&conn, &query, Some(limit))?;
   593→
   594→            match format {
   595→                OutputFormat::Json => {
   596→                    println!("{}", serde_json::to_string(&tasks).unwrap_or_default());
   597→                }
   598→                OutputFormat::Quiet => {
   599→                    for task in &tasks {
   600→                        println!("{}", task.id);
   601→                    }
   602→                }
   603→                OutputFormat::Text => {
   604→                    if tasks.is_empty() {
   605→                        println!("No tasks found matching '{}'.", query);
   606→                    } else {
   607→                        println!("Found {} task(s) matching '{}':", tasks.len(), query);
   608→                        println!();
   609→                        println!("{:<26}  {:<8}  {}", "ID", "STATUS", "TITLE");
   610→                        println!("{}", "-".repeat(70));
   611→                        for task in &tasks {
   612→                            println!(
   613→                                "{:<26}  {:<8}  {}",
   614→                                task.id,
   615→                                task.status.as_str(),
   616→                                truncate(&task.title, 32)
   617→                            );
   618→                        }
   619→                    }
   620→                }
   621→            }
   622→        }
   623→
   624→        TaskCommands::Get { id } => {
   625→            let task = db::get_task(&conn, &id)?
   626→                .ok_or_else(|| CliError::NotFound(format!("Task {}", id)))?;
   627→
   628→            match format {
   629→                OutputFormat::Json => {
   630→                    println!("{}", serde_json::to_string(&task).unwrap_or_default());
   631→                }
   632→                OutputFormat::Quiet => {
   633→                    println!("{}", task.id);
   634→                }
   635→                OutputFormat::Text => {
   636→                    println!("ID:          {}", task.id);
   637→                    println!("Title:       {}", task.title);
   638→                    println!("Status:      {}", task.status.as_str());
   639→                    println!("Category:    {}", task.category.as_deref().unwrap_or("-"));
   640→                    println!("Description: {}", task.description.as_deref().unwrap_or("-"));
   641→                    println!("Passed:      {}", task.is_passed);
   642→                    println!("Order:       {:.2}", task.sort_order);
   643→                    println!("Created:     {}", task.created_at);
   644→                    println!("Updated:     {}", task.updated_at);
   645→
   646→                    // Show steps
   647→                    let steps = db::list_steps(&conn, &id)?;
   648→                    if !steps.is_empty() {
   649→                        println!();
   650→                        println!("Steps ({}):", steps.len());
   651→                        for step in &steps {
   652→                            let check = if step.is_completed { "[x]" } else { "[ ]" };
   653→                            println!("  {} {}", check, step.description);
   654→                        }
   655→                    }
   656→                }
   657→            }
   658→        }
   659→
   660→        TaskCommands::Update {
   661→            id,
   662→            title,
   663→            description,
   664→            category,
   665→            status,
   666→            passed,
   667→        } => {
   668→            let mut update = db::TaskUpdate::default();
   669→            update.title = title;
   670→            update.description = description.map(Some);
   671→            update.category = category.map(Some);
   672→            update.status = status.map(|s| s.into());
   673→
   674→            let updated = db::update_task(&conn, &id, update)?;
   675→
   676→            if let Some(p) = passed {
   677→                db::set_task_passed(&conn, &id, p)?;
   678→            }
   679→
   680→            if !updated && passed.is_none() {
   681→                return Err(CliError::NotFound(format!("Task {}", id)));
   682→            }
   683→
   684→            match format {
   685→                OutputFormat::Json => {
   686→                    if let Some(task) = db::get_task(&conn, &id)? {
   687→                        println!("{}", serde_json::to_string(&task).unwrap_or_default());
   688→                    }
   689→                }
   690→                OutputFormat::Quiet => {
   691→                    println!("{}", id);
   692→                }
   693→                OutputFormat::Text => {
   694→                    println!("Updated task: {}", id);
   695→                }
   696→            }
   697→        }
   698→
   699→        TaskCommands::Delete { id } => {
