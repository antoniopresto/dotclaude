     1→//! Database types for the Harness storage layer.
     2→//!
     3→//! This module defines the core data structures used in the SQLite database:
     4→//! - `TaskStatus` enum with 6 Kanban states
     5→//! - `ProjectTask` for features/tasks
     6→//! - `TaskStep` for checklist items (composite ID)
     7→//! - `WorkSession` for work periods
     8→//! - `ProjectProgress` for session log entries (composite ID)
     9→//! - `DbError` for error handling
    10→
    11→use chrono::{DateTime, Utc};
    12→use serde::{Deserialize, Serialize};
    13→use thiserror::Error;
    14→

    19→/// Database operation errors.
    20→#[derive(Debug, Error)]
    21→pub enum DbError {
    22→    #[error("Database error: {0}")]
    23→    Sqlite(#[from] rusqlite::Error),
    24→
    25→    #[error("Not found: {0}")]
    26→    NotFound(String),
    27→
    28→    #[error("Invalid ID format: {0}")]
    29→    InvalidId(String),
    30→
    31→    #[error("Validation error: {0}")]
    32→    Validation(String),
    33→
    34→    #[error("IO error: {0}")]
    35→    Io(#[from] std::io::Error),
    36→
    37→    #[error("Embedding error: {0}")]
    38→    Embedding(String),
    39→}
    40→
    41→pub type DbResult<T> = Result<T, DbError>;

    47→/// Task status representing Kanban columns.
    48→///
    49→/// Serializes to UPPERCASE strings for SQL compatibility.
    50→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
    51→#[serde(rename_all = "SCREAMING_SNAKE_CASE")]
    52→pub enum TaskStatus {
    53→    Backlog,
    54→    Todo,
    55→    Doing,
    56→    Review,
    57→    Qa,
    58→    Done,
    59→}
    60→
    61→impl TaskStatus {
    62→    /// Returns the SQL string representation (UPPERCASE).
    63→    pub fn as_str(&self) -> &'static str {
    64→        match self {
    65→            Self::Backlog => "BACKLOG",
    66→            Self::Todo => "TODO",
    67→            Self::Doing => "DOING",
    68→            Self::Review => "REVIEW",
    69→            Self::Qa => "QA",
    70→            Self::Done => "DONE",
    71→        }
    72→    }
    73→
    74→    /// Parse from SQL string representation.
    75→    pub fn from_str(s: &str) -> Option<Self> {
    76→        match s {
    77→            "BACKLOG" => Some(Self::Backlog),
    78→            "TODO" => Some(Self::Todo),
    79→            "DOING" => Some(Self::Doing),
    80→            "REVIEW" => Some(Self::Review),
    81→            "QA" => Some(Self::Qa),
    82→            "DONE" => Some(Self::Done),
    83→            _ => None,
    84→        }
    85→    }
    86→
    87→    /// Returns all status values in order.
    88→    pub fn all() -> &'static [TaskStatus] {
    89→        &[
    90→            Self::Backlog,
    91→            Self::Todo,
    92→            Self::Doing,
    93→            Self::Review,
    94→            Self::Qa,
    95→            Self::Done,
    96→        ]
    97→    }
    98→}
    99→
   100→impl std::fmt::Display for TaskStatus {
   101→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   102→        write!(f, "{}", self.as_str())
   103→    }
   104→}
   105→
   106→impl Default for TaskStatus {
   107→    fn default() -> Self {
   108→        Self::Backlog
   109→    }
   110→}

   116→/// A project task (replaces features.json entries).
   117→///
   118→/// Uses ULID for ID (26 characters, timestamp-sortable).
   119→#[derive(Debug, Clone, Serialize, Deserialize)]
   120→pub struct ProjectTask {
   121→    /// ULID identifier (26 chars).
   122→    pub id: String,
   123→    /// Task title.
   124→    pub title: String,
   125→    /// Optional description.
   126→    pub description: Option<String>,
   127→    /// Current status (Kanban column).
   128→    pub status: TaskStatus,
   129→    /// Optional category for grouping.
   130→    pub category: Option<String>,
   131→    /// Sort order within status column (float for fractional ordering).
   132→    pub sort_order: f64,
   133→    /// Whether the task has passed validation/testing.
   134→    pub is_passed: bool,
   135→    /// Creation timestamp (ISO 8601).
   136→    pub created_at: DateTime<Utc>,
   137→    /// Last update timestamp (ISO 8601).
   138→    pub updated_at: DateTime<Utc>,
   139→}
   140→
   141→impl ProjectTask {
   142→    /// Creates a new ProjectTask with the given title.
   143→    ///
   144→    /// Generates a new ULID and sets timestamps to now.
   145→    pub fn new(title: impl Into<String>) -> Self {
   146→        let now = Utc::now();
   147→        Self {
   148→            id: ulid::Ulid::new().to_string(),
   149→            title: title.into(),
   150→            description: None,
   151→            status: TaskStatus::default(),
   152→            category: None,
   153→            sort_order: 0.0,
   154→            is_passed: false,
   155→            created_at: now,
   156→            updated_at: now,
   157→        }
   158→    }
   159→
   160→    /// Builder method to set description.
   161→    pub fn with_description(mut self, desc: impl Into<String>) -> Self {
   162→        self.description = Some(desc.into());
   163→        self
   164→    }
   165→
   166→    /// Builder method to set status.
   167→    pub fn with_status(mut self, status: TaskStatus) -> Self {
   168→        self.status = status;
   169→        self
   170→    }
   171→
   172→    /// Builder method to set category.
   173→    pub fn with_category(mut self, category: impl Into<String>) -> Self {
   174→        self.category = Some(category.into());
   175→        self
   176→    }
   177→
   178→    /// Builder method to set sort order.
   179→    pub fn with_sort_order(mut self, order: f64) -> Self {
   180→        self.sort_order = order;
   181→        self
   182→    }
   183→}

   185→/// A checklist item within a task.
   186→///
   187→/// Uses composite ID: `task#{task_ulid}#{step_ulid}`
   188→#[derive(Debug, Clone, Serialize, Deserialize)]
   189→pub struct TaskStep {
   190→    /// Composite ID: task#{task_ulid}#{step_ulid}
   191→    pub id: String,
   192→    /// Step description.
   193→    pub description: String,
   194→    /// Whether the step is completed.
   195→    pub is_completed: bool,
   196→    /// Creation timestamp (ISO 8601).
   197→    pub created_at: DateTime<Utc>,
   198→}
   199→
   200→impl TaskStep {
   201→    /// Prefix for composite ID (parent table name, singular).
   202→    pub const PREFIX: &'static str = "task";
   203→
   204→    /// Creates a new TaskStep for a given task.
   205→    pub fn new(task_id: &str, description: impl Into<String>) -> DbResult<Self> {
   206→        // Validate task_id is a valid ULID (26 chars)
   207→        if task_id.len() != 26 {
   208→            return Err(DbError::InvalidId(format!(
   209→                "Task ID must be 26 characters (ULID), got: {}",
   210→                task_id
   211→            )));
   212→        }
   213→
   214→        let step_ulid = ulid::Ulid::new().to_string();
   215→        Ok(Self {
   216→            id: Self::make_id(task_id, &step_ulid),
   217→            description: description.into(),
   218→            is_completed: false,
   219→            created_at: Utc::now(),
   220→        })
   221→    }
   222→
   223→    /// Generate composite ID for a step.
   224→    pub fn make_id(task_id: &str, step_ulid: &str) -> String {
   225→        format!("{}#{}#{}", Self::PREFIX, task_id, step_ulid)
   226→    }
   227→
   228→    /// Generate GLOB pattern to find all steps for a task.
   229→    pub fn glob_pattern(task_id: &str) -> String {
   230→        format!("{}#{}#*", Self::PREFIX, task_id)
   231→    }
   232→
   233→    /// Extract parent task ULID from composite ID.
   234→    pub fn task_id(&self) -> Option<&str> {
   235→        self.extract_task_id(&self.id)
   236→    }
   237→
   238→    /// Extract task ID from any composite step ID.
   239→    pub fn extract_task_id<'a>(&self, id: &'a str) -> Option<&'a str> {
   240→        let parts: Vec<&str> = id.split('#').collect();
   241→        if parts.len() == 3 && parts[0] == Self::PREFIX {
   242→            Some(parts[1])
   243→        } else {
   244→            None
   245→        }
   246→    }
   247→
   248→    /// Extract step ULID from composite ID.
   249→    pub fn step_ulid(&self) -> Option<&str> {
   250→        let parts: Vec<&str> = self.id.split('#').collect();
   251→        if parts.len() == 3 {
   252→            Some(parts[2])
   253→        } else {
   254→            None
   255→        }
   256→    }
   257→
   258→    /// Validate that an ID is a valid TaskStep composite ID.
   259→    pub fn validate_id(id: &str) -> bool {
   260→        let parts: Vec<&str> = id.split('#').collect();
   261→        parts.len() == 3
   262→            && parts[0] == Self::PREFIX
   263→            && parts[1].len() == 26
   264→            && parts[2].len() == 26
   265→    }
   266→}

   268→/// A work session (AI/human work period).
   269→///
   270→/// Uses ULID for ID.
   271→#[derive(Debug, Clone, Serialize, Deserialize)]
   272→pub struct WorkSession {
   273→    /// ULID identifier (26 chars).
   274→    pub id: String,
   275→    /// Optional session summary.
   276→    pub summary: Option<String>,
   277→    /// Session start timestamp (ISO 8601).
   278→    pub started_at: DateTime<Utc>,
   279→    /// Session end timestamp (ISO 8601), None if ongoing.
   280→    pub ended_at: Option<DateTime<Utc>>,
   281→}
   282→
   283→impl WorkSession {
   284→    /// Creates a new WorkSession starting now.
   285→    pub fn new() -> Self {
   286→        Self {
   287→            id: ulid::Ulid::new().to_string(),
   288→            summary: None,
   289→            started_at: Utc::now(),
   290→            ended_at: None,
   291→        }
   292→    }
   293→
   294→    /// Builder method to set summary.
   295→    pub fn with_summary(mut self, summary: impl Into<String>) -> Self {
   296→        self.summary = Some(summary.into());
   297→        self
   298→    }
   299→
   300→    /// End the session with current timestamp.
   301→    pub fn end(&mut self) {
   302→        self.ended_at = Some(Utc::now());
   303→    }
   304→
   305→    /// Check if session is still active.
   306→    pub fn is_active(&self) -> bool {
   307→        self.ended_at.is_none()
   308→    }
   309→}
   310→
   311→impl Default for WorkSession {
   312→    fn default() -> Self {
   313→        Self::new()
   314→    }
   315→}

   401→/// Paginated result for list queries.
   402→#[derive(Debug, Clone, Serialize, Deserialize)]
   403→pub struct PaginatedResult<T> {
   404→    /// Items in current page.
   405→    pub items: Vec<T>,
   406→    /// Whether there are more items after current page.
   407→    pub has_next: bool,
   408→    /// Whether there are items before current page.
   409→    pub has_prev: bool,
   410→    /// Cursor for next page (base64 encoded).
   411→    pub next_cursor: Option<String>,
   412→    /// Cursor for previous page (base64 encoded).
   413→    pub prev_cursor: Option<String>,
   414→}
   415→
   416→impl<T> PaginatedResult<T> {
   417→    /// Create an empty result.
   418→    pub fn empty() -> Self {
   419→        Self {
   420→            items: Vec::new(),
   421→            has_next: false,
   422→            has_prev: false,
   423→            next_cursor: None,
   424→            prev_cursor: None,
   425→        }
   426→    }
   427→
   428→    /// Create result from items with pagination info.
   429→    pub fn new(
   430→        items: Vec<T>,
   431→        has_next: bool,
   432→        has_prev: bool,
   433→        next_cursor: Option<String>,
   434→        prev_cursor: Option<String>,
   435→    ) -> Self {
   436→        Self {
   437→            items,
   438→            has_next,
   439→            has_prev,
   440→            next_cursor,
   441→            prev_cursor,
   442→        }
   443→    }
   444→}
   445→
   446→impl<T> Default for PaginatedResult<T> {
   447→    fn default() -> Self {
   448→        Self::empty()
   449→    }
   450→}
