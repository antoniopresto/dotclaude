    25→/// Create a new task.
    26→///
    27→/// Automatically generates and stores a vector embedding for semantic search.
    28→pub fn create_task(conn: &Connection, task: &ProjectTask) -> DbResult<()> {
    29→    conn.execute(
    30→        r#"
    31→        INSERT INTO project_tasks (id, title, description, status, category, sort_order, is_passed, created_at, updated_at)
    32→        VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8, ?9)
    33→        "#,
    34→        params![
    35→            task.id,
    36→            task.title,
    37→            task.description,
    38→            task.status.as_str(),
    39→            task.category,
    40→            task.sort_order,
    41→            if task.is_passed { "true" } else { "false" },
    42→            task.created_at.to_rfc3339(),
    43→            task.updated_at.to_rfc3339(),
    44→        ],
    45→    )?;
    46→
    47→    // Generate and store embedding for semantic search
    48→    let rowid = conn.last_insert_rowid();
    49→    let text = format!(
    50→        "{} {}",
    51→        task.title,
    52→        task.description.as_deref().unwrap_or("")
    53→    );
    54→    let embedding = embeddings::generate_embedding(&text)?;
    55→    embeddings::store_task_embedding(conn, rowid, &embedding)?;
    56→
    57→    Ok(())
    58→}

    64→/// Get a task by ID.

    74→            |row| row_to_task(row),

    80→/// List all tasks with pagination.
    81→///
    82→/// Returns tasks ordered by sort_order, then created_at.
    83→pub fn list_tasks(
    84→    conn: &Connection,
    85→    cursor: Option<&str>,
    86→    limit: Option<usize>,
    87→) -> DbResult<PaginatedResult<ProjectTask>> {
    88→    let limit = limit.unwrap_or(DEFAULT_PAGE_SIZE);
    89→
    90→    // Decode cursor if present
    91→    let after_id = cursor.map(|c| decode_cursor(c)).transpose()?;
    92→
    93→    let mut tasks = Vec::new();
    94→
    95→    if let Some(after_id) = after_id {
    96→        // Get the sort_order of the cursor task
    97→        let cursor_order: Option<f64> = conn
    98→            .query_row(
    99→                "SELECT sort_order FROM project_tasks WHERE id = ?1",
   100→                [&after_id],
   101→                |row| row.get(0),
   102→            )
   103→            .optional()?;
   104→
   105→        if let Some(order) = cursor_order {
   106→            // Get tasks after the cursor
   107→            let mut stmt = conn.prepare(
   108→                r#"
   109→                SELECT id, title, description, status, category, sort_order, is_passed, created_at, updated_at
   110→                FROM project_tasks
   111→                WHERE (sort_order > ?1) OR (sort_order = ?1 AND id > ?2)
   112→                ORDER BY sort_order ASC, id ASC
   113→                LIMIT ?3
   114→                "#,
   115→            )?;
   116→
   117→            let rows = stmt.query_map(params![order, after_id, (limit + 1) as i64], |row| row_to_task(row))?;
   118→            for row in rows {
   119→                tasks.push(row?);
   120→            }
   121→        }
   122→    } else {
   123→        // No cursor, get from the beginning
   124→        let mut stmt = conn.prepare(
   125→            r#"
   126→            SELECT id, title, description, status, category, sort_order, is_passed, created_at, updated_at
   127→            FROM project_tasks
   128→            ORDER BY sort_order ASC, id ASC
   129→            LIMIT ?1
   130→            "#,
   131→        )?;
   132→
   133→        let rows = stmt.query_map([(limit + 1) as i64], |row| row_to_task(row))?;
   134→        for row in rows {
   135→            tasks.push(row?);
   136→        }
   137→    }
   138→
   139→    // Check if there are more results
   140→    let has_next = tasks.len() > limit;
   141→    if has_next {
   142→        tasks.pop(); // Remove the extra item
   143→    }
   144→
   145→    // Has prev if we used a cursor
   146→    let has_prev = cursor.is_some();
   147→
   148→    // Generate cursors
   149→    let next_cursor = if has_next {
   150→        tasks.last().map(|t| encode_cursor(&t.id))
   151→    } else {
   152→        None
   153→    };
   154→
   155→    let prev_cursor = if has_prev {
   156→        tasks.first().map(|t| encode_cursor(&t.id))
   157→    } else {
   158→        None
   159→    };
   160→
   161→    Ok(PaginatedResult::new(tasks, has_next, has_prev, next_cursor, prev_cursor))
   162→}

   164→/// List tasks by status with pagination.
   165→pub fn list_tasks_by_status(
   166→    conn: &Connection,
   167→    status: TaskStatus,
   168→    cursor: Option<&str>,
   169→    limit: Option<usize>,
   170→) -> DbResult<PaginatedResult<ProjectTask>> {
   171→    let limit = limit.unwrap_or(DEFAULT_PAGE_SIZE);
   172→    let status_str = status.as_str();
   173→
   174→    let after_id = cursor.map(|c| decode_cursor(c)).transpose()?;
   175→
   176→    let mut tasks = Vec::new();
   177→
   178→    if let Some(after_id) = after_id {
   179→        let cursor_order: Option<f64> = conn
   180→            .query_row(
   181→                "SELECT sort_order FROM project_tasks WHERE id = ?1",
   182→                [&after_id],
   183→                |row| row.get(0),
   184→            )
   185→            .optional()?;
   186→
   187→        if let Some(order) = cursor_order {
   188→            let mut stmt = conn.prepare(
   189→                r#"
   190→                SELECT id, title, description, status, category, sort_order, is_passed, created_at, updated_at
   191→                FROM project_tasks
   192→                WHERE status = ?1 AND ((sort_order > ?2) OR (sort_order = ?2 AND id > ?3))
   193→                ORDER BY sort_order ASC, id ASC
   194→                LIMIT ?4
   195→                "#,
   196→            )?;
   197→
   198→            let rows = stmt.query_map(params![status_str, order, after_id, (limit + 1) as i64], |row| {
   199→                row_to_task(row)
   200→            })?;
   201→            for row in rows {
   202→                tasks.push(row?);
   203→            }
   204→        }
   205→    } else {
   206→        let mut stmt = conn.prepare(
   207→            r#"
   208→            SELECT id, title, description, status, category, sort_order, is_passed, created_at, updated_at
   209→            FROM project_tasks
   210→            WHERE status = ?1
   211→            ORDER BY sort_order ASC, id ASC
   212→            LIMIT ?2
   213→            "#,
   214→        )?;
   215→
   216→        let rows = stmt.query_map(params![status_str, (limit + 1) as i64], |row| row_to_task(row))?;
   217→        for row in rows {
   218→            tasks.push(row?);
   219→        }
   220→    }
   221→
   222→    let has_next = tasks.len() > limit;
   223→    if has_next {
   224→        tasks.pop();
   225→    }
   226→
   227→    let has_prev = cursor.is_some();
   228→
   229→    let next_cursor = if has_next {
   230→        tasks.last().map(|t| encode_cursor(&t.id))
   231→    } else {
   232→        None
   233→    };
   234→
   235→    let prev_cursor = if has_prev {
   236→        tasks.first().map(|t| encode_cursor(&t.id))
   237→    } else {
   238→        None
   239→    };
   240→
   241→    Ok(PaginatedResult::new(tasks, has_next, has_prev, next_cursor, prev_cursor))
   242→}

   244→/// Search tasks using full-text search.
   245→pub fn search_tasks(
   246→    conn: &Connection,
   247→    query: &str,
   248→    limit: Option<usize>,
   249→) -> DbResult<Vec<ProjectTask>> {
   250→    let limit = limit.unwrap_or(DEFAULT_PAGE_SIZE);
   251→
   252→    let mut stmt = conn.prepare(
   253→        r#"
   254→        SELECT p.id, p.title, p.description, p.status, p.category, p.sort_order, p.is_passed, p.created_at, p.updated_at
   255→        FROM project_tasks p
   256→        JOIN project_tasks_fts fts ON p.rowid = fts.rowid
   257→        WHERE project_tasks_fts MATCH ?1
   258→        ORDER BY rank
   259→        LIMIT ?2
   260→        "#,
   261→    )?;
   262→
   263→    let rows = stmt.query_map(params![query, limit as i64], |row| row_to_task(row))?;
   264→
   265→    let mut tasks = Vec::new();
   266→    for row in rows {
   267→        tasks.push(row?);
   268→    }
   269→
   270→    Ok(tasks)
   271→}

   273→/// Count tasks by status.
   274→pub fn count_tasks_by_status(conn: &Connection) -> DbResult<std::collections::HashMap<String, i64>> {
   275→    let mut stmt = conn.prepare(
   276→        "SELECT status, COUNT(*) FROM project_tasks GROUP BY status",
   277→    )?;
   278→
   279→    let rows = stmt.query_map([], |row| {
   280→        Ok((row.get::<_, String>(0)?, row.get::<_, i64>(1)?))
   281→    })?;
   282→
   283→    let mut counts = std::collections::HashMap::new();
   284→    for row in rows {
   285→        let (status, count) = row?;
   286→        counts.insert(status, count);
   287→    }
   288→
   289→    Ok(counts)
   290→}

   296→/// Update fields for a task.
   297→#[derive(Default)]
   298→pub struct TaskUpdate {
   299→    pub title: Option<String>,
   300→    pub description: Option<Option<String>>,
   301→    pub status: Option<TaskStatus>,
   302→    pub category: Option<Option<String>>,
   303→    pub sort_order: Option<f64>,
   304→    pub is_passed: Option<bool>,
   305→}

   307→/// Update a task with the given changes.
   308→///
   309→/// Automatically regenerates the vector embedding if title or description changes.
   310→pub fn update_task(conn: &Connection, id: &str, update: TaskUpdate) -> DbResult<bool> {
   311→    // Track if we need to regenerate the embedding
   312→    let needs_reembed = update.title.is_some() || update.description.is_some();
   313→
   314→    let mut updates = Vec::new();
   315→    let mut values: Vec<Box<dyn rusqlite::ToSql>> = Vec::new();
   316→
   317→    if let Some(ref title) = update.title {
   318→        updates.push("title = ?");
   319→        values.push(Box::new(title.clone()));
   320→    }
   321→
   322→    if let Some(ref desc) = update.description {
   323→        updates.push("description = ?");
   324→        values.push(Box::new(desc.clone()));
   325→    }
   326→
   327→    if let Some(status) = update.status {
   328→        updates.push("status = ?");
   329→        values.push(Box::new(status.as_str().to_string()));
   330→    }
   331→
   332→    if let Some(ref category) = update.category {
   333→        updates.push("category = ?");
   334→        values.push(Box::new(category.clone()));
   335→    }
   336→
   337→    if let Some(sort_order) = update.sort_order {
   338→        updates.push("sort_order = ?");
   339→        values.push(Box::new(sort_order));
   340→    }
   341→
   342→    if let Some(is_passed) = update.is_passed {
   343→        updates.push("is_passed = ?");
   344→        values.push(Box::new(if is_passed { "true" } else { "false" }));
   345→    }
   346→
   347→    if updates.is_empty() {
   348→        return Ok(false);
   349→    }
   350→
   351→    // Always update updated_at
   352→    updates.push("updated_at = ?");
   353→    values.push(Box::new(Utc::now().to_rfc3339()));
   354→
   355→    // Add ID at the end
   356→    values.push(Box::new(id.to_string()));
   357→
   358→    let sql = format!(
   359→        "UPDATE project_tasks SET {} WHERE id = ?",
   360→        updates.join(", ")
   361→    );
   362→
   363→    let params: Vec<&dyn rusqlite::ToSql> = values.iter().map(|v| v.as_ref()).collect();
   364→    let rows_affected = conn.execute(&sql, params.as_slice())?;
   365→
   366→    // Regenerate embedding if title or description changed
   367→    if needs_reembed && rows_affected > 0 {
   368→        if let Some(rowid) = embeddings::get_task_rowid(conn, id)? {
   369→            // Get current title and description
   370→            let (title, desc): (String, Option<String>) = conn.query_row(
   371→                "SELECT title, description FROM project_tasks WHERE id = ?1",
   372→                [id],
   373→                |row| Ok((row.get(0)?, row.get(1)?)),
   374→            )?;
   375→
   376→            let text = format!("{} {}", title, desc.as_deref().unwrap_or(""));
   377→            let embedding = embeddings::generate_embedding(&text)?;
   378→            embeddings::update_task_embedding(conn, rowid, &embedding)?;
   379→        }
   380→    }
   381→
   382→    Ok(rows_affected > 0)
   383→}

   405→/// Delete a task by ID.

   407→/// Also removes the associated vector embedding.
   408→/// Note: This also needs to delete related task_steps (handled by caller or cascade).
   409→pub fn delete_task(conn: &Connection, id: &str) -> DbResult<bool> {
   410→    // Get rowid before deleting (needed to remove embedding)
   411→    let rowid = embeddings::get_task_rowid(conn, id)?;
   412→
   413→    let rows_affected = conn.execute("DELETE FROM project_tasks WHERE id = ?1", [id])?;
   414→
   415→    // Delete embedding if task existed
   416→    if let Some(rowid) = rowid {
   417→        embeddings::delete_task_embedding(conn, rowid)?;
   418→    }
   419→
   420→    Ok(rows_affected > 0)
   421→}

   427→/// Get the sort order for positioning after another task.
   428→///
   429→/// If after_id is None, positions at the beginning.
   430→pub fn get_order_after(conn: &Connection, after_id: Option<&str>) -> DbResult<f64> {
   431→    if let Some(after_id) = after_id {
   432→        // Get the order of the after task
   433→        let after_order: f64 = conn.query_row(
   434→            "SELECT sort_order FROM project_tasks WHERE id = ?1",
   435→            [after_id],
   436→            |row| row.get(0),
   437→        )?;
   438→
   439→        // Get the next task's order (if any)
   440→        let next_order: Option<f64> = conn
   441→            .query_row(
   442→                "SELECT sort_order FROM project_tasks WHERE sort_order > ?1 ORDER BY sort_order ASC LIMIT 1",
   443→                [after_order],
   444→                |row| row.get(0),
   445→            )
   446→            .optional()?;
   447→
   448→        // Position between after and next
   449→        match next_order {
   450→            Some(next) => Ok((after_order + next) / 2.0),
   451→            None => Ok(after_order + 1.0),
   452→        }
   453→    } else {
   454→        // Position at the beginning
   455→        let min_order: Option<f64> = conn
   456→            .query_row(
   457→                "SELECT MIN(sort_order) FROM project_tasks",
   458→                [],
   459→                |row| row.get(0),
   460→            )
   461→            .optional()?
   462→            .flatten();
   463→
   464→        match min_order {
   465→            Some(min) => Ok(min - 1.0),
   466→            None => Ok(0.0),
   467→        }
   468→    }
   469→}
   470→
   471→/// Reorder a task to be after another task.
   472→pub fn reorder_task(conn: &Connection, id: &str, after_id: Option<&str>) -> DbResult<bool> {
   473→    let new_order = get_order_after(conn, after_id)?;
   474→    update_task(conn, id, TaskUpdate {
   475→        sort_order: Some(new_order),
   476→        ..Default::default()
   477→    })
   478→}

   484→fn row_to_task(row: &rusqlite::Row) -> rusqlite::Result<ProjectTask> {
   485→    let status_str: String = row.get(3)?;
   486→    let is_passed_str: String = row.get(6)?;
   487→    let created_at_str: String = row.get(7)?;
   488→    let updated_at_str: String = row.get(8)?;
   489→
   490→    Ok(ProjectTask {
   491→        id: row.get(0)?,
   492→        title: row.get(1)?,
   493→        description: row.get(2)?,
   494→        status: TaskStatus::from_str(&status_str).unwrap_or_default(),
   495→        category: row.get(4)?,
   496→        sort_order: row.get(5)?,
   497→        is_passed: is_passed_str == "true",
   498→        created_at: chrono::DateTime::parse_from_rfc3339(&created_at_str)
   499→            .map(|dt| dt.with_timezone(&Utc))
   500→            .unwrap_or_else(|_| Utc::now()),
   501→        updated_at: chrono::DateTime::parse_from_rfc3339(&updated_at_str)
   502→            .map(|dt| dt.with_timezone(&Utc))
   503→            .unwrap_or_else(|_| Utc::now()),
   504→    })
   505→}

   507→pub(crate) fn encode_cursor(id: &str) -> String {
   508→    use std::io::Write;
   509→    let mut encoder = base64::write::EncoderStringWriter::new(&base64::engine::general_purpose::URL_SAFE_NO_PAD);
   510→    encoder.write_all(id.as_bytes()).unwrap();
   511→    encoder.into_inner()
   512→}
   513→
   514→pub(crate) fn decode_cursor(cursor: &str) -> DbResult<String> {
   515→    use base64::Engine;
   516→    let bytes = base64::engine::general_purpose::URL_SAFE_NO_PAD
   517→        .decode(cursor)
   518→        .map_err(|e| super::DbError::InvalidId(format!("Invalid cursor: {}", e)))?;
   519→    String::from_utf8(bytes)
   520→        .map_err(|e| super::DbError::InvalidId(format!("Invalid cursor UTF-8: {}", e)))
   521→}
