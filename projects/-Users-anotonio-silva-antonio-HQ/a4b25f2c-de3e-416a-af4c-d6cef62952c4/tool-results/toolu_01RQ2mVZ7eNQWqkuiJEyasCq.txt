     1→//! Database schema initialization and migrations.
     2→//!
     3→//! This module contains the SQLite schema definition for the Harness storage layer.
     4→//! Key features:
     5→//! - WAL mode for better concurrency
     6→//! - FTS5 virtual tables for full-text search
     7→//! - Triggers to sync FTS5 with source tables
     8→//! - Schema versioning for migrations
     9→
    10→use rusqlite::Connection;
    11→
    12→use super::DbResult;
    13→
    14→/// Current schema version (semantic versioning).
    15→pub const SCHEMA_VERSION: &str = "1.0.0";
    16→
    17→/// SQLite pragmas to run on every connection.
    18→pub const PRAGMAS: &str = r#"
    19→PRAGMA journal_mode = WAL;
    20→PRAGMA synchronous = NORMAL;
    21→PRAGMA foreign_keys = OFF;
    22→"#;
    23→
    24→/// Core schema creation SQL.
    25→pub const SCHEMA_SQL: &str = r#"
    26→-- ============================================================================
    27→-- SCHEMA METADATA
    28→-- ============================================================================
    29→CREATE TABLE IF NOT EXISTS schema_metadata (
    30→    key TEXT PRIMARY KEY,
    31→    value TEXT NOT NULL
    32→) WITHOUT ROWID;
    33→
    34→-- ============================================================================
    35→-- CORE TABLES
    36→-- ============================================================================
    37→
    38→-- Project Tasks (replaces features.json)
    39→CREATE TABLE IF NOT EXISTS project_tasks (
    40→    id TEXT PRIMARY KEY CHECK(length(id) = 26),
    41→    title TEXT NOT NULL,
    42→    description TEXT,
    43→    status TEXT NOT NULL DEFAULT 'BACKLOG'
    44→        CHECK(status IN ('BACKLOG', 'TODO', 'DOING', 'REVIEW', 'QA', 'DONE')),
    45→    category TEXT,
    46→    sort_order REAL NOT NULL DEFAULT 0.0,
    47→    is_passed TEXT NOT NULL DEFAULT 'false'
    48→        CHECK(is_passed IN ('true', 'false')),
    49→    created_at TEXT NOT NULL,
    50→    updated_at TEXT NOT NULL
    51→);
    52→
    53→-- Task Steps (checklist items within a task)
    54→-- ID format: task#{task_ulid}#{step_ulid}
    55→CREATE TABLE IF NOT EXISTS task_steps (
    56→    id TEXT PRIMARY KEY,
    57→    description TEXT NOT NULL,
    58→    is_completed TEXT NOT NULL DEFAULT 'false'
    59→        CHECK(is_completed IN ('true', 'false')),
    60→    created_at TEXT NOT NULL
    61→) WITHOUT ROWID;
    62→
    63→-- Work Sessions (AI/human work periods)
    64→CREATE TABLE IF NOT EXISTS work_sessions (
    65→    id TEXT PRIMARY KEY CHECK(length(id) = 26),
    66→    summary TEXT,
    67→    started_at TEXT NOT NULL,
    68→    ended_at TEXT
    69→);
    70→
    71→-- Project Progress (session log entries)
    72→-- ID format: session#{session_ulid}#{progress_ulid}
    73→-- Note: No WITHOUT ROWID - needs implicit rowid for FTS5 sync
    74→CREATE TABLE IF NOT EXISTS project_progress (
    75→    id TEXT PRIMARY KEY,
    76→    content TEXT NOT NULL,
    77→    created_at TEXT NOT NULL
    78→);
    79→
    80→-- Test Screenshots (F015)
    81→-- Stores metadata for visual test screenshots
    82→CREATE TABLE IF NOT EXISTS test_screenshots (
    83→    id TEXT PRIMARY KEY CHECK(length(id) = 26),
    84→    task_id TEXT,
    85→    step_id TEXT,
    86→    file_name TEXT NOT NULL,
    87→    caption TEXT,
    88→    created_at TEXT NOT NULL,
    89→    FOREIGN KEY (task_id) REFERENCES project_tasks(id) ON DELETE CASCADE
    90→);
    91→
    92→-- ============================================================================
    93→-- INDEXES
    94→-- ============================================================================
    95→
    96→CREATE INDEX IF NOT EXISTS idx_test_screenshots_task_id ON test_screenshots(task_id);
    97→CREATE INDEX IF NOT EXISTS idx_test_screenshots_created_at ON test_screenshots(created_at);
    98→CREATE INDEX IF NOT EXISTS idx_project_tasks_status ON project_tasks(status);
    99→CREATE INDEX IF NOT EXISTS idx_project_tasks_sort_order ON project_tasks(sort_order);
   100→CREATE INDEX IF NOT EXISTS idx_project_tasks_created_at ON project_tasks(created_at);
   101→CREATE INDEX IF NOT EXISTS idx_work_sessions_started_at ON work_sessions(started_at);
   102→"#;

   104→/// FTS5 virtual tables for full-text search.
   105→pub const FTS5_SCHEMA_SQL: &str = r#"
   106→-- ============================================================================
   107→-- FULL-TEXT SEARCH (FTS5)
   108→-- ============================================================================
   109→
   110→CREATE VIRTUAL TABLE IF NOT EXISTS project_tasks_fts USING fts5(
   111→    title,
   112→    description,
   113→    category,
   114→    content='project_tasks',
   115→    content_rowid='rowid'
   116→);
   117→
   118→CREATE VIRTUAL TABLE IF NOT EXISTS project_progress_fts USING fts5(
   119→    content,
   120→    content='project_progress',
   121→    content_rowid='rowid'
   122→);
   123→"#;
   124→
   125→/// Triggers to keep FTS5 tables in sync with source tables.
   126→pub const FTS5_TRIGGERS_SQL: &str = r#"
   127→-- ============================================================================
   128→-- FTS SYNC TRIGGERS: project_tasks
   129→-- ============================================================================
   130→
   131→CREATE TRIGGER IF NOT EXISTS project_tasks_fts_insert AFTER INSERT ON project_tasks BEGIN
   132→    INSERT INTO project_tasks_fts(rowid, title, description, category)
   133→    VALUES (NEW.rowid, NEW.title, NEW.description, NEW.category);
   134→END;
   135→
   136→CREATE TRIGGER IF NOT EXISTS project_tasks_fts_delete AFTER DELETE ON project_tasks BEGIN
   137→    INSERT INTO project_tasks_fts(project_tasks_fts, rowid, title, description, category)
   138→    VALUES('delete', OLD.rowid, OLD.title, OLD.description, OLD.category);
   139→END;
   140→
   141→CREATE TRIGGER IF NOT EXISTS project_tasks_fts_update AFTER UPDATE ON project_tasks BEGIN
   142→    INSERT INTO project_tasks_fts(project_tasks_fts, rowid, title, description, category)
   143→    VALUES('delete', OLD.rowid, OLD.title, OLD.description, OLD.category);
   144→    INSERT INTO project_tasks_fts(rowid, title, description, category)
   145→    VALUES (NEW.rowid, NEW.title, NEW.description, NEW.category);
   146→END;
   147→
   148→-- ============================================================================
   149→-- FTS SYNC TRIGGERS: project_progress
   150→-- ============================================================================
   151→
   152→CREATE TRIGGER IF NOT EXISTS project_progress_fts_insert AFTER INSERT ON project_progress BEGIN
   153→    INSERT INTO project_progress_fts(rowid, content)
   154→    VALUES (NEW.rowid, NEW.content);
   155→END;
   156→
   157→CREATE TRIGGER IF NOT EXISTS project_progress_fts_delete AFTER DELETE ON project_progress BEGIN
   158→    INSERT INTO project_progress_fts(project_progress_fts, rowid, content)
   159→    VALUES('delete', OLD.rowid, OLD.content);
   160→END;
   161→
   162→CREATE TRIGGER IF NOT EXISTS project_progress_fts_update AFTER UPDATE ON project_progress BEGIN
   163→    INSERT INTO project_progress_fts(project_progress_fts, rowid, content)
   164→    VALUES('delete', OLD.rowid, OLD.content);
   165→    INSERT INTO project_progress_fts(rowid, content)
   166→    VALUES (NEW.rowid, NEW.content);
   167→END;
   168→"#;

   170→/// Vector embedding tables for semantic search (sqlite-vec).
   171→/// Uses AllMiniLML6V2 model which produces 384-dimensional embeddings.
   172→pub const VEC_SCHEMA_SQL: &str = r#"
   173→-- ============================================================================
   174→-- VECTOR EMBEDDINGS (sqlite-vec)
   175→-- ============================================================================
   176→
   177→-- Task embeddings (384 dims = AllMiniLML6V2)
   178→-- rowid matches project_tasks.rowid for JOIN
   179→CREATE VIRTUAL TABLE IF NOT EXISTS task_embeddings USING vec0(
   180→    embedding float[384]
   181→);
   182→
   183→-- Progress embeddings (384 dims = AllMiniLML6V2)
   184→-- rowid matches project_progress.rowid for JOIN
   185→CREATE VIRTUAL TABLE IF NOT EXISTS progress_embeddings USING vec0(
   186→    embedding float[384]
   187→);
   188→"#;

   190→/// Initialize database schema.
   191→///
   192→/// Creates all tables, indexes, FTS5 virtual tables, vec0 tables, and triggers.
   193→/// Safe to call multiple times (uses IF NOT EXISTS).
   194→pub fn init_schema(conn: &Connection) -> DbResult<()> {
   195→    // Execute pragmas
   196→    conn.execute_batch(PRAGMAS)?;
   197→
   198→    // Create core tables and indexes
   199→    conn.execute_batch(SCHEMA_SQL)?;
   200→
   201→    // Create FTS5 virtual tables
   202→    conn.execute_batch(FTS5_SCHEMA_SQL)?;
   203→
   204→    // Create FTS sync triggers
   205→    conn.execute_batch(FTS5_TRIGGERS_SQL)?;
   206→
   207→    // Create vec0 embedding tables (sqlite-vec)
   208→    conn.execute_batch(VEC_SCHEMA_SQL)?;
   209→
   210→    // Insert or update schema version
   211→    conn.execute(
   212→        "INSERT OR REPLACE INTO schema_metadata (key, value) VALUES ('version', ?1)",
   213→        [SCHEMA_VERSION],
   214→    )?;
   215→
   216→    Ok(())
   217→}
   218→
   219→/// Get the current schema version from the database.
   220→pub fn get_schema_version(conn: &Connection) -> DbResult<Option<String>> {
   221→    let result = conn.query_row(
   222→        "SELECT value FROM schema_metadata WHERE key = 'version'",
   223→        [],
   224→        |row| row.get::<_, String>(0),
   225→    );
   226→
   227→    match result {
   228→        Ok(version) => Ok(Some(version)),
   229→        Err(rusqlite::Error::QueryReturnedNoRows) => Ok(None),
   230→        Err(e) => Err(e.into()),
   231→    }
   232→}
   233→
   234→/// Check if schema is initialized.
   235→pub fn is_initialized(conn: &Connection) -> bool {
   236→    get_schema_version(conn).unwrap_or(None).is_some()
   237→}

   239→/// Get list of all tables in the database.
   240→pub fn list_tables(conn: &Connection) -> DbResult<Vec<String>> {
   241→    let mut stmt = conn.prepare(
   242→        "SELECT name FROM sqlite_master WHERE type='table' ORDER BY name",
   243→    )?;
   244→
   245→    let tables = stmt
   246→        .query_map([], |row| row.get::<_, String>(0))?
   247→        .collect::<Result<Vec<_>, _>>()?;
   248→
   249→    Ok(tables)
   250→}
   251→
   252→/// Get list of all indexes in the database.
   253→pub fn list_indexes(conn: &Connection) -> DbResult<Vec<String>> {
   254→    let mut stmt = conn.prepare(
   255→        "SELECT name FROM sqlite_master WHERE type='index' AND name NOT LIKE 'sqlite_%' ORDER BY name",
   256→    )?;
   257→
   258→    let indexes = stmt
   259→        .query_map([], |row| row.get::<_, String>(0))?
   260→        .collect::<Result<Vec<_>, _>>()?;
   261→
   262→    Ok(indexes)
   263→}
   264→
   265→/// Get list of all triggers in the database.
   266→pub fn list_triggers(conn: &Connection) -> DbResult<Vec<String>> {
   267→    let mut stmt = conn.prepare(
   268→        "SELECT name FROM sqlite_master WHERE type='trigger' ORDER BY name",
   269→    )?;
   270→
   271→    let triggers = stmt
   272→        .query_map([], |row| row.get::<_, String>(0))?
   273→        .collect::<Result<Vec<_>, _>>()?;
   274→
   275→    Ok(triggers)
   276→}
