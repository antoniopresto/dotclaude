    46→/**
    47→ * Dashboard KPIs Component
    48→ *
    49→ * Displays the four main fleet KPIs:
    50→ * - Total Trains
    51→ * - Available Trains
    52→ * - In Maintenance
    53→ * - Km Variance (with traffic-light indicator)
    54→ */
    55→export function DashboardKPIs() {
    56→  const router = useRouter();
    57→  const [summary, setSummary] = useState<DashboardSummary | null>(null);
    58→  const [loading, setLoading] = useState(true);
    59→  const [error, setError] = useState<string | null>(null);
    60→
    61→  useEffect(() => {
    62→    async function fetchSummary() {
    63→      setLoading(true);
    64→      const { data, error } = await dashboardApi.getSummary();
    65→
    66→      if (error) {
    67→        setError(error);
    68→      } else if (data) {
    69→        setSummary(data);
    70→      }
    71→      setLoading(false);
    72→    }
    73→
    74→    fetchSummary();
    75→  }, []);
    76→
    77→  // Determine variance border color and icon based on status
    78→  const getVarianceConfig = (status?: 'green' | 'yellow' | 'red') => {
    79→    switch (status) {
    80→      case 'green':
    81→        return {
    82→          borderColor: 'border-variance-green',
    83→          icon: <CheckIcon />,
    84→          subtitle: 'Target met',
    85→        };
    86→      case 'yellow':
    87→        return {
    88→          borderColor: 'border-variance-yellow',
    89→          icon: <WarningIcon />,
    90→          subtitle: 'Attention needed',
    91→        };
    92→      case 'red':
    93→        return {
    94→          borderColor: 'border-variance-red',
    95→          icon: <AlertIcon />,
    96→          subtitle: 'Critical variance',
    97→        };
    98→      default:
    99→        return {
   100→          borderColor: 'border-gray-300',
   101→          icon: null,
   102→          subtitle: undefined,
   103→        };
   104→    }
   105→  };
   106→
   107→  // Calculate available percentage
   108→  const availablePercentage = summary
   109→    ? Math.round((summary.availableTrains / summary.totalTrains) * 100)
   110→    : 0;
   111→
   112→  const varianceConfig = getVarianceConfig(summary?.varianceStatus);
   113→
   114→  if (error) {
   115→    return (
   116→      <div className="bg-red-50 border border-red-200 rounded-lg p-4">
   117→        <p className="text-red-800">
   118→          <strong>Error loading dashboard:</strong> {error}
   119→        </p>
   120→        <p className="mt-2 text-red-600 text-sm">
   121→          Make sure the API server is running at{' '}
   122→          <code className="bg-red-100 px-1 rounded">http://localhost:4000</code>
   123→        </p>
   124→      </div>
   125→    );
   126→  }
   127→
   128→  // Navigation handlers
   129→  const handleNavigateFleet = () => router.push('/fleet');
   130→  const handleNavigateAvailable = () => router.push('/fleet?status=fit');
   131→  const handleNavigateMaintenance = () => router.push('/maintenance');
   132→  const handleNavigateBalancing = () => router.push('/balancing');
   133→
   134→  return (
   135→    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
   136→      <KPICard
   137→        title="Total Trains"
   138→        value={summary?.totalTrains ?? '--'}
   139→        borderColor="border-primary-500"
   140→        icon={<TrainIcon />}
   141→        loading={loading}
   142→        onClick={handleNavigateFleet}
   143→        tooltip="Click to view all trains"
   144→      />
   145→      <KPICard
   146→        title="Available"
   147→        value={summary ? `${summary.availableTrains}` : '--'}
   148→        subtitle={summary ? `${availablePercentage}% of fleet` : undefined}
   149→        borderColor="border-status-available"
   150→        icon={<AvailableIcon />}
   151→        loading={loading}
   152→        onClick={handleNavigateAvailable}
   153→        tooltip="Click to view available trains"
   154→      />
   155→      <KPICard
   156→        title="In Maintenance"
   157→        value={summary?.inMaintenanceCount ?? '--'}
   158→        borderColor="border-status-maintenance"
   159→        icon={<MaintenanceIcon />}
   160→        loading={loading}
   161→        onClick={handleNavigateMaintenance}
   162→        tooltip="Click to view maintenance schedule"
   163→      />
   164→      <KPICard
   165→        title="Km Variance"
   166→        value={summary ? `${summary.kmVariance}%` : '--%'}
   167→        subtitle={varianceConfig.subtitle}
   168→        borderColor={varianceConfig.borderColor}
   169→        icon={varianceConfig.icon}
   170→        tooltip={summary ? `Fleet mileage variance: ${summary.kmVariance}% (target: ≤10%) - Click for details` : undefined}
   171→        loading={loading}
   172→        onClick={handleNavigateBalancing}
   173→      />
   174→    </div>
   175→  );
   176→}
