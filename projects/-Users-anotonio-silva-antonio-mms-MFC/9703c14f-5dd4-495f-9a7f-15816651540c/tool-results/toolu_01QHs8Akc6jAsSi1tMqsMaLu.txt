     1→/**
     2→ * MMS - Metro Mileage Management System
     3→ * Fastify API Server
     4→ *
     5→ * Main entry point for the backend API
     6→ */
     7→
     8→import Fastify from 'fastify';
     9→import cors from '@fastify/cors';
    10→
    11→// Import route handlers
    12→import { dashboardRoutes } from './routes/dashboard';
    13→import { trainsRoutes } from './routes/trains';
    14→import { depotsRoutes } from './routes/depots';
    15→import { servicesRoutes } from './routes/services';
    16→import { allocationsRoutes } from './routes/allocations';
    17→import { balancingRoutes } from './routes/balancing';
    18→import { simulationRoutes } from './routes/simulation';
    19→import { maintenanceRoutes } from './routes/maintenance';
    20→import { checkDatabaseConnection } from './db/client';
    21→
    22→const fastify = Fastify({
    23→  logger: {
    24→    level: process.env.LOG_LEVEL || 'info',
    25→    transport: {
    26→      target: 'pino-pretty',
    27→      options: {
    28→        translateTime: 'HH:MM:ss Z',
    29→        ignore: 'pid,hostname',
    30→      },
    31→    },
    32→  },
    33→});
    34→
    35→// Register CORS for frontend access
    36→await fastify.register(cors, {
    37→  origin: ['http://localhost:3000'],
    38→  methods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH'],
    39→});
    40→
    41→// Health check endpoint
    42→fastify.get('/health', async () => {
    43→  const dbConnected = await checkDatabaseConnection();
    44→
    45→  return {
    46→    status: dbConnected ? 'ok' : 'degraded',
    47→    timestamp: new Date().toISOString(),
    48→    service: 'mms-api',
    49→    database: dbConnected ? 'connected' : 'disconnected',
    50→  };
    51→});
    52→
    53→// Register API routes
    54→await fastify.register(dashboardRoutes, { prefix: '/api/dashboard' });
    55→await fastify.register(trainsRoutes, { prefix: '/api/trains' });
    56→await fastify.register(depotsRoutes, { prefix: '/api/depots' });
    57→await fastify.register(servicesRoutes, { prefix: '/api/services' });
    58→await fastify.register(allocationsRoutes, { prefix: '/api/allocations' });
    59→await fastify.register(balancingRoutes, { prefix: '/api/balancing' });
    60→await fastify.register(simulationRoutes, { prefix: '/api/simulation' });
    61→await fastify.register(maintenanceRoutes, { prefix: '/api/maintenance' });
    62→
    63→// Start server
    64→const start = async () => {
    65→  try {
    66→    const port = Number(process.env.PORT) || 4000;
    67→    const host = process.env.HOST || '0.0.0.0';
    68→
    69→    await fastify.listen({ port, host });
    70→
    71→    console.log(`
    72→╔════════════════════════════════════════════╗
    73→║  MMS API Server                            ║
    74→╠════════════════════════════════════════════╣
    75→║  Status:   Running                         ║
    76→║  Address:  http://${host}:${port}              ║
    77→╚════════════════════════════════════════════╝
    78→
    79→Available Endpoints:
    80→  GET  /health                    Health check
    81→  GET  /api/dashboard/summary     Dashboard KPIs
    82→  GET  /api/dashboard/depot-capacity  Depot occupancy
    83→  GET  /api/trains                Train list
    84→  GET  /api/trains/:id            Train details
    85→  GET  /api/trains/:id/history    Mileage history
    86→  GET  /api/depots                Depot list
    87→  GET  /api/services              Service list
    88→  GET  /api/services/pending      Pending allocations
    89→  POST /api/allocations/recommend Get recommendations
    90→  POST /api/allocations           Create allocation
    91→  GET  /api/balancing/distribution Km distribution
    92→  GET  /api/balancing/ranking     Train ranking
    93→  POST /api/simulation/create     Create scenario
    94→  POST /api/simulation/:id/run    Run simulation
    95→`);
    96→  } catch (err) {
    97→    fastify.log.error(err);
    98→    process.exit(1);
    99→  }
   100→};
   101→
   102→start();
