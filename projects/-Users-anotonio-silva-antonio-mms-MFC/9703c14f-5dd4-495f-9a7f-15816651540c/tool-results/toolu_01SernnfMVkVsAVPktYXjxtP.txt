     1→'use client';
     2→
     3→import React, { useState } from 'react';
     4→import { apiFetch } from '@/lib/api';
     5→
     6→interface Scenario {
     7→  id: string;
     8→  name: string;
     9→  baseDate: string;
    10→  changeCount: number;
    11→  createdAt: string;
    12→}
    13→
    14→interface ComparisonMetric {
    15→  metric: string;
    16→  current: number;
    17→  projected: number;
    18→  change: number;
    19→  improved: boolean;
    20→}
    21→
    22→interface SimulationResult {
    23→  scenarioId: string;
    24→  projectedVariance: number;
    25→  projectedDeadRunning: number;
    26→  comparison: ComparisonMetric[];
    27→}
    28→
    29→/**
    30→ * Simulation Panel Component
    31→ *
    32→ * Enables what-if scenario creation and comparison.
    33→ */
    34→export function SimulationPanel() {
    35→  const [scenarios, setScenarios] = useState<Scenario[]>([]);
    36→  const [selectedScenario, setSelectedScenario] = useState<Scenario | null>(null);
    37→  const [simulationResult, setSimulationResult] = useState<SimulationResult | null>(null);
    38→  const [creating, setCreating] = useState(false);
    39→  const [running, setRunning] = useState(false);
    40→  const [newScenarioName, setNewScenarioName] = useState('');
    41→  const [error, setError] = useState<string | null>(null);
    42→
    43→  const handleCreateScenario = async () => {
    44→    if (!newScenarioName.trim()) return;
    45→
    46→    setCreating(true);
    47→    setError(null);
    48→
    49→    const { data, error } = await apiFetch<{ scenarioId: string } & Scenario>(
    50→      '/api/simulation/create',
    51→      {
    52→        method: 'POST',
    53→        body: JSON.stringify({
    54→          name: newScenarioName,
    55→          baseDate: new Date().toISOString().split('T')[0],
    56→        }),
    57→      }
    58→    );
    59→
    60→    if (error) {
    61→      setError(error);
    62→    } else if (data) {
    63→      const newScenario: Scenario = {
    64→        id: data.scenarioId,
    65→        name: newScenarioName,
    66→        baseDate: new Date().toISOString().split('T')[0],
    67→        changeCount: 0,
    68→        createdAt: new Date().toISOString(),
    69→      };
    70→      setScenarios((prev) => [...prev, newScenario]);
    71→      setSelectedScenario(newScenario);
    72→      setNewScenarioName('');
    73→    }
    74→
    75→    setCreating(false);
    76→  };
    77→
    78→  const handleRunSimulation = async () => {
    79→    if (!selectedScenario) return;
    80→
    81→    setRunning(true);
    82→    setError(null);
    83→
    84→    // Run simulation with sample changes
    85→    const { data, error } = await apiFetch<SimulationResult>(
    86→      `/api/simulation/${selectedScenario.id}/run`,
    87→      {
    88→        method: 'POST',
    89→        body: JSON.stringify({
    90→          changes: [
    91→            // Sample allocation changes for demo
    92→            { serviceId: 'demo-1', originalTrainId: null, newTrainId: 'demo-train-1' },
    93→            { serviceId: 'demo-2', originalTrainId: null, newTrainId: 'demo-train-2' },
    94→          ],
    95→        }),
    96→      }
    97→    );
    98→
    99→    if (error) {
   100→      setError(error);
   101→    } else if (data) {
   102→      setSimulationResult(data);
   103→    }
   104→
   105→    setRunning(false);
   106→  };
   107→
   108→  return (
   109→    <div className="space-y-6">
   110→      {/* Create New Scenario */}
   111→      <div className="bg-white shadow rounded-lg p-6">
   112→        <h2 className="text-lg font-semibold text-gray-900 mb-4">
   113→          What-If Simulation
   114→        </h2>
   115→        <p className="text-sm text-gray-600 mb-4">
   116→          Create scenarios to test different allocation strategies and see projected impact on km variance.
   117→        </p>
   118→
   119→        <div className="flex gap-4">
   120→          <input
   121→            type="text"
   122→            value={newScenarioName}
   123→            onChange={(e) => setNewScenarioName(e.target.value)}
   124→            placeholder="Scenario name (e.g., Q1 Rebalancing)"
   125→            className="flex-1 px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-primary-500 focus:border-primary-500"
   126→          />
   127→          <button
   128→            onClick={handleCreateScenario}
   129→            disabled={creating || !newScenarioName.trim()}
   130→            className="px-6 py-2 bg-primary-600 text-white rounded-lg hover:bg-primary-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
   131→          >
   132→            {creating ? 'Creating...' : 'Create Scenario'}
   133→          </button>
   134→        </div>
   135→
   136→        {error && (
   137→          <div className="mt-4 p-3 bg-red-50 border border-red-200 rounded text-red-800 text-sm">
   138→            {error}
   139→          </div>
   140→        )}
   141→      </div>
   142→
   143→      {/* Scenarios List */}
   144→      {scenarios.length > 0 && (
   145→        <div className="bg-white shadow rounded-lg overflow-hidden">
   146→          <div className="p-4 border-b border-gray-200 bg-gray-50">
   147→            <h3 className="font-semibold text-gray-900">Your Scenarios</h3>
   148→          </div>
   149→          <div className="divide-y divide-gray-100">
   150→            {scenarios.map((scenario) => (
   151→              <div
   152→                key={scenario.id}
   153→                className={`p-4 cursor-pointer transition-colors ${
   154→                  selectedScenario?.id === scenario.id
   155→                    ? 'bg-primary-50 border-l-4 border-primary-500'
   156→                    : 'hover:bg-gray-50'
   157→                }`}
   158→                onClick={() => setSelectedScenario(scenario)}
   159→              >
   160→                <div className="flex items-center justify-between">
   161→                  <div>
   162→                    <div className="font-medium text-gray-900">{scenario.name}</div>
   163→                    <div className="text-sm text-gray-500">
   164→                      Base: {scenario.baseDate} • Changes: {scenario.changeCount}
   165→                    </div>
   166→                  </div>
   167→                  {selectedScenario?.id === scenario.id && (
   168→                    <button
   169→                      onClick={(e) => {
   170→                        e.stopPropagation();
   171→                        handleRunSimulation();
   172→                      }}
   173→                      disabled={running}
   174→                      className="px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700 disabled:opacity-50 text-sm transition-colors"
   175→                    >
   176→                      {running ? 'Running...' : 'Run Simulation'}
   177→                    </button>
   178→                  )}
   179→                </div>
   180→              </div>
   181→            ))}
   182→          </div>
   183→        </div>
   184→      )}
   185→
   186→      {/* Simulation Results */}
   187→      {simulationResult && (
   188→        <div className="bg-white shadow rounded-lg overflow-hidden">
   189→          <div className="p-4 border-b border-gray-200 bg-green-50">
   190→            <h3 className="font-semibold text-green-800">Simulation Results</h3>
   191→            <p className="text-sm text-green-600">
   192→              Projected impact of {selectedScenario?.name}
   193→            </p>
   194→          </div>
   195→
   196→          {/* Comparison Table */}
   197→          <div className="overflow-x-auto">
   198→            <table className="w-full">
   199→              <thead className="bg-gray-50">
   200→                <tr>
   201→                  <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase">
   202→                    Metric
   203→                  </th>
   204→                  <th className="px-4 py-3 text-right text-xs font-medium text-gray-500 uppercase">
   205→                    Current
   206→                  </th>
   207→                  <th className="px-4 py-3 text-right text-xs font-medium text-gray-500 uppercase">
   208→                    Projected
   209→                  </th>
   210→                  <th className="px-4 py-3 text-right text-xs font-medium text-gray-500 uppercase">
   211→                    Change
   212→                  </th>
   213→                </tr>
   214→              </thead>
   215→              <tbody className="divide-y divide-gray-100">
   216→                {simulationResult.comparison.map((metric) => (
   217→                  <tr key={metric.metric}>
   218→                    <td className="px-4 py-3 text-sm font-medium text-gray-900">
   219→                      {metric.metric}
   220→                    </td>
   221→                    <td className="px-4 py-3 text-sm text-right text-gray-500">
   222→                      {metric.current}
   223→                    </td>
   224→                    <td className="px-4 py-3 text-sm text-right text-gray-900 font-medium">
   225→                      {metric.projected}
   226→                    </td>
   227→                    <td className="px-4 py-3 text-sm text-right">
   228→                      <span
   229→                        className={`font-medium ${
   230→                          metric.improved ? 'text-green-600' : 'text-red-600'
   231→                        }`}
   232→                      >
   233→                        {metric.change > 0 ? '+' : ''}{metric.change}
   234→                        {metric.improved && ' ✓'}
   235→                      </span>
   236→                    </td>
   237→                  </tr>
   238→                ))}
   239→              </tbody>
   240→            </table>
   241→          </div>
   242→
   243→          {/* Summary */}
   244→          <div className="p-4 bg-gray-50 border-t">
   245→            <div className="flex items-center justify-between">
   246→              <span className="text-sm text-gray-600">
   247→                Projected Variance: <strong>{simulationResult.projectedVariance}%</strong>
   248→              </span>
   249→              <span className="text-sm text-gray-600">
   250→                Dead Running: <strong>{simulationResult.projectedDeadRunning} km</strong>
   251→              </span>
   252→            </div>
   253→          </div>
   254→        </div>
   255→      )}
   256→
   257→      {/* Empty State */}
   258→      {scenarios.length === 0 && (
   259→        <div className="bg-gray-100 border border-gray-200 rounded-lg p-8 text-center">
   260→          <div className="text-4xl mb-4">🔬</div>
   261→          <h3 className="text-lg font-medium text-gray-900 mb-2">
   262→            No Scenarios Yet
   263→          </h3>
   264→          <p className="text-gray-600">
   265→            Create a scenario above to start testing allocation strategies
   266→          </p>
   267→        </div>
   268→      )}
   269→    </div>
   270→  );
   271→}
