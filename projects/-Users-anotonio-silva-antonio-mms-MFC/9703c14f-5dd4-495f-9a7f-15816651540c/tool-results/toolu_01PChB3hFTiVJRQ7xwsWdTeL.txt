    78→function randomInt(min: number, max: number): number {
    79→  return Math.floor(Math.random() * (max - min + 1)) + min;
    80→}
    81→
    82→function randomFloat(min: number, max: number, decimals: number = 2): number {
    83→  const value = Math.random() * (max - min) + min;
    84→  return Number(value.toFixed(decimals));
    85→}
    86→
    87→function randomChoice<T>(arr: T[]): T {
    88→  return arr[Math.floor(Math.random() * arr.length)];
    89→}
    90→
    91→function formatTime(hours: number, minutes: number): string {
    92→  return `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:00`;
    93→}
    94→
    95→function addMinutes(time: string, minutes: number): string {
    96→  const [h, m] = time.split(':').map(Number);
    97→  const totalMinutes = h * 60 + m + minutes;
    98→  const newH = Math.floor(totalMinutes / 60) % 24;
    99→  const newM = totalMinutes % 60;
   100→  return formatTime(newH, newM);
   101→}

   170→      const depot = randomChoice(depotNames);
   171→      const depotData = DEPOTS.find(d => d.name === depot)!;
   172→
   173→      // Randomize position slightly from depot
   174→      const lat = depotData.lat + randomFloat(-0.05, 0.05, 6);

   176→

   178→      const fitness = Math.random() > 0.05 ? 'fit' : 'unfit';

   185→          train_id, fleet_id, base_depot_id, current_km, target_km,

   202→async function seedServices(): Promise<string[]> {
   203→  console.log('Seeding services...');
   204→  const serviceIds: string[] = [];
   205→  let serviceNumber = 100;
   206→
   207→  // Create ~250 services
   208→  const startHour = 5; // First train at 05:00
   209→  const endHour = 23; // Last train at 23:00
   210→  const interval = 30; // Minutes between services
   211→
   212→  for (let hour = startHour; hour <= endHour; hour++) {
   213→    for (let minute = 0; minute < 60; minute += interval) {
   214→      // Pick a random route
   215→      const route = randomChoice(SERVICE_ROUTES);
   216→      serviceNumber++;
   217→
   218→      const departureTime = formatTime(hour, minute);
   219→      // Calculate arrival based on distance (average 100km/h for intercity, 60km/h for regional)
   220→      const speed = route.type === 'intercity' ? 100 : 60;
   221→      const travelMinutes = Math.round((route.distance / speed) * 60);
   222→      const arrivalTime = addMinutes(departureTime, travelMinutes);
   223→
   224→      const result = await sql`
   225→        INSERT INTO services (
   226→          service_code, origin, destination, distance_km, departure_time, arrival_time, service_type
   227→        )
   228→        VALUES (
   229→          ${`${route.code} ${serviceNumber}`}, ${route.origin}, ${route.destination},
   230→          ${route.distance}, ${departureTime}, ${arrivalTime}, ${route.type}
   231→        )
   232→        RETURNING id
   233→      `;
   234→      serviceIds.push(result[0].id);
   235→
   236→      // Don't exceed 300 services
   237→      if (serviceIds.length >= 280) break;
   238→    }
   239→    if (serviceIds.length >= 280) break;
   240→  }
   241→
   242→  console.log(`  ✓ ${serviceIds.length} services created`);
   243→  return serviceIds;
   244→}

   246→async function seedMileageHistory(trainIds: string[]): Promise<void> {
   247→  console.log('Seeding mileage history (6 months)...');
   248→  const today = new Date();
   249→  let recordCount = 0;
   250→
   251→  for (const trainId of trainIds) {
   252→    // Get train's current km
   253→    const train = await sql`SELECT current_km FROM trains WHERE id = ${trainId}`;
   254→    let currentKm = train[0].current_km;
   255→
   256→    // Go back 180 days and build history
   257→    for (let daysAgo = 180; daysAgo >= 0; daysAgo -= 7) {
   258→      // Weekly mileage between 2000-4000 km
   259→      const weeklyKm = randomInt(2000, 4000);
   260→      const kmValue = Math.max(0, currentKm - (daysAgo / 7) * weeklyKm);
   261→
   262→      const recordDate = new Date(today);
   263→      recordDate.setDate(recordDate.getDate() - daysAgo);
   264→      const dateStr = recordDate.toISOString().split('T')[0];
   265→
   266→      await sql`
   267→        INSERT INTO mileage_history (train_id, recorded_date, km_value)
   268→        VALUES (${trainId}, ${dateStr}, ${Math.round(kmValue)})
   269→      `;
   270→      recordCount++;
   271→    }
   272→  }
   273→
   274→  console.log(`  ✓ ${recordCount} mileage history records created`);
   275→}

   277→async function seedMaintenanceSchedules(
   278→  trainIds: string[],
   279→  depotIds: Map<string, string>
   280→): Promise<void> {
   281→  console.log('Seeding maintenance schedules...');
   282→  const depotIdList = Array.from(depotIds.values());
   283→  let scheduleCount = 0;
   284→  const today = new Date();
   285→
   286→  // Create some past completed maintenance and future scheduled
   287→  for (const trainId of trainIds) {
   288→    // 1-3 maintenance events per train
   289→    const eventCount = randomInt(1, 3);
   290→
   291→    for (let i = 0; i < eventCount; i++) {
   292→      const maintenanceType = randomChoice(['PM1', 'PM1', 'PM1', 'PM2', 'HM']); // PM1 more common
   293→      const daysOffset = randomInt(-60, 30); // Past 60 days to next 30 days
   294→      const scheduleDate = new Date(today);
   295→      scheduleDate.setDate(scheduleDate.getDate() + daysOffset);
   296→      const dateStr = scheduleDate.toISOString().split('T')[0];
   297→
   298→      const status = daysOffset < -5 ? 'completed' : daysOffset < 0 ? 'in_progress' : 'scheduled';
   299→      const duration = maintenanceType === 'PM1' ? 1 : maintenanceType === 'PM2' ? 2 : 5;
   300→
   301→      await sql`
   302→        INSERT INTO maintenance_schedules (
   303→          train_id, depot_id, maintenance_type, scheduled_date, duration_days, status
   304→        )
   305→        VALUES (
   306→          ${trainId}, ${randomChoice(depotIdList)}, ${maintenanceType},
   307→          ${dateStr}, ${duration}, ${status}
   308→        )
   309→      `;
   310→      scheduleCount++;
   311→    }
   312→  }
   313→
   314→  console.log(`  ✓ ${scheduleCount} maintenance schedules created`);
   315→}
