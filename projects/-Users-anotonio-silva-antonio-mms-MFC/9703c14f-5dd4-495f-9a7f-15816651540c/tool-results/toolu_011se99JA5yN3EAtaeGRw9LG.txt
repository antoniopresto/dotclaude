     8→import {
     9→  pgTable,
    10→  uuid,
    11→  varchar,
    12→  integer,
    13→  decimal,
    14→  timestamp,
    15→  date,
    16→  time,
    17→  jsonb,
    18→  index,
    19→  uniqueIndex,
    20→} from 'drizzle-orm/pg-core';
    21→import { relations } from 'drizzle-orm';
    22→
    23→// =============================================================================
    24→// Fleets Table
    25→// =============================================================================
    26→export const fleets = pgTable('fleets', {
    27→  id: uuid('id').primaryKey().defaultRandom(),
    28→  name: varchar('name', { length: 100 }).notNull(),
    29→  manufacturer: varchar('manufacturer', { length: 50 }),
    30→  pm1IntervalKm: integer('pm1_interval_km').notNull(),
    31→  pm2IntervalKm: integer('pm2_interval_km').notNull(),
    32→  hmIntervalKm: integer('hm_interval_km').notNull(),
    33→  createdAt: timestamp('created_at').defaultNow(),
    34→});
    35→
    36→// =============================================================================
    37→// Depots Table
    38→// =============================================================================
    39→export const depots = pgTable(
    40→  'depots',
    41→  {
    42→    id: uuid('id').primaryKey().defaultRandom(),
    43→    name: varchar('name', { length: 100 }).notNull(),
    44→    code: varchar('code', { length: 10 }).notNull(),
    45→    latitude: decimal('latitude', { precision: 10, scale: 6 }),
    46→    longitude: decimal('longitude', { precision: 10, scale: 6 }),
    47→    capacityTracks: integer('capacity_tracks').default(5),
    48→    capacityManhours: integer('capacity_manhours').default(80),
    49→    createdAt: timestamp('created_at').defaultNow(),
    50→  },
    51→  (table) => [
    52→    uniqueIndex('depots_code_idx').on(table.code),
    53→  ]
    54→);
    55→
    56→// =============================================================================
    57→// Trains Table
    58→// =============================================================================
    59→export const trains = pgTable(
    60→  'trains',
    61→  {
    62→    id: uuid('id').primaryKey().defaultRandom(),
    63→    trainId: varchar('train_id', { length: 20 }).notNull(), // Business ID: UTE-3400-012
    64→    fleetId: uuid('fleet_id').references(() => fleets.id),
    65→    baseDepotId: uuid('base_depot_id').references(() => depots.id),
    66→    currentKm: integer('current_km').default(0),
    67→    targetKm: integer('target_km').default(0),
    68→    fitnessStatus: varchar('fitness_status', { length: 10 }).default('fit'),
    69→    lastMaintenanceDate: date('last_maintenance_date'),
    70→    nextMaintenanceKm: integer('next_maintenance_km'),
    71→    latitude: decimal('latitude', { precision: 10, scale: 6 }),
    72→    longitude: decimal('longitude', { precision: 10, scale: 6 }),
    73→    createdAt: timestamp('created_at').defaultNow(),
    74→    updatedAt: timestamp('updated_at').defaultNow(),
    75→  },
    76→  (table) => [
    77→    uniqueIndex('trains_train_id_idx').on(table.trainId),
    78→    index('trains_fleet_idx').on(table.fleetId),
    79→    index('trains_depot_idx').on(table.baseDepotId),
    80→  ]
    81→);
    82→
    83→// =============================================================================
    84→// Services Table
    85→// =============================================================================
    86→export const services = pgTable(
    87→  'services',
    88→  {
    89→    id: uuid('id').primaryKey().defaultRandom(),
    90→    serviceCode: varchar('service_code', { length: 20 }).notNull(), // IC 501
    91→    origin: varchar('origin', { length: 100 }).notNull(),
    92→    destination: varchar('destination', { length: 100 }).notNull(),
    93→    distanceKm: integer('distance_km').notNull(),
    94→    departureTime: time('departure_time').notNull(),
    95→    arrivalTime: time('arrival_time').notNull(),
    96→    serviceType: varchar('service_type', { length: 20 }).default('regional'),
    97→    createdAt: timestamp('created_at').defaultNow(),
    98→  },
    99→  (table) => [
   100→    uniqueIndex('services_code_idx').on(table.serviceCode),
   101→  ]
   102→);
   103→
   104→// =============================================================================
   105→// Allocations Table
   106→// =============================================================================
   107→export const allocations = pgTable(
   108→  'allocations',
   109→  {
   110→    id: uuid('id').primaryKey().defaultRandom(),
   111→    serviceId: uuid('service_id').references(() => services.id),
   112→    trainId: uuid('train_id').references(() => trains.id),
   113→    allocationDate: date('allocation_date').notNull(),
   114→    status: varchar('status', { length: 20 }).default('pending'),
   115→    score: integer('score'),
   116→    justifications: jsonb('justifications').$type<string[]>(),
   117→    createdBy: varchar('created_by', { length: 100 }),
   118→    createdAt: timestamp('created_at').defaultNow(),
   119→  },
   120→  (table) => [
   121→    uniqueIndex('allocations_service_date_idx').on(table.serviceId, table.allocationDate),
   122→    index('allocations_date_idx').on(table.allocationDate),
   123→    index('allocations_status_idx').on(table.status),
   124→  ]
   125→);
   126→
   127→// =============================================================================
   128→// Mileage History Table
   129→// =============================================================================
   130→export const mileageHistory = pgTable(
   131→  'mileage_history',
   132→  {
   133→    id: uuid('id').primaryKey().defaultRandom(),
   134→    trainId: uuid('train_id').references(() => trains.id),
   135→    recordedDate: date('recorded_date').notNull(),
   136→    kmValue: integer('km_value').notNull(),
   137→    createdAt: timestamp('created_at').defaultNow(),
   138→  },
   139→  (table) => [
   140→    index('mileage_train_date_idx').on(table.trainId, table.recordedDate),
   141→  ]
   142→);
   143→
   144→// =============================================================================
   145→// Maintenance Schedules Table
   146→// =============================================================================
   147→export const maintenanceSchedules = pgTable(
   148→  'maintenance_schedules',
   149→  {
   150→    id: uuid('id').primaryKey().defaultRandom(),
   151→    trainId: uuid('train_id').references(() => trains.id),
   152→    depotId: uuid('depot_id').references(() => depots.id),
   153→    maintenanceType: varchar('maintenance_type', { length: 10 }).notNull(), // PM1, PM2, HM, OVH
   154→    scheduledDate: date('scheduled_date').notNull(),
   155→    durationDays: integer('duration_days').default(1),
   156→    status: varchar('status', { length: 20 }).default('scheduled'),
   157→    createdAt: timestamp('created_at').defaultNow(),
   158→  },
   159→  (table) => [
   160→    index('maintenance_train_idx').on(table.trainId),
   161→    index('maintenance_depot_idx').on(table.depotId),
   162→    index('maintenance_date_idx').on(table.scheduledDate),
   163→  ]
   164→);

   197→export const allocationsRelations = relations(allocations, ({ one }) => ({

   211→    references: [trains.id],

   215→export const maintenanceSchedulesRelations = relations(maintenanceSchedules, ({ one }) => ({

   218→    references: [trains.id],
