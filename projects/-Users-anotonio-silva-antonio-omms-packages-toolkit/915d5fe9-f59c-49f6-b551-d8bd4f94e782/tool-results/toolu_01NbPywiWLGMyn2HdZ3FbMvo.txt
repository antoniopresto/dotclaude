    36→    const { getInitialState, reducer, machineID } = params;

    38→    let currentReducer = reducer;
    39→    let currentState = getInitialState();
    40→    let currentListeners: Listener<State>[] = [];
    41→
    42→    function getState() {
    43→      return currentState;
    44→    }
    45→

    47→      invariantType('function', listener, subscribe);
    48→      currentListeners.push(listener);
    49→
    50→      return function unsubscribe() {
    51→        const index = currentListeners.indexOf(listener);
    52→        currentListeners.splice(index, 1);
    53→      };
    54→    }

    61→        throw new Error('Actions may not have an undefined "type" property.');
    62→      }
    63→
    64→      currentState = currentReducer(currentState, action);
    65→
    66→      for (const listener of currentListeners) {
    67→        listener(currentState, action);
    68→      }
    69→
    70→      return action;
    71→    }
