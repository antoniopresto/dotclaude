     2→ * TypeBox abstraction layer.
     3→ * This is the ONLY file that should import from 'typebox' directly.
     4→ */
     5→
     6→import { Static, TSchema, Type } from 'typebox';
     7→import { Value } from 'typebox/value';
     8→
     9→/** Custom kind identifier for Date type (TypeBox 1.x uses ~kind property) */
    10→const CUSTOM_DATE_KIND = 'CustomDate';
    11→
    12→/**
    13→ * Check if a schema is our custom Date type.
    14→ */
    15→function isCustomDateSchema(schema: TSchema): boolean {
    16→  return (schema as any)['~customKind'] === CUSTOM_DATE_KIND;
    17→}
    18→
    19→/**
    20→ * Validate a value as a Date object.
    21→ */
    22→function isValidDate(value: unknown): value is Date {
    23→  return value instanceof Date && !isNaN(value.getTime());
    24→}
    25→
    26→/**
    27→ * Creates a TypeBox schema for validating Date objects.
    28→ */
    29→export function createDateType(): TSchema {
    30→  return Type.Unsafe<Date>({ '~customKind': CUSTOM_DATE_KIND });
    31→}
    32→
    33→/**
    34→ * Infers the TypeScript type from a TypeBox schema.
    35→ */
    36→export type TypeBoxInfer<T extends TSchema> = Static<T>;
    37→
    38→/**
    39→ * Result of createBaseType - contains the schema and validation utilities.
    40→ */
    41→export type BaseType<T extends TSchema> = {
    42→  /** The raw TypeBox schema */
    43→  schema: T;
    44→  /** Check if a value matches the schema */
    45→  check(value: unknown): value is Static<T>;
    46→  /** Parse and validate a value, throws on failure */
    47→  parse(value: unknown): Static<T>;
    48→  /** Get validation errors for a value */
    49→  errors(value: unknown): string[];
    50→  /** Decode a value (applies transforms) */
    51→  decode(value: unknown): Static<T>;
    52→};
    53→
    54→/**
    55→ * Creates a wrapped TypeBox type with validation utilities.
    56→ * Similar to Zod's API but using TypeBox under the hood.
    57→ */
    58→export function createBaseType<T extends TSchema>(schema: T): BaseType<T> {
    59→  const isDateSchema = isCustomDateSchema(schema);
    60→
    61→  return {
    62→    schema,
    63→    check(value: unknown): value is Static<T> {
    64→      if (isDateSchema) {
    65→        return isValidDate(value);
    66→      }
    67→      return Value.Check(schema, value);
    68→    },
    69→    parse(value: unknown): Static<T> {
    70→      if (isDateSchema) {
    71→        if (isValidDate(value)) {
    72→          return value as Static<T>;
    73→        }
    74→        throw new TypeError('Expected a valid Date object');
    75→      }
    76→      if (Value.Check(schema, value)) {
    77→        return Value.Decode(schema, value);
    78→      }
    79→      const errors = [...Value.Errors(schema, value)];
    80→      throw new TypeError(errors.map(e => e.message).join(', '));
    81→    },
    82→    errors(value: unknown): string[] {
    83→      if (isDateSchema) {
    84→        return isValidDate(value) ? [] : ['Expected a valid Date object'];
    85→      }
    86→      return [...Value.Errors(schema, value)].map(e => e.message);
    87→    },
    88→    decode(value: unknown): Static<T> {
    89→      if (isDateSchema) {
    90→        return value as Static<T>;
    91→      }
    92→      return Value.Decode(schema, value);
    93→    },
    94→  };
    95→}
    96→

    98→export { Type };
    99→export type { TSchema };
