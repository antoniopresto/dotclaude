    11→export function Hooks(params: {
    12→  machineID: string;
    13→  store: Store.Instance;
    14→  getContext(): SliceContext;

    19→  function useSelector<T>(selector: (state: any) => T): T {
    20→    return React.useSyncExternalStore(store.subscribe, () => {
    21→      const fullState = store.getState();
    22→      return selector(fullState);
    23→    });
    24→  }
    25→
    26→  // transforming action creators into action dispatchers
    27→  const runtimeActions = {} as {
    28→    [K: StringTypes.SliceKey]: {
    29→      [K: StringTypes.SliceReducerName]: (payload: unknown) => UnknownAction;
    30→    };
    31→  };
    32→
    33→  mappedSlices.info.forEach(({ sliceKey }) => {
    34→    const actions = mappedSlices.actionCreatorListBySliceKey[sliceKey];
    35→    actions.forEach(actionCreator => {
    36→      function dispatch(payload: unknown) {
    37→        const context = getContext();
    38→        const actionObject = actionCreator(payload, context);
    39→        return store.dispatch(actionObject);
    40→      }
    41→
    42→      defineFunctionName(
    43→        dispatch,
    44→        `${machineID}Runtime${capitalize(actionCreator.sliceReducerName)}`,
    45→      );
    46→      runtimeActions[sliceKey] = runtimeActions[sliceKey] || {};
    47→      runtimeActions[sliceKey][actionCreator.sliceReducerName] = dispatch;
    48→    });
    49→  });
    50→
    51→  /**
    52→   * Custom React hook to interact with a specific slice of the state machine.
    53→   * @param path The key of the slice or selector to access.
    54→   * @returns The slice's state and associated actions
    55→   */
    56→  function useMachine(path: string) {
    57→    const isSliceKey = Boolean(
    58→      mappedSlices.reducerBySliceKey[path as StringTypes.SliceKey],
    59→    );
    60→
    61→    const [initialState] = React.useState(() => {
    62→      const selection = Object.create(null);
    63→      const actions = isSliceKey
    64→        ? { ...runtimeActions[path as StringTypes.SliceKey] }
    65→        : undefined;
    66→      return Object.assign(
    67→        [
    68→          selection,
    69→          actions,
    70→        ],
    71→        actions,
    72→        selection,
    73→      );
    74→    });
    75→
    76→    const ref = React.useRef(initialState);
    77→
    78→    return useSelector(state => {
    79→      const selection = getProperty(state, path, true);
    80→      const [current, actions] = ref.current;
    81→      if (selection !== current) {
    82→        ref.current = Object.assign(
    83→          [
    84→            selection,
    85→            actions,
    86→          ],
    87→          actions,
    88→          selection,
    89→        );
    90→      }
    91→      return isSliceKey ? ref.current : ref.current[0];
    92→    });
    93→  }
    94→
    95→  return {
    96→    useSelector,
    97→    useMachine,
    98→  };
    99→}
