     1→# @mms/toolkit
     2→
     3→A TypeScript utility library providing runtime type validation with full type inference, state management, and common helpers.
     4→
     5→## Installation
     6→
     7→```bash
     8→bun add @mms/toolkit
     9→```
    10→
    11→## Quick Start
    12→
    13→```typescript
    14→import { createType, createModel, createMachine, createSlice, ms, invariant } from '@mms/toolkit';
    15→
    16→// Define a validated type with full TypeScript inference
    17→const UserSchema = createModel({
    18→  id: 'number',
    19→  name: 'string',
    20→  email: 'string?',
    21→  roles: 'string[]',
    22→});
    23→
    24→// Parse and validate data
    25→const user = UserSchema.parse(data); // Throws if invalid
    26→```

    30→## Schema System
    31→
    32→Runtime type validation with compile-time TypeScript inference.
    33→
    34→### createType / createModel
    35→
    36→```typescript
    37→import { createType, createModel, InferType } from '@mms/toolkit';
    38→
    39→// String-based definition
    40→const StringType = createType('string');
    41→const NumberArrayType = createType('number[]');
    42→const OptionalStringType = createType('string?');
    43→
    44→// Object schema with createModel (shorthand)
    45→const UserSchema = createModel({
    46→  id: 'number',
    47→  name: 'string',
    48→  email: 'string?',
    49→});
    50→
    51→// Equivalent using createType
    52→const UserSchema2 = createType({
    53→  properties: {
    54→    id: 'number',
    55→    name: 'string',
    56→    email: 'string?',
    57→  },
    58→});
    59→
    60→// Get TypeScript type from schema
    61→type User = InferType<typeof UserSchema>;
    62→// => { id: number; name: string; email?: string }
    63→```
    64→
    65→### Type Definitions
    66→
    67→#### Primitive Types
    68→
    69→| Definition | TypeScript Type | Description |
    70→|------------|-----------------|-------------|
    71→| `'string'` | `string` | String values |
    72→| `'number'` | `number` | Numeric values (rejects NaN) |
    73→| `'boolean'` | `boolean` | Boolean values |
    74→| `'null'` | `null` | Null only |
    75→| `'undefined'` | `undefined` | Undefined only |
    76→| `'date'` | `Date` | Valid Date objects |
    77→| `'function'` | `Function` | Any function |
    78→| `'object'` | `object` | Plain objects |
    79→| `'unknown'` | `unknown` | Any value (no validation) |
    80→| `'any'` | `any` | Any value (no validation) |
    81→
    82→#### Modifiers
    83→
    84→| Modifier | Example | Result |
    85→|----------|---------|--------|
    86→| `?` (optional) | `'string?'` | `string \| undefined` |
    87→| `[]` (array) | `'string[]'` | `string[]` |
    88→| `[]?` (optional array) | `'string[]?'` | `string[] \| undefined` |

   270→## State Machine
   271→
   272→Redux-like state management with slices, caching, and React hooks.
   273→
   274→### createMachine
   275→
   276→```typescript
   277→import { createMachine, createSlice, createAsyncSlice } from '@mms/toolkit';
   278→
   279→const machine = createMachine({
   280→  id: 'app',
   281→  slices: {
   282→    counter: createSlice({
   283→      id: 'counter',
   284→      initial: { count: 0 },
   285→      cache: 'session',
   286→      reducers: {
   287→        increment: (draft) => { draft.count += 1; },
   288→        decrement: (draft) => { draft.count -= 1; },
   289→        set: (draft, value: number) => { draft.count = value; },
   290→      },
   291→    }),
   292→    user: createAsyncSlice({
   293→      id: 'user',
   294→      cache: 'user',
   295→      context: { userID: '' },
   296→      load: async (ctx) => {
   297→        const res = await fetch(`/api/users/${ctx.userID}`);
   298→        return res.json();
   299→      },
   300→    }),
   301→  },
   302→  selectors: {
   303→    doubleCount: (state) => state.counter.count * 2,
   304→  },
   305→  context: { userID: 'user-123' },
   306→});
   307→```
   308→
   309→### createSlice (Sync)
   310→
   311→```typescript
   312→const counterSlice = createSlice({
   313→  id: 'counter',
   314→  initial: { count: 0, history: [] as number[] },
   315→  cache: 'session', // 'session' | 'user' | 'global'
   316→  reducers: {
   317→    // Reducers receive Immer draft - mutate directly
   318→    increment: (draft) => {
   319→      draft.count += 1;
   320→    },
   321→    add: (draft, amount: number) => {
   322→      draft.count += amount;
   323→      draft.history.push(draft.count);
   324→    },
   325→    reset: (draft) => {
   326→      draft.count = 0;
   327→      draft.history = [];
   328→    },
   329→  },
   330→});
   331→```
   332→
   333→### createAsyncSlice
   334→
   335→```typescript
   336→const usersSlice = createAsyncSlice({
   337→  id: 'users',
   338→  cache: 'session',
   339→  context: { page: 1 }, // Required context for load function
   340→  initial: { value: [], loading: false }, // Optional initial override
   341→  load: async (context) => {
   342→    const response = await fetch(`/api/users?page=${context.page}`);
   343→    return response.json();
   344→  },
   345→});
   346→
   347→// Async slices automatically generate these actions:
   348→// - load: Start loading (sets loading=true)
   349→// - success: Set value on success
   350→// - fail: Set error on failure
   351→// - clear: Reset to initial state
   352→```
   353→
   354→### Async Slice State Shape
   355→
   356→```typescript
   357→type AsyncSliceState<T> = {
   358→  loading: boolean;
   359→  loaded: boolean;
   360→  failed: boolean;
   361→  error: string | null;
   362→  value: T | null;
   363→  changes: number;
   364→};
   365→```

   367→### Cache Strategies
   368→
   369→| Strategy | Description |
   370→|----------|-------------|
   371→| `'session'` | Invalidate when sessionID changes |
   372→| `'user'` | Invalidate when userID changes |
   373→| `'global'` | Never automatically invalidate |

   375→### Machine API
   376→
   377→```typescript
   378→const machine = createMachine({ id: 'app', slices: { ... } });
   379→
   380→// Dispatch actions
   381→machine.dispatch(machine.actions.counter.increment());
   382→machine.dispatch(machine.actions.counter.add(5));
   383→
   384→// Get current state
   385→const state = machine.getState();
   386→
   387→// Subscribe to changes
   388→const unsubscribe = machine.subscribe((state, prevState) => {
   389→  console.log('State changed:', state);
   390→});
   391→
   392→// Reset to initial state
   393→machine.reset();
   394→
   395→// Access action creators and types
   396→machine.actions.counter.increment(); // => { type: '@app/counter/increment', payload: undefined }
   397→machine.actionTypes.counter.increment; // => '@app/counter/increment'
   398→
   399→// Context management
   400→machine.getContext(); // => { userID: 'user-123' }
   401→machine.provideContext({ userID: 'new-user' });
   402→
   403→// Direct reducer access (for Redux integration)
   404→const rootReducer = machine.reducer;
   405→```

   436→### Error Utilities
   437→
   438→```typescript
   439→import {
   440→  InvariantError,
   441→  invariant,
   442→  tryCatch,
   443→  moveStackTrace,
   444→  createErrorList
   445→} from '@mms/toolkit';
   446→
   447→// invariant - Assert truthy or throw
   448→invariant(user, 'User is required');
   449→invariant(user.id, (typeName) => `Expected id, got ${typeName}`);
   450→
   451→// tryCatch - Safe execution returning tuple
   452→const [error, result] = tryCatch(() => JSON.parse(data));
   453→if (error) {
   454→  console.error('Parse failed:', error.message);
   455→} else {
   456→  console.log('Parsed:', result);
   457→}
   458→
   459→// InvariantError - Custom error with clean stack traces
   460→throw new InvariantError('Invalid input', myFunction, { debug: data });
   461→
   462→// moveStackTrace - Clean error origins
   463→function validate(data: unknown) {
   464→  return moveStackTrace(() => {
   465→    if (!data) throw new Error('Data required');
   466→    return data;
   467→  }, validate); // Error points to validate() caller
   468→}
   469→
   470→// createErrorList - Collect multiple errors
   471→const errors = createErrorList('Validation failed: ');
   472→if (!name) errors.add('name is required');
   473→if (!email) errors.add('email is required');
   474→errors.check(); // Throws if any errors added
   475→```

   477→### Object Utilities
   478→
   479→```typescript
   480→import {
   481→  getProperty,
   482→  setProperty,
   483→  objectKeys,
   484→  objectEntries,
   485→  objectValues,
   486→  isObject,
   487→  assertObject,
   488→  keyBy,
   489→  reduceObject,
   490→  requireAll,
   491→  deepFreeze,
   492→} from '@mms/toolkit';
   493→
   494→// getProperty - Type-safe deep access
   495→const user = { profile: { name: 'John', age: 30 } };
   496→const name = getProperty(user, 'profile.name'); // 'John' (throws if falsy)
   497→const age = getProperty(user, 'profile.age', true); // 30 (allows falsy)
   498→
   499→// setProperty - Type-safe deep set
   500→setProperty(user, 'profile.name', 'Jane');
   501→
   502→// Typed object iteration
   503→const obj = { a: 1, b: 2 };
   504→objectKeys(obj);    // ('a' | 'b')[]
   505→objectEntries(obj); // ['a', 1] | ['b', 2][]
   506→objectValues(obj);  // number[]
   507→
   508→// isObject / assertObject
   509→if (isObject(data)) { /* data is Record<string, unknown> */ }
   510→assertObject(data); // throws if not plain object
   511→
   512→// keyBy - Create lookup from array
   513→const users = [{ id: 1, name: 'A' }, { id: 2, name: 'B' }];
   514→const byId = keyBy(users, 'id'); // { 1: {...}, 2: {...} }
   515→const byName = keyBy(users, (u) => u.name); // { A: {...}, B: {...} }
   516→
   517→// reduceObject - Reduce over entries
   518→const doubled = reduceObject({ a: 1, b: 2 }, ([k, v]) => ({ [k]: v * 2 }));
   519→// => { a: 2, b: 4 }
   520→
   521→// requireAll - Assert all properties are truthy
   522→const config = requireAll({ host, port, db }); // throws if any falsy
   523→
   524→// deepFreeze - Recursively freeze object
   525→const frozen = deepFreeze({ nested: { value: 1 } });
   526→```

   528→### Type Utilities
   529→
   530→```typescript
   531→import { stringEnum } from '@mms/toolkit';
   532→import type {
   533→  AnyRecord,
   534→  UnknownRecord,
   535→  Simplify,
   536→  Merge,
   537→  Paths,
   538→  PathType,
   539→  PromiseValue,
   540→  MaybePromise,
   541→  MaybeArray,
   542→  Writable,
   543→  PartialFields,
   544→  UnionToIntersection,
   545→  Primitive,
   546→  Serializable,
   547→  AnyFunction,
   548→  IsAny,
   549→  IsNever,
   550→  IsOptional,
   551→  Cast,
   552→  Join,
   553→} from '@mms/toolkit';
   554→
   555→// stringEnum - Create typed string array for enums
   556→const Status = stringEnum('pending', 'active', 'done');
   557→type StatusType = (typeof Status)[number]; // 'pending' | 'active' | 'done'
   558→
   559→// Simplify - Flatten intersection types for readability
   560→type Flat = Simplify<{ a: 1 } & { b: 2 }>; // { a: 1; b: 2 }
   561→
   562→// Merge - Deep merge two types
   563→type Merged = Merge<{ a: 1; b: 2 }, { b: 3; c: 4 }>; // { a: 1; b: 3; c: 4 }
   564→
   565→// Paths - Get all valid paths of an object
   566→type UserPaths = Paths<{ user: { name: string } }>; // 'user' | 'user.name'
   567→
   568→// PathType - Get type at path
   569→type NameType = PathType<{ user: { name: string } }, 'user.name'>; // string
   570→
   571→// PromiseValue - Extract resolved type
   572→type Value = PromiseValue<Promise<string>>; // string
   573→
   574→// MaybePromise - T | Promise<T>
   575→type MaybeString = MaybePromise<string>; // string | Promise<string>
   576→
   577→// MaybeArray - T | T[]
   578→type MaybeStrings = MaybeArray<string>; // string | string[]
   579→
   580→// Writable - Remove readonly modifiers
   581→type Mutable = Writable<{ readonly a: 1 }>; // { a: 1 }
   582→
   583→// PartialFields - Make specific fields optional
   584→type Partial = PartialFields<{ a: 1; b: 2; c: 3 }, ['a', 'b']>;
   585→// { a?: 1; b?: 2; c: 3 }
   586→
   587→// UnionToIntersection
   588→type Inter = UnionToIntersection<{ a: 1 } | { b: 2 }>; // { a: 1 } & { b: 2 }
   589→
   590→// Primitive types
   591→type P = Primitive; // string | number | boolean | null | undefined | bigint | symbol
   592→
   593→// Join - Join array as string type
   594→type Joined = Join<['a', 'b', 'c'], '.'>; // 'a.b.c'
   595→```
