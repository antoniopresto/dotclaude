   667→// Schema validation (replaces Zod)
   668→import { createSchema, createType, InferType } from '@belt/tools';
   669→
   670→const TrainModel = createSchema({
   671→  id: 'string',
   672→  trainNumber: 'string',
   673→  mileageKm: 'number',
   674→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   675→  tags: 'string[]',
   676→  metadata: 'object?',
   677→});
   678→type Train = InferType<typeof TrainModel>;
   679→const train = TrainModel.parse(data);
   680→
   681→// State machine (replaces React hooks + zustand)
   682→import { createSlice, createMachine } from '@belt/tools';
   683→
   684→const fleetSlice = createSlice({
   685→  id: 'fleet',
   686→  initial: { trains: [], selected: null },
   687→  cache: { policy: 'session' },
   688→  reducers: {
   689→    setTrains: (state, trains) => { state.trains = trains; },
   690→    selectTrain: (state, id) => { state.selected = id; },
   691→  },
   692→});
   693→
   694→const machine = createMachine({
   695→  id: 'mms',
   696→  slices: { fleet: fleetSlice },
   697→  trackMetadata: false, // Critical for 60fps
   698→});
   699→
   700→// In React component (NO useState/useEffect!)
   701→const { state, actions } = machine.useMachine('fleet');
