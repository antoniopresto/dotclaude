   311→#### Reference: @belt/tools Patterns
   312→```typescript
   313→// Schema validation (replaces Zod)
   314→import { createSchema, createType, InferType } from '@belt/tools';
   315→
   316→const TrainModel = createSchema({
   317→  id: 'string',
   318→  trainNumber: 'string',
   319→  mileageKm: 'number',
   320→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   321→  tags: 'string[]',
   322→  metadata: 'object?',
   323→});
   324→type Train = InferType<typeof TrainModel>;
   325→const train = TrainModel.parse(data);
   326→
   327→// State machine (replaces React hooks + zustand)
   328→import { createSlice, createMachine } from '@belt/tools';
   329→
   330→const fleetSlice = createSlice({
   331→  id: 'fleet',
   332→  initial: { trains: [], selected: null },
   333→  cache: { policy: 'session' },
   334→  reducers: {
   335→    setTrains: (state, trains) => { state.trains = trains; },
   336→    selectTrain: (state, id) => { state.selected = id; },
   337→  },
   338→});
   339→
   340→const machine = createMachine({
   341→  id: 'mms',
   342→  slices: { fleet: fleetSlice },
   343→  trackMetadata: false, // Critical for 60fps
   344→});
   345→
   346→// In React component (NO useState/useEffect!)
   347→const { state, actions } = machine.useMachine('fleet');
   348→```
