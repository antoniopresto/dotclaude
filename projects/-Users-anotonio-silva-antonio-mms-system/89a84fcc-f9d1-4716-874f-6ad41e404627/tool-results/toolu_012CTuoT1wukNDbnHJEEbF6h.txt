   624→// Schema validation (replaces Zod)
   625→import { createSchema, createType, InferType } from '@belt/tools';
   626→
   627→const TrainModel = createSchema({

   629→  trainNumber: 'string',
   630→  mileageKm: 'number',
   631→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   632→  tags: 'string[]',
   633→  metadata: 'object?',

   635→type Train = InferType<typeof TrainModel>;
   636→const train = TrainModel.parse(data);

   639→import { createSlice, createMachine } from '@belt/tools';
   640→

   642→  id: 'fleet',
   643→  initial: { trains: [], selected: null },
   644→  cache: { policy: 'session' },
   645→  reducers: {
   646→    setTrains: (state, trains) => { state.trains = trains; },
   647→    selectTrain: (state, id) => { state.selected = id; },
   648→  },
   649→});
   650→
   651→const machine = createMachine({
   652→  id: 'mms',
   653→  slices: { fleet: fleetSlice },
   654→  trackMetadata: false, // Critical for 60fps
   655→});
   656→
   657→// In React component (NO useState/useEffect!)
   658→const { state, actions } = machine.useMachine('fleet');
