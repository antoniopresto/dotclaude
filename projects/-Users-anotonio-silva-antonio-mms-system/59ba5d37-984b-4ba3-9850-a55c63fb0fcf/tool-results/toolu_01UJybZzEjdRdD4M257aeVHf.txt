   103→| **ECM-F3** | Fleet Maintenance Management | Real-time fleet status, conflict detection | Conflict indicators |
   104→| **ECM-F4** | Maintenance Delivery | Work order visualization, depot allocation | Visual representation |

   117→### 1.2 RAMS Compliance (EN 50126-1)
   118→
   119→**RAMS** = Reliability, Availability, Maintainability, Safety
   120→
   121→EN 50126-1:2017 defines the railway RAMS lifecycle. MMS visualizes RAMS metrics.
   122→
   123→#### RAMS Metric Definitions
   124→
   125→| Aspect | Definition | MMS V1.0 Feature |
   126→|--------|------------|------------------|
   127→| **Reliability** | Probability of intended function under stated conditions | MTBF display |
   128→| **Availability** | Ability to perform required function at given instant | Fleet availability % |
   129→| **Maintainability** | Probability of maintenance within stated time | MTTR display |
   130→| **Safety** | Freedom from unacceptable risk | Critical task priority |
   131→
   132→#### RAMS Targets
   133→
   134→| Metric | Formula | Target | Critical Threshold |
   135→|--------|---------|--------|-------------------|
   136→| Fleet Availability | (Fit Trains / Total Trains) x 100 | >= 97% | < 95% |
   137→| MTBF | Total Operating Hours / Number of Failures | Operator-defined | Operator-defined |
   138→| MTTR | Total Downtime Hours / Number of Repairs | <= 24 hours | > 48 hours |
   139→| Schedule Adherence | (On-Time / Scheduled) x 100 | >= 95% | < 90% |

   144→
   145→All user-facing labels MUST use EN 15380 compliant terminology.

   147→#### Mandatory Term Usage
   148→
   149→| EN 15380 Term | Definition | NOT Acceptable |

   336→### 3.3 TaskRender (Computed)
   337→
   338→```typescript
   339→/**
   340→ * Color mapping for task types.
   341→ */
   342→export type TaskColor = 'blue' | 'orange' | 'gray';
   343→
   344→export const TYPE_TO_COLOR: Record<TaskType, TaskColor> = {
   345→  maintenance: 'blue',
   346→  overhaul: 'orange',
   347→  inspection: 'gray',
   348→};
   349→
   350→/**
   351→ * TaskRender - computed values for canvas rendering.
   352→ * NEVER persisted - recomputed on data load.
   353→ */
   354→export interface TaskRender {
   355→  id: string;
   356→  trainIndex: number;     // Y position (0-based index)
   357→  startDay: number;       // X position (days since start)
   358→  duration: number;       // Width in days
   359→  color: TaskColor;       // Derived from type
   360→  hasConflict: boolean;
   361→  type: TaskType;
   362→  predecessorId?: string;
   363→}
   364→
   365→/**
   366→ * Transform persistent Task to render-ready TaskRender.
   367→ */
   368→function transformToRender(
   369→  task: Task,
   370→  trainIndexMap: Map<string, number>,
   371→  datasetStartDate: string
   372→): TaskRender {
   373→  const trainIndex = trainIndexMap.get(task.trainId);
   374→  if (trainIndex === undefined) {
   375→    throw new Error(`Train not found: ${task.trainId}`);
   376→  }
   377→
   378→  const startDay = daysBetween(datasetStartDate, task.startDate);
   379→  const duration = daysBetween(task.startDate, task.endDate);
   380→
   381→  return {
   382→    id: task.id,
   383→    trainIndex,
   384→    startDay,
   385→    duration,
   386→    color: TYPE_TO_COLOR[task.type],
   387→    hasConflict: task.hasConflict,
   388→    type: task.type,
   389→    predecessorId: task.predecessorId,
   390→  };
   391→}
   392→```

   478→## 4. Mathematical Formulas
   479→
   480→### 4.1 Km Variance (Coefficient of Variation)
   481→
   482→Primary metric for fleet mileage balance. Lower variance = better distribution.
   483→
   484→```
   485→Km Variance % = (Standard Deviation / Mean) * 100
   486→
   487→Where:
   488→  mean = sum(all train km) / n
   489→  stddev = sqrt(sum((x_i - mean)^2) / n)
   490→```
   491→
   492→```typescript
   493→function calculateKmVariance(kmValues: number[]): number {
   494→  if (kmValues.length < 2) return 0;
   495→
   496→  const mean = kmValues.reduce((a, b) => a + b, 0) / kmValues.length;
   497→  if (mean === 0) return 0;
   498→
   499→  const squaredDiffs = kmValues.map(km => Math.pow(km - mean, 2));
   500→  const variance = squaredDiffs.reduce((a, b) => a + b, 0) / kmValues.length;
   501→  const stdDev = Math.sqrt(variance);
   502→
   503→  return (stdDev / mean) * 100;
   504→}
   505→```
   506→
   507→| Status | Variance | Indicator |
   508→|--------|----------|-----------|
   509→| Green (Healthy) | <= 10% | Green badge |
   510→| Yellow (Warning) | 10% - 15% | Yellow badge |
   511→| Red (Critical) | > 15% | Red badge |
   512→

   514→
   515→### 4.2 Fleet Availability
   516→
   517→```
   518→Fleet Availability % = (Fit Trains / Total Trains) * 100
   519→
   520→Targets:
   521→  Target: >= 97%
   522→  Critical: < 95%
   523→```
   524→
   525→```typescript
   526→function calculateFleetAvailability(fitTrains: number, totalTrains: number): number {
   527→  if (totalTrains === 0) return 0;
   528→  return (fitTrains / totalTrains) * 100;
   529→}
   530→```
   531→
   532→---
   533→
   534→### 4.3 MTBF / MTTR / MDBF
   535→
   536→```
   537→MTBF = Total Operating Hours / Number of Failures (Hours)
   538→MDBF = Total Operating km / Number of Failures (Kilometers)
   539→MTTR = Total Downtime Hours / Number of Repairs (Hours)
   540→       Target: <= 24 hours, Critical: > 48 hours
   541→```
   542→
   543→```typescript
   544→interface FailureMetrics {
   545→  mtbf: number;
   546→  mdbf: number;
   547→  mttr: number;
   548→}
   549→
   550→function calculateFailureMetrics(
   551→  operatingHours: number,
   552→  operatingKm: number,
   553→  downtimeHours: number,
   554→  failureCount: number,
   555→  repairCount: number
   556→): FailureMetrics {
   557→  return {
   558→    mtbf: failureCount > 0 ? operatingHours / failureCount : Infinity,
   559→    mdbf: failureCount > 0 ? operatingKm / failureCount : Infinity,
   560→    mttr: repairCount > 0 ? downtimeHours / repairCount : 0,
   561→  };
   562→}
   563→```

   581→const PX_PER_DAY = 10;

   591→
   592→function screenToWorld(

   614→
   615→```typescript
   616→const BLEND_ZONE_SIZE = 0.15;
   617→
   618→function calculateOpacity(
   619→  zoom: number,
   620→  threshold: number,
   621→  direction: 'in' | 'out'
   622→): number {
   623→  const t = (zoom - threshold) / BLEND_ZONE_SIZE;
   624→  const clamped = Math.max(0, Math.min(1, t));
   625→  return direction === 'in' ? clamped : 1 - clamped;
   626→}
   627→```

   631→### 4.5 Km Penalty Calculation (Solver)
   632→
   633→Used by allocation solver to prioritize under-utilized trains.
   634→
   635→```
   636→deviation_pct = |current_km - target_km| / max(target_km, 1) * 100
   637→
   638→If current_km < target_km:
   639→  penalty = -deviation_pct * 10  (BONUS)
   640→If current_km >= target_km:
   641→  penalty = +deviation_pct * 10  (PENALTY)
   642→```
   643→
   644→```typescript
   645→function calculateKmPenalty(
   646→  currentKm: number,
   647→  targetKm: number,
   648→  fleetAvgKm: number
   649→): number {
   650→  const effectiveTarget = targetKm || fleetAvgKm;
   651→  if (effectiveTarget === 0) return 0;
   652→
   653→  const deviationPct = Math.abs(currentKm - effectiveTarget) / effectiveTarget * 100;
   654→
   655→  if (currentKm < effectiveTarget) {
   656→    return -Math.round(deviationPct * 10);
   657→  } else {
   658→    return Math.round(deviationPct * 10);
   659→  }
   660→}
   661→```

   665→### 4.6 Haversine Distance (Dead Running)
   666→
   667→Great circle distance between two geographic points.
   668→
   669→```
   670→a = sin^2(delta_lat/2) + cos(lat1) * cos(lat2) * sin^2(delta_lon/2)
   671→c = 2 * atan2(sqrt(a), sqrt(1-a))
   672→distance = R * c
   673→
   674→Where: R = 6371 km (Earth's radius)
   675→```
   676→
   677→```typescript
   678→function haversineDistance(
   679→  lat1: number, lon1: number,
   680→  lat2: number, lon2: number
   681→): number {
   682→  const R = 6371;
   683→  const toRad = (deg: number) => deg * Math.PI / 180;
   684→
   685→  const lat1Rad = toRad(lat1);
   686→  const lat2Rad = toRad(lat2);
   687→  const deltaLat = toRad(lat2 - lat1);
   688→  const deltaLon = toRad(lon2 - lon1);
   689→
   690→  const a = Math.sin(deltaLat / 2) ** 2 +
   691→            Math.cos(lat1Rad) * Math.cos(lat2Rad) *
   692→            Math.sin(deltaLon / 2) ** 2;
   693→
   694→  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
   695→
   696→  return R * c;
   697→}
   698→```

  1232→### 12.2 Crossfade Transitions
  1233→
  1234→```typescript
  1235→const LOD_THRESHOLDS = {
  1236→  macroToMeso: 0.5,
  1237→  mesoToMicro: 2.0,
  1238→};
  1239→const BLEND_ZONE_SIZE = 0.15;
  1240→
  1241→function calculateLayerOpacities(zoom: number): {
  1242→  macro: number; meso: number; micro: number;
  1243→} {
  1244→  const { macroToMeso, mesoToMicro } = LOD_THRESHOLDS;
  1245→
  1246→  const macroEnd = macroToMeso - BLEND_ZONE_SIZE;
  1247→  const mesoStart = macroToMeso + BLEND_ZONE_SIZE;
  1248→  const mesoEnd = mesoToMicro - BLEND_ZONE_SIZE;
  1249→  const microStart = mesoToMicro + BLEND_ZONE_SIZE;
  1250→
  1251→  let macro = 0, meso = 0, micro = 0;
  1252→
  1253→  if (zoom <= macroEnd) macro = 1;
  1254→  else if (zoom < mesoStart) {
  1255→    const t = (zoom - macroEnd) / (mesoStart - macroEnd);
  1256→    macro = 1 - t;
  1257→    meso = t;
  1258→  } else if (zoom <= mesoEnd) {
  1259→    meso = 1;
  1260→  } else if (zoom < microStart) {
  1261→    const t = (zoom - mesoEnd) / (microStart - mesoEnd);
  1262→    meso = 1 - t;
  1263→    micro = t;
  1264→  } else {
  1265→    micro = 1;
  1266→  }
  1267→
  1268→  return { macro, meso, micro };
  1269→}
  1270→```

  1413→**Problem:** Dynamic count causes GPU buffer reallocation.

  1429→### 13.3 RBush Spatial Indexing
  1430→
  1431→**MANDATORY: All viewport queries must use RBush. Array.filter() is FORBIDDEN.**
  1432→
  1433→```typescript
  1434→import RBush from 'rbush';
  1435→
  1436→interface RBushItem {
  1437→  minX: number; minY: number;
  1438→  maxX: number; maxY: number;
  1439→  id: string;
  1440→}
  1441→
  1442→export class RBushSpatialIndex {
  1443→  private tree: RBush<RBushItem>;
  1444→  private itemsById: Map<string, RBushItem> = new Map();
  1445→
  1446→  constructor() { this.tree = new RBush(); }
  1447→
  1448→  load(tasks: TaskRender[]): void {
  1449→    this.tree.clear();
  1450→    this.itemsById.clear();
  1451→
  1452→    const items: RBushItem[] = tasks.map(task => ({
  1453→      minX: task.startDay * PX_PER_DAY,
  1454→      maxX: (task.startDay + task.duration) * PX_PER_DAY,
  1455→      minY: -task.trainIndex * PX_PER_TRAIN - PX_PER_TRAIN,
  1456→      maxY: -task.trainIndex * PX_PER_TRAIN,
  1457→      id: task.id,
  1458→    }));
  1459→
  1460→    for (const item of items) {
  1461→      this.itemsById.set(item.id, item);
  1462→    }
  1463→
  1464→    this.tree.load(items);  // Bulk load O(n log n)
  1465→  }
  1466→
  1467→  queryViewport(bounds: ViewportBounds): string[] {
  1468→    const OVERSCAN = 100;
  1469→    const results = this.tree.search({
  1470→      minX: bounds.minX - OVERSCAN,
  1471→      maxX: bounds.maxX + OVERSCAN,
  1472→      minY: bounds.minY - OVERSCAN,
  1473→      maxY: bounds.maxY + OVERSCAN,
  1474→    });
  1475→    return results.map(item => item.id);
  1476→  }
  1477→
  1478→  queryNearest(x: number, y: number, maxDistance = 50): { id: string; distance: number } | null {
  1479→    let radius = 20;
  1480→    let nearest: { id: string; distance: number } | null = null;
  1481→
  1482→    while (radius <= maxDistance) {
  1483→      const candidates = this.tree.search({
  1484→        minX: x - radius, maxX: x + radius,
  1485→        minY: y - radius, maxY: y + radius,
  1486→      });
  1487→
  1488→      for (const item of candidates) {
  1489→        const centerX = (item.minX + item.maxX) / 2;
  1490→        const centerY = (item.minY + item.maxY) / 2;
  1491→        const dist = Math.sqrt((x - centerX) ** 2 + (y - centerY) ** 2);
  1492→
  1493→        if (dist <= maxDistance && (!nearest || dist < nearest.distance)) {
  1494→          nearest = { id: item.id, distance: dist };
  1495→        }
  1496→      }
  1497→
  1498→      if (nearest) return nearest;
  1499→      radius *= 2;
  1500→    }
  1501→
  1502→    return nearest;
  1503→  }
  1504→}
  1505→```
