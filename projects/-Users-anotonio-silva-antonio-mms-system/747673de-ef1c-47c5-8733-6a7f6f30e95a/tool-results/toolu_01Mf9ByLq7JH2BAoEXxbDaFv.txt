   188→function formatDate(dateStr: string): string {
   189→  const date = new Date(dateStr);
   190→  return date.toLocaleDateString('en-US', {
   191→    weekday: 'short',
   192→    month: 'short',
   193→    day: 'numeric',
   194→    year: 'numeric',
   195→  });
   196→}

   198→function formatDateTime(dateStr: string): string {
   199→  const date = new Date(dateStr);
   200→  return date.toLocaleDateString('en-US', {
   201→    month: 'short',
   202→    day: 'numeric',
   203→    hour: '2-digit',
   204→    minute: '2-digit',
   205→  });
   206→}
   207→
   208→function getDays(fromDate: Date, toDate: Date): Date[] {
   209→  const days: Date[] = [];
   210→  const current = new Date(fromDate);
   211→  while (current <= toDate) {
   212→    days.push(new Date(current));
   213→    current.setDate(current.getDate() + 1);
   214→  }
   215→  return days;
   216→}
   217→
   218→function dateDiffDays(a: Date, b: Date): number {
   219→  return Math.floor((a.getTime() - b.getTime()) / (1000 * 60 * 60 * 24));
   220→}
   221→
   222→function parseDate(dateStr: string): Date {
   223→  return new Date(dateStr + 'T00:00:00');
   224→}

   226→function buildTrainGroups(pmBlocks: DetailPmBlock[]): Map<string, DetailPmBlock[]> {
   227→  const groups = new Map<string, DetailPmBlock[]>();
   228→  for (const pm of pmBlocks) {
   229→    if (!groups.has(pm.trainId)) {
   230→      groups.set(pm.trainId, []);
   231→    }
   232→    groups.get(pm.trainId)!.push(pm);
   233→  }
   234→  return groups;
   235→}

   237→function calculateComplianceProgress(pm: DetailPmBlock): {
   238→  progress: number;
   239→  daysRemaining: number;
   240→  status: 'ok' | 'approaching' | 'violation';
   241→} {
   242→  const constraintStart = parseDate(pm.constraintStart);
   243→  const constraintEnd = parseDate(pm.constraintEnd);
   244→  const targetDate = parseDate(pm.targetDate);
   245→  const today = new Date();
   246→  today.setHours(0, 0, 0, 0);
   247→
   248→  const totalWindow = dateDiffDays(constraintEnd, constraintStart) || 1;
   249→  const positionInWindow = dateDiffDays(targetDate, constraintStart);
   250→  const daysRemaining = dateDiffDays(constraintEnd, today);
   251→  const progress = Math.min(100, Math.max(0, (positionInWindow / totalWindow) * 100));
   252→
   253→  let status: 'ok' | 'approaching' | 'violation' = 'ok';
   254→  if (progress > 90 || daysRemaining < 0) {
   255→    status = 'violation';
   256→  } else if (progress > 70 || daysRemaining < 7) {
   257→    status = 'approaching';
   258→  }
   259→
   260→  return { progress, daysRemaining, status };
   261→}

   271→function ComplianceTimeline({ pm }: ComplianceTimelineProps) {
   272→  const { progress, daysRemaining, status } = calculateComplianceProgress(pm);
   273→
   274→  let progressIntent: Intent = Intent.SUCCESS;
   275→  let statusColor: string = CONSTRAINT_COLORS.ok;
   276→  if (status === 'violation') {
   277→    progressIntent = Intent.DANGER;
   278→    statusColor = CONSTRAINT_COLORS.hardViolation;
   279→  } else if (status === 'approaching') {
   280→    progressIntent = Intent.WARNING;
   281→    statusColor = CONSTRAINT_COLORS.approaching;
   282→  }
   283→
   284→  return (
   285→    <div style={{ marginTop: 12 }}>
   286→      <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: 4 }}>
   287→        <span style={{ fontSize: 11, color: '#64748b' }}>Compliance Window</span>
   288→        <span style={{ fontSize: 11, color: statusColor, fontWeight: 500 }}>
   289→          {daysRemaining > 0 ? `${daysRemaining} days remaining` : 'Past deadline'}
   290→        </span>
   291→      </div>
   292→      <ProgressBar
   293→        value={progress / 100}
   294→        intent={progressIntent}
   295→        animate={false}
   296→        stripes={false}
   297→      />
   298→      <div style={{ display: 'flex', justifyContent: 'space-between', marginTop: 4 }}>
   299→        <span style={{ fontSize: 10, color: '#94a3b8' }}>
   300→          {formatDate(pm.constraintStart)}
   301→        </span>
   302→        <span style={{ fontSize: 10, color: '#1e293b', fontWeight: 500 }}>
   303→          Scheduled: {formatDate(pm.targetDate)}
   304→        </span>
   305→        <span style={{ fontSize: 10, color: '#94a3b8' }}>
   306→          {formatDate(pm.constraintEnd)}
   307→        </span>
   308→      </div>
   309→    </div>
   310→  );
   311→}

   366→function ChangeHistorySection({ history }: ChangeHistorySectionProps) {
   367→  if (!history || history.length === 0) {
   368→    return (
   369→      <div style={{ padding: '12px 0', color: '#94a3b8', fontStyle: 'italic', fontSize: 12 }}>
   370→        No change history available
   371→      </div>
   372→    );
   373→  }
   374→
   375→  const getChangeIcon = (type: ChangeHistoryEntry['changeType']) => {
   376→    switch (type) {
   377→      case 'CREATE':
   378→        return 'add';
   379→      case 'MOVE':
   380→        return 'move';
   381→      case 'REASSIGN':
   382→        return 'swap-horizontal';
   383→      case 'LOCK':
   384→        return 'lock';
   385→      case 'UNLOCK':
   386→        return 'unlock';
   387→      case 'STATUS_CHANGE':
   388→        return 'refresh';
   389→      default:
   390→        return 'history';
   391→    }
   392→  };
   393→
   394→  return (
   395→    <div style={{ maxHeight: 200, overflowY: 'auto' }}>
   396→      {history.slice(0, 10).map((entry) => (
   397→        <div
   398→          key={entry.id}
   399→          style={{
   400→            display: 'flex',
   401→            alignItems: 'flex-start',
   402→            padding: '8px 0',
   403→            borderBottom: '1px solid #f1f5f9',
   404→          }}
   405→        >
   406→          <Icon
   407→            icon={getChangeIcon(entry.changeType) as 'add' | 'move' | 'swap-horizontal' | 'lock' | 'unlock' | 'refresh' | 'history'}
   408→            size={14}
   409→            style={{ marginRight: 8, marginTop: 2, color: '#64748b' }}
   410→          />
   411→          <div style={{ flex: 1 }}>
   412→            <div style={{ fontSize: 12, color: '#1e293b' }}>
   413→              <strong>{entry.changeType.replace('_', ' ')}</strong>
   414→              {entry.field && ` - ${entry.field}`}
   415→            </div>
   416→            {entry.oldValue && entry.newValue && (
   417→              <div style={{ fontSize: 11, color: '#64748b' }}>
   418→                {entry.oldValue} → {entry.newValue}
   419→              </div>
   420→            )}
   421→            <div style={{ fontSize: 10, color: '#94a3b8' }}>
   422→              {formatDateTime(entry.timestamp)} by {entry.userName}
   423→            </div>
   424→          </div>
   425→        </div>
   426→      ))}
   427→    </div>
   428→  );
   429→}

   435→function ConstraintValidationSection({ validation }: ConstraintValidationSectionProps) {
   436→  if (!validation) {
   437→    return (
   438→      <div style={{ padding: '12px 0' }}>
   439→        <Tag intent={Intent.SUCCESS} icon="tick">
   440→          All constraints satisfied
   441→        </Tag>
   442→      </div>
   443→    );
   444→  }
   445→
   446→  if (validation.isValid && validation.warnings.length === 0) {
   447→    return (
   448→      <div style={{ padding: '12px 0' }}>
   449→        <Tag intent={Intent.SUCCESS} icon="tick">
   450→          All constraints satisfied
   451→        </Tag>
   452→      </div>
   453→    );
   454→  }
   455→
   456→  return (
   457→    <div style={{ padding: '8px 0' }}>
   458→      {validation.violations.map((v, i) => (
   459→        <div
   460→          key={i}
   461→          style={{
   462→            display: 'flex',
   463→            alignItems: 'center',
   464→            padding: '6px 8px',
   465→            marginBottom: 4,
   466→            backgroundColor: v.type === 'HARD' ? '#fef2f2' : '#fefce8',
   467→            borderRadius: 4,
   468→            border: `1px solid ${v.type === 'HARD' ? '#fecaca' : '#fef08a'}`,
   469→          }}
   470→        >
   471→          <Icon
   472→            icon={v.type === 'HARD' ? 'error' : 'warning-sign'}
   473→            intent={v.type === 'HARD' ? Intent.DANGER : Intent.WARNING}
   474→            size={14}
   475→            style={{ marginRight: 8 }}
   476→          />
   477→          <div style={{ flex: 1 }}>
   478→            <div style={{ fontSize: 12, fontWeight: 500, color: v.type === 'HARD' ? '#dc2626' : '#ca8a04' }}>
   479→              {v.constraint}
   480→            </div>
   481→            <div style={{ fontSize: 11, color: '#64748b' }}>{v.message}</div>
   482→          </div>
   483→          {v.canOverride && (
   484→            <Button small minimal intent={Intent.WARNING}>
   485→              Override
   486→            </Button>
   487→          )}
   488→        </div>
   489→      ))}
   490→      {validation.warnings.map((w, i) => (
   491→        <div
   492→          key={`warn-${i}`}
   493→          style={{
   494→            display: 'flex',
   495→            alignItems: 'center',
   496→            padding: '4px 8px',
   497→            marginBottom: 4,
   498→            backgroundColor: '#f0f9ff',
   499→            borderRadius: 4,
   500→          }}
   501→        >
   502→          <Icon icon="info-sign" intent={Intent.PRIMARY} size={12} style={{ marginRight: 8 }} />
   503→          <span style={{ fontSize: 11, color: '#0369a1' }}>{w.message}</span>
   504→        </div>
   505→      ))}
   506→    </div>
   507→  );
   508→}

   822→            const isWeekend = day.getDay() === 0 || day.getDay() === 6;
