     1→/**
     2→ * MMS Fleet Service
     3→ *
     4→ * Core domain service for fleet asset management.
     5→ * Provides CRUD operations for depots, fleets, trains, routes, and tracks.
     6→ * Manages train lifecycle states and mileage tracking.
     7→ *
     8→ * Business rules:
     9→ * - Train fitness transitions: FIT <-> RESTRICTED <-> GROUNDED
    10→ * - Mileage accumulation is additive (never decreases except for corrections)
    11→ * - Fleet mileage balancing: target = fleet average, variance = current - target
    12→ * - Active entities can be deactivated but not deleted (audit compliance)
    13→ *
    14→ * @see specs/003-database-schema.md Section 3.2 (Core Fleet Domain)
    15→ */

   461→  async listTrains(
   462→    filters?: TrainListFilters,
   463→    pagination?: PaginationOptions
   464→  ): Promise<PaginatedResult<Train>> {
   465→    return this.traced('listTrains', async (span) => {
   466→      const conditions: SQL<unknown>[] = [];
   467→
   468→      if (filters?.fleetId) {
   469→        conditions.push(eq(trains.fleetId, filters.fleetId));
   470→      }
   471→      if (filters?.depotId) {
   472→        conditions.push(eq(trains.currentDepotId, filters.depotId));
   473→      }
   474→      if (filters?.fitnessStatus) {
   475→        if (Array.isArray(filters.fitnessStatus)) {
   476→          conditions.push(inArray(trains.fitnessStatus, filters.fitnessStatus));
   477→        } else {
   478→          conditions.push(eq(trains.fitnessStatus, filters.fitnessStatus));
   479→        }
   480→      }
   481→      if (filters?.active !== undefined) {
   482→        conditions.push(eq(trains.active, filters.active));
   483→      }
   484→      if (filters?.trainNumbers && filters.trainNumbers.length > 0) {
   485→        conditions.push(inArray(trains.trainNumber, filters.trainNumbers));
   486→      }
   487→
   488→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   489→
   490→      // Get total count
   491→      const [{ totalCount }] = await this.db
   492→        .select({ totalCount: count() })
   493→        .from(trains)
   494→        .where(whereClause);
   495→
   496→      // Apply pagination
   497→      const page = pagination?.page ?? 1;
   498→      const pageSize = Math.min(pagination?.pageSize ?? 50, 500);
   499→      const offset = (page - 1) * pageSize;
   500→
   501→      // Determine sort order
   502→      const sortDirection = pagination?.sortOrder === 'desc' ? desc : asc;
   503→      const getOrderBy = () => {
   504→        switch (pagination?.sortBy) {
   505→          case 'mileage':
   506→            return sortDirection(trains.currentMileageKm);
   507→          case 'fitnessStatus':
   508→            return sortDirection(trains.fitnessStatus);
   509→          default:
   510→            return sortDirection(trains.trainNumber);
   511→        }
   512→      };
   513→
   514→      const data = await this.db.query.trains.findMany({
   515→        where: whereClause,
   516→        orderBy: getOrderBy(),
   517→        limit: pageSize,
   518→        offset,
   519→      });
   520→
   521→      span.setAttribute('result.count', data.length);
   522→      span.setAttribute('result.total', totalCount);
   523→
   524→      return {
   525→        data,
   526→        total: totalCount,
   527→        page,
   528→        pageSize,
   529→        totalPages: Math.ceil(totalCount / pageSize),
   530→      };
   531→    });
   532→  }

   640→  /**
   641→   * Update train fitness status with validation.
   642→   *
   643→   * Valid transitions:
   644→   * - FIT -> RESTRICTED, GROUNDED
   645→   * - RESTRICTED -> FIT, GROUNDED
   646→   * - GROUNDED -> RESTRICTED, FIT
   647→   *
   648→   * Note: Transitions from GROUNDED to FIT should typically go through RESTRICTED
   649→   * for inspection, but this is not enforced (operational flexibility).
   650→   */
   651→  async updateFitnessStatus(request: FitnessStatusChangeRequest): Promise<Train> {
   652→    return this.inTransaction('updateFitnessStatus', async (tx, span) => {
   653→      const { trainId, newStatus, notes, changedBy } = request;
   654→
   655→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   656→      span.setAttribute('fitness.newStatus', newStatus);
   657→
   658→      const train = await tx.query.trains.findFirst({

   762→  /**
   763→   * Update train mileage (typically called by ingestion service)
   764→   *
   765→   * Validates:
   766→   * - New mileage must be >= current (no rollback unless meter event)
   767→   * - Maximum daily increase: 2000 km (configurable)
   768→   */
   769→  async updateMileage(request: MileageUpdateRequest): Promise<Train> {
   770→    return this.inTransaction('updateMileage', async (tx, span) => {
   771→      const { trainId, newMileageKm, source } = request;
   772→
   773→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   774→      span.setAttribute(MmsAttributes.MILEAGE_KM, newMileageKm);
   775→
   776→      const train = await tx.query.trains.findFirst({
   777→        where: eq(trains.id, trainId),
   778→      });
   779→
   780→      if (!train) {
   781→        throw new TrainNotFoundError(trainId);
   782→      }
   783→
   784→      const oldMileage = train.currentMileageKm;
   785→      const delta = newMileageKm - oldMileage;
   786→
   787→      span.setAttribute('mileage.old', oldMileage);
   788→      span.setAttribute('mileage.delta', delta);
   789→
   790→      // Validate mileage increase
   791→      if (delta < 0) {
   792→        throw new BusinessRuleError(
   793→          `Mileage cannot decrease (${oldMileage} -> ${newMileageKm}). Use meter event for corrections.`,
   794→          {
   795→            rule: 'mileage_no_decrease',
   796→            context: { trainId, oldMileage, newMileageKm },
   797→          }
   798→        );
   799→      }
   800→
   801→      // Warn on large delta (>2000km in single update)
   802→      if (delta > 2000) {
   803→        this.log.warn(
   804→          {
   805→            trainId,
   806→            trainNumber: train.trainNumber,
   807→            oldMileage,
   808→            newMileageKm,
   809→            delta,
   810→            source,
   811→          },
   812→          'Large mileage delta detected - possible anomaly'
   813→        );
   814→      }
   815→
   816→      const [result] = await tx
   817→        .update(trains)
   818→        .set({
   819→          currentMileageKm: newMileageKm,
   820→          updatedAt: new Date(),
   821→        })
   822→        .where(eq(trains.id, trainId))
   823→        .returning();
   824→
   825→      this.log.info(
   826→        {
   827→          trainId,
   828→          trainNumber: train.trainNumber,
   829→          oldMileage,
   830→          newMileageKm,
   831→          delta,
   832→          source,
   833→        },
   834→        `Train mileage updated: ${oldMileage} -> ${newMileageKm} (+${delta} km)`
   835→      );
   836→
   837→      return result;
   838→    });
   839→  }
