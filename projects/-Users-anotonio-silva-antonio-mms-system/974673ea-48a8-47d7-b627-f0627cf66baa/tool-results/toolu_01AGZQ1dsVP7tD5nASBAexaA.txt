   304→#### Reference: @belt/tools Patterns
   305→```typescript
   306→// Schema validation (replaces Zod)
   307→import { createSchema, createType, InferType } from '@belt/tools';
   308→
   309→const TrainModel = createSchema({
   310→  id: 'string',
   311→  trainNumber: 'string',
   312→  mileageKm: 'number',
   313→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   314→  tags: 'string[]',
   315→  metadata: 'object?',
   316→});
   317→type Train = InferType<typeof TrainModel>;
   318→const train = TrainModel.parse(data);
   319→
   320→// State machine (replaces React hooks + zustand)
   321→import { createSlice, createMachine } from '@belt/tools';
   322→
   323→const fleetSlice = createSlice({
   324→  id: 'fleet',
   325→  initial: { trains: [], selected: null },
   326→  cache: { policy: 'session' },
   327→  reducers: {
   328→    setTrains: (state, trains) => { state.trains = trains; },
   329→    selectTrain: (state, id) => { state.selected = id; },
   330→  },
   331→});
   332→
   333→const machine = createMachine({
   334→  id: 'mms',
   335→  slices: { fleet: fleetSlice },
   336→  trackMetadata: false, // Critical for 60fps
   337→});
   338→
   339→// In React component (NO useState/useEffect!)
   340→const { state, actions } = machine.useMachine('fleet');
   341→```
