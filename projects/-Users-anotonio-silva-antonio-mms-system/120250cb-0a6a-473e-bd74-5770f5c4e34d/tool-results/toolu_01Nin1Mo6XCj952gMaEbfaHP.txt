    52→    case 'IN_PROGRESS':
    53→      return PM_STATUS_COLORS.inProgress;
    54→    case 'COMPLETED':
    55→      return PM_STATUS_COLORS.completed;

    60→    default:

   103→function getDays(fromDate: Date, toDate: Date): DayData[] {
   104→  const days: DayData[] = [];
   105→  const current = new Date(fromDate);
   106→  const today = new Date();
   107→  today.setHours(0, 0, 0, 0);
   108→
   109→  while (current <= toDate) {
   110→    const dayOfWeek = current.getDay();
   111→    days.push({
   112→      date: new Date(current),
   113→      dateStr: current.toISOString().split('T')[0],
   114→      dayOfWeek,
   115→      isWeekStart: dayOfWeek === 0,
   116→      isToday: current.getTime() === today.getTime(),
   117→    });
   118→    current.setDate(current.getDate() + 1);
   119→  }
   120→  return days;
   121→}
   122→
   123→function getWeeks(days: DayData[]): WeekData[] {
   124→  const weeks: WeekData[] = [];
   125→  let currentWeek: WeekData | null = null;
   126→
   127→  for (const day of days) {
   128→    if (day.isWeekStart || currentWeek === null) {
   129→      if (currentWeek) {
   130→        weeks.push(currentWeek);
   131→      }
   132→      const weekNum = getWeekNumber(day.date);
   133→      currentWeek = { weekStart: day.date, weekNum, daysInWeek: 1 };
   134→    } else {
   135→      currentWeek.daysInWeek++;
   136→    }
   137→  }
   138→  if (currentWeek) {
   139→    weeks.push(currentWeek);
   140→  }
   141→  return weeks;
   142→}
   143→
   144→function getWeekNumber(date: Date): number {
   145→  const d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));
   146→  const dayNum = d.getUTCDay() || 7;
   147→  d.setUTCDate(d.getUTCDate() + 4 - dayNum);
   148→  const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
   149→  return Math.ceil(((d.getTime() - yearStart.getTime()) / 86400000 + 1) / 7);
   150→}
   151→
   152→function dateDiffDays(a: Date, b: Date): number {
   153→  return Math.floor((a.getTime() - b.getTime()) / (1000 * 60 * 60 * 24));
   154→}
   155→
   156→function parseDate(dateStr: string): Date {
   157→  return new Date(dateStr + 'T00:00:00');
   158→}
   159→
   160→function buildTrainIndexMap(trainOrder: string[]): Map<string, number> {
   161→  const map = new Map<string, number>();
   162→  trainOrder.forEach((id, i) => map.set(id, i));
   163→  return map;
   164→}
   165→
   166→function buildTrainMetadata(pmBlocks: PmBlock[]): { numbers: Map<string, string>; fleets: Map<string, string> } {
   167→  const numbers = new Map<string, string>();
   168→  const fleets = new Map<string, string>();
   169→
   170→  for (const pm of pmBlocks) {
   171→    if (!numbers.has(pm.trainId)) {
   172→      numbers.set(pm.trainId, pm.trainNumber);
   173→      const code = pm.trainNumber.split('-')[0] || 'KS';
   174→      fleets.set(pm.trainId, code);
   175→    }
   176→  }
   177→
   178→  return { numbers, fleets };
   179→}

   323→function GanttGrid({ days, rowCount, offsetX }: GanttGridProps) {
   324→  const weekStartDays = days.filter((d) => d.isWeekStart);
   325→  const todayIndex = days.findIndex((d) => d.isToday);
   326→
   327→  return (
   328→    <Group x={offsetX} y={LAYOUT.HEADER_HEIGHT}>
   329→      {/* Row backgrounds */}
   330→      {Array.from({ length: rowCount }).map((_, i) => (
   331→        <Rect
   332→          key={i}
   333→          y={i * LAYOUT.ROW_HEIGHT}
   334→          width={days.length * LAYOUT.DAY_WIDTH}
   335→          height={LAYOUT.ROW_HEIGHT}
   336→          fill={i % 2 === 0 ? '#ffffff' : '#fafafa'}
   337→          stroke={LAYOUT.GRID_LINE_COLOR}
   338→          strokeWidth={0.5}
   339→        />
   340→      ))}
   341→
   342→      {/* Day columns - today highlight */}
   343→      {todayIndex >= 0 && (
   344→        <Rect
   345→          x={todayIndex * LAYOUT.DAY_WIDTH}
   346→          y={0}
   347→          width={LAYOUT.DAY_WIDTH}
   348→          height={rowCount * LAYOUT.ROW_HEIGHT}
   349→          fill="rgba(37, 99, 235, 0.05)"
   350→        />
   351→      )}
   352→
   353→      {/* Week separators */}
   354→      {weekStartDays.map((day) => {
   355→        const x = days.indexOf(day) * LAYOUT.DAY_WIDTH;
   356→        return (
   357→          <Line
   358→            key={`week-${day.dateStr}`}
   359→            points={[x, 0, x, rowCount * LAYOUT.ROW_HEIGHT]}
   360→            stroke="#cbd5e1"
   361→            strokeWidth={1}
   362→          />
   363→        );
   364→      })}
   365→
   366→      {/* Today line */}
   367→      {todayIndex >= 0 && (
   368→        <Line
   369→          points={[
   370→            todayIndex * LAYOUT.DAY_WIDTH,
   371→            0,
   372→            todayIndex * LAYOUT.DAY_WIDTH,
   373→            rowCount * LAYOUT.ROW_HEIGHT,
   374→          ]}
   375→          stroke="#2563eb"
   376→          strokeWidth={2}
   377→          dash={[4, 4]}
   378→        />
   379→      )}
   380→    </Group>
   381→  );
   382→}

   384→interface PmBlockRectProps {
   385→  pm: PmBlock;
   386→  x: number;
   387→  y: number;
   388→  width: number;
   389→  isSelected: boolean;
   390→  onHover?: (pm: PmBlock | null) => void;
   391→  onClick?: (pmId: string) => void;
   392→  onDragStart?: (pmId: string) => void;
   393→  onDragEnd?: (pmId: string, newDate: string) => void;
   394→  fromDate: Date;
   395→}
   396→
   397→function PmBlockRect({
   398→  pm,
   399→  x,
   400→  y,
   401→  width,
   402→  isSelected,
   403→  onHover,
   404→  onClick,
   405→  onDragStart,
   406→  onDragEnd,
   407→  fromDate,
   408→}: PmBlockRectProps) {
   409→  const color = pm.displayColor || getPmColor(pm.status);
   410→  const finalWidth = Math.max(width, LAYOUT.MIN_PM_WIDTH);
   411→
   412→  const handleDragEnd = (e: KonvaEventObject<DragEvent>) => {
   413→    if (onDragEnd) {
   414→      const newX = e.target.x();
   415→      const dayOffset = Math.round(newX / LAYOUT.DAY_WIDTH);
   416→      const newDate = new Date(fromDate);
   417→      newDate.setDate(newDate.getDate() + dayOffset);
   418→      onDragEnd(pm.pmId, newDate.toISOString().split('T')[0]);
   419→    }
   420→  };
   421→
   422→  return (
   423→    <Group
   424→      x={x}
   425→      y={y}
   426→      draggable={!pm.isLocked}
   427→      onDragStart={() => onDragStart?.(pm.pmId)}
   428→      onDragEnd={handleDragEnd}
   429→      onMouseEnter={() => onHover?.(pm)}
   430→      onMouseLeave={() => onHover?.(null)}
   431→      onClick={() => onClick?.(pm.pmId)}
   432→    >
   433→      {/* Shadow for selected */}
   434→      {isSelected && (
   435→        <Rect
   436→          x={-2}
   437→          y={-2}
   438→          width={finalWidth + 4}
   439→          height={LAYOUT.PM_BLOCK_HEIGHT + 4}
   440→          fill="transparent"
   441→          stroke="#3b82f6"
   442→          strokeWidth={2}
   443→          cornerRadius={6}
   444→        />
   445→      )}
   446→
   447→      {/* PM block */}
   448→      <Rect
   449→        width={finalWidth}
   450→        height={LAYOUT.PM_BLOCK_HEIGHT}
   451→        fill={color}
   452→        stroke={isSelected ? '#1d4ed8' : '#00000015'}
   453→        strokeWidth={1}
   454→        cornerRadius={4}
   455→        shadowColor="#000"
   456→        shadowBlur={2}
   457→        shadowOpacity={0.1}
   458→        shadowOffsetY={1}
   459→      />
   460→
   461→      {/* Lock indicator */}
   462→      {pm.isLocked && (
   463→        <Text
   464→          x={4}
   465→          y={7}
   466→          text="🔒"
   467→          fontSize={10}
   468→        />
   469→      )}
   470→
   471→      {/* PM code label */}
   472→      <Text
   473→        x={pm.isLocked ? 18 : 6}
   474→        y={7}
   475→        text={pm.pmCode}
   476→        fontSize={11}
   477→        fontFamily={LAYOUT.FONT_FAMILY}
   478→        fill="#ffffff"
   479→        fontStyle="bold"
   480→      />
   481→
   482→      {/* Duration label (if space) */}
   483→      {finalWidth > 50 && pm.estimatedHours && (
   484→        <Text
   485→          x={finalWidth - 30}
   486→          y={8}
   487→          text={`${pm.estimatedHours}h`}
   488→          fontSize={9}
   489→          fontFamily={LAYOUT.FONT_FAMILY}
   490→          fill="rgba(255,255,255,0.8)"
   491→        />
   492→      )}
   493→    </Group>
   494→  );
   495→}

   497→interface ConstraintWallsLayerProps {
   498→  walls: ConstraintWall[];
   499→  trainOrder: string[];
   500→  days: DayData[];
   501→  pmBlocks: PmBlock[];
   502→  offsetX: number;

   505→function ConstraintWallsLayer({ walls, trainOrder, days, pmBlocks, offsetX }: ConstraintWallsLayerProps) {
   506→  const trainIndexMap = buildTrainIndexMap(trainOrder);
   507→  const pmBlockMap = new Map<string, PmBlock>();
   508→  pmBlocks.forEach((pm) => pmBlockMap.set(pm.pmId, pm));
   509→
   510→  return (
   511→    <Group x={offsetX} y={LAYOUT.HEADER_HEIGHT}>
   512→      {walls.map((wall) => {
   513→        const pm = pmBlockMap.get(wall.pmId);
   514→        if (!pm) return null;
   515→
   516→        const trainIndex = trainIndexMap.get(pm.trainId);
   517→        if (trainIndex === undefined) return null;
   518→
   519→        const dayIndex = days.findIndex((d) => d.dateStr === wall.date);
   520→        if (dayIndex < 0) return null;
   521→
   522→        const x = dayIndex * LAYOUT.DAY_WIDTH;
   523→        const y = trainIndex * LAYOUT.ROW_HEIGHT;
   524→        const color = wall.type === 'COMPLIANCE_START'
   525→          ? CONSTRAINT_COLORS.approaching
   526→          : CONSTRAINT_COLORS.hardViolation;
   527→
   528→        return (
   529→          <Line
   530→            key={`${wall.pmId}-${wall.type}`}
   531→            points={[x, y + 4, x, y + LAYOUT.ROW_HEIGHT - 4]}
   532→            stroke={color}
   533→            strokeWidth={LAYOUT.CONSTRAINT_WALL_WIDTH}
   534→            dash={[4, 2]}
   535→          />
   536→        );
   537→      })}
   538→    </Group>
   539→  );
   540→}

   545→
   546→export function TacticalGanttLayer({
   547→  data,
   548→  trainOrder,
   549→  fromDate,
   550→  toDate,
   551→  width,
   552→  height,
   553→  selectedPmIds,
   554→  onPmClick,
   555→  onPmHover,
   556→  onPmDragStart,
   557→  onPmDragEnd,
   558→  showConstraintWalls = true,
   559→  showGrid = true,
   560→}: TacticalGanttLayerProps) {
   561→  const stageRef = useRef<StageType>(null);
   562→
   563→  if (!data) {
   564→    return null;
   565→  }
   566→
   567→  // Compute derived data inline (no useMemo - pure functions)
   568→  const days = getDays(fromDate, toDate);
   569→  const weeks = getWeeks(days);
   570→  const trainIndexMap = buildTrainIndexMap(trainOrder);
   571→  const { numbers: trainNumbers, fleets: fleetCodes } = buildTrainMetadata(data.pmBlocks);
   572→  const selectedSet = new Set(selectedPmIds);
   573→
   574→  const gridOffsetX = LAYOUT.TRAIN_LABEL_WIDTH;
   575→  const contentWidth = days.length * LAYOUT.DAY_WIDTH + gridOffsetX;
   576→  const contentHeight = trainOrder.length * LAYOUT.ROW_HEIGHT + LAYOUT.HEADER_HEIGHT;
   577→
   578→  return (
   579→    <Stage
   580→      ref={stageRef}
   581→      width={width}
   582→      height={height}
   583→      style={{ backgroundColor: '#f8fafc' }}
   584→    >
   585→      {/* Background layer */}
   586→      <Layer>
   587→        <Rect width={contentWidth} height={contentHeight} fill="#ffffff" />
   588→      </Layer>
   589→
   590→      {/* Grid layer */}
   591→      {showGrid && (
   592→        <Layer>
   593→          <GanttGrid days={days} rowCount={trainOrder.length} offsetX={gridOffsetX} />
   594→        </Layer>
   595→      )}
   596→
   597→      {/* Constraint walls layer */}
   598→      {showConstraintWalls && data.constraintWalls.length > 0 && (
   599→        <Layer>
   600→          <ConstraintWallsLayer
   601→            walls={data.constraintWalls}
   602→            trainOrder={trainOrder}
   603→            days={days}
   604→            pmBlocks={data.pmBlocks}
   605→            offsetX={gridOffsetX}
   606→          />
   607→        </Layer>
   608→      )}
   609→
   610→      {/* PM blocks layer */}
   611→      <Layer>
   612→        <Group x={gridOffsetX} y={LAYOUT.HEADER_HEIGHT}>
   613→          {data.pmBlocks.map((pm) => {
   614→            const trainIndex = trainIndexMap.get(pm.trainId);
   615→            if (trainIndex === undefined) return null;
   616→
   617→            const pmDate = parseDate(pm.targetDate);
   618→            const dayOffset = dateDiffDays(pmDate, fromDate);
   619→            if (dayOffset < 0 || dayOffset >= days.length) return null;
   620→
   621→            const x = dayOffset * LAYOUT.DAY_WIDTH;
   622→            const y = trainIndex * LAYOUT.ROW_HEIGHT + LAYOUT.PM_BLOCK_MARGIN;
   623→            const blockWidth = pm.durationDays * LAYOUT.DAY_WIDTH - 4;
   624→
   625→            return (
   626→              <PmBlockRect
   627→                key={pm.pmId}
   628→                pm={pm}
   629→                x={x}
   630→                y={y}
