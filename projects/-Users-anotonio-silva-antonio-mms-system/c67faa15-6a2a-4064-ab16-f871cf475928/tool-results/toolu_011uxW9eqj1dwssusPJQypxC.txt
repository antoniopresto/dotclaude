     9‚Üí## 5.1 API Overview
    10‚Üí
    11‚Üí### 5.1.1 Base Configuration
    12‚Üí
    13‚Üí```yaml
    14‚ÜíBase URL: /api/v1
    15‚ÜíContent-Type: application/json
    16‚ÜíAuthentication: Bearer JWT
    17‚ÜíRate Limiting: 100 requests/minute per user
    18‚Üí```
    19‚Üí
    20‚Üí### 5.1.2 Standard Response Format
    21‚Üí
    22‚Üí```typescript
    23‚Üí// Success Response
    24‚Üíinterface SuccessResponse<T> {
    25‚Üí  success: true;
    26‚Üí  data: T;
    27‚Üí  meta?: {
    28‚Üí    pagination?: PaginationMeta;
    29‚Üí    timing?: { durationMs: number };
    30‚Üí  };
    31‚Üí}
    32‚Üí
    33‚Üí// Error Response
    34‚Üíinterface ErrorResponse {
    35‚Üí  success: false;
    36‚Üí  error: {
    37‚Üí    code: string;           // e.g., 'VALIDATION_ERROR', 'NOT_FOUND'
    38‚Üí    message: string;        // Human-readable message
    39‚Üí    details?: any;          // Additional error context
    40‚Üí    requestId: string;      // For support/debugging
    41‚Üí  };
    42‚Üí}
    43‚Üí
    44‚Üí// Pagination
    45‚Üíinterface PaginationMeta {
    46‚Üí  page: number;
    47‚Üí  pageSize: number;
    48‚Üí  totalItems: number;
    49‚Üí  totalPages: number;
    50‚Üí  hasNext: boolean;
    51‚Üí  hasPrev: boolean;
    52‚Üí}
    53‚Üí```
    54‚Üí
    55‚Üí### 5.1.3 Standard Query Parameters
    56‚Üí
    57‚Üí| Parameter | Type | Description |
    58‚Üí|-----------|------|-------------|
    59‚Üí| `page` | number | Page number (1-indexed) |
    60‚Üí| `pageSize` | number | Items per page (default: 50, max: 200) |
    61‚Üí| `sort` | string | Sort field with direction (e.g., `targetDate:asc`) |
    62‚Üí| `filter` | string | JSON-encoded filter object |
    63‚Üí| `fields` | string | Comma-separated fields to include |
