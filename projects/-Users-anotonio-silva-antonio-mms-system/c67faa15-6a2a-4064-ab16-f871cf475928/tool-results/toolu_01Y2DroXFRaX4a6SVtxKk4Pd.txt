    11в†’The database schema is organized into the following domains:
    12в†’
    13в†’| Domain | Tables | Purpose |
    14в†’|--------|--------|---------|
    15в†’| Core Fleet | `fleets`, `trains`, `depots`, `routes` | Asset master data |
    16в†’| Mileage Tracking | `mileage_readings`, `meter_events` | Time-series odometer data |
    17в†’| Component Tracking | `train_components`, `component_mileage` | Component-level lifecycle |
    18в†’| PM Management | `pm_types`, `pm_schedule`, `pm_material_requirements` | Maintenance planning |

    22в†’| Scenario Management | `scenarios`, `scenario_deltas` | What-if analysis |
    23в†’| Costing | `labor_rates`, `material_costs`, `cost_baselines` | Financial tracking |
    24в†’| Audit & History | `pm_schedule_history`, `audit_log` | Change tracking |
    25в†’| Users & Auth | `users`, `roles`, `sessions` | Access control |

   145в†’
   146в†’```sql
   147в†’-- Meter events (odometer replacements, resets, swaps)
   148в†’CREATE TABLE meter_events (
   149в†’    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   150в†’    train_id UUID NOT NULL REFERENCES trains(id),
   151в†’    event_date DATE NOT NULL,
   152в†’    event_type VARCHAR(30) NOT NULL,  -- 'REPLACEMENT', 'RESET', 'CALIBRATION', 'SWAP'
   153в†’    
   154в†’    old_meter_serial VARCHAR(50),
   155в†’    new_meter_serial VARCHAR(50),
   156в†’    old_meter_reading INTEGER,
   157в†’    new_meter_reading INTEGER,
   158в†’    adjustment_km INTEGER,  -- Cumulative mileage adjustment to apply
   159в†’    
   160в†’    reason TEXT,
   161в†’    approved_by UUID REFERENCES users(id),
   162в†’    approved_at TIMESTAMPTZ,
   163в†’    
   164в†’    created_at TIMESTAMPTZ DEFAULT NOW(),
   165в†’    created_by UUID REFERENCES users(id)
   166в†’);
   167в†’
   168в†’CREATE INDEX idx_meter_events_train ON meter_events(train_id, event_date DESC);
   169в†’
   170в†’-- Daily mileage readings (TimescaleDB hypertable)
   171в†’CREATE TABLE mileage_readings (
   172в†’    train_id UUID NOT NULL REFERENCES trains(id),
   173в†’    reading_date DATE NOT NULL,
   174в†’    
   175в†’    -- Raw and adjusted values
   176в†’    raw_reading INTEGER NOT NULL,        -- Odometer value from CSV
   177в†’    adjusted_reading INTEGER NOT NULL,   -- After meter event adjustments
   178в†’    daily_km INTEGER,                    -- Calculated: today - yesterday
   179в†’    
   180в†’    -- Validation
   181в†’    validation_status VARCHAR(20) DEFAULT 'PENDING',  -- 'PENDING', 'VALID', 'ANOMALY', 'CORRECTED'
   182в†’    anomaly_type VARCHAR(30),            -- 'NEGATIVE', 'SPIKE', 'MISSING', 'DUPLICATE'
   183в†’    anomaly_notes TEXT,
   184в†’    
   185в†’    -- Source tracking
   186в†’    source_file VARCHAR(255),
   187в†’    meter_event_id UUID REFERENCES meter_events(id),
   188в†’    
   189в†’    ingested_at TIMESTAMPTZ DEFAULT NOW(),
   190в†’    
   191в†’    PRIMARY KEY (train_id, reading_date)
   192в†’);
   193в†’
   194в†’-- Convert to TimescaleDB hypertable
   195в†’SELECT create_hypertable('mileage_readings', 'reading_date', 
   196в†’    chunk_time_interval => INTERVAL '7 days',
   197в†’    if_not_exists => TRUE
   198в†’);
   199в†’
   200в†’-- Compression policy (compress data older than 30 days)
   201в†’SELECT add_compression_policy('mileage_readings', INTERVAL '30 days');
   202в†’
   203в†’-- Retention policy (keep 13 years for contract + buffer)
   204в†’SELECT add_retention_policy('mileage_readings', INTERVAL '13 years');
   205в†’
   206в†’-- Indexes for common queries
   207в†’CREATE INDEX idx_mileage_train_date ON mileage_readings (train_id, reading_date DESC);
   208в†’CREATE INDEX idx_mileage_validation ON mileage_readings (validation_status, reading_date DESC) 
   209в†’    WHERE validation_status != 'VALID';
   210в†’
   211в†’-- Continuous aggregate for dashboard queries
   212в†’CREATE MATERIALIZED VIEW daily_fleet_mileage
   213в†’WITH (timescaledb.continuous) AS
   214в†’SELECT 
   215в†’    time_bucket('1 day', reading_date) AS bucket,
   216в†’    t.fleet_id,
   217в†’    COUNT(*) as train_count,
   218в†’    SUM(daily_km) as total_daily_km,
   219в†’    AVG(daily_km) as avg_daily_km,
   220в†’    MIN(daily_km) as min_daily_km,
   221в†’    MAX(daily_km) as max_daily_km,
   222в†’    STDDEV(daily_km) as stddev_daily_km
   223в†’FROM mileage_readings mr
   224в†’JOIN trains t ON mr.train_id = t.id
   225в†’WHERE mr.validation_status = 'VALID'
   226в†’GROUP BY bucket, t.fleet_id
   227в†’WITH NO DATA;
   228в†’
   229в†’SELECT add_continuous_aggregate_policy('daily_fleet_mileage',
   230в†’    start_offset => INTERVAL '3 days',
   231в†’    end_offset => INTERVAL '1 hour',
   232в†’    schedule_interval => INTERVAL '1 hour'
   233в†’);
   234в†’```

   316в†’## 3.5 PM Management Tables
   317в†’
   318в†’```sql
   319в†’-- PM types (maintenance task definitions)
   320в†’CREATE TABLE pm_types (
   321в†’    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   322в†’    fleet_id UUID NOT NULL REFERENCES fleets(id),
   323в†’    pm_code VARCHAR(20) NOT NULL,      -- 'C1', 'C2', 'C3', 'C4', 'OVERHAUL'
   324в†’    pm_name VARCHAR(100) NOT NULL,
   325в†’    version INTEGER DEFAULT 1,
   326в†’    
   327в†’    -- Interval thresholds (whichever comes first)
   328в†’    interval_km INTEGER,               -- Mileage-based trigger
   329в†’    interval_months INTEGER,           -- Time-based trigger (industry standard)
   330в†’    
   331в†’    -- PM Hierarchy: which lower-level PMs does this PM satisfy?
   332в†’    -- When a C4 is completed, it satisfies C1/C2/C3 requirements
   333в†’    -- This resets their mileage counters from the C4 completion point
   334в†’    -- Example: C4 row has satisfies_codes = ['C1', 'C2', 'C3']
   335в†’    satisfies_codes VARCHAR(20)[],
   336в†’    
   337в†’    -- Resource requirements
   338в†’    estimated_hours DECIMAL(6,2) NOT NULL,
   339в†’    track_type track_type NOT NULL,
   340в†’    min_track_days INTEGER DEFAULT 1,  -- How many days track is occupied
   341в†’    
   342в†’    -- Compliance window (tolerance for scheduling)
   343в†’    early_tolerance_pct DECIMAL(5,2) DEFAULT 10.0,   -- Can do 10% early
   344в†’    late_tolerance_pct DECIMAL(5,2) DEFAULT 10.0,    -- Can do 10% late
   345в†’    
   346в†’    -- Certification requirements
   347в†’    required_certifications VARCHAR(50)[],
   348в†’    
   349в†’    -- Canvas display
   350в†’    display_color VARCHAR(7),
   351в†’    display_priority INTEGER DEFAULT 50,
   352в†’    
   353в†’    active BOOLEAN DEFAULT TRUE,
   354в†’    created_at TIMESTAMPTZ DEFAULT NOW(),
   355в†’    updated_at TIMESTAMPTZ DEFAULT NOW(),
   356в†’    
   357в†’    UNIQUE(fleet_id, pm_code, version)
   358в†’);
   359в†’
   360в†’-- Example PM hierarchy data:

   368в†’
