   354→#### Reference: @belt/tools Patterns
   355→```typescript
   356→// Schema validation (replaces Zod)
   357→import { createSchema, createType, InferType } from '@belt/tools';
   358→
   359→const TrainModel = createSchema({
   360→  id: 'string',
   361→  trainNumber: 'string',
   362→  mileageKm: 'number',
   363→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   364→  tags: 'string[]',
   365→  metadata: 'object?',
   366→});
   367→type Train = InferType<typeof TrainModel>;
   368→const train = TrainModel.parse(data);
   369→
   370→// State machine (replaces React hooks + zustand)
   371→import { createSlice, createMachine } from '@belt/tools';
   372→
   373→const fleetSlice = createSlice({
   374→  id: 'fleet',
   375→  initial: { trains: [], selected: null },
