    69→/**
    70→ * Maximo Webhook Event Types
    71→ */
    72→type MaximoWebhookEventType =
    73→  | 'wo_status_changed'
    74→  | 'wo_created'
    75→  | 'wo_updated'
    76→  | 'wo_cancelled';
    77→
    78→/**
    79→ * Maximo Webhook Payload
    80→ *
    81→ * Payload sent by Maximo when a work order status changes.
    82→ */
    83→interface MaximoWebhookPayload {
    84→  /** Event type */
    85→  event: MaximoWebhookEventType;
    86→  /** Work order number (e.g., "WO-2026-ABC123") */
    87→  wonum: string;
    88→  /** New status (e.g., "INPRG", "COMP") */
    89→  status?: MaximoWoStatus;
    90→  /** Timestamp of the change (ISO 8601) */
    91→  changedate: string;
    92→  /** Optional: Site ID */
    93→  siteid?: string;
    94→  /** Optional: Asset number (train number) */
    95→  assetnum?: string;
    96→  /** Optional: Actual start date */
    97→  actstart?: string;
    98→  /** Optional: Actual finish date */
    99→  actfinish?: string;
   100→  /** Optional: Actual labor hours */
   101→  actlabhrs?: number;
   102→  /** Optional: MMS schedule ID backreference */
   103→  mms_schedule_id?: string;
   104→}
   105→
   106→/**
   107→ * Validate Maximo webhook payload
   108→ */
   109→function validateWebhookPayload(
   110→  payload: unknown
   111→): { valid: true; data: MaximoWebhookPayload } | { valid: false; error: string } {
   112→  if (typeof payload !== 'object' || payload === null) {
   113→    return { valid: false, error: 'Payload must be an object' };
   114→  }
   115→
   116→  const p = payload as Record<string, unknown>;
   117→
   118→  // Required: event
   119→  if (!p.event || typeof p.event !== 'string') {
   120→    return { valid: false, error: 'Missing or invalid event field' };
   121→  }
   122→
   123→  const validEvents: MaximoWebhookEventType[] = [
   124→    'wo_status_changed',
   125→    'wo_created',
   126→    'wo_updated',
   127→    'wo_cancelled',
   128→  ];
   129→  if (!validEvents.includes(p.event as MaximoWebhookEventType)) {
   130→    return { valid: false, error: `Invalid event type: ${p.event}` };
   131→  }
   132→
   133→  // Required: wonum
   134→  if (!p.wonum || typeof p.wonum !== 'string') {
   135→    return { valid: false, error: 'Missing or invalid wonum field' };
   136→  }
   137→
   138→  // Required: changedate
   139→  if (!p.changedate || typeof p.changedate !== 'string') {
   140→    return { valid: false, error: 'Missing or invalid changedate field' };
   141→  }
   142→
   143→  // Validate changedate is valid ISO 8601
   144→  const changeDate = new Date(p.changedate);
   145→  if (isNaN(changeDate.getTime())) {
   146→    return { valid: false, error: 'changedate must be valid ISO 8601 format' };
   147→  }
   148→
   149→  // Optional: status (required for wo_status_changed)
   150→  if (p.event === 'wo_status_changed') {
   151→    if (!p.status || typeof p.status !== 'string') {
   152→      return { valid: false, error: 'status is required for wo_status_changed event' };
   153→    }
   154→
   155→    const validStatuses: MaximoWoStatus[] = ['WAPPR', 'APPR', 'INPRG', 'COMP', 'CLOSE', 'CAN'];
   156→    if (!validStatuses.includes(p.status as MaximoWoStatus)) {
   157→      return { valid: false, error: `Invalid status: ${p.status}` };
   158→    }
   159→  }
   160→
   161→  return {
   162→    valid: true,
   163→    data: {
   164→      event: p.event as MaximoWebhookEventType,
   165→      wonum: p.wonum as string,
   166→      status: p.status as MaximoWoStatus | undefined,
   167→      changedate: p.changedate as string,
   168→      siteid: typeof p.siteid === 'string' ? p.siteid : undefined,
   169→      assetnum: typeof p.assetnum === 'string' ? p.assetnum : undefined,
   170→      actstart: typeof p.actstart === 'string' ? p.actstart : undefined,
   171→      actfinish: typeof p.actfinish === 'string' ? p.actfinish : undefined,
   172→      actlabhrs: typeof p.actlabhrs === 'number' ? p.actlabhrs : undefined,
   173→      mms_schedule_id: typeof p.mms_schedule_id === 'string' ? p.mms_schedule_id : undefined,
   174→    },
   175→  };
   176→}
   177→
   178→/**
   179→ * POST /integration/webhooks/maximo
   180→ *
   181→ * Receive CM/WO status updates from Maximo.
   182→ *
   183→ * This endpoint is called by Maximo when:
   184→ * - Work order status changes (WAPPR → APPR → INPRG → COMP → CLOSE)
   185→ * - Work order is cancelled (CAN)
   186→ * - Actual labor hours are recorded
   187→ *
   188→ * The webhook updates the shadow record and locks the PM if necessary.
   189→ *
   190→ * @security This endpoint should be protected by Maximo webhook authentication
   191→ * (e.g., shared secret, IP allowlist, or mutual TLS)
   192→ *
   193→ * @see specs/007-integration.md Section 7.4.7
   194→ */
   195→integrationRoutes.post('/integration/webhooks/maximo', async (c) => {
   196→  const requestId = getRequestId(c);
   197→  const startTime = Date.now();
   198→
   199→  let payload: MaximoWebhookPayload;
   200→
   201→  try {
   202→    const body = await c.req.json();
   203→    const validation = validateWebhookPayload(body);
   204→
   205→    if (!validation.valid) {
   206→      logger.warn({ requestId, error: validation.error }, 'Invalid Maximo webhook payload');
   207→      return c.json(
   208→        error(ErrorCodes.VALIDATION_ERROR, validation.error, requestId),
   209→        400
   210→      );
   211→    }
   212→
   213→    payload = validation.data;
   214→  } catch (err) {
   215→    logger.warn({ requestId, error: err }, 'Failed to parse Maximo webhook payload');
   216→    return c.json(
   217→      error(ErrorCodes.VALIDATION_ERROR, 'Invalid JSON payload', requestId),
   218→      400
   219→    );
   220→  }
   221→
   222→  logger.info(
   223→    { requestId, event: payload.event, wonum: payload.wonum, status: payload.status },
   224→    'Received Maximo webhook'
   225→  );
   226→
   227→  // Using imported db singleton
   228→
   229→  // Find shadow record by Maximo WO number
   230→  const shadowRecord = await db.query.maximoShadowRecords.findFirst({
   231→    where: eq(maximoShadowRecords.maximoWoNum, payload.wonum),
   232→  });
   233→
   234→  if (!shadowRecord) {
   235→    // Check if we have an mms_schedule_id backreference
   236→    if (payload.mms_schedule_id && isValidUuid(payload.mms_schedule_id)) {
   237→      // Try to find shadow record by PM schedule ID
   238→      const shadowByPm = await db.query.maximoShadowRecords.findFirst({
   239→        where: eq(maximoShadowRecords.pmScheduleId, payload.mms_schedule_id),
   240→      });
   241→
   242→      if (shadowByPm) {
   243→        // Update shadow record with WO number
   244→        await db
   245→          .update(maximoShadowRecords)
   246→          .set({
   247→            maximoWoNum: payload.wonum,
   248→            updatedAt: new Date(),
   249→          })
   250→          .where(eq(maximoShadowRecords.id, shadowByPm.id));
   251→
   252→        logger.info(
   253→          { requestId, shadowRecordId: shadowByPm.id, wonum: payload.wonum },
   254→          'Associated Maximo WO number with existing shadow record'
   255→        );
   256→      }
   257→    }
   258→
   259→    // If still no shadow record, log and return success (idempotent)
   260→    if (!shadowRecord) {
   261→      logger.info(
   262→        { requestId, wonum: payload.wonum },
   263→        'No shadow record found for Maximo WO - may be untracked in MMS'
   264→      );
   265→
   266→      return c.json(
   267→        success({
   268→          received: true,
   269→          processed: false,
   270→          reason: 'No matching shadow record found',
   271→          wonum: payload.wonum,
   272→        }),
   273→        200
   274→      );
   275→    }
   276→  }
   277→
   278→  // Process based on event type
   279→  try {
   280→    let action = 'none';
   281→    let mmsStatusUpdated = false;
   282→    let pmLocked = false;
   283→
   284→    if (payload.event === 'wo_status_changed' && payload.status) {
   285→      // Get the current shadow record (refreshed)
   286→      const currentShadow = await db.query.maximoShadowRecords.findFirst({
   287→        where: eq(maximoShadowRecords.maximoWoNum, payload.wonum),
   288→      });
   289→
   290→      if (currentShadow) {
   291→        const previousMaximoData = currentShadow.maximoData as { status?: string } | null;
   292→        const previousStatus = previousMaximoData?.status;
   293→
   294→        // Update shadow record with new Maximo data
   295→        await db
   296→          .update(maximoShadowRecords)
   297→          .set({
   298→            maximoData: {
   299→              status: payload.status,
   300→              actualStart: payload.actstart,
   301→              actualFinish: payload.actfinish,
   302→              actualLabor: payload.actlabhrs,
   303→              changeDate: payload.changedate,
   304→            },
   305→            lastSyncAt: new Date(),
   306→            syncStatus: 'SYNCED',
   307→            updatedAt: new Date(),
   308→          })
   309→          .where(eq(maximoShadowRecords.id, currentShadow.id));
   310→
   311→        // Map Maximo status to MMS status
   312→        const mmsStatus = MAXIMO_TO_MMS_STATUS[payload.status];
   313→
   314→        // Update PM schedule status
   315→        const pmRecord = await db.query.pmSchedule.findFirst({
   316→          where: eq(pmSchedule.id, currentShadow.pmScheduleId),
   317→        });
   318→
   319→        if (pmRecord) {
   320→          const isLocked = LOCKED_MAXIMO_STATUSES.includes(payload.status);
   321→
   322→          // Update PM schedule
   323→          await db
   324→            .update(pmSchedule)
   325→            .set({
   326→              status: mmsStatus,
   327→              isLocked: isLocked,
   328→              actualStartDate: payload.actstart ? payload.actstart.split('T')[0] : pmRecord.actualStartDate,
   329→              actualEndDate: payload.actfinish ? payload.actfinish.split('T')[0] : pmRecord.actualEndDate,
   330→              actualHours: payload.actlabhrs !== undefined
   331→                ? payload.actlabhrs.toString()
   332→                : pmRecord.actualHours,
   333→              updatedAt: new Date(),
   334→            })
   335→            .where(eq(pmSchedule.id, pmRecord.id));
   336→
   337→          mmsStatusUpdated = true;
   338→          pmLocked = isLocked;
   339→
   340→          logger.info(
   341→            {
   342→              requestId,
   343→              pmScheduleId: pmRecord.id,
   344→              previousStatus,
   345→              newStatus: payload.status,
   346→              mmsStatus,
   347→              isLocked,
   348→            },
   349→            'Updated PM schedule from Maximo webhook'
   350→          );
   351→        }
   352→
   353→        action = 'status_updated';
   354→      }
   355→    } else if (payload.event === 'wo_cancelled') {
   356→      // Handle work order cancellation
   357→      const currentShadow = await db.query.maximoShadowRecords.findFirst({
   358→        where: eq(maximoShadowRecords.maximoWoNum, payload.wonum),
   359→      });
   360→
   361→      if (currentShadow) {
   362→        await db
   363→          .update(maximoShadowRecords)
   364→          .set({
   365→            maximoData: {
   366→              status: 'CAN',
   367→              changeDate: payload.changedate,
   368→            },
   369→            syncStatus: 'SYNCED',
   370→            updatedAt: new Date(),
   371→          })
   372→          .where(eq(maximoShadowRecords.id, currentShadow.id));
   373→
   374→        // Update PM schedule to CANCELLED and locked
   375→        await db
   376→          .update(pmSchedule)
   377→          .set({
   378→            status: 'CANCELLED',
   379→            isLocked: true,
   380→            updatedAt: new Date(),
   381→          })
   382→          .where(eq(pmSchedule.id, currentShadow.pmScheduleId));
   383→
   384→        action = 'wo_cancelled';
   385→        mmsStatusUpdated = true;
   386→        pmLocked = true;
   387→
   388→        logger.info(
   389→          { requestId, pmScheduleId: currentShadow.pmScheduleId, wonum: payload.wonum },
   390→          'PM cancelled from Maximo webhook'
   391→        );
   392→      }
   393→    }
   394→
   395→    const durationMs = Date.now() - startTime;
   396→
   397→    return c.json(
   398→      success({
   399→        received: true,
   400→        processed: true,
   401→        action,
   402→        wonum: payload.wonum,
   403→        event: payload.event,
   404→        mmsStatusUpdated,
   405→        pmLocked,
   406→        durationMs,
   407→      }),
   408→      200
   409→    );
   410→  } catch (err) {
   411→    logger.error({ requestId, error: err, wonum: payload.wonum }, 'Failed to process Maximo webhook');
   412→    return c.json(
   413→      error(
   414→        ErrorCodes.INTERNAL_ERROR,
   415→        'Failed to process webhook',
   416→        requestId,
   417→        { wonum: payload.wonum }
   418→      ),
   419→      500
   420→    );
   421→  }
   422→});

   582→          processed: pullResult.processed,
   583→          updated: pullResult.updated,
   584→          conflicts: pullResult.conflicts,
   585→          errors: pullResult.errors,
   586→        },
   587→      },
   588→      'Full sync completed'
   589→    );
   590→
   591→    return c.json(
   592→      success({
   593→        operation: 'full',
   594→        push: {
   595→          pushed: pushResult.pushed,
   596→          errors: pushResult.errors,
   597→        },
   598→        pull: {
   599→          processed: pullResult.processed,
   600→          updated: pullResult.updated,
   601→          conflicts: pullResult.conflicts,
   602→          errors: pullResult.errors,
   603→        },
   604→        success: pushResult.errors === 0 && pullResult.errors === 0,
   605→      }, { timing: { durationMs: getRequestDuration(c) } })
   606→    );
   607→  } catch (err) {
   608→    logger.error({ requestId, error: err }, 'Full sync failed');
   609→    return c.json(
   610→      error(ErrorCodes.INTERNAL_ERROR, 'Full sync failed', requestId),
   611→      500
   612→    );
   613→  }
   614→});

   620→/**
   621→ * GET /integration/conflicts
   622→ *
   623→ * List all shadow records with sync conflicts.
   624→ *
   625→ * Query Parameters:
   626→ * - page: Page number (default: 1)
   627→ * - pageSize: Items per page (default: 50, max: 200)
   628→ *
   629→ * @returns Paginated list of conflicts with PM details
   630→ */
   631→integrationRoutes.get('/integration/conflicts', async (c) => {
   632→  const requestId = getRequestId(c);
   633→  const service = getService();
   634→
   635→  const pagination = parsePaginationQuery({
   636→    page: c.req.query('page') ? Number(c.req.query('page')) : undefined,
   637→    pageSize: c.req.query('pageSize') ? Number(c.req.query('pageSize')) : undefined,
   638→  });
   639→
   640→  try {
   641→    const result = await service.listShadowRecords(
   642→      { syncStatus: 'CONFLICT' },
   643→      { page: pagination.page, pageSize: pagination.pageSize }
   644→    );
   645→
   646→    // Transform to API response format
   647→    const conflicts = result.data.map((item) => ({
   648→      id: item.shadowRecord.id,
