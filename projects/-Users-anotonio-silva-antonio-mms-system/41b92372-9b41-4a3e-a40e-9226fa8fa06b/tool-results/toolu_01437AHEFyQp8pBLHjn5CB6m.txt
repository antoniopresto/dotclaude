    10→## 3.1 Schema Overview
    11→
    12→The database schema is organized into the following domains:
    13→
    14→| Domain | Tables | Purpose |
    15→|--------|--------|---------|
    16→| Core Fleet | `fleets`, `trains`, `depots`, `routes` | Asset master data |
    17→| **Operational Constraints** | `service_demand`, `route_utilization_profiles`, `train_route_compatibility` | **Operational requirements and constraints** |
    18→| **Unavailability** | `unavailability_events` | **Corrective maintenance and incident tracking** |
    19→| Mileage Tracking | `mileage_readings`, `meter_events` | Time-series odometer data |
    20→| Component Tracking | `train_components`, `component_mileage` | Component-level lifecycle |
    21→| PM Management | `pm_types`, `pm_schedule`, `pm_material_requirements` | Maintenance planning |
    22→| **Track Capabilities** | `track_capabilities`, `pm_capability_requirements` | **Qualified resource matching** |
    23→| **Data Migration** | `pm_baseline_assumptions` | **Cold start and history inference** |
    24→| Material Management | `inventory_positions`, `material_reservations` | Parts availability |
    25→| Man-hour Planning | `manhour_capacity`, `manhour_projections`, `manhour_adjustments` | Workforce balancing |
    26→| Baseline & Drift | `plan_baselines`, `baseline_pm_schedule`, `baseline_mileage_projection` | Plan stability |
    27→| Scenario Management | `scenarios`, `scenario_deltas` | What-if analysis |
    28→| Costing | `labor_rates`, `material_costs`, `cost_baselines` | Financial tracking |
    29→| Audit & History | `pm_schedule_history`, `audit_log` | Change tracking |
    30→| Users & Auth | `users`, `roles`, `sessions` | Access control |
    31→
    32→> **v2.1 Changes:** Added operational constraints domain (service demand, route utilization, train-route compatibility), unavailability events, track capabilities with PM requirements, and data migration assumptions tables.

   168→## 3.2.1 Service Demand Tables (NEW v2.1)
   169→
   170→Service demand defines the minimum number of trains required for passenger operations by day type. This is a **hard constraint** for the optimizer - it cannot schedule maintenance that would violate availability targets.
   171→
   172→```sql
   173→-- Service demand by fleet and day type
   174→CREATE TABLE service_demand (
   175→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   176→    fleet_id UUID NOT NULL REFERENCES fleets(id),
   177→    effective_date DATE NOT NULL,
   178→    day_type day_type NOT NULL,  -- 'WEEKDAY', 'SATURDAY', 'SUNDAY', 'HOLIDAY'
   179→    
   180→    -- Minimum availability requirements
   181→    min_trains_required INTEGER NOT NULL,
   182→    peak_am_requirement INTEGER,       -- 6:00-9:00 requirement
   183→    midday_requirement INTEGER,        -- 9:00-16:00 requirement  
   184→    peak_pm_requirement INTEGER,       -- 16:00-19:00 requirement
   185→    evening_requirement INTEGER,       -- 19:00-23:00 requirement
   186→    standby_buffer INTEGER DEFAULT 2,  -- Additional trains for contingency
   187→    
   188→    -- Metadata
   189→    source VARCHAR(50),                -- 'OPERATIONS', 'CONTRACT', 'MANUAL'
   190→    notes TEXT,
   191→    
   192→    created_at TIMESTAMPTZ DEFAULT NOW(),
   193→    updated_at TIMESTAMPTZ DEFAULT NOW(),
   194→    created_by UUID REFERENCES users(id),
   195→    
   196→    UNIQUE(fleet_id, effective_date, day_type)
   197→);
   198→
   199→CREATE INDEX idx_service_demand_fleet ON service_demand(fleet_id, effective_date DESC);
   200→CREATE INDEX idx_service_demand_lookup ON service_demand(fleet_id, day_type, effective_date DESC);
   201→
   202→-- Example data:
   203→-- INSERT INTO service_demand (fleet_id, effective_date, day_type, min_trains_required, peak_am_requirement, standby_buffer)
   204→-- VALUES ('fleet-ks-uuid', '2026-01-01', 'WEEKDAY', 45, 52, 3);
   205→-- VALUES ('fleet-ks-uuid', '2026-01-01', 'SATURDAY', 32, 35, 2);
   206→-- VALUES ('fleet-ks-uuid', '2026-01-01', 'SUNDAY', 28, 30, 2);
   207→```

   211→## 3.2.2 Route Utilization Profiles (NEW v2.1)
   212→
   213→Route utilization profiles capture the expected daily mileage by route, enabling accurate long-term projections. Without this, mileage projection uses fleet-wide averages which can be off by 60,000+ km over 12 years.
   214→
   215→```sql
   216→-- Route-specific utilization profiles
   217→CREATE TABLE route_utilization_profiles (
   218→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   219→    route_id UUID NOT NULL REFERENCES routes(id),
   220→    effective_date DATE NOT NULL DEFAULT CURRENT_DATE,
   221→    
   222→    -- Daily mileage by day type (km)
   223→    weekday_km INTEGER NOT NULL,
   224→    saturday_km INTEGER,
   225→    sunday_km INTEGER,
   226→    holiday_km INTEGER,
   227→    
   228→    -- Seasonal adjustments
   229→    peak_season_multiplier DECIMAL(3,2) DEFAULT 1.0,  -- e.g., 1.15 for CNY
   230→    off_peak_multiplier DECIMAL(3,2) DEFAULT 1.0,
   231→    
   232→    -- Service characteristics
   233→    service_hours_per_day DECIMAL(4,1),               -- Operating hours
   234→    round_trip_km DECIMAL(6,2),                       -- Single round trip distance
   235→    trips_per_day INTEGER,                            -- Expected service runs
   236→    
   237→    -- Metadata
   238→    source VARCHAR(50),                               -- 'MEASURED', 'ESTIMATED', 'CONTRACT'
   239→    confidence confidence_level DEFAULT 'MEDIUM',
   240→    notes TEXT,
   241→    
   242→    created_at TIMESTAMPTZ DEFAULT NOW(),
   243→    updated_at TIMESTAMPTZ DEFAULT NOW(),
   244→    
   245→    UNIQUE(route_id, effective_date)
   246→);
   247→
   248→CREATE INDEX idx_route_util_route ON route_utilization_profiles(route_id, effective_date DESC);
   249→

   251→-- North-South Line: High utilization (350 km/day weekday)

   254→-- East-West Line: Medium utilization (220 km/day weekday)
   255→-- VALUES ('route-ewl-uuid', '2026-01-01', 220, 180, 150, 'HIGH');
   256→```

   260→## 3.2.3 Unavailability Events (NEW v2.1)
   261→
   262→Unavailability events track trains that cannot be scheduled due to corrective maintenance, incidents, or other blocking factors. This is a **critical integration point** with Maximo CMMS.
   263→
   264→```sql
   265→-- Unavailability events (corrective maintenance, incidents, etc.)
   266→CREATE TABLE unavailability_events (
   267→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   268→    train_id UUID NOT NULL REFERENCES trains(id),
   269→    
   270→    -- Event classification
   271→    event_type unavailability_type NOT NULL,  -- 'CORRECTIVE', 'INCIDENT', etc.
   272→    status unavailability_status NOT NULL DEFAULT 'ACTIVE',
   273→    restriction_level fitness_status NOT NULL, -- Uses existing ENUM: 'RESTRICTED' or 'GROUNDED'
   274→    
   275→    -- Timing
   276→    reported_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
   277→    estimated_repair_hours INTEGER,
   278→    estimated_return_date DATE,
   279→    actual_return_date DATE,
   280→    
   281→    -- External system linkage
   282→    source_system VARCHAR(30),        -- 'MAXIMO', 'OCC', 'MANUAL'
   283→    external_reference VARCHAR(50),   -- Maximo WO number (e.g., 'WO-2026-001234')
   284→    external_status VARCHAR(30),      -- Maximo status (e.g., 'INPRG', 'WAPPR')
   285→    
   286→    -- Description
   287→    description TEXT,
   288→    failure_code VARCHAR(30),         -- Standard failure classification
   289→    affected_system VARCHAR(50),      -- 'TRACTION', 'BRAKES', 'DOORS', 'HVAC', etc.
   290→    
   291→    -- Scheduling impact flags
   292→    blocks_pm_scheduling BOOLEAN DEFAULT TRUE,
   293→    blocks_run_assignment BOOLEAN DEFAULT TRUE,
   294→    
   295→    -- Resolution
   296→    resolved_at TIMESTAMPTZ,
   297→    resolved_by UUID REFERENCES users(id),
   298→    resolution_notes TEXT,
   299→    
   300→    created_at TIMESTAMPTZ DEFAULT NOW(),
   301→    updated_at TIMESTAMPTZ DEFAULT NOW(),
   302→    
   303→    -- Constraint: Cannot have resolved data if status is ACTIVE
   304→    CONSTRAINT chk_resolved_status CHECK (
   305→        (status = 'ACTIVE' AND resolved_at IS NULL) OR
   306→        (status = 'RESOLVED' AND resolved_at IS NOT NULL)
   307→    )
   308→);
   309→
   310→-- Primary query: Find active unavailability for a train
   311→CREATE INDEX idx_unavailability_active ON unavailability_events(train_id, status) 
   312→    WHERE status = 'ACTIVE';
   313→-- Dashboard query: All active unavailability by estimated return
   314→CREATE INDEX idx_unavailability_return ON unavailability_events(estimated_return_date) 
   315→    WHERE status = 'ACTIVE';
   316→-- Maximo sync: Find by external reference
   317→CREATE INDEX idx_unavailability_external ON unavailability_events(source_system, external_reference);
   318→-- Audit: Find by type and date range
   319→CREATE INDEX idx_unavailability_audit ON unavailability_events(event_type, reported_at DESC);
   320→
   321→-- Trigger to update train fitness_status when unavailability changes
   322→CREATE OR REPLACE FUNCTION sync_train_fitness_from_unavailability()
   323→RETURNS TRIGGER AS $$
   324→BEGIN
   325→    IF TG_OP = 'INSERT' OR TG_OP = 'UPDATE' THEN
   326→        IF NEW.status = 'ACTIVE' THEN
   327→            UPDATE trains 
   328→            SET fitness_status = NEW.restriction_level,
   329→                fitness_notes = COALESCE(NEW.description, 'Unavailability event: ' || NEW.event_type),
   330→                updated_at = NOW()
   331→            WHERE id = NEW.train_id;
   332→        ELSIF NEW.status = 'RESOLVED' THEN
   333→            -- Only reset to FIT if no other active unavailability exists
   334→            IF NOT EXISTS (
   335→                SELECT 1 FROM unavailability_events 
   336→                WHERE train_id = NEW.train_id 
   337→                AND status = 'ACTIVE' 
   338→                AND id != NEW.id
   339→            ) THEN
   340→                UPDATE trains 
   341→                SET fitness_status = 'FIT',
   342→                    fitness_notes = NULL,
   343→                    updated_at = NOW()
   344→                WHERE id = NEW.train_id;
   345→            END IF;
   346→        END IF;
   347→    END IF;
   348→    RETURN NEW;
   349→END;
   350→$$ LANGUAGE plpgsql;
   351→
   352→CREATE TRIGGER trg_sync_train_fitness
   353→AFTER INSERT OR UPDATE ON unavailability_events
   354→FOR EACH ROW EXECUTE FUNCTION sync_train_fitness_from_unavailability();
   355→```

   359→## 3.2.4 Train-Route Compatibility (NEW v2.1)
   360→
   361→Defines which trains can operate on which routes based on technical constraints (signaling, gauge, power supply, etc.). Prevents the optimizer from assigning trains to incompatible routes.
   362→
   363→```sql
   364→-- Train-route compatibility matrix
   365→CREATE TABLE train_route_compatibility (
   366→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   367→    
   368→    -- Can be at train or fleet level
   369→    train_id UUID REFERENCES trains(id),
   370→    fleet_id UUID REFERENCES fleets(id),
   371→    route_id UUID NOT NULL REFERENCES routes(id),
   372→    
   373→    -- Compatibility determination
   374→    is_compatible BOOLEAN NOT NULL DEFAULT TRUE,
   375→    constraint_type constraint_type,   -- 'SIGNALING', 'GAUGE', 'POWER_SUPPLY', etc.
   376→    restriction_reason TEXT,           -- Human-readable explanation
   377→    
   378→    -- Validity period (for temporary restrictions)
   379→    effective_date DATE DEFAULT CURRENT_DATE,
   380→    expiry_date DATE,                  -- NULL = permanent
   381→    
   382→    -- Metadata
   383→    verified_by UUID REFERENCES users(id),
   384→    verified_at TIMESTAMPTZ,
   385→    notes TEXT,
   386→    
   387→    created_at TIMESTAMPTZ DEFAULT NOW(),
   388→    updated_at TIMESTAMPTZ DEFAULT NOW(),
   389→    
   390→    -- Must specify either train_id OR fleet_id, not both
   391→    CONSTRAINT chk_train_or_fleet CHECK (
   392→        (train_id IS NOT NULL AND fleet_id IS NULL) OR
   393→        (train_id IS NULL AND fleet_id IS NOT NULL)
   394→    )
   395→);
   396→
   397→CREATE INDEX idx_compatibility_route ON train_route_compatibility(route_id, is_compatible);
   398→CREATE INDEX idx_compatibility_train ON train_route_compatibility(train_id, is_compatible) 
   399→    WHERE train_id IS NOT NULL;
   400→CREATE INDEX idx_compatibility_fleet ON train_route_compatibility(fleet_id, is_compatible) 
   401→    WHERE fleet_id IS NOT NULL;
   402→CREATE INDEX idx_compatibility_active ON train_route_compatibility(effective_date, expiry_date)
   403→    WHERE is_compatible = FALSE;
   404→
   405→-- Example data:
   406→-- Fleet KS cannot run on Line 4 (no CBTC equipment)
   407→-- INSERT INTO train_route_compatibility (fleet_id, route_id, is_compatible, constraint_type, restriction_reason)
   408→-- VALUES ('fleet-ks-uuid', 'route-line4-uuid', FALSE, 'SIGNALING', 'KS fleet lacks CBTC - only ATC equipped');
   409→
   410→-- Train 1042 temporarily restricted from NEL (oversized pantograph)
   411→-- INSERT INTO train_route_compatibility (train_id, route_id, is_compatible, constraint_type, restriction_reason, effective_date, expiry_date)
   412→-- VALUES ('train-1042-uuid', 'route-nel-uuid', FALSE, 'TUNNEL_CLEARANCE', 'Temporary - pantograph modification pending', '2026-01-01', '2026-03-31');
   413→```

   417→## 3.2.5 Track Capabilities (NEW v2.1)
   418→
   419→Defines the specific equipment and features available on each maintenance track. Enables constraint-based scheduling that matches PM requirements to track capabilities.
   420→
   421→```sql
   422→-- Track capability definitions
   423→CREATE TABLE track_capabilities (
   424→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   425→    track_id UUID NOT NULL REFERENCES tracks(id),
   426→    capability track_capability NOT NULL,
   427→    -- 'PIT', 'CRANE', 'LIFTING_JACKS', 'WHEEL_LATHE', 
   428→    -- 'CATENARY_ISOLATION', 'ROOF_ACCESS', 'WASH', 'DROP_TABLE'
   429→    
   430→    -- Capacity limits for this capability
   431→    capacity_limit JSONB,  -- e.g., {"max_axle_load_tonnes": 17, "max_lift_tonnes": 25}
   432→    
   433→    -- Availability
   434→    is_operational BOOLEAN DEFAULT TRUE,
   435→    maintenance_schedule TEXT,         -- e.g., "Unavailable Tuesdays 06:00-10:00"
   436→    
   437→    created_at TIMESTAMPTZ DEFAULT NOW(),
   438→    
   439→    UNIQUE(track_id, capability)
   440→);
   441→
   442→CREATE INDEX idx_track_cap_track ON track_capabilities(track_id);
   443→CREATE INDEX idx_track_cap_type ON track_capabilities(capability) WHERE is_operational = TRUE;
   444→
   445→-- PM type capability requirements
   446→CREATE TABLE pm_capability_requirements (
   447→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   448→    pm_type_id UUID NOT NULL REFERENCES pm_types(id),
   449→    capability track_capability NOT NULL,
   450→    is_mandatory BOOLEAN DEFAULT TRUE,  -- FALSE = preferred but not required
   451→    
   452→    notes TEXT,                         -- Explanation of why this capability is needed
   453→    
   454→    created_at TIMESTAMPTZ DEFAULT NOW(),
   455→    
   456→    UNIQUE(pm_type_id, capability)
   457→);
   458→
   459→CREATE INDEX idx_pm_cap_req_pm ON pm_capability_requirements(pm_type_id);
   460→CREATE INDEX idx_pm_cap_req_cap ON pm_capability_requirements(capability);
   461→

   478→## 3.2.6 PM Baseline Assumptions (NEW v2.1)
   479→
   480→Handles cold start scenarios where PM history is incomplete during data migration. Tracks assumptions with confidence levels and flags records requiring manual review.
   481→
   482→```sql
   483→-- PM history baseline assumptions for cold start scenarios
   484→CREATE TABLE pm_baseline_assumptions (
   485→    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
   486→    train_id UUID NOT NULL REFERENCES trains(id),
   487→    pm_type_id UUID NOT NULL REFERENCES pm_types(id),
   488→    
   489→    -- Assumption details
   490→    assumption_type assumption_type NOT NULL,
   491→    -- 'INFERRED_FROM_COMMISSIONING' - assume first PM at commissioning date + interval
   492→    -- 'INFERRED_FROM_HIGHER_PM' - derived from C4/OH completion implies C1/C2/C3 done
   493→    -- 'MANUAL_OVERRIDE' - operator entered value
   494→    -- 'IMPORTED' - from legacy system with known quality
   495→    -- 'NO_HISTORY' - no data available, using conservative estimate
   496→    
   497→    -- Assumed baseline values
   498→    assumed_last_pm_date DATE,
   499→    assumed_last_pm_mileage INTEGER,
   500→    confidence_level confidence_level NOT NULL,  -- 'HIGH', 'MEDIUM', 'LOW', 'ESTIMATED'
   501→    
   502→    -- Inference source documentation
   503→    inference_source TEXT,             -- Explanation of how assumption was derived
   504→    inference_data JSONB,              -- Supporting data (e.g., higher PM dates)
   505→    
   506→    -- Review workflow
   507→    requires_review BOOLEAN DEFAULT FALSE,
   508→    reviewed_by UUID REFERENCES users(id),
   509→    reviewed_at TIMESTAMPTZ,
   510→    review_notes TEXT,
   511→    
   512→    -- When applied
   513→    applied_at TIMESTAMPTZ,
   514→    applied_by UUID REFERENCES users(id),
   515→    
   516→    created_at TIMESTAMPTZ DEFAULT NOW(),
   517→    updated_at TIMESTAMPTZ DEFAULT NOW(),
   518→    
   519→    UNIQUE(train_id, pm_type_id)
   520→);
   521→
   522→CREATE INDEX idx_assumptions_train ON pm_baseline_assumptions(train_id);
   523→CREATE INDEX idx_assumptions_review ON pm_baseline_assumptions(requires_review) 
   524→    WHERE requires_review = TRUE;
   525→CREATE INDEX idx_assumptions_confidence ON pm_baseline_assumptions(confidence_level)
   526→    WHERE confidence_level IN ('LOW', 'ESTIMATED');
   527→
   528→-- Example inference rules (implemented in application logic):
   529→-- 1. If train commissioned 2020-01-01 and C1 interval is 6 months:
   530→--    assumed_last_pm_date = 2020-01-01, confidence = 'LOW'
   531→-- 2. If C4 completed 2024-06-15 and C4 satisfies C1/C2/C3:
   532→--    For C1/C2/C3: assumed_last_pm_date = 2024-06-15, confidence = 'HIGH'
   533→-- 3. If no data at all:
   534→--    assumed_last_pm_date = commissioning_date, confidence = 'ESTIMATED', requires_review = TRUE

   608→WITH (timescaledb.continuous) AS

  1430→## 3.13 Canvas-Optimized Views
  1431→
  1432→```sql
  1433→-- Strategic zoom level (yearly aggregates)
  1434→CREATE MATERIALIZED VIEW v_canvas_strategic AS
  1435→SELECT 
  1436→    t.id as train_id,
  1437→    t.train_number,
  1438→    t.fleet_id,
  1439→    date_trunc('year', ps.target_date) as year,
  1440→    COUNT(*) as pm_count,
  1441→    SUM(pt.estimated_hours) as total_hours,
  1442→    COUNT(*) FILTER (WHERE ps.status = 'COMPLETED') as completed_count,
  1443→    COUNT(*) FILTER (WHERE ps.is_locked) as locked_count,
  1444→    SUM(pt.estimated_hours) FILTER (WHERE ps.status = 'SCHEDULED') as pending_hours
  1445→FROM trains t
  1446→LEFT JOIN pm_schedule ps ON t.id = ps.train_id AND ps.scenario_id IS NULL
  1447→LEFT JOIN pm_types pt ON ps.pm_type_id = pt.id
  1448→WHERE t.active = TRUE
  1449→GROUP BY t.id, t.train_number, t.fleet_id, date_trunc('year', ps.target_date);
  1450→
  1451→CREATE INDEX idx_v_canvas_strategic_train ON v_canvas_strategic(train_id, year);
  1452→
  1453→-- Tactical zoom level (weekly aggregates with PM details)
  1454→CREATE MATERIALIZED VIEW v_canvas_tactical AS
  1455→SELECT 
  1456→    t.id as train_id,
  1457→    t.train_number,
  1458→    date_trunc('week', ps.target_date) as week_start,
  1459→    ps.id as pm_id,
  1460→    ps.target_date,
  1461→    ps.status,
  1462→    ps.constraint_start,
  1463→    ps.constraint_end,
  1464→    ps.is_locked,
  1465→    pt.pm_code,
  1466→    pt.estimated_hours,
  1467→    pt.display_color,
  1468→    tr.track_number,
  1469→    d.code as depot_code
  1470→FROM trains t
  1471→JOIN pm_schedule ps ON t.id = ps.train_id AND ps.scenario_id IS NULL
  1472→JOIN pm_types pt ON ps.pm_type_id = pt.id
  1473→LEFT JOIN tracks tr ON ps.assigned_track_id = tr.id
  1474→LEFT JOIN depots d ON ps.assigned_depot_id = d.id
  1475→WHERE t.active = TRUE
  1476→  AND ps.status IN ('SCHEDULED', 'IN_PROGRESS');
  1477→
  1478→CREATE INDEX idx_v_canvas_tactical_train ON v_canvas_tactical(train_id, week_start);
  1479→CREATE INDEX idx_v_canvas_tactical_date ON v_canvas_tactical(target_date);
  1480→
  1481→-- Man-hour canvas overlay
  1482→CREATE VIEW v_manhour_canvas AS
  1483→SELECT 
  1484→    mp.depot_id,
  1485→    d.code as depot_code,
  1486→    mp.period_start,
  1487→    mp.period_end,
  1488→    mp.required_total_hours,
  1489→    mp.available_total_hours,
  1490→    mp.variance_hours,
  1491→    mp.utilization_pct,
  1492→    mp.is_overloaded,
  1493→    mp.is_underutilized,
  1494→    COALESCE(adj.adjustment_count, 0) as pending_adjustments,
  1495→    COALESCE(adj.total_adjustment_hours, 0) as proposed_adjustment_hours
  1496→FROM manhour_projections mp
  1497→JOIN depots d ON mp.depot_id = d.id
  1498→LEFT JOIN (
  1499→    SELECT 
  1500→        projection_id,
  1501→        COUNT(*) as adjustment_count,
  1502→        SUM(hours_adjusted) as total_adjustment_hours
  1503→    FROM manhour_adjustments
  1504→    WHERE status = 'PROPOSED'
  1505→    GROUP BY projection_id
  1506→) adj ON mp.id = adj.projection_id
  1507→WHERE mp.projection_run_id = (
  1508→    SELECT projection_run_id 
  1509→    FROM manhour_projections 
  1510→    ORDER BY created_at DESC 
  1511→    LIMIT 1
  1512→);
  1513→
  1514→-- Drift overlay (comparison to baseline)
  1515→CREATE VIEW v_drift_canvas AS
  1516→SELECT 
  1517→    da.train_id,
  1518→    t.train_number,
  1519→    da.baseline_id,
  1520→    pb.name as baseline_name,
  1521→    da.analysis_date,
  1522→    da.mileage_drift_km,
  1523→    da.mileage_drift_pct,
  1524→    da.avg_pm_drift_days,
  1525→    da.requires_replan,
  1526→    da.replan_reason,
  1527→    -- Color coding for canvas
  1528→    CASE 
  1529→        WHEN da.requires_replan THEN 'RED'
  1530→        WHEN ABS(da.mileage_drift_pct) > 3 THEN 'YELLOW'
  1531→        ELSE 'GREEN'
  1532→    END as drift_severity
  1533→FROM drift_analysis da
  1534→JOIN trains t ON da.train_id = t.id
  1535→JOIN plan_baselines pb ON da.baseline_id = pb.id
  1536→WHERE pb.status = 'PUBLISHED'
  1537→  AND da.analysis_date = (
  1538→      SELECT MAX(analysis_date) 
  1539→      FROM drift_analysis 
  1540→      WHERE baseline_id = da.baseline_id
  1541→  );
  1542→
  1543→-- Refresh materialized views (scheduled job)
  1544→-- REFRESH MATERIALIZED VIEW CONCURRENTLY v_canvas_strategic;
  1545→-- REFRESH MATERIALIZED VIEW CONCURRENTLY v_canvas_tactical;
