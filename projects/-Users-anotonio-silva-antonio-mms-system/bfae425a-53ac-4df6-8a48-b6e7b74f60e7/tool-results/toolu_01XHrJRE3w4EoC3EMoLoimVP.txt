    49‚Üí * Map drift severity based on whether replan is required
    50‚Üí */
    51‚Üífunction getDriftSeverity(
    52‚Üí  mileageDriftPct: number,
    53‚Üí  maxPmDriftDays: number,
    54‚Üí  requiresReplan: boolean,
    55‚Üí  thresholds: { mileageDriftPct: number; scheduleDriftDays: number }
    56‚Üí): 'GREEN' | 'YELLOW' | 'RED' {
    57‚Üí  if (requiresReplan) {
    58‚Üí    return 'RED';
    59‚Üí  }
    60‚Üí
    61‚Üí  // Approaching threshold (>70% of threshold)
    62‚Üí  const mileageApproaching = Math.abs(mileageDriftPct) > thresholds.mileageDriftPct * 0.7;
    63‚Üí  const scheduleApproaching = maxPmDriftDays > thresholds.scheduleDriftDays * 0.7;
    64‚Üí
    65‚Üí  if (mileageApproaching || scheduleApproaching) {
    66‚Üí    return 'YELLOW';
    67‚Üí  }
    68‚Üí
    69‚Üí  return 'GREEN';
    70‚Üí}
