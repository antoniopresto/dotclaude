   314→#### Reference: @belt/tools Patterns
   315→```typescript
   316→// Schema validation (replaces Zod)
   317→import { createSchema, createType, InferType } from '@belt/tools';
   318→
   319→const TrainModel = createSchema({
   320→  id: 'string',
   321→  trainNumber: 'string',
   322→  mileageKm: 'number',
   323→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   324→  tags: 'string[]',
   325→  metadata: 'object?',
   326→});
   327→type Train = InferType<typeof TrainModel>;
   328→const train = TrainModel.parse(data);
   329→
   330→// State machine (replaces React hooks + zustand)
   331→import { createSlice, createMachine } from '@belt/tools';
   332→
   333→const fleetSlice = createSlice({
   334→  id: 'fleet',
   335→  initial: { trains: [], selected: null },
   336→  cache: { policy: 'session' },
   337→  reducers: {
   338→    setTrains: (state, trains) => { state.trains = trains; },
   339→    selectTrain: (state, id) => { state.selected = id; },
   340→  },
   341→});
   342→
   343→const machine = createMachine({
   344→  id: 'mms',
   345→  slices: { fleet: fleetSlice },
   346→  trackMetadata: false, // Critical for 60fps
   347→});
   348→
   349→// In React component (NO useState/useEffect!)
   350→const { state, actions } = machine.useMachine('fleet');
   351→```
