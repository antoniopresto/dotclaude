    68→/**
    69→ * Create a success response
    70→ */
    71→export function success<T>(data: T, meta?: ResponseMeta): SuccessResponse<T> {
    72→  return {
    73→    success: true,
    74→    data,
    75→    ...(meta && { meta }),
    76→  };
    77→}
    78→
    79→/**
    80→ * Create a paginated success response
    81→ */
    82→export function paginatedSuccess<T>(
    83→  data: T[],
    84→  pagination: {
    85→    page: number;
    86→    pageSize: number;
    87→    totalItems: number;
    88→  },
    89→  timing?: TimingMeta
    90→): SuccessResponse<T[]> {
    91→  const totalPages = Math.ceil(pagination.totalItems / pagination.pageSize);
    92→  return {
    93→    success: true,
    94→    data,
    95→    meta: {
    96→      pagination: {
    97→        page: pagination.page,
    98→        pageSize: pagination.pageSize,
    99→        totalItems: pagination.totalItems,
   100→        totalPages,
   101→        hasNext: pagination.page < totalPages,
   102→        hasPrev: pagination.page > 1,
   103→      },
   104→      ...(timing && { timing }),
   105→    },
   106→  };
   107→}

   109→/**
   110→ * Create an error response
   111→ */
   112→export function error(
   113→  code: string,
   114→  message: string,
   115→  requestId: string,
   116→  details?: unknown
   117→): ErrorResponse {
   118→  return {
   119→    success: false,
   120→    error: {
   121→      code,
   122→      message,
   123→      requestId,
   124→      ...(details !== undefined && { details }),

   126→  };

   169→/**
   170→ * Parsed pagination parameters with defaults applied
   171→ */
   172→export interface ParsedPagination {
   173→  page: number;
   174→  pageSize: number;
   175→  offset: number;
   176→  sort?: { field: string; direction: 'asc' | 'desc' };
   177→}
   178→
   179→/**
   180→ * Parse pagination query parameters with defaults
   181→ */
   182→export function parsePaginationQuery(params: ListQueryParams): ParsedPagination {
   183→  const page = Math.max(1, Number(params.page) || 1);
   184→  const pageSize = Math.min(200, Math.max(1, Number(params.pageSize) || 50));
   185→  const offset = (page - 1) * pageSize;
   186→
   187→  let sort: { field: string; direction: 'asc' | 'desc' } | undefined;
   188→  if (params.sort) {
   189→    const [field, dir] = params.sort.split(':');
   190→    sort = {
   191→      field,
   192→      direction: dir === 'desc' ? 'desc' : 'asc',
   193→    };
   194→  }
   195→
   196→  return { page, pageSize, offset, sort };
   197→}
