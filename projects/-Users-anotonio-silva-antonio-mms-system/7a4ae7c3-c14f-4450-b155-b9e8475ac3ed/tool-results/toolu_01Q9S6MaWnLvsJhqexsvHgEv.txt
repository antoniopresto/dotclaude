     1→/**
     2→ * MMS Material Service
     3→ *
     4→ * Manages inventory positions and material reservations for PM scheduling.
     5→ * Materials are a constraint for PM scheduling - PMs cannot proceed without
     6→ * required blocking parts.
     7→ *
     8→ * Key responsibilities:
     9→ * - Query inventory positions with filtering (depot, part number, low stock)
    10→ * - Query material reservations with filtering (PM, status)
    11→ * - Fulfill material reservations (mark as delivered)
    12→ * - Support soft/hard reservation lifecycle
    13→ *
    14→ * Integration:
    15→ * - Inventory snapshots come from Maximo/ERP daily
    16→ * - Reservations are created when PMs are scheduled
    17→ * - Soft reservations auto-expire if PM moves
    18→ *
    19→ * Constraint enforcement:
    20→ * - BLOCKING parts with SOFT reservation: Warning, PM can proceed
    21→ * - BLOCKING parts without reservation: PM cannot be scheduled
    22→ *
    23→ * @see specs/004-domain-services.md
    24→ * @see specs/005-api.md Section 5.11
    25→ */

   175→  /**
   176→   * List inventory positions with filtering
   177→   *
   178→   * Supports filtering by:
   179→   * - depotId: Filter by specific depot
   180→   * - partNumber: Filter by exact or partial part number
   181→   * - lowStock: Only show items where available <= reorder point
   182→   *
   183→   * @param filters - Query filters
   184→   * @param pagination - Pagination options
   185→   * @returns Paginated list of inventory positions with depot info
   186→   */
   187→  async listInventory(
   188→    filters: InventoryFilters = {},
   189→    pagination: PaginationOptions = { page: 1, pageSize: 50 }
   190→  ): Promise<PaginatedResult<InventoryWithDepot>> {
   191→    return this.traced('listInventory', async (span) => {
   192→      span.setAttribute('filters.depotId', filters.depotId || '');
   193→      span.setAttribute('filters.partNumber', filters.partNumber || '');
   194→      span.setAttribute('filters.lowStock', filters.lowStock || false);
   195→      span.setAttribute('pagination.page', pagination.page);
   196→      span.setAttribute('pagination.pageSize', pagination.pageSize);
   197→
   198→      const offset = (pagination.page - 1) * pagination.pageSize;
   199→
   200→      // Build dynamic where conditions
   201→      const conditions: SQL<unknown>[] = [];
   202→
   203→      if (filters.depotId) {
   204→        conditions.push(eq(inventoryPositions.depotId, filters.depotId));
   205→      }
   206→
   207→      if (filters.partNumber) {
   208→        conditions.push(
   209→          sql`${inventoryPositions.partNumber} ILIKE ${`%${filters.partNumber}%`}`
   210→        );
   211→      }
   212→
   213→      // Execute query with joins
   214→      const query = this.db
   215→        .select({
   216→          id: inventoryPositions.id,
   217→          depotId: inventoryPositions.depotId,
   218→          depotCode: depots.code,
   219→          depotName: depots.name,
   220→          partNumber: inventoryPositions.partNumber,
   221→          partDescription: inventoryPositions.partDescription,
   222→          quantityOnHand: inventoryPositions.quantityOnHand,
   223→          quantityReserved: inventoryPositions.quantityReserved,
   224→          quantityOnOrder: inventoryPositions.quantityOnOrder,
   225→          reorderPoint: inventoryPositions.reorderPoint,
   226→          reorderQuantity: inventoryPositions.reorderQuantity,
   227→          nextDeliveryDate: inventoryPositions.nextDeliveryDate,
   228→          nextDeliveryQuantity: inventoryPositions.nextDeliveryQuantity,
   229→          unitCost: inventoryPositions.unitCost,
   230→          currency: inventoryPositions.currency,
   231→          lastSyncAt: inventoryPositions.lastSyncAt,
   232→          sourceSystem: inventoryPositions.sourceSystem,
   233→        })
   234→        .from(inventoryPositions)
   235→        .innerJoin(depots, eq(inventoryPositions.depotId, depots.id))
   236→        .where(conditions.length > 0 ? and(...conditions) : undefined);
   237→
   238→      // For lowStock filter, we need to do it post-query since it's a computed field
   239→      const allResults = await query;
   240→
   241→      // Calculate available quantity and filter by low stock if needed
   242→      let filteredResults = allResults.map((row) => ({
   243→        ...row,
   244→        quantityAvailable: row.quantityOnHand - row.quantityReserved,
   245→        isLowStock:
   246→          row.reorderPoint !== null &&
   247→          row.quantityOnHand - row.quantityReserved <= row.reorderPoint,
   248→      }));
   249→
   250→      if (filters.lowStock) {
   251→        filteredResults = filteredResults.filter((r) => r.isLowStock);
   252→      }
   253→
   254→      // Apply pagination
   255→      const total = filteredResults.length;
   256→      const paginatedItems = filteredResults.slice(offset, offset + pagination.pageSize);
   257→
   258→      // Format response
   259→      const items: InventoryWithDepot[] = paginatedItems.map((row) => ({
   260→        id: row.id,
   261→        depotId: row.depotId,
   262→        depotCode: row.depotCode,
   263→        depotName: row.depotName,
   264→        partNumber: row.partNumber,
   265→        partDescription: row.partDescription,
   266→        quantityOnHand: row.quantityOnHand,
   267→        quantityReserved: row.quantityReserved,
   268→        quantityAvailable: row.quantityAvailable,
   269→        quantityOnOrder: row.quantityOnOrder,
   270→        reorderPoint: row.reorderPoint,
   271→        reorderQuantity: row.reorderQuantity,
   272→        nextDeliveryDate: row.nextDeliveryDate,
   273→        nextDeliveryQuantity: row.nextDeliveryQuantity,
   274→        unitCost: row.unitCost ? Number(row.unitCost) : null,
   275→        currency: row.currency,
   276→        lastSyncAt: row.lastSyncAt?.toISOString() || null,
   277→        sourceSystem: row.sourceSystem,
   278→        isLowStock: row.isLowStock,
   279→      }));
   280→
   281→      span.setAttribute('result.count', items.length);
   282→      span.setAttribute('result.total', total);
   283→
   284→      return {
   285→        items,
   286→        total,
   287→        page: pagination.page,
   288→        pageSize: pagination.pageSize,
   289→        totalPages: Math.ceil(total / pagination.pageSize),
   290→      };
   291→    });
   292→  }

   302→    return this.traced('getInventoryById', async (span) => {
   303→      span.setAttribute('inventory.id', id);
   304→
   305→      const result = await this.db
   306→        .select({
   307→          id: inventoryPositions.id,
   308→          depotId: inventoryPositions.depotId,
   309→          depotCode: depots.code,

   311→          partNumber: inventoryPositions.partNumber,

   339→      return {

   341→        depotId: row.depotId,

   351→        reorderQuantity: row.reorderQuantity,
   352→        nextDeliveryDate: row.nextDeliveryDate,
   353→        nextDeliveryQuantity: row.nextDeliveryQuantity,
   354→        unitCost: row.unitCost ? Number(row.unitCost) : null,
   355→        currency: row.currency,
   356→        lastSyncAt: row.lastSyncAt?.toISOString() || null,

   363→  /**
   364→   * List material reservations with filtering
   365→   *
   366→   * Supports filtering by:
   367→   * - pmScheduleId: Filter by specific PM
   368→   * - status: PENDING, FULFILLED, or EXPIRED
   369→   * - depotId: Filter by depot
   370→   * - partNumber: Filter by part number
   371→   *
   372→   * @param filters - Query filters
   373→   * @param pagination - Pagination options
   374→   * @returns Paginated list of reservations with PM and depot details
   375→   */
   376→  async listReservations(
   377→    filters: ReservationFilters = {},
   378→    pagination: PaginationOptions = { page: 1, pageSize: 50 }
   379→  ): Promise<PaginatedResult<ReservationWithDetails>> {
   380→    return this.traced('listReservations', async (span) => {
   381→      span.setAttribute('filters.pmScheduleId', filters.pmScheduleId || '');
   382→      span.setAttribute('filters.status', filters.status || '');
   383→      span.setAttribute('pagination.page', pagination.page);
   384→      span.setAttribute('pagination.pageSize', pagination.pageSize);
   385→
   386→      const offset = (pagination.page - 1) * pagination.pageSize;
   387→      const now = new Date();
   388→
   389→      // Build dynamic where conditions
   390→      const conditions: SQL<unknown>[] = [];
   391→
   392→      if (filters.pmScheduleId) {
   393→        conditions.push(eq(materialReservations.pmScheduleId, filters.pmScheduleId));
   394→      }
   395→
   396→      if (filters.partNumber) {
   397→        conditions.push(eq(materialReservations.partNumber, filters.partNumber));
   398→      }
   399→
   400→      // Execute query with joins
   401→      const results = await this.db
   402→        .select({
   403→          id: materialReservations.id,
   404→          pmScheduleId: materialReservations.pmScheduleId,
   405→          inventoryPositionId: materialReservations.inventoryPositionId,
   406→          partNumber: materialReservations.partNumber,
   407→          quantityReserved: materialReservations.quantityReserved,
   408→          reservationType: materialReservations.reservationType,
   409→          reservedAt: materialReservations.reservedAt,
   410→          reservedBy: materialReservations.reservedBy,
   411→          expiresAt: materialReservations.expiresAt,
   412→          fulfilledAt: materialReservations.fulfilledAt,
   413→          fulfilledQuantity: materialReservations.fulfilledQuantity,
   414→          pmTargetDate: pmSchedule.targetDate,
   415→          trainId: pmSchedule.trainId,
   416→          trainNumber: trains.trainNumber,
   417→          pmTypeId: pmSchedule.pmTypeId,
   418→          pmCode: pmTypes.pmCode,
   419→          depotId: pmSchedule.assignedDepotId,
   420→          depotCode: depots.code,
   421→          depotName: depots.name,
   422→          reservedByFirstName: users.firstName,
   423→          reservedByLastName: users.lastName,
   424→        })
   425→        .from(materialReservations)
   426→        .innerJoin(pmSchedule, eq(materialReservations.pmScheduleId, pmSchedule.id))
   427→        .innerJoin(pmTypes, eq(pmSchedule.pmTypeId, pmTypes.id))
   428→        .innerJoin(trains, eq(pmSchedule.trainId, trains.id))
   429→        .leftJoin(depots, eq(pmSchedule.assignedDepotId, depots.id))
   430→        .leftJoin(users, eq(materialReservations.reservedBy, users.id))
   431→        .where(conditions.length > 0 ? and(...conditions) : undefined)
   432→        .orderBy(desc(materialReservations.reservedAt));
   433→
   434→      // Calculate status and filter
   435→      let processedResults = results.map((row) => {
   436→        let status: ReservationStatus;
   437→        if (row.fulfilledAt) {
   438→          status = 'FULFILLED';
   439→        } else if (row.expiresAt && new Date(row.expiresAt) < now) {
   440→          status = 'EXPIRED';
   441→        } else {
   442→          status = 'PENDING';
   443→        }
   444→
   445→        return {
   446→          ...row,
   447→          status,
   448→        };
   449→      });
   450→
   451→      // Filter by status if specified
   452→      if (filters.status) {
   453→        processedResults = processedResults.filter((r) => r.status === filters.status);
   454→      }
   455→
   456→      // Filter by depot if specified
   457→      if (filters.depotId) {
   458→        processedResults = processedResults.filter((r) => r.depotId === filters.depotId);
   459→      }
   460→
   461→      // Apply pagination
   462→      const total = processedResults.length;
   463→      const paginatedItems = processedResults.slice(offset, offset + pagination.pageSize);
   464→
   465→      // Format response
   466→      const items: ReservationWithDetails[] = paginatedItems.map((row) => ({
   467→        id: row.id,
   468→        pmScheduleId: row.pmScheduleId,
   469→        inventoryPositionId: row.inventoryPositionId,
   470→        partNumber: row.partNumber,
   471→        quantityReserved: row.quantityReserved,
   472→        reservationType: row.reservationType as 'SOFT' | 'HARD',
   473→        reservedAt: row.reservedAt?.toISOString() || '',
   474→        reservedBy: row.reservedBy,
   475→        reservedByName:
   476→          row.reservedByFirstName && row.reservedByLastName
   477→            ? `${row.reservedByFirstName} ${row.reservedByLastName}`
   478→            : null,
   479→        expiresAt: row.expiresAt?.toISOString() || null,
   480→        fulfilledAt: row.fulfilledAt?.toISOString() || null,
   481→        fulfilledQuantity: row.fulfilledQuantity,
   482→        status: row.status,
   483→        pm: {
   484→          id: row.pmScheduleId,
   485→          trainNumber: row.trainNumber,
   486→          pmCode: row.pmCode,
   487→          targetDate: row.pmTargetDate,
   488→        },
   489→        depot: {
   490→          id: row.depotId || '',
   491→          code: row.depotCode || '',
   492→          name: row.depotName || '',
   493→        },
   494→      }));
   495→
   496→      span.setAttribute('result.count', items.length);
   497→      span.setAttribute('result.total', total);
   498→
   499→      return {
   500→        items,
   501→        total,
   502→        page: pagination.page,
   503→        pageSize: pagination.pageSize,
   504→        totalPages: Math.ceil(total / pagination.pageSize),
   505→      };
   506→    });

   607→   * quantity than reserved (partial fulfillment).
   608→   *
   609→   * @param request - Fulfillment request with reservation ID
   610→   * @returns Updated reservation details
   611→   * @throws NotFoundError if reservation not found
   612→   * @throws ValidationError if already fulfilled or expired
   613→   */
   614→  async fulfillReservation(
   615→    request: FulfillReservationRequest
   616→  ): Promise<FulfillReservationResult> {

   624→        .select()

   626→        .where(eq(materialReservations.id, request.reservationId))

   631→      }

   633→      const reservation = existing[0];

   635→      // Check if already fulfilled
   636→      if (reservation.fulfilledAt) {

   640→        );
   641→      }
   642→
   643→      // Check if expired
   644→      if (reservation.expiresAt && new Date(reservation.expiresAt) < now) {
   645→        throw new ValidationError(
   646→          'Cannot fulfill an expired reservation',
   647→          { context: { code: 'RESERVATION_EXPIRED', reservationId: request.reservationId } }
   648→        );
   649→      }
   650→
   651→      // Determine fulfilled quantity
   652→      const fulfilledQuantity =
   653→        request.fulfilledQuantity ?? reservation.quantityReserved;
   654→

   656→      await this.db
   657→        .update(materialReservations)
   658→        .set({
   659→          fulfilledAt: now,
   660→          fulfilledQuantity,
   661→        })
   662→        .where(eq(materialReservations.id, request.reservationId));

   665→        {
   666→          reservationId: request.reservationId,

   676→        partNumber: reservation.partNumber,

   713→
   714→      // Aggregate by depot

   717→        { depotCode: string; itemCount: number; lowStockCount: number }

   721→
   722→      for (const item of items) {

   724→        const isLowStock =

   732→        if (existing) {

   734→          if (isLowStock) existing.lowStockCount++;

   736→          depotMap.set(item.depotId, {
   737→            depotCode: item.depotCode,
   738→            itemCount: 1,
   739→            lowStockCount: isLowStock ? 1 : 0,
   740→          });
   741→        }
   742→      }
   743→
   744→      const byDepot = Array.from(depotMap.entries()).map(([depotId, stats]) => ({
   745→        depotId,
   746→        ...stats,

   748→
   749→      span.setAttribute('result.totalItems', items.length);
   750→      span.setAttribute('result.lowStockCount', totalLowStock);

   754→        lowStockCount: totalLowStock,
