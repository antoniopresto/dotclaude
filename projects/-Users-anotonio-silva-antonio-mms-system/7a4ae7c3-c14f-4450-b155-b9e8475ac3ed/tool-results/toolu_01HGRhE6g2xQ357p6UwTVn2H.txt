     1→/**
     2→ * MMS Maximo Sync Service
     3→ *
     4→ * Implements bidirectional synchronization between MMS and IBM Maximo CMMS.
     5→ * Uses the shadow record pattern to enable offline operation and conflict resolution.
     6→ *
     7→ * Key responsibilities:
     8→ * - Create and manage shadow records for PM schedules
     9→ * - Push MMS changes to Maximo (create/update work orders)
    10→ * - Pull Maximo status changes to MMS
    11→ * - Detect and resolve conflicts
    12→ * - Validate edit operations against Maximo status locks
    13→ *
    14→ * Integration flow:
    15→ * 1. When a PM is scheduled in MMS, a shadow record is created (status: PENDING)
    16→ * 2. Sync job pushes shadow records to Maximo (creates work orders)
    17→ * 3. Maximo status changes (via polling or webhook) update shadow records
    18→ * 4. INPRG/COMP/CLOSE/CAN status locks PM in MMS (read-only)
    19→ *
    20→ * @see specs/007-integration.md Section 7.4
    21→ */

   209→// =============================================================================
   210→// MaximoSyncService Implementation
   211→// =============================================================================
   212→
   213→/**
   214→ * Maximo Sync Service for bidirectional PM synchronization
   215→ *
   216→ * Implements the shadow record pattern from spec 007.4.2:
   217→ * - MMS is source of truth for scheduling
   218→ * - Shadow records track sync state with Maximo
   219→ * - Status-based locking prevents conflicts
   220→ *
   221→ * @example
   222→ * ```typescript
   223→ * const maximoSync = getMaximoSyncService();
   224→ *
   225→ * // Validate before editing a PM
   226→ * const validation = await maximoSync.validateEdit(pmId);
   227→ * if (!validation.valid) {
   228→ *   throw new MaximoLockError(validation.message);
   229→ * }
   230→ *
   231→ * // Create shadow record for new PM
   232→ * await maximoSync.createShadowRecord(pmId);
   233→ *
   234→ * // Push changes to Maximo
   235→ * const result = await maximoSync.pushToMaximo(pmId);
   236→ *
   237→ * // Pull status updates from Maximo
   238→ * await maximoSync.pullFromMaximo();
   239→ * ```
   240→ */
   241→export class MaximoSyncService extends BaseService {
   242→  private readonly circuitBreaker: CircuitBreaker;
   243→  private lastPullTimestamp: Date | null = null;
   244→  private maximoClient: MaximoClient;
   245→  private clientInitialized = false;
   246→
   247→  constructor() {
   248→    super({ name: 'maximo-sync-service' });
   249→
   250→    // Initialize Maximo client
   251→    this.maximoClient = getMaximoClient();
   252→
   253→    // Circuit breaker for Maximo API calls
   254→    this.circuitBreaker = new CircuitBreaker(
   255→      {
   256→        failureThreshold: 5,
   257→        successThreshold: 2,
   258→        resetTimeoutMs: 60000, // 60 seconds in open state
   259→        isFailure: (error) => {
   260→          // Only count network/availability errors as circuit breaker failures
   261→          if (error instanceof MaximoConnectionError) return true;
   262→          if (error instanceof Error) {
   263→            const msg = error.message.toLowerCase();
   264→            return (
   265→              msg.includes('econnrefused') ||
   266→              msg.includes('etimedout') ||
   267→              msg.includes('econnreset') ||
   268→              msg.includes('503') ||
   269→              msg.includes('504')
   270→            );
   271→          }
   272→          return false;
   273→        },
   274→      },
   275→      'maximo-sync'
   276→    );
   277→  }

   289→  // ===========================================================================
   290→  // Edit Validation
   291→  // ===========================================================================
   292→
   293→  /**
   294→   * Validate PM edit request against Maximo status
   295→   *
   296→   * MUST be called before any PM schedule modification.
   297→   * Prevents race conditions where a planner reschedules a PM that
   298→   * technicians are actively working on.
   299→   *
   300→   * @see specs/007-integration.md Section 7.4.6
   301→   */
   302→  async validateEdit(pmScheduleId: string): Promise<EditValidationResult> {
   303→    return this.traced('validateEdit', async (span) => {
   304→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   305→
   306→      // Get PM with shadow record
   307→      const pm = await this.getPmWithShadow(pmScheduleId);
   308→
   309→      if (!pm) {
   310→        return {
   311→          valid: false,
   312→          reason: 'NOT_FOUND',
   313→          message: 'PM schedule not found',
   314→        };
   315→      }
   316→
   317→      // Check 1: Maximo status lock
   318→      if (pm.shadowRecord?.maximoData) {
   319→        const maximoData = pm.shadowRecord.maximoData as { status?: string };
   320→        const maximoStatus = maximoData.status as MaximoWoStatus | undefined;
   321→
   322→        if (maximoStatus && LOCKED_MAXIMO_STATUSES.includes(maximoStatus)) {
   323→          span.setAttribute('maximo.status', maximoStatus);
   324→          span.setAttribute('maximo.locked', true);
   325→
   326→          return {
   327→            valid: false,
   328→            reason: 'MAXIMO_STATUS_LOCK',
   329→            message: `Cannot modify: Work order is ${MAXIMO_STATUS_LABELS[maximoStatus]} in Maximo`,
   330→            maximoStatus,
   331→            maximoWoNum: pm.shadowRecord.maximoWoNum ?? undefined,
   332→            suggestion:
   333→              maximoStatus === 'INPRG'
   334→                ? 'Wait for technicians to complete work, or contact supervisor to cancel in Maximo first'
   335→                : 'This PM has been completed/cancelled. Create a new PM if rescheduling is needed.',
   336→          };
   337→        }
   338→      }
   339→
   340→      // Check 2: Pending sync conflict
   341→      if (pm.shadowRecord?.syncStatus === 'CONFLICT') {
   342→        span.setAttribute('sync.conflict', true);
   343→
   344→        return {
   345→          valid: false,
   346→          reason: 'SYNC_CONFLICT',
   347→          message: 'Unresolved sync conflict exists. Resolve before editing.',
   348→          suggestion: 'Go to Integration > Conflicts to resolve',
   349→        };
   350→      }
   351→
   352→      // Check 3: Baseline lock (if PM is locked)
   353→      if (pm.pmSchedule.isLocked) {
   354→        span.setAttribute('baseline.locked', true);
   355→
   356→        return {
   357→          valid: false,
   358→          reason: 'BASELINE_LOCK',
   359→          message: 'PM is locked in published baseline',
   360→          suggestion: 'Unlock in baseline editor or create scenario to test changes',
   361→        };
   362→      }
   363→
   364→      span.setAttribute('edit.valid', true);
   365→      return { valid: true };
   366→    });
   367→  }

   381→  /**
   382→   * Create a shadow record for a PM schedule
   383→   *
   384→   * Called when a new PM is scheduled or when sync is first enabled.
   385→   */
   386→  async createShadowRecord(pmScheduleId: string): Promise<MaximoShadowRecord> {
   387→    return this.traced('createShadowRecord', async (span) => {
   388→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   389→
   390→      // Verify PM exists
   391→      const pm = await this.db.query.pmSchedule.findFirst({
   392→        where: eq(pmSchedule.id, pmScheduleId),
   393→        with: {
   394→          train: true,
   395→          pmType: true,
   396→        },
   397→      });
   398→
   399→      if (!pm) {
   400→        throw new NotFoundError(`PM schedule not found: ${pmScheduleId}`, {
   401→          entityType: 'PmSchedule',
   402→          entityId: pmScheduleId,
   403→        });
   404→      }
   405→
   406→      // Check if shadow record already exists
   407→      const existing = await this.db.query.maximoShadowRecords.findFirst({
   408→        where: eq(maximoShadowRecords.pmScheduleId, pmScheduleId),
   409→      });
   410→
   411→      if (existing) {
   412→        this.log.info(
   413→          { pmScheduleId, shadowRecordId: existing.id },
   414→          'Shadow record already exists'
   415→        );
   416→        return existing;
   417→      }
   418→
   419→      // Create MMS data snapshot
   420→      const mmsDataSnapshot = {
   421→        trainId: pm.trainId,
   422→        pmTypeCode: pm.pmType?.pmCode ?? 'UNKNOWN',
   423→        targetDate: pm.targetDate,
   424→        status: pm.status ?? 'SCHEDULED',
   425→        assignedDepotId: pm.assignedDepotId ?? undefined,
   426→        assignedHours: pm.assignedHours ?? undefined,
   427→      };
   428→
   429→      // Insert shadow record
   430→      const [shadowRecord] = await this.db
   431→        .insert(maximoShadowRecords)
   432→        .values({
   433→          pmScheduleId,
   434→          syncStatus: 'PENDING',
   435→          mmsDataSnapshot,
   436→          syncAttempts: 0,
   437→          nextSyncAt: new Date(), // Sync immediately
   438→        })
   439→        .returning();
   440→
   441→      span.setAttribute('shadow.record.id', shadowRecord.id);
   442→      this.log.info(
   443→        { pmScheduleId, shadowRecordId: shadowRecord.id },
   444→        'Created shadow record for PM schedule'
   445→      );
   446→
   447→      return shadowRecord;
   448→    });
   449→  }

   501→  /**
   502→   * List shadow records with filters
   503→   */
   504→  async listShadowRecords(
   505→    filters: ShadowRecordFilters = {},
   506→    pagination: PaginationOptions = {}
   507→  ): Promise<PaginatedShadowRecords> {
   508→    return this.traced('listShadowRecords', async (span) => {
   509→      const { page = 1, pageSize = 50 } = pagination;
   510→      const offset = (page - 1) * pageSize;
   511→
   512→      // Build where conditions
   513→      const conditions: SQL<unknown>[] = [];
   514→
   515→      if (filters.syncStatus) {
   516→        const statuses = Array.isArray(filters.syncStatus)
   517→          ? filters.syncStatus
   518→          : [filters.syncStatus];
   519→        conditions.push(inArray(maximoShadowRecords.syncStatus, statuses));
   520→      }
   521→
   522→      if (filters.pmScheduleId) {
   523→        conditions.push(eq(maximoShadowRecords.pmScheduleId, filters.pmScheduleId));
   524→      }
   525→
   526→      if (filters.maximoWoNum) {
   527→        conditions.push(eq(maximoShadowRecords.maximoWoNum, filters.maximoWoNum));
   528→      }
   529→
   530→      if (filters.hasConflict) {
   531→        conditions.push(eq(maximoShadowRecords.syncStatus, 'CONFLICT'));
   532→      }
   533→
   534→      if (filters.needsSync) {
   535→        const needsSyncCondition = or(
   536→          eq(maximoShadowRecords.syncStatus, 'PENDING'),
   537→          and(
   538→            eq(maximoShadowRecords.syncStatus, 'ERROR'),
   539→            lt(maximoShadowRecords.syncAttempts, 5)
   540→          )
   541→        );
   542→        if (needsSyncCondition) {
   543→          conditions.push(needsSyncCondition);
   544→        }
   545→      }
   546→
   547→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   548→
   549→      // Get total count
   550→      const [{ value: total }] = await this.db
   551→        .select({ value: count() })
   552→        .from(maximoShadowRecords)
   553→        .where(whereClause);
   554→
   555→      // Get records
   556→      const records = await this.db.query.maximoShadowRecords.findMany({
   557→        where: whereClause,
   558→        orderBy: [desc(maximoShadowRecords.updatedAt)],
   559→        limit: pageSize,
   560→        offset,
   561→        with: {
   562→          pmScheduleEntry: {
   563→            with: {
   564→              train: true,
   565→              pmType: true,
   566→              assignedDepot: true,
   567→            },
   568→          },
   569→        },
   570→      });
   571→
   572→      // Transform to ShadowRecordWithPm
   573→      const data: ShadowRecordWithPm[] = records.map((r) => ({
   574→        shadowRecord: r,
   575→        pmSchedule: {
   576→          ...r.pmScheduleEntry,
   577→          train: r.pmScheduleEntry.train
   578→            ? {
   579→                id: r.pmScheduleEntry.train.id,
   580→                trainNumber: r.pmScheduleEntry.train.trainNumber,
   581→              }
   582→            : undefined,
   583→          pmType: r.pmScheduleEntry.pmType
   584→            ? {
   585→                id: r.pmScheduleEntry.pmType.id,
   586→                pmCode: r.pmScheduleEntry.pmType.pmCode,
   587→                pmName: r.pmScheduleEntry.pmType.pmName,
   588→              }
   589→            : undefined,
   590→          depot: r.pmScheduleEntry.assignedDepot
   591→            ? {
   592→                id: r.pmScheduleEntry.assignedDepot.id,
   593→                code: r.pmScheduleEntry.assignedDepot.code,
   594→              }
   595→            : undefined,
   596→        },
   597→      }));
   598→
   599→      span.setAttribute('query.total', total);
   600→      span.setAttribute('query.returned', data.length);
   601→
   602→      return {
   603→        data,
   604→        total,
   605→        page,
   606→        pageSize,
   607→        totalPages: Math.ceil(total / pageSize),
   608→      };
   609→    });
   610→  }

   612→  // ===========================================================================
   613→  // Push Sync (MMS → Maximo)
   614→  // ===========================================================================
   615→
   616→  /**
   617→   * Push MMS changes to Maximo for a specific PM
   618→   *
   619→   * Creates or updates the Maximo work order based on shadow record state.
   620→   *
   621→   * @see specs/007-integration.md Section 7.4.7
   622→   */
   623→  async pushToMaximo(pmScheduleId: string): Promise<PushResult> {
   624→    return this.traced('pushToMaximo', async (span) => {
   625→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   626→
   627→      const pmWithShadow = await this.getPmWithShadow(pmScheduleId);
   628→
   629→      if (!pmWithShadow) {
   630→        return { action: 'ERROR', error: 'PM schedule not found' };
   631→      }
   632→
   633→      const { pmSchedule: pm, shadowRecord } = pmWithShadow;
   634→
   635→      // Create shadow record if doesn't exist
   636→      let shadow = shadowRecord;
   637→      if (!shadow?.id) {
   638→        shadow = await this.createShadowRecord(pmScheduleId);
   639→      }
   640→
   641→      // Check circuit breaker state
   642→      if (this.circuitBreaker.getState() === 'OPEN') {
   643→        this.log.warn({ pmScheduleId }, 'Circuit breaker is open, skipping push');
   644→        return { action: 'SKIPPED', reason: 'Circuit breaker open' };
   645→      }
   646→
   647→      const syncAttempts = shadow.syncAttempts ?? 0;
   648→
   649→      try {
   650→        // No Maximo WO yet - create new
   651→        if (!shadow.maximoWoNum) {
   652→          // Use circuit breaker for Maximo API call
   653→          const woNum = await this.circuitBreaker.execute(() =>
   654→            this.createMaximoWorkOrder(pm)
   655→          );
   656→
   657→          await this.updateShadowRecord(shadow.id, {
   658→            maximoWoNum: woNum,
   659→            syncStatus: 'SYNCED',
   660→            lastSyncAt: new Date(),
   661→            lastSyncError: null,
   662→            syncAttempts: syncAttempts + 1,
   663→            mmsDataSnapshot: this.createMmsSnapshot(pm),
   664→          });
   665→
   666→          await this.recordSyncHistory(shadow.id, 'PUSH', {
   667→            previousStatus: shadow.syncStatus,
   668→            newStatus: 'SYNCED',
   669→            success: true,
   670→          });
   671→
   672→          span.setAttribute('maximo.wo.created', woNum);
   673→
   674→          return { action: 'CREATED', woNum };
   675→        }
   676→
   677→        // Check if Maximo WO is in locked state
   678→        const maximoWo = await this.circuitBreaker.execute(() =>
   679→          this.fetchMaximoWorkOrder(shadow.maximoWoNum!)
   680→        );
   681→
   682→        if (maximoWo && LOCKED_MAXIMO_STATUSES.includes(maximoWo.status)) {
   683→          return {
   684→            action: 'SKIPPED',
   685→            reason: `Work order is ${MAXIMO_STATUS_LABELS[maximoWo.status]}`,
   686→            woNum: shadow.maximoWoNum,
   687→          };
   688→        }
   689→
   690→        // Check for conflicts
   691→        if (maximoWo?.changedate && shadow.lastSyncAt) {
   692→          const maximoChangeDate = new Date(maximoWo.changedate);
   693→          if (maximoChangeDate > shadow.lastSyncAt) {
   694→            await this.updateShadowRecord(shadow.id, {
   695→              syncStatus: 'CONFLICT',
   696→              maximoData: {
   697→                status: maximoWo.status,
   698→                schedStart: maximoWo.schedstart,
   699→                schedFinish: maximoWo.schedfinish,
   700→                actualStart: maximoWo.actstart,
   701→                actualFinish: maximoWo.actfinish,
   702→                actualLabor: maximoWo.actlabhrs,
   703→                changeDate: maximoWo.changedate,
   704→              },
   705→            });
   706→
   707→            await this.recordSyncHistory(shadow.id, 'PUSH', {
   708→              previousStatus: shadow.syncStatus,
   709→              newStatus: 'CONFLICT',
   710→              success: false,
   711→              errorMessage: 'Conflict detected - Maximo changed after last sync',
   712→            });
   713→
   714→            return { action: 'CONFLICT', woNum: shadow.maximoWoNum };
   715→          }
   716→        }
   717→
   718→        // Update existing work order
   719→        await this.circuitBreaker.execute(() =>
   720→          this.updateMaximoWorkOrder(shadow.maximoWoNum!, pm)
   721→        );
   722→
   723→        await this.updateShadowRecord(shadow.id, {
   724→          syncStatus: 'SYNCED',
   725→          lastSyncAt: new Date(),
   726→          lastSyncError: null,
   727→          syncAttempts: syncAttempts + 1,
   728→          mmsDataSnapshot: this.createMmsSnapshot(pm),
   729→          pendingChanges: null,
   730→        });
   731→
   732→        await this.recordSyncHistory(shadow.id, 'PUSH', {
   733→          previousStatus: shadow.syncStatus,
   734→          newStatus: 'SYNCED',
   735→          success: true,
   736→        });
   737→
   738→        span.setAttribute('maximo.wo.updated', shadow.maximoWoNum);
   739→
   740→        return { action: 'UPDATED', woNum: shadow.maximoWoNum };
   741→      } catch (error) {
   742→        const errorMessage = error instanceof Error ? error.message : String(error);
   743→
   744→        await this.updateShadowRecord(shadow.id, {
   745→          syncStatus: 'ERROR',
   746→          lastSyncError: errorMessage,
   747→          syncAttempts: syncAttempts + 1,
   748→          nextSyncAt: this.calculateNextRetry(syncAttempts + 1),
   749→        });
   750→
   751→        await this.recordSyncHistory(shadow.id, 'PUSH', {
   752→          previousStatus: shadow.syncStatus,
   753→          newStatus: 'ERROR',
   754→          success: false,
   755→          errorMessage,
   756→        });
   757→
   758→        span.setAttribute('error', true);
   759→        span.setAttribute('error.message', errorMessage);
   760→
   761→        return { action: 'ERROR', error: errorMessage };
   762→      }
   763→    });
   764→  }

   766→  /**
   767→   * Push all pending changes to Maximo (batch sync)
   768→   */
   769→  async pushAllPending(): Promise<{ pushed: number; errors: number }> {
   770→    return this.traced('pushAllPending', async (span) => {
   771→      // Get all pending shadow records
   772→      const pending = await this.db.query.maximoShadowRecords.findMany({
   773→        where: or(
   774→          eq(maximoShadowRecords.syncStatus, 'PENDING'),
   775→          and(
   776→            eq(maximoShadowRecords.syncStatus, 'ERROR'),
   777→            lt(maximoShadowRecords.syncAttempts, 5),
   778→            or(
   779→              isNull(maximoShadowRecords.nextSyncAt),
   780→              lt(maximoShadowRecords.nextSyncAt, new Date())
   781→            )
   782→          )
   783→        ),
   784→        limit: 100, // Process in batches
   785→      });
   786→
   787→      span.setAttribute('pending.count', pending.length);
   788→
   789→      let pushed = 0;
   790→      let errors = 0;
   791→
   792→      for (const record of pending) {
   793→        try {
   794→          const result = await this.pushToMaximo(record.pmScheduleId);
   795→          if (result.action === 'CREATED' || result.action === 'UPDATED') {
   796→            pushed++;
   797→          } else if (result.action === 'ERROR') {
   798→            errors++;
   799→          }
   800→        } catch (error) {
   801→          errors++;
   802→          this.log.error(
   803→            { pmScheduleId: record.pmScheduleId, error },
   804→            'Failed to push to Maximo'
   805→          );
   806→        }
   807→      }
   808→
   809→      span.setAttribute('pushed.count', pushed);
   810→      span.setAttribute('errors.count', errors);
   811→
   812→      return { pushed, errors };
   813→    });
   814→  }

   817→  // Pull Sync (Maximo → MMS)
   818→  // ===========================================================================
   819→
   820→  /**
   821→   * Pull Maximo status changes to MMS
   822→   *
   823→   * Queries Maximo for work orders modified since last pull and updates
   824→   * MMS PM statuses accordingly.
   825→   *
   826→   * @see specs/007-integration.md Section 7.4.7
   827→   */
   828→  async pullFromMaximo(): Promise<PullResult> {
   829→    return this.traced('pullFromMaximo', async (span) => {
   830→      const result: PullResult = {
   831→        processed: 0,
   832→        updated: 0,
   833→        conflicts: 0,
   834→        errors: 0,
   835→        details: [],
   836→      };
   837→
   838→      // Check circuit breaker state
   839→      if (this.circuitBreaker.getState() === 'OPEN') {
   840→        this.log.warn('Circuit breaker is open, skipping pull');
   841→        return result;
   842→      }
   843→
   844→      // Get all synced shadow records
   845→      const syncedRecords = await this.db.query.maximoShadowRecords.findMany({
   846→        where: and(
   847→          eq(maximoShadowRecords.syncStatus, 'SYNCED'),
   848→          sql`${maximoShadowRecords.maximoWoNum} IS NOT NULL`
   849→        ),
   850→      });
   851→
   852→      span.setAttribute('synced.records', syncedRecords.length);
   853→
   854→      for (const record of syncedRecords) {
   855→        result.processed++;
   856→
   857→        try {
   858→          if (!record.maximoWoNum) continue;
   859→
   860→          // Use circuit breaker for Maximo API call
   861→          const maximoWo = await this.circuitBreaker.execute(() =>
   862→            this.fetchMaximoWorkOrder(record.maximoWoNum!)
   863→          );
   864→
   865→          if (!maximoWo) {
   866→            result.details.push({
   867→              pmScheduleId: record.pmScheduleId,
   868→              maximoWoNum: record.maximoWoNum,
   869→              action: 'SKIPPED',
   870→              error: 'Work order not found in Maximo',
   871→            });
   872→            continue;
   873→          }
   874→
   875→          const currentMaximoData = record.maximoData as { status?: string } | null;
   876→          const previousStatus = currentMaximoData?.status;
   877→
   878→          // Status changed
   879→          if (maximoWo.status !== previousStatus) {
   880→            await this.handleStatusChange(record, maximoWo);
   881→            result.updated++;
   882→            result.details.push({
   883→              pmScheduleId: record.pmScheduleId,
   884→              maximoWoNum: record.maximoWoNum,
   885→              previousStatus,
   886→              newStatus: maximoWo.status,
   887→              action: 'UPDATED',
   888→            });
   889→          }
   890→
   891→          // Record actuals if completed
   892→          if (
   893→            (maximoWo.status === 'COMP' || maximoWo.status === 'CLOSE') &&
   894→            maximoWo.actlabhrs
   895→          ) {
   896→            await this.recordActuals(record.pmScheduleId, maximoWo);
   897→          }
   898→        } catch (error) {
   899→          result.errors++;
   900→          result.details.push({
   901→            pmScheduleId: record.pmScheduleId,
   902→            maximoWoNum: record.maximoWoNum ?? '',
   903→            action: 'ERROR',
   904→            error: error instanceof Error ? error.message : String(error),
   905→          });
   906→        }
   907→      }
   908→
   909→      this.lastPullTimestamp = new Date();
   910→      span.setAttribute('pull.updated', result.updated);
   911→      span.setAttribute('pull.errors', result.errors);
   912→
   913→      return result;
   914→    });
   915→  }

   921→  /**
   922→   * Resolve sync conflict
   923→   *
   924→   * @param pmScheduleId - PM schedule ID with conflict
   925→   * @param resolution - Strategy: USE_MMS (force push) or USE_MAXIMO (apply Maximo data)
   926→   * @param userId - User resolving the conflict
   927→   */
   928→  async resolveConflict(
   929→    pmScheduleId: string,
   930→    resolution: ConflictResolution,
   931→    userId?: string
   932→  ): Promise<void> {
   933→    return this.traced('resolveConflict', async (span) => {
   934→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   935→      span.setAttribute('resolution', resolution);
   936→
   937→      const shadow = await this.db.query.maximoShadowRecords.findFirst({
   938→        where: eq(maximoShadowRecords.pmScheduleId, pmScheduleId),
   939→      });
   940→
   941→      if (!shadow) {
   942→        throw new NotFoundError(`Shadow record not found for PM: ${pmScheduleId}`, {
   943→          entityType: 'MaximoShadowRecord',
   944→          entityId: pmScheduleId,
   945→        });
   946→      }
   947→
   948→      if (shadow.syncStatus !== 'CONFLICT') {
   949→        throw new ValidationError('Shadow record is not in conflict state');
   950→      }
   951→
   952→      if (!shadow.maximoWoNum) {
   953→        throw new ValidationError('Shadow record has no Maximo work order');
   954→      }
   955→
   956→      if (resolution === 'USE_MMS') {
   957→        // Force push MMS data to Maximo
   958→        const pm = await this.db.query.pmSchedule.findFirst({
   959→          where: eq(pmSchedule.id, pmScheduleId),
   960→          with: { train: true, pmType: true, assignedDepot: true },
   961→        });
   962→
   963→        if (!pm) {
   964→          throw new NotFoundError(`PM schedule not found: ${pmScheduleId}`, {
   965→            entityType: 'PmSchedule',
   966→            entityId: pmScheduleId,
   967→          });
   968→        }
   969→
   970→        await this.updateMaximoWorkOrder(shadow.maximoWoNum, pm, { force: true });
   971→      } else {
   972→        // Apply Maximo data to MMS
   973→        const maximoWo = await this.fetchMaximoWorkOrder(shadow.maximoWoNum);
   974→
   975→        if (!maximoWo) {
   976→          throw new MaximoConnectionError('Failed to fetch Maximo work order');
   977→        }
   978→
   979→        await this.applyMaximoDataToMms(pmScheduleId, maximoWo);
   980→      }
   981→
   982→      // Update shadow record
   983→      await this.updateShadowRecord(shadow.id, {
   984→        syncStatus: 'SYNCED',
   985→        lastSyncAt: new Date(),
   986→        conflictResolvedAt: new Date(),
   987→        conflictResolvedBy: userId ?? undefined,
   988→        conflictResolution: resolution,
   989→      });
   990→
   991→      await this.recordSyncHistory(shadow.id, 'CONFLICT_RESOLVE', {
   992→        previousStatus: 'CONFLICT',
   993→        newStatus: 'SYNCED',
   994→        success: true,
   995→      });
   996→
   997→      this.log.info(
   998→        { pmScheduleId, resolution, userId },
   999→        'Conflict resolved'
  1000→      );
  1001→    });
  1002→  }
