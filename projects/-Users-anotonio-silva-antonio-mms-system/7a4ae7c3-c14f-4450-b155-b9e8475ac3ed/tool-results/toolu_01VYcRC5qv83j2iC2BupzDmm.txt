   481→/**
   482→ * Render a template with Handlebars-style variable interpolation
   483→ */
   484→function renderTemplate(template: string, data: Record<string, unknown>): string {
   485→  let result = template;
   486→
   487→  // Simple variable replacement: {{variable}}
   488→  result = result.replace(/\{\{([^#/}]+)\}\}/g, (match, key) => {
   489→    const trimmedKey = key.trim();
   490→    const value = getNestedValue(data, trimmedKey);
   491→    return value !== undefined ? String(value) : match;
   492→  });
   493→
   494→  // Array iteration: {{#each items}}...{{/each}}
   495→  result = result.replace(
   496→    /\{\{#each\s+(\w+)\}\}([\s\S]*?)\{\{\/each\}\}/g,
   497→    (match, arrayKey, itemTemplate) => {
   498→      const array = data[arrayKey];
   499→      if (!Array.isArray(array)) return '';
   500→      return array
   501→        .map((item) => {
   502→          let itemResult = itemTemplate;
   503→          // Replace item properties
   504→          itemResult = itemResult.replace(/\{\{(\w+)\}\}/g, (m, k) => {
   505→            const val = typeof item === 'object' && item !== null ? item[k] : undefined;
   506→            return val !== undefined ? String(val) : m;
   507→          });
   508→          return itemResult;
   509→        })
   510→        .join('');
   511→    }
   512→  );
   513→
   514→  return result;
   515→}

   517→/**
   518→ * Get nested property value from object
   519→ */
   520→function getNestedValue(obj: Record<string, unknown>, path: string): unknown {
   521→  const keys = path.split('.');
   522→  let current: unknown = obj;
   523→  for (const key of keys) {
   524→    if (current === null || current === undefined) return undefined;
   525→    if (typeof current !== 'object') return undefined;
   526→    current = (current as Record<string, unknown>)[key];
   527→  }
   528→  return current;
   529→}

   596→  /**
   597→   * Get a notification template by name
   598→   *
   599→   * Template lookup hierarchy (first match wins):
   600→   * 1. Depot-specific template (if depotId provided)
   601→   * 2. Global template (depotId = NULL)
   602→   * 3. Default hardcoded template
   603→   *
   604→   * @param templateName Template name (e.g., 'drift_alert', 'pm_reminder')
   605→   * @param depotId Optional depot ID for depot-specific override
   606→   * @returns Template content and subject, or defaults if not found
   607→   */
   608→  async getTemplate(
   609→    templateName: string,
   610→    depotId?: string
   611→  ): Promise<{ subjectTemplate: string; bodyHtml: string; bodyText?: string }> {
   612→    return this.traced('getTemplate', async (span) => {
   613→      span.setAttribute('template.name', templateName);
   614→      if (depotId) span.setAttribute('depot.id', depotId);
   615→
   616→      // Try depot-specific template first
   617→      if (depotId) {
   618→        const depotTemplate = await this.db.query.notificationTemplates.findFirst({
   619→          where: and(
   620→            eq(notificationTemplates.name, templateName),
   621→            eq(notificationTemplates.depotId, depotId),
   622→            eq(notificationTemplates.isActive, true)
   623→          ),
   624→        });
   625→
   626→        if (depotTemplate) {
   627→          span.setAttribute('template.source', 'depot');
   628→          return {
   629→            subjectTemplate: depotTemplate.subjectTemplate,
   630→            bodyHtml: depotTemplate.bodyHtml,
   631→            bodyText: depotTemplate.bodyText ?? undefined,
   632→          };
   633→        }
   634→      }
   635→
   636→      // Try global template
   637→      const globalTemplate = await this.db.query.notificationTemplates.findFirst({
   638→        where: and(
   639→          eq(notificationTemplates.name, templateName),
   640→          isNull(notificationTemplates.depotId),
   641→          eq(notificationTemplates.isActive, true)
   642→        ),
   643→      });
   644→
   645→      if (globalTemplate) {
   646→        span.setAttribute('template.source', 'global');
   647→        return {
   648→          subjectTemplate: globalTemplate.subjectTemplate,
   649→          bodyHtml: globalTemplate.bodyHtml,
   650→          bodyText: globalTemplate.bodyText ?? undefined,
   651→        };
   652→      }
   653→
   654→      // Fall back to hardcoded defaults
   655→      const defaultBody = DEFAULT_TEMPLATES[templateName] ?? DEFAULT_TEMPLATES['drift_alert'];
   656→      span.setAttribute('template.source', 'default');
   657→      return {
   658→        subjectTemplate: 'MMS {{type}} Notification',
   659→        bodyHtml: defaultBody,
   660→      };
   661→    });
   662→  }

   869→  /**
   870→   * Send notifications to recipients
   871→   *
   872→   * Handles:
   873→   * - User preference checking (enabled, severity filter, depot filter)
   874→   * - Digest batching based on user preferences
   875→   * - Immediate delivery for CRITICAL notifications
   876→   * - Template rendering
   877→   */
   878→  async sendNotification(request: SendNotificationRequest): Promise<SendNotificationResult> {
   879→    return this.traced('sendNotification', async (span) => {
   880→      span.setAttribute('notification.type', request.type);
   881→      span.setAttribute('notification.severity', request.severity);
   882→      span.setAttribute('notification.recipients', request.recipients.length);
   883→
   884→      const result: SendNotificationResult = {
   885→        notificationIds: [],
   886→        queued: 0,
   887→        suppressed: 0,
   888→        immediate: 0,
   889→        digested: 0,
   890→      };
   891→
   892→      // Resolve recipients (convert user IDs to emails, fetch preferences)
   893→      const resolvedRecipients = await this.resolveRecipients(request.recipients);
   894→
   895→      for (const recipient of resolvedRecipients) {
   896→        // Check if notification should be sent based on preferences
   897→        const { shouldSend, reason } = this.shouldSendNotification(
   898→          recipient.preferences,
   899→          request.type,
   900→          request.severity,
   901→          request.depotId
   902→        );
   903→
   904→        if (!shouldSend) {
   905→          // Record suppressed notification
   906→          const [logEntry] = await this.db
   907→            .insert(notificationLog)
   908→            .values({
   909→              userId: recipient.userId,
   910→              recipientEmail: recipient.email,
   911→              type: request.type,
   912→              severity: request.severity,
   913→              template: request.template,
   914→              subject: request.subject,
   915→              data: request.data,
   916→              status: 'SUPPRESSED',
   917→              trainId: request.trainId,
   918→              depotId: request.depotId,
   919→              baselineId: request.baselineId,
   920→              pmScheduleId: request.pmScheduleId,
   921→              errorMessage: reason,
   922→            })
   923→            .returning();
   924→
   925→          result.notificationIds.push(logEntry.id);
   926→          result.suppressed++;
   927→          continue;
   928→        }
   929→
   930→        // Determine delivery mode
   931→        const isCritical = request.severity === 'CRITICAL';
   932→        const isImmediate = isCritical || recipient.preferences?.digestMode === 'IMMEDIATE';
   933→
   934→        if (isImmediate) {
   935→          // Queue for immediate delivery
   936→          const [logEntry] = await this.db
   937→            .insert(notificationLog)
   938→            .values({
   939→              userId: recipient.userId,
   940→              recipientEmail: recipient.email,
   941→              type: request.type,
   942→              severity: request.severity,
   943→              template: request.template,
   944→              subject: request.subject,
   945→              data: request.data,
   946→              status: 'QUEUED',
   947→              trainId: request.trainId,
   948→              depotId: request.depotId,
   949→              baselineId: request.baselineId,
   950→              pmScheduleId: request.pmScheduleId,
   951→            })
   952→            .returning();
   953→
   954→          // Queue the notification job
   955→          await addJob<NotificationJobData>(
   956→            QUEUE_NAMES.NOTIFICATION,
   957→            'send-email',
   958→            {
   959→              type: 'email',
   960→              template: request.template,
   961→              recipients: [recipient.email],
   962→              data: {
   963→                ...request.data,
   964→                notificationLogId: logEntry.id,
   965→              },
   966→              priority: request.severity === 'CRITICAL' ? 'high' : 'normal',
   967→            },
   968→            {
   969→              priority: request.severity === 'CRITICAL' ? 1 : 10,
   970→            }
   971→          );
   972→
   973→          result.notificationIds.push(logEntry.id);
   974→          result.immediate++;
   975→          result.queued++;
   976→        } else {
   977→          // Add to digest batch
   978→          const digestBatch = await this.getOrCreateDigestBatch(
   979→            recipient.userId!,
   980→            recipient.preferences!.digestMode,
   981→            recipient.preferences!.digestTime,
   982→            recipient.preferences!.timezone
   983→          );
   984→
   985→          const [logEntry] = await this.db
   986→            .insert(notificationLog)
   987→            .values({
   988→              userId: recipient.userId,
   989→              recipientEmail: recipient.email,
   990→              type: request.type,
   991→              severity: request.severity,
   992→              template: request.template,
   993→              subject: request.subject,
   994→              data: request.data,
   995→              status: 'PENDING',
   996→              digestBatchId: digestBatch.id,
   997→              isDigested: true,
   998→              trainId: request.trainId,
   999→              depotId: request.depotId,
  1000→              baselineId: request.baselineId,
  1001→              pmScheduleId: request.pmScheduleId,
  1002→            })
  1003→            .returning();
  1004→
  1005→          // Update batch count
  1006→          await this.db
  1007→            .update(digestBatches)
  1008→            .set({
  1009→              notificationCount: sql`${digestBatches.notificationCount} + 1`,
  1010→            })
  1011→            .where(eq(digestBatches.id, digestBatch.id));
  1012→
  1013→          result.notificationIds.push(logEntry.id);
  1014→          result.digested++;
  1015→        }
  1016→      }
  1017→
  1018→      // Update metrics
  1019→      notificationMetrics.notificationsQueued.inc({ type: request.type, severity: request.severity }, result.queued);
  1020→
  1021→      return result;
  1022→    });
  1023→  }

  1025→  /**
  1026→   * Resolve recipients from user IDs or email addresses
  1027→   */
  1028→  private async resolveRecipients(
  1029→    recipients: string[]
  1030→  ): Promise<Array<{ userId?: string; email: string; preferences?: UserPreferences }>> {
  1031→    const result: Array<{ userId?: string; email: string; preferences?: UserPreferences }> = [];
  1032→
  1033→    for (const recipient of recipients) {
  1034→      // Check if it's a UUID (user ID) or email
  1035→      const isEmail = recipient.includes('@');
  1036→
  1037→      if (isEmail) {
  1038→        // Direct email - no user preferences
  1039→        result.push({ email: recipient });
  1040→      } else {
  1041→        // User ID - fetch user and preferences
  1042→        const user = await this.db.query.users.findFirst({
  1043→          where: eq(users.id, recipient),
  1044→        });
  1045→
  1046→        if (!user) {
  1047→          this.log.warn({ userId: recipient }, 'User not found for notification');
  1048→          continue;
  1049→        }
  1050→
  1051→        const prefs = await this.getUserPreferences(recipient);
  1052→        result.push({
  1053→          userId: recipient,
  1054→          email: prefs?.notificationEmail ?? user.email,
  1055→          preferences: prefs ?? undefined,
  1056→        });
  1057→      }
  1058→    }
  1059→
  1060→    return result;
  1061→  }

  1063→  /**
  1064→   * Check if notification should be sent based on preferences
  1065→   */
  1066→  private shouldSendNotification(
  1067→    preferences: UserPreferences | undefined,
  1068→    type: NotificationType,
  1069→    severity: NotificationSeverity,
  1070→    depotId?: string
  1071→  ): { shouldSend: boolean; reason?: string } {
  1072→    // No preferences = send everything
  1073→    if (!preferences) {
  1074→      return { shouldSend: true };
  1075→    }
  1076→
  1077→    const typePref = preferences.preferences[type];
  1078→    if (!typePref) {
  1079→      return { shouldSend: true };
  1080→    }
  1081→
  1082→    // Check if type is enabled
  1083→    if (!typePref.enabled) {
  1084→      return { shouldSend: false, reason: `Notification type ${type} is disabled` };
  1085→    }
  1086→
  1087→    // Check severity filter
  1088→    if (typePref.minSeverity) {
  1089→      const severityOrder: NotificationSeverity[] = ['LOW', 'INFO', 'MEDIUM', 'HIGH', 'CRITICAL'];
  1090→      const minIndex = severityOrder.indexOf(typePref.minSeverity);
  1091→      const currentIndex = severityOrder.indexOf(severity);
  1092→      if (currentIndex < minIndex) {
  1093→        return {
  1094→          shouldSend: false,
  1095→          reason: `Severity ${severity} below minimum ${typePref.minSeverity}`,
  1096→        };
  1097→      }
  1098→    }
  1099→
  1100→    // Check depot filter (for OVERLOAD_WARNING)
  1101→    if (typePref.depots && typePref.depots.length > 0 && depotId) {
  1102→      if (!typePref.depots.includes(depotId)) {
  1103→        return { shouldSend: false, reason: `Depot ${depotId} not in allowed list` };
  1104→      }
  1105→    }
  1106→
  1107→    return { shouldSend: true };
  1108→  }

  1118→  ): Promise<DigestBatch> {

  1121→

  1148→  /**
  1149→   * Calculate next digest delivery time
  1150→   */
  1151→  private calculateNextDigestTime(
  1152→    digestMode: DigestMode,
  1153→    digestTime: string,
  1154→    // eslint-disable-next-line @typescript-eslint/no-unused-vars
  1155→    timezone: string
  1156→  ): Date {
  1157→    const now = new Date();
  1158→    const [hours, minutes] = digestTime.split(':').map(Number);
  1159→
  1160→    // Start with today at digest time
  1161→    const scheduled = new Date(now);
  1162→    scheduled.setHours(hours, minutes, 0, 0);
  1163→
  1164→    // If time has passed today, move to next period
  1165→    if (scheduled <= now) {
  1166→      if (digestMode === 'DAILY_DIGEST') {
  1167→        scheduled.setDate(scheduled.getDate() + 1);
  1168→      } else if (digestMode === 'WEEKLY_DIGEST') {
  1169→        // Move to next Monday
  1170→        const daysUntilMonday = (8 - scheduled.getDay()) % 7 || 7;
  1171→        scheduled.setDate(scheduled.getDate() + daysUntilMonday);
  1172→      }
  1173→    } else if (digestMode === 'WEEKLY_DIGEST') {
  1174→      // Find next Monday at digest time
  1175→      const dayOfWeek = scheduled.getDay();
  1176→      if (dayOfWeek !== 1) {

  1298→  /**
  1299→   * Process a single digest batch
  1300→   */
  1301→  private async processDigestBatch(batch: DigestBatch): Promise<void> {
  1302→    // Get all notifications in this batch
  1303→    const notifications = await this.db.query.notificationLog.findMany({
  1304→      where: eq(notificationLog.digestBatchId, batch.id),
  1305→      orderBy: desc(notificationLog.createdAt),
  1306→    });
  1307→
  1308→    if (notifications.length === 0) {
  1309→      // Empty batch, mark as processed
  1310→      await this.db
  1311→        .update(digestBatches)
  1312→        .set({ status: 'SENT', processedAt: new Date() })
  1313→        .where(eq(digestBatches.id, batch.id));
  1314→      return;
  1315→    }
  1316→
  1317→    // Get user email
  1318→    const user = await this.db.query.users.findFirst({
  1319→      where: eq(users.id, batch.userId),
  1320→    });
  1321→
  1322→    if (!user) {
  1323→      throw new NotificationError(`User not found for digest batch: ${batch.userId}`);
  1324→    }
  1325→
  1326→    // Get preferences for email address
  1327→    const prefs = await this.getUserPreferences(batch.userId);
  1328→    const email = prefs?.notificationEmail ?? user.email;
  1329→
  1330→    // Group notifications by type
  1331→    const sections: Array<{
  1332→      type: string;
  1333→      count: number;
  1334→      items: Array<{ summary: string }>;
  1335→    }> = [];
  1336→
  1337→    const byType = new Map<string, NotificationLogEntry[]>();
  1338→    for (const n of notifications) {
  1339→      const existing = byType.get(n.type) ?? [];
  1340→      existing.push(n);
  1341→      byType.set(n.type, existing);
  1342→    }
  1343→
  1344→    for (const [type, items] of byType) {
  1345→      sections.push({
  1346→        type,
  1347→        count: items.length,
  1348→        items: items.map((item) => ({
  1349→          summary: (item.data as Record<string, unknown>)['summary'] as string ?? item.subject ?? type,
  1350→        })),
  1351→      });
  1352→    }
  1353→
  1354→    // Render digest template
  1355→    const digestPeriod = batch.digestMode === 'DAILY_DIGEST' ? 'Last 24 hours' : 'Last 7 days';
  1356→    const html = renderTemplate(DEFAULT_TEMPLATES['digest'], {
  1357→      digestPeriod,
  1358→      totalCount: notifications.length,
  1359→      sections,
  1360→      dashboardUrl: process.env['MMS_DASHBOARD_URL'] ?? 'https://mms.metro.example',
  1361→    });
  1362→
  1363→    // Send digest email
  1364→    await sendEmail(
  1365→      {
  1366→        to: email,
  1367→        from: this.smtpConfig.from,
  1368→        subject: `MMS ${digestPeriod} Digest - ${notifications.length} notifications`,
  1369→        html,
  1370→      },
  1371→      this.smtpConfig,
  1372→      this.retryConfig
  1373→    );
  1374→
  1375→    // Update all notifications as sent
  1376→    await this.db
  1377→      .update(notificationLog)
  1378→      .set({ status: 'SENT', sentAt: new Date() })
  1379→      .where(eq(notificationLog.digestBatchId, batch.id));
  1380→
  1381→    // Mark batch as processed
  1382→    await this.db
  1383→      .update(digestBatches)
  1384→      .set({ status: 'SENT', processedAt: new Date() })
  1385→      .where(eq(digestBatches.id, batch.id));
  1386→  }
