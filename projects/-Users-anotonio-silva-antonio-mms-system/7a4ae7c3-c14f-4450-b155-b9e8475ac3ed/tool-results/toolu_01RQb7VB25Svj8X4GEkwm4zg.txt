     1→/**
     2→ * MMS Baseline Assumption Service
     3→ *
     4→ * Manages PM history assumptions for cold start scenarios during data migration.
     5→ * Implements inference rules, manual review queue, and approval workflow.
     6→ *
     7→ * Key features:
     8→ * - Inference rules for missing PM history (from commissioning, higher PM, etc.)
     9→ * - Manual review queue for LOW/ESTIMATED confidence assumptions
    10→ * - Review/approval workflow with audit trail
    11→ * - Integration with PM scheduler for baseline calculation
    12→ *
    13→ * @see specs/003-database-schema.md Section 3.2.6
    14→ * @see ADR-014: PM History Assumptions for Cold Start
    15→ */

   195→  /**
   196→   * Create a new baseline assumption
   197→   */
   198→  async createAssumption(request: CreateAssumptionRequest): Promise<AssumptionWithDetails> {
   199→    return this.traced('createAssumption', async (span) => {
   200→      span.setAttribute(MmsAttributes.TRAIN_ID, request.trainId);
   201→
   202→      // Validate train exists
   203→      const train = await this.db.query.trains.findFirst({
   204→        where: eq(trains.id, request.trainId),
   205→      });
   206→      if (!train) {
   207→        throw new NotFoundError('Train not found', {
   208→          entityType: 'train',
   209→          entityId: request.trainId,
   210→        });
   211→      }
   212→
   213→      // Validate PM type exists
   214→      const pmType = await this.db.query.pmTypes.findFirst({
   215→        where: eq(pmTypes.id, request.pmTypeId),
   216→      });
   217→      if (!pmType) {
   218→        throw new NotFoundError('PM type not found', {
   219→          entityType: 'pmType',
   220→          entityId: request.pmTypeId,
   221→        });
   222→      }
   223→
   224→      // Check for existing assumption
   225→      const existing = await this.db.query.pmBaselineAssumptions.findFirst({
   226→        where: and(
   227→          eq(pmBaselineAssumptions.trainId, request.trainId),
   228→          eq(pmBaselineAssumptions.pmTypeId, request.pmTypeId)
   229→        ),
   230→      });
   231→      if (existing) {
   232→        throw new ConflictError(
   233→          `Assumption already exists for train ${train.trainNumber} and PM type ${pmType.pmCode}`,
   234→          { existingId: existing.id }
   235→        );
   236→      }
   237→
   238→      // Determine if review is required based on confidence
   239→      const requiresReview =
   240→        request.requiresReview ??
   241→        (request.confidenceLevel === 'LOW' || request.confidenceLevel === 'ESTIMATED');
   242→
   243→      const newAssumption: NewPmBaselineAssumption = {
   244→        trainId: request.trainId,
   245→        pmTypeId: request.pmTypeId,
   246→        assumptionType: request.assumptionType,
   247→        assumedLastPmDate: request.assumedLastPmDate ?? null,
   248→        assumedLastPmMileage: request.assumedLastPmMileage ?? null,
   249→        confidenceLevel: request.confidenceLevel,
   250→        inferenceSource: request.inferenceSource ?? null,
   251→        inferenceData: request.inferenceData ?? null,
   252→        requiresReview,
   253→      };
   254→
   255→      const [created] = await this.db
   256→        .insert(pmBaselineAssumptions)
   257→        .values(newAssumption)
   258→        .returning();
   259→
   260→      this.log.info(
   261→        { assumptionId: created.id, trainId: request.trainId, pmTypeCode: pmType.pmCode },
   262→        'Created baseline assumption'
   263→      );
   264→
   265→      return this.getAssumptionById(created.id);
   266→    });
   267→  }

   297→  /**
   298→   * List assumptions with filtering and pagination
   299→   */
   300→  async listAssumptions(
   301→    filters: AssumptionFilters = {},
   302→    pagination: PaginationOptions = {}
   303→  ): Promise<PaginatedResult<AssumptionWithDetails>> {
   304→    return this.traced('listAssumptions', async (span) => {
   305→      const page = pagination.page ?? 1;
   306→      const pageSize = pagination.pageSize ?? 20;
   307→      const offset = (page - 1) * pageSize;
   308→
   309→      // Build where conditions
   310→      const conditions: SQL<unknown>[] = [];
   311→
   312→      if (filters.trainId) {
   313→        conditions.push(eq(pmBaselineAssumptions.trainId, filters.trainId));
   314→      }
   315→      if (filters.pmTypeId) {
   316→        conditions.push(eq(pmBaselineAssumptions.pmTypeId, filters.pmTypeId));
   317→      }
   318→      if (filters.confidenceLevel) {
   319→        const levels = Array.isArray(filters.confidenceLevel)
   320→          ? filters.confidenceLevel
   321→          : [filters.confidenceLevel];
   322→        conditions.push(inArray(pmBaselineAssumptions.confidenceLevel, levels));
   323→      }
   324→      if (filters.assumptionType) {
   325→        const types = Array.isArray(filters.assumptionType)
   326→          ? filters.assumptionType
   327→          : [filters.assumptionType];
   328→        conditions.push(inArray(pmBaselineAssumptions.assumptionType, types));
   329→      }
   330→      if (filters.requiresReview !== undefined) {
   331→        conditions.push(eq(pmBaselineAssumptions.requiresReview, filters.requiresReview));
   332→      }
   333→      if (filters.reviewed !== undefined) {
   334→        if (filters.reviewed) {
   335→          conditions.push(sql`${pmBaselineAssumptions.reviewedAt} IS NOT NULL`);
   336→        } else {
   337→          conditions.push(isNull(pmBaselineAssumptions.reviewedAt));
   338→        }
   339→      }
   340→      if (filters.applied !== undefined) {
   341→        if (filters.applied) {
   342→          conditions.push(sql`${pmBaselineAssumptions.appliedAt} IS NOT NULL`);
   343→        } else {
   344→          conditions.push(isNull(pmBaselineAssumptions.appliedAt));
   345→        }
   346→      }
   347→
   348→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   349→
   350→      // Get total count
   351→      const [{ total }] = await this.db
   352→        .select({ total: count() })
   353→        .from(pmBaselineAssumptions)
   354→        .where(whereClause);
   355→
   356→      // Build order by
   357→      let orderBy: SQL;
   358→      const sortOrder = pagination.sortOrder === 'asc' ? asc : desc;
   359→      switch (pagination.sortBy) {
   360→        case 'confidenceLevel':
   361→          orderBy = sortOrder(pmBaselineAssumptions.confidenceLevel);
   362→          break;
   363→        case 'trainNumber':
   364→          orderBy = sortOrder(trains.trainNumber);
   365→          break;
   366→        default:
   367→          orderBy = sortOrder(pmBaselineAssumptions.createdAt);
   368→      }
   369→
   370→      // Query with joins
   371→      const results = await this.db.query.pmBaselineAssumptions.findMany({
   372→        where: whereClause,
   373→        with: {
   374→          train: true,
   375→          pmType: true,
   376→          reviewedByUser: true,
   377→          appliedByUser: true,
   378→        },
   379→        orderBy: [orderBy],
   380→        limit: pageSize,
   381→        offset,
   382→      });
   383→
   384→      span.setAttribute('results.count', results.length);
   385→      span.setAttribute('results.total', total);
   386→
   387→      return {
   388→        items: results.map((r) => this.mapToAssumptionWithDetails(r)),
   389→        total,
   390→        page,
   391→        pageSize,
   392→        totalPages: Math.ceil(total / pageSize),
   393→      };
   394→    });
   395→  }

   397→  /**
   398→   * Update an assumption
   399→   */
   400→  async updateAssumption(
   401→    id: string,
   402→    request: UpdateAssumptionRequest
   403→  ): Promise<AssumptionWithDetails> {
   404→    return this.traced('updateAssumption', async (span) => {
   405→      span.setAttribute('assumption.id', id);
   406→
   407→      const existing = await this.db.query.pmBaselineAssumptions.findFirst({
   408→        where: eq(pmBaselineAssumptions.id, id),
   409→      });
   410→
   411→      if (!existing) {
   412→        throw new NotFoundError('Assumption not found', {
   413→          entityType: 'pmBaselineAssumption',
   414→          entityId: id,
   415→        });
   416→      }
   417→
   418→      // Cannot update applied assumptions
   419→      if (existing.appliedAt) {
   420→        throw new ConflictError('Cannot update an assumption that has already been applied');
   421→      }
   422→
   423→      const updates: Partial<NewPmBaselineAssumption> = {
   424→        updatedAt: new Date(),
   425→      };
   426→
   427→      if (request.assumedLastPmDate !== undefined) {
   428→        updates.assumedLastPmDate = request.assumedLastPmDate;
   429→      }
   430→      if (request.assumedLastPmMileage !== undefined) {
   431→        updates.assumedLastPmMileage = request.assumedLastPmMileage;
   432→      }
   433→      if (request.confidenceLevel !== undefined) {
   434→        updates.confidenceLevel = request.confidenceLevel;
   435→        // Update review requirement based on new confidence
   436→        updates.requiresReview =
   437→          request.confidenceLevel === 'LOW' || request.confidenceLevel === 'ESTIMATED';
   438→      }
   439→      if (request.inferenceSource !== undefined) {
   440→        updates.inferenceSource = request.inferenceSource;
   441→      }
   442→      if (request.inferenceData !== undefined) {
   443→        updates.inferenceData = request.inferenceData;
   444→      }
   445→      if (request.requiresReview !== undefined) {
   446→        updates.requiresReview = request.requiresReview;
   447→      }
   448→
   449→      await this.db
   450→        .update(pmBaselineAssumptions)
   451→        .set(updates)
   452→        .where(eq(pmBaselineAssumptions.id, id));
   453→
   454→      this.log.info({ assumptionId: id }, 'Updated baseline assumption');
   455→
   456→      return this.getAssumptionById(id);
   457→    });
   458→  }

   512→  /**
   513→   * Get review queue summary statistics
   514→   */
   515→  async getReviewQueueSummary(): Promise<ReviewQueueSummary> {
   516→    return this.traced('getReviewQueueSummary', async () => {
   517→      // Get pending review assumptions
   518→      const pendingReview = await this.db.query.pmBaselineAssumptions.findMany({
   519→        where: and(
   520→          eq(pmBaselineAssumptions.requiresReview, true),
   521→          isNull(pmBaselineAssumptions.reviewedAt)
   522→        ),
   523→      });
   524→
   525→      // Count by confidence level
   526→      const byConfidenceLevel: Record<ConfidenceLevel, number> = {
   527→        HIGH: 0,
   528→        MEDIUM: 0,
   529→        LOW: 0,
   530→        ESTIMATED: 0,
   531→      };
   532→
   533→      // Count by assumption type
   534→      const byAssumptionType: Record<AssumptionType, number> = {
   535→        INFERRED_FROM_COMMISSIONING: 0,
   536→        INFERRED_FROM_HIGHER_PM: 0,
   537→        MANUAL_OVERRIDE: 0,
   538→        IMPORTED: 0,
   539→        NO_HISTORY: 0,
   540→      };
   541→
   542→      let oldestDate: Date | null = null;
   543→
   544→      for (const assumption of pendingReview) {
   545→        byConfidenceLevel[assumption.confidenceLevel as ConfidenceLevel]++;
   546→        byAssumptionType[assumption.assumptionType as AssumptionType]++;
   547→
   548→        if (!oldestDate || (assumption.createdAt && assumption.createdAt < oldestDate)) {
   549→          oldestDate = assumption.createdAt;
   550→        }
   551→      }
   552→
   553→      return {
   554→        totalPendingReview: pendingReview.length,
   555→        byConfidenceLevel,
   556→        byAssumptionType,
   557→        oldestPendingDate: oldestDate,
   558→      };
   559→    });
   560→  }

   562→  /**
   563→   * Review an assumption (approve or reject)
   564→   */
   565→  async reviewAssumption(
   566→    id: string,
   567→    request: ReviewAssumptionRequest,
   568→    userId: string
   569→  ): Promise<AssumptionWithDetails> {
   570→    return this.traced('reviewAssumption', async (span) => {
   571→      span.setAttribute('assumption.id', id);
   572→      span.setAttribute('review.approved', request.approved);
   573→
   574→      const existing = await this.db.query.pmBaselineAssumptions.findFirst({
   575→        where: eq(pmBaselineAssumptions.id, id),
   576→      });
   577→
   578→      if (!existing) {
   579→        throw new NotFoundError('Assumption not found', {
   580→          entityType: 'pmBaselineAssumption',
   581→          entityId: id,
   582→        });
   583→      }
   584→
   585→      // Already applied cannot be reviewed again
   586→      if (existing.appliedAt) {
   587→        throw new ConflictError('Cannot review an assumption that has already been applied');
   588→      }
   589→
   590→      const updates: Partial<NewPmBaselineAssumption> = {
   591→        reviewedBy: userId,
   592→        reviewedAt: new Date(),
   593→        reviewNotes: request.reviewNotes ?? null,
   594→        updatedAt: new Date(),
   595→      };
   596→
   597→      if (request.approved) {
   598→        // If corrections provided, update the assumed values
   599→        if (request.correctedDate !== undefined) {
   600→          updates.assumedLastPmDate = request.correctedDate;
   601→        }
   602→        if (request.correctedMileage !== undefined) {
   603→          updates.assumedLastPmMileage = request.correctedMileage;
   604→        }
   605→        // Upgrade confidence if corrections were made
   606→        if (request.correctedDate !== undefined || request.correctedMileage !== undefined) {
   607→          updates.confidenceLevel = 'HIGH';
   608→          updates.assumptionType = 'MANUAL_OVERRIDE';
   609→        }
   610→        // Clear the review flag
   611→        updates.requiresReview = false;
   612→      } else {
   613→        // Rejected - keep in review queue or mark for deletion
   614→        updates.reviewNotes = `REJECTED: ${request.reviewNotes ?? 'No reason provided'}`;
   615→      }
   616→
   617→      await this.db
   618→        .update(pmBaselineAssumptions)
   619→        .set(updates)
   620→        .where(eq(pmBaselineAssumptions.id, id));
   621→
   622→      this.log.info(
   623→        { assumptionId: id, approved: request.approved, reviewedBy: userId },
   624→        `Assumption ${request.approved ? 'approved' : 'rejected'}`
   625→      );
   626→
   627→      return this.getAssumptionById(id);
   628→    });
   629→  }

   644→      const existing = await this.db.query.pmBaselineAssumptions.findFirst({
   645→        where: eq(pmBaselineAssumptions.id, id),
   646→        with: {
   647→          train: true,
   648→          pmType: true,
   649→        },
   650→      });
   651→
   652→      if (!existing) {
   653→        throw new NotFoundError('Assumption not found', {
   654→          entityType: 'pmBaselineAssumption',
   655→          entityId: id,
   656→        });
   657→      }
   658→
   659→      // Must be reviewed first (unless HIGH confidence)
   660→      if (
   661→        existing.requiresReview &&
   662→        !existing.reviewedAt &&
   663→        existing.confidenceLevel !== 'HIGH'
   664→      ) {
   665→        throw new ValidationError(
   666→          'Assumption must be reviewed before it can be applied',
   667→          { field: 'reviewedAt' }
   668→        );
   669→      }
   670→
   671→      // Already applied
   672→      if (existing.appliedAt) {
   673→        throw new ConflictError('Assumption has already been applied');
   674→      }
   675→
   676→      // Determine final values
   677→      const finalDate = request.overrideDate ?? existing.assumedLastPmDate;
   678→      const finalMileage = request.overrideMileage ?? existing.assumedLastPmMileage;
   679→
   680→      if (!finalDate && !finalMileage) {
   681→        throw new ValidationError(
   682→          'Cannot apply assumption without a date or mileage value',
   683→          { field: 'assumedLastPmDate' }
   684→        );
   685→      }
   686→
   687→      // Update the assumption as applied
   688→      await this.db
   689→        .update(pmBaselineAssumptions)
   690→        .set({
   691→          appliedAt: new Date(),
   692→          appliedBy: userId,
   693→          assumedLastPmDate: finalDate,
   694→          assumedLastPmMileage: finalMileage,
   695→          updatedAt: new Date(),
   696→        })
   697→        .where(eq(pmBaselineAssumptions.id, id));
   698→
   699→      this.log.info(
   700→        {
   701→          assumptionId: id,
   702→          trainId: existing.trainId,
   703→          pmTypeId: existing.pmTypeId,
   704→          appliedDate: finalDate,
   705→          appliedMileage: finalMileage,
   706→          appliedBy: userId,
   707→        },
   708→        'Applied baseline assumption'
   709→      );
   710→
   711→      return this.getAssumptionById(id);
   712→    });
   713→  }

   719→  /**
   720→   * Infer assumptions from train commissioning date
   721→   *
   722→   * For each PM type, assumes the first PM occurred at commissioning date + interval.
   723→   * Sets confidence to LOW and requires review.
   724→   */
   725→  async inferFromCommissioning(request: InferFromCommissioningRequest): Promise<InferenceResult> {
   726→    return this.traced('inferFromCommissioning', async (span) => {
   727→      span.setAttribute(MmsAttributes.TRAIN_ID, request.trainId);
   728→
   729→      const train = await this.db.query.trains.findFirst({
   730→        where: eq(trains.id, request.trainId),
   731→      });
   732→
   733→      if (!train) {
   734→        throw new NotFoundError('Train not found', {
   735→          entityType: 'train',
   736→          entityId: request.trainId,
   737→        });
   738→      }
   739→
   740→      if (!train.commissioningDate) {
   741→        throw new ValidationError(
   742→          'Train does not have a commissioning date set',
   743→          { field: 'commissioningDate' }
   744→        );
   745→      }
   746→
   747→      // Get PM types for this train's fleet
   748→      const fleetPmTypes = await this.db.query.pmTypes.findMany({
   749→        where: train.fleetId ? eq(pmTypes.fleetId, train.fleetId) : undefined,
   750→      });
   751→
   752→      // Filter by requested codes if provided
   753→      const targetPmTypes = request.pmTypeCodes
   754→        ? fleetPmTypes.filter((pt) => request.pmTypeCodes!.includes(pt.pmCode))
   755→        : fleetPmTypes;
   756→
   757→      const result: InferenceResult = {
   758→        created: 0,
   759→        skipped: 0,
   760→        errors: [],
   761→        assumptions: [],
   762→      };
   763→
   764→      const commissioningDate = new Date(train.commissioningDate);
   765→
   766→      for (const pmType of targetPmTypes) {
   767→        try {
   768→          // Check if assumption already exists
   769→          const existing = await this.db.query.pmBaselineAssumptions.findFirst({
   770→            where: and(
   771→              eq(pmBaselineAssumptions.trainId, train.id),
   772→              eq(pmBaselineAssumptions.pmTypeId, pmType.id)
   773→            ),
   774→          });
   775→
   776→          if (existing) {
   777→            result.skipped++;
   778→            continue;
   779→          }
   780→
   781→          // Calculate assumed first PM date
   782→          const intervalMonths =
   783→            pmType.intervalMonths ?? DEFAULT_PM_INTERVALS[pmType.pmCode] ?? 12;
   784→          const assumedFirstPmDate = addMonths(commissioningDate, intervalMonths);
   785→
   786→          // Calculate assumed mileage if interval is known
   787→          let assumedMileage: number | undefined;
   788→          if (pmType.intervalKm && train.currentMileageKm) {
   789→            // Rough estimate based on interval
   790→            assumedMileage = pmType.intervalKm;
   791→          }
   792→
   793→          const assumption = await this.createAssumption(
   794→            {
   795→              trainId: train.id,
   796→              pmTypeId: pmType.id,
   797→              assumptionType: 'INFERRED_FROM_COMMISSIONING',
   798→              assumedLastPmDate: format(assumedFirstPmDate, 'yyyy-MM-dd'),
   799→              assumedLastPmMileage: assumedMileage,
   800→              confidenceLevel: 'LOW',
   801→              inferenceSource: `Inferred from commissioning date ${train.commissioningDate}`,
   802→              inferenceData: {
   803→                commissioningDate: train.commissioningDate,
   804→                intervalMonths,
   805→                calculationMethod: 'commission_date_plus_interval',
   806→              },
   807→              requiresReview: true,
   808→            });
   809→
   810→          result.created++;
   811→          result.assumptions.push(assumption);
   812→        } catch (error) {
   813→          result.errors.push({
   814→            trainId: train.id,
   815→            pmTypeId: pmType.id,
   816→            error: error instanceof Error ? error.message : String(error),
   817→          });
   818→        }
   819→      }
   820→
   821→      this.log.info(
   822→        {
   823→          trainId: train.id,
   824→          created: result.created,
   825→          skipped: result.skipped,
   826→          errors: result.errors.length,
   827→        },
   828→        'Completed commissioning date inference'
   829→      );
   830→
   831→      return result;
   832→    });
   833→  }
   834→
   835→  /**
   836→   * Infer assumptions from a known higher-level PM
   837→   *
   838→   * When a higher PM (like C4) is known, it satisfies and resets counters
   839→   * for lower PMs (C1, C2, C3).
   840→   */
   841→  async inferFromHigherPm(request: InferFromHigherPmRequest): Promise<InferenceResult> {
   842→    return this.traced('inferFromHigherPm', async (span) => {
   843→      span.setAttribute(MmsAttributes.TRAIN_ID, request.trainId);
   844→
   845→      const train = await this.db.query.trains.findFirst({
   846→        where: eq(trains.id, request.trainId),
   847→      });
   848→
   849→      if (!train) {
   850→        throw new NotFoundError('Train not found', {
   851→          entityType: 'train',
   852→          entityId: request.trainId,
   853→        });
   854→      }
   855→
   856→      // Find the known PM type
   857→      const knownPmType = await this.db.query.pmTypes.findFirst({
   858→        where: and(
   859→          eq(pmTypes.pmCode, request.knownPmTypeCode),
   860→          train.fleetId ? eq(pmTypes.fleetId, train.fleetId) : undefined
   861→        ),
   862→      });
   863→
   864→      if (!knownPmType) {
   865→        throw new NotFoundError('PM type not found', {
   866→          entityType: 'pmType',
   867→          entityId: request.knownPmTypeCode,
   868→        });
   869→      }
   870→
   871→      // Get satisfied PM codes from hierarchy
   872→      const satisfiedCodes = PM_HIERARCHY[request.knownPmTypeCode] ?? [];
   873→
   874→      if (satisfiedCodes.length === 0) {
   875→        return {
   876→          created: 0,
   877→          skipped: 0,
   878→          errors: [],
   879→          assumptions: [],
   880→        };
   881→      }
   882→
   883→      // Get PM types for satisfied codes
   884→      const satisfiedPmTypes = await this.db.query.pmTypes.findMany({
   885→        where: and(
   886→          inArray(pmTypes.pmCode, satisfiedCodes),
   887→          train.fleetId ? eq(pmTypes.fleetId, train.fleetId) : undefined
   888→        ),
   889→      });
   890→
   891→      const result: InferenceResult = {
   892→        created: 0,
   893→        skipped: 0,
   894→        errors: [],
   895→        assumptions: [],
   896→      };
   897→
   898→      for (const pmType of satisfiedPmTypes) {
   899→        try {
   900→          // Check if assumption already exists
   901→          const existing = await this.db.query.pmBaselineAssumptions.findFirst({
   902→            where: and(
   903→              eq(pmBaselineAssumptions.trainId, train.id),
   904→              eq(pmBaselineAssumptions.pmTypeId, pmType.id)
   905→            ),
   906→          });
   907→
   908→          if (existing) {
   909→            result.skipped++;
   910→            continue;
   911→          }
   912→
   913→          const assumption = await this.createAssumption(
   914→            {
   915→              trainId: train.id,
   916→              pmTypeId: pmType.id,
   917→              assumptionType: 'INFERRED_FROM_HIGHER_PM',
   918→              assumedLastPmDate: request.knownPmDate,
   919→              assumedLastPmMileage: request.knownPmMileage,
   920→              confidenceLevel: 'HIGH', // High confidence because derived from actual PM
   921→              inferenceSource: `Counters reset by ${request.knownPmTypeCode} on ${request.knownPmDate}`,
   922→              inferenceData: {
   923→                higherPmCode: request.knownPmTypeCode,
   924→                higherPmDate: request.knownPmDate,
   925→                higherPmMileage: request.knownPmMileage,
   926→                satisfiedBy: request.knownPmTypeCode,
   927→              },
   928→              requiresReview: false, // High confidence, no review needed
   929→            });
   930→
   931→          result.created++;
   932→          result.assumptions.push(assumption);
   933→        } catch (error) {
   934→          result.errors.push({
   935→            trainId: train.id,
   936→            pmTypeId: pmType.id,
   937→            error: error instanceof Error ? error.message : String(error),
   938→          });
   939→        }
   940→      }
   941→
   942→      // Also create/update assumption for the known PM itself
   943→      try {
   944→        const existingKnown = await this.db.query.pmBaselineAssumptions.findFirst({
   945→          where: and(
   946→            eq(pmBaselineAssumptions.trainId, train.id),
   947→            eq(pmBaselineAssumptions.pmTypeId, knownPmType.id)
   948→          ),
   949→        });
   950→
   951→        if (!existingKnown) {
   952→          const assumption = await this.createAssumption(
   953→            {
   954→              trainId: train.id,
   955→              pmTypeId: knownPmType.id,
   956→              assumptionType: 'IMPORTED',
   957→              assumedLastPmDate: request.knownPmDate,
   958→              assumedLastPmMileage: request.knownPmMileage,
   959→              confidenceLevel: 'HIGH',
   960→              inferenceSource: 'Imported known PM completion',
   961→              inferenceData: {
   962→                source: 'inferFromHigherPm',
   963→              },
   964→              requiresReview: false,
   965→            });
   966→          result.created++;
   967→          result.assumptions.push(assumption);
   968→        }
   969→      } catch (error) {
   970→        result.errors.push({
   971→          trainId: train.id,
   972→          pmTypeId: knownPmType.id,
   973→          error: error instanceof Error ? error.message : String(error),
   974→        });
   975→      }
   976→
   977→      this.log.info(
   978→        {
   979→          trainId: train.id,
   980→          knownPmCode: request.knownPmTypeCode,
   981→          created: result.created,
   982→          skipped: result.skipped,
   983→          errors: result.errors.length,
   984→        },
   985→        'Completed higher PM inference'
   986→      );
   987→
   988→      return result;
   989→    });
   990→  }

  1029→  /**
  1030→   * Map database result to AssumptionWithDetails
  1031→   */
  1032→  private mapToAssumptionWithDetails(
  1033→    assumption: PmBaselineAssumption & {
  1034→      train?: Train | null;
  1035→      pmType?: PmType | null;
  1036→      reviewedByUser?: { id: string; email: string } | null;
  1037→      appliedByUser?: { id: string; email: string } | null;
  1038→    }
  1039→  ): AssumptionWithDetails {
  1040→    // Determine review status
  1041→    let reviewStatus: ReviewStatus;
  1042→    if (assumption.appliedAt) {
  1043→      reviewStatus = 'APPLIED';
  1044→    } else if (assumption.reviewedAt) {
  1045→      if (assumption.reviewNotes?.startsWith('REJECTED:')) {
  1046→        reviewStatus = 'REJECTED';
  1047→      } else {
  1048→        reviewStatus = 'REVIEWED';
  1049→      }
  1050→    } else {
  1051→      reviewStatus = 'PENDING_REVIEW';
  1052→    }
  1053→
  1054→    return {
  1055→      ...assumption,
  1056→      train: assumption.train
  1057→        ? {
  1058→            id: assumption.train.id,
  1059→            trainNumber: assumption.train.trainNumber,
  1060→            currentMileageKm: assumption.train.currentMileageKm,
  1061→            commissioningDate: assumption.train.commissioningDate,
  1062→          }
  1063→        : ({
  1064→            id: assumption.trainId,
  1065→            trainNumber: 'Unknown',
  1066→            currentMileageKm: 0,
  1067→            commissioningDate: null,
  1068→          } as Pick<Train, 'id' | 'trainNumber' | 'currentMileageKm' | 'commissioningDate'>),
  1069→      pmType: assumption.pmType
  1070→        ? {
  1071→            id: assumption.pmType.id,
  1072→            pmCode: assumption.pmType.pmCode,
  1073→            intervalKm: assumption.pmType.intervalKm,
  1074→            intervalMonths: assumption.pmType.intervalMonths,
  1075→          }
  1076→        : ({
  1077→            id: assumption.pmTypeId,
  1078→            pmCode: 'Unknown',
  1079→            intervalKm: null,
  1080→            intervalMonths: null,
  1081→          } as Pick<PmType, 'id' | 'pmCode' | 'intervalKm' | 'intervalMonths'>),
  1082→      reviewedByUser: assumption.reviewedByUser ?? null,
  1083→      appliedByUser: assumption.appliedByUser ?? null,
  1084→      reviewStatus,
  1085→    };
  1086→  }
