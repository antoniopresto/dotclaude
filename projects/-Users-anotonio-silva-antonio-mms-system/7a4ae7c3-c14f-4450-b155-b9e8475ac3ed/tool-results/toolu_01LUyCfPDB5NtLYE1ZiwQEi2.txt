     1→/**
     2→ * MMS Maximo Polling Service
     3→ *
     4→ * Implements polling fallback mechanism for Maximo integration.
     5→ * When webhooks become unreliable (no delivery for configured timeout),
     6→ * this service automatically activates scheduled polling.
     7→ *
     8→ * Key responsibilities:
     9→ * - Monitor webhook health and detect stale connections
    10→ * - Schedule and manage BullMQ repeatable polling jobs
    11→ * - Deduplicate webhook vs polling updates
    12→ * - Track sync mode transitions (WEBHOOK → POLLING → HYBRID)
    13→ *
    14→ * Sync modes:
    15→ * - WEBHOOK: Primary mode, real-time updates via webhooks
    16→ * - POLLING: Fallback mode, scheduled pulls every 15 minutes
    17→ * - HYBRID: Both active during transition periods
    18→ *
    19→ * @see specs/007-integration.md Section 7.4.7
    20→ */
    21→

   116→/**
   117→ * Maximo Polling Service for fallback synchronization
   118→ *
   119→ * Monitors webhook health and automatically activates polling when
   120→ * webhooks become unreliable. Uses BullMQ repeatable jobs for scheduling.
   121→ *
   122→ * @example
   123→ * ```typescript
   124→ * const pollingService = getMaximoPollingService();
   125→ *
   126→ * // Record successful webhook
   127→ * await pollingService.recordWebhookReceived('wo_status_changed');
   128→ *
   129→ * // Check if polling should be activated
   130→ * const health = await pollingService.getWebhookHealth();
   131→ * if (!health.isHealthy) {
   132→ *   await pollingService.activatePolling();
   133→ * }
   134→ *
   135→ * // Execute polling job
   136→ * const result = await pollingService.executePolling();
   137→ * ```
   138→ */
   139→export class MaximoPollingService extends BaseService {
   140→  private queue: Queue<MaximoSyncJobData>;
   141→  private pollingJobId: string | null = null;
   142→
   143→  constructor() {
   144→    super({ name: 'maximo-polling-service' });
   145→    this.queue = getQueue(QUEUE_NAMES.MAXIMO_SYNC);
   146→  }
   147→
   148→  // ===========================================================================
   149→  // Webhook Health Tracking
   150→  // ===========================================================================
   151→
   152→  /**
   153→   * Record a successful webhook receipt
   154→   *
   155→   * Updates the webhook health record and potentially deactivates polling
   156→   * if the system was in fallback mode.
   157→   */
   158→  async recordWebhookReceived(
   159→    eventType: string,
   160→    depotId?: string
   161→  ): Promise<void> {
   162→    return this.traced('recordWebhookReceived', async (span) => {
   163→      span.setAttribute('webhook.event_type', eventType);
   164→      if (depotId) span.setAttribute(MmsAttributes.DEPOT_ID, depotId);
   165→
   166→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   167→      const now = new Date();
   168→
   169→      await this.db
   170→        .update(maximoWebhookHealth)
   171→        .set({
   172→          lastWebhookReceivedAt: now,
   173→          lastWebhookEventType: eventType,
   174→          webhookConsecutiveFailures: 0,
   175→          totalWebhooksReceived: sql`${maximoWebhookHealth.totalWebhooksReceived} + 1`,
   176→          updatedAt: now,
   177→        })
   178→        .where(eq(maximoWebhookHealth.id, healthRecord.id));
   179→
   180→      // Check if we should transition from POLLING back to WEBHOOK
   181→      if (healthRecord.syncMode === 'POLLING' || healthRecord.syncMode === 'HYBRID') {
   182→        const config = await this.getPollingConfig(depotId);
   183→
   184→        // Enter HYBRID mode for grace period before fully switching back
   185→        if (healthRecord.syncMode === 'POLLING') {
   186→          await this.setSyncMode(healthRecord.id, 'HYBRID', depotId);
   187→          this.log.info(
   188→            { depotId, eventType },
   189→            'Webhook received during polling mode, entering HYBRID mode'
   190→          );
   191→        }
   192→
   193→        // Schedule deactivation after grace period
   194→        setTimeout(async () => {
   195→          const currentHealth = await this.getWebhookHealth(depotId);
   196→          if (currentHealth.isHealthy && currentHealth.syncMode === 'HYBRID') {
   197→            await this.deactivatePolling(depotId);
   198→          }
   199→        }, config.gracePeriodMinutes * 60 * 1000);
   200→      }
   201→
   202→      this.log.debug({ eventType, depotId }, 'Webhook received recorded');
   203→    });
   204→  }
   205→
   206→  /**
   207→   * Record a webhook delivery failure
   208→   */
   209→  async recordWebhookFailure(
   210→    error: string,
   211→    depotId?: string
   212→  ): Promise<void> {
   213→    return this.traced('recordWebhookFailure', async (span) => {
   214→      if (depotId) span.setAttribute(MmsAttributes.DEPOT_ID, depotId);
   215→
   216→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   217→      const now = new Date();
   218→      const newFailureCount = (healthRecord.webhookConsecutiveFailures ?? 0) + 1;
   219→
   220→      await this.db
   221→        .update(maximoWebhookHealth)
   222→        .set({
   223→          webhookConsecutiveFailures: newFailureCount,
   224→          webhookLastFailureAt: now,
   225→          webhookLastError: error,
   226→          updatedAt: now,
   227→        })
   228→        .where(eq(maximoWebhookHealth.id, healthRecord.id));
   229→
   230→      // Check if we should activate polling
   231→      if (newFailureCount >= 3) {
   232→        this.log.warn(
   233→          { depotId, failures: newFailureCount, error },
   234→          'Multiple webhook failures detected, activating polling'
   235→        );
   236→        await this.activatePolling(depotId, 'circuit_breaker');
   237→      }
   238→    });
   239→  }
   240→
   241→  /**
   242→   * Get current webhook health status
   243→   */
   244→  async getWebhookHealth(depotId?: string): Promise<WebhookHealthStatus> {
   245→    return this.traced('getWebhookHealth', async (span) => {
   246→      if (depotId) span.setAttribute(MmsAttributes.DEPOT_ID, depotId);
   247→
   248→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   249→      const config = await this.getPollingConfig(depotId);
   250→
   251→      const now = new Date();
   252→      const lastReceived = healthRecord.lastWebhookReceivedAt;
   253→      let minutesSinceLastWebhook: number | null = null;
   254→      let isHealthy = false;
   255→
   256→      if (lastReceived) {
   257→        minutesSinceLastWebhook = Math.floor(
   258→          (now.getTime() - lastReceived.getTime()) / (60 * 1000)
   259→        );
   260→        isHealthy = minutesSinceLastWebhook < config.webhookTimeoutMinutes;
   261→      }
   262→
   263→      const status: WebhookHealthStatus = {
   264→        isHealthy,
   265→        lastReceivedAt: lastReceived,
   266→        minutesSinceLastWebhook,
   267→        consecutiveFailures: healthRecord.webhookConsecutiveFailures ?? 0,
   268→        syncMode: (healthRecord.syncMode as SyncMode) ?? 'WEBHOOK',
   269→        pollingActive: healthRecord.pollingEnabled ?? false,
   270→      };
   271→
   272→      span.setAttribute('webhook.healthy', isHealthy);
   273→      span.setAttribute('webhook.sync_mode', status.syncMode);
   274→
   275→      return status;
   276→    });
   277→  }
   278→
   279→  /**
   280→   * Check webhook health and activate polling if needed
   281→   *
   282→   * This should be called periodically (e.g., from a scheduled job)
   283→   * to detect webhook timeouts and activate polling automatically.
   284→   */
   285→  async checkAndActivatePollingIfNeeded(depotId?: string): Promise<boolean> {
   286→    return this.traced('checkAndActivatePollingIfNeeded', async (span) => {
   287→      const health = await this.getWebhookHealth(depotId);
   288→
   289→      if (!health.isHealthy && !health.pollingActive) {
   290→        this.log.info(
   291→          {
   292→            depotId,
   293→            minutesSinceLastWebhook: health.minutesSinceLastWebhook,
   294→            consecutiveFailures: health.consecutiveFailures,
   295→          },
   296→          'Webhook timeout detected, activating polling fallback'
   297→        );
   298→
   299→        await this.activatePolling(depotId, 'webhook_timeout');
   300→        span.setAttribute('polling.activated', true);
   301→        return true;
   302→      }
   303→
   304→      span.setAttribute('polling.activated', false);
   305→      return false;
   306→    });
   307→  }

   313→  /**
   314→   * Activate polling fallback
   315→   *
   316→   * Creates a BullMQ repeatable job that executes polling at configured intervals.
   317→   */
   318→  async activatePolling(
   319→    depotId?: string,
   320→    reason: 'webhook_timeout' | 'manual' | 'circuit_breaker' = 'manual'
   321→  ): Promise<void> {
   322→    return this.traced('activatePolling', async (span) => {
   323→      if (depotId) span.setAttribute(MmsAttributes.DEPOT_ID, depotId);
   324→      span.setAttribute('polling.activation_reason', reason);
   325→
   326→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   327→      const config = await this.getPollingConfig(depotId);
   328→
   329→      // Update health record
   330→      await this.db
   331→        .update(maximoWebhookHealth)
   332→        .set({
   333→          pollingEnabled: true,
   334→          pollingActivatedAt: new Date(),
   335→          syncMode: 'POLLING',
   336→          totalFallbackActivations: sql`${maximoWebhookHealth.totalFallbackActivations} + 1`,
   337→          updatedAt: new Date(),
   338→        })
   339→        .where(eq(maximoWebhookHealth.id, healthRecord.id));
   340→
   341→      // Create repeatable job
   342→      const jobKey = depotId ? `${POLLING_JOB_KEY}-${depotId}` : POLLING_JOB_KEY;
   343→
   344→      // Remove existing job if any
   345→      await this.removePollingJob(depotId);
   346→
   347→      // Add new repeatable job
   348→      const job = await this.queue.add(
   349→        jobKey,
   350→        {
   351→          operation: 'polling',
   352→          depotId,
   353→          isPollingFallback: true,
   354→          deduplicateWithWebhooks: true,
   355→        },
   356→        {
   357→          repeat: {
   358→            pattern: `*/${config.intervalMinutes} * * * *`, // Every N minutes
   359→          },
   360→          jobId: jobKey,
   361→        }
   362→      );
   363→
   364→      this.pollingJobId = job.id ?? null;
   365→
   366→      // Record metrics
   367→      recordPollingFallbackActivation(reason);
   368→      updateSyncModeMetric(depotId ?? 'global', 'POLLING');
   369→
   370→      this.log.info(
   371→        {
   372→          depotId,
   373→          reason,
   374→          intervalMinutes: config.intervalMinutes,
   375→          jobId: job.id,
   376→        },
   377→        'Polling fallback activated'
   378→      );
   379→    });
   380→  }
   381→
   382→  /**
   383→   * Deactivate polling fallback
   384→   */
   385→  async deactivatePolling(depotId?: string): Promise<void> {
   386→    return this.traced('deactivatePolling', async (span) => {
   387→      if (depotId) span.setAttribute(MmsAttributes.DEPOT_ID, depotId);
   388→
   389→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   390→
   391→      // Update health record
   392→      await this.db
   393→        .update(maximoWebhookHealth)
   394→        .set({
   395→          pollingEnabled: false,
   396→          syncMode: 'WEBHOOK',
   397→          updatedAt: new Date(),
   398→        })
   399→        .where(eq(maximoWebhookHealth.id, healthRecord.id));
   400→
   401→      // Remove repeatable job
   402→      await this.removePollingJob(depotId);
   403→
   404→      // Update metrics
   405→      updateSyncModeMetric(depotId ?? 'global', 'WEBHOOK');
   406→
   407→      this.log.info({ depotId }, 'Polling fallback deactivated');
   408→    });
   409→  }
   410→
   411→  /**
   412→   * Remove the polling repeatable job
   413→   */
   414→  private async removePollingJob(depotId?: string): Promise<void> {
   415→    const jobKey = depotId ? `${POLLING_JOB_KEY}-${depotId}` : POLLING_JOB_KEY;
   416→
   417→    try {
   418→      // Get all repeatable jobs and remove matching ones
   419→      const repeatableJobs = await this.queue.getRepeatableJobs();
   420→      for (const job of repeatableJobs) {
   421→        if (job.name === jobKey || job.id === jobKey) {
   422→          await this.queue.removeRepeatableByKey(job.key);
   423→          this.log.debug({ jobKey, repeatKey: job.key }, 'Removed polling repeatable job');
   424→        }
   425→      }
   426→    } catch (error) {
   427→      this.log.warn({ error, jobKey }, 'Failed to remove polling job');
   428→    }
   429→  }

   434→
   435→  /**
   436→   * Execute a polling sync operation
   437→   *
   438→   * Pulls changes from Maximo and deduplicates against recent webhooks.
   439→   * Returns detailed results including duplicates prevented.
   440→   */
   441→  async executePolling(depotId?: string): Promise<PollingResult> {
   442→    return this.traced('executePolling', async (span) => {
   443→      if (depotId) span.setAttribute(MmsAttributes.DEPOT_ID, depotId);
   444→
   445→      const startTime = Date.now();
   446→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   447→      const config = await this.getPollingConfig(depotId);
   448→
   449→      // Get last processed timestamp for deduplication
   450→      const lastProcessedAt = healthRecord.lastProcessedChangeTimestamp;
   451→      let duplicatesPrevented = 0;
   452→
   453→      // Execute pull from Maximo
   454→      const maximoService = getMaximoSyncService();
   455→      const pullResult: PullResult = await maximoService.pullFromMaximo();
   456→
   457→      // Deduplicate: check if any pulled records were recently updated by webhooks
   458→      if (lastProcessedAt && config.deduplicationWindowMs > 0) {
   459→        const windowStart = new Date(Date.now() - config.deduplicationWindowMs);
   460→
   461→        // For each updated record, check if it was updated within the deduplication window
   462→        // If so, it was likely already processed by a webhook
   463→        for (const detail of pullResult.details) {
   464→          if (detail.action === 'UPDATED') {
   465→            const shadowRecord = await this.db.query.maximoShadowRecords.findFirst({
   466→              where: eq(maximoShadowRecords.pmScheduleId, detail.pmScheduleId),
   467→            });
   468→
   469→            if (shadowRecord?.updatedAt && shadowRecord.updatedAt > windowStart) {
   470→              duplicatesPrevented++;
   471→              maximoMetrics.duplicatesPrevented.inc();
   472→            }
   473→          }
   474→        }
   475→      }
   476→
   477→      // Update health record with poll results
   478→      const now = new Date();
   479→      await this.db
   480→        .update(maximoWebhookHealth)
   481→        .set({
   482→          lastPollAt: now,
   483→          lastPollResult: {
   484→            processed: pullResult.processed,
   485→            updated: pullResult.updated,
   486→            conflicts: pullResult.conflicts,
   487→            errors: pullResult.errors,
   488→            durationMs: Date.now() - startTime,
   489→          },
   490→          lastProcessedChangeTimestamp: now,
   491→          totalPollsExecuted: sql`${maximoWebhookHealth.totalPollsExecuted} + 1`,
   492→          totalDuplicatesPrevented: sql`${maximoWebhookHealth.totalDuplicatesPrevented} + ${duplicatesPrevented}`,
   493→          pollingConsecutiveFailures: pullResult.errors > 0
   494→            ? sql`${maximoWebhookHealth.pollingConsecutiveFailures} + 1`
   495→            : sql`0`,
   496→          updatedAt: now,
   497→        })
   498→        .where(eq(maximoWebhookHealth.id, healthRecord.id));
   499→
   500→      // Record metrics
   501→      const durationMs = Date.now() - startTime;
   502→      maximoMetrics.pollingDuration.observe(durationMs / 1000);
   503→      maximoMetrics.pollingRecordsFound.inc(pullResult.processed);
   504→
   505→      const result: PollingResult = {
   506→        processed: pullResult.processed,
   507→        updated: pullResult.updated - duplicatesPrevented,
   508→        conflicts: pullResult.conflicts,
   509→        errors: pullResult.errors,
   510→        duplicatesPrevented,
   511→        durationMs,
   512→        syncMode: (healthRecord.syncMode as SyncMode) ?? 'POLLING',
   513→      };
   514→
   515→      span.setAttribute('polling.processed', result.processed);
   516→      span.setAttribute('polling.updated', result.updated);
   517→      span.setAttribute('polling.duplicates_prevented', duplicatesPrevented);
   518→
   519→      this.log.info(
   520→        {
   521→          depotId,
   522→          processed: result.processed,
   523→          updated: result.updated,
   524→          duplicatesPrevented,
   525→          durationMs,
   526→        },
   527→        'Polling sync completed'
   528→      );
   529→
   530→      return result;
   531→    });
   532→  }

   538→  /**
   539→   * Get polling configuration for a depot (or global default)
   540→   */
   541→  async getPollingConfig(depotId?: string): Promise<PollingConfig> {
   542→    return this.traced('getPollingConfig', async () => {
   543→      // Try depot-specific config first
   544→      if (depotId) {
   545→        const depotConfig = await this.db.query.maximoIntegrationConfig.findFirst({
   546→          where: eq(maximoIntegrationConfig.depotId, depotId),
   547→        });
   548→
   549→        if (depotConfig) {
   550→          return {
   551→            enabled: depotConfig.syncEnabled ?? true,
   552→            intervalMinutes: depotConfig.pollIntervalMinutes ?? DEFAULT_POLLING_CONFIG.intervalMinutes,
   553→            webhookTimeoutMinutes: DEFAULT_POLLING_CONFIG.webhookTimeoutMinutes,
   554→            gracePeriodMinutes: DEFAULT_POLLING_CONFIG.gracePeriodMinutes,
   555→            deduplicationWindowMs: DEFAULT_POLLING_CONFIG.deduplicationWindowMs,
   556→          };
   557→        }
   558→      }
   559→
   560→      // Try global config
   561→      const globalConfig = await this.db.query.maximoIntegrationConfig.findFirst({
   562→        where: isNull(maximoIntegrationConfig.depotId),
   563→      });
   564→
   565→      if (globalConfig) {
   566→        return {
   567→          enabled: globalConfig.syncEnabled ?? true,
   568→          intervalMinutes: globalConfig.pollIntervalMinutes ?? DEFAULT_POLLING_CONFIG.intervalMinutes,
   569→          webhookTimeoutMinutes: DEFAULT_POLLING_CONFIG.webhookTimeoutMinutes,
   570→          gracePeriodMinutes: DEFAULT_POLLING_CONFIG.gracePeriodMinutes,
   571→          deduplicationWindowMs: DEFAULT_POLLING_CONFIG.deduplicationWindowMs,
   572→        };
   573→      }
   574→
   575→      return DEFAULT_POLLING_CONFIG;
   576→    });
   577→  }

   579→  /**
   580→   * Update polling configuration
   581→   */
   582→  async updatePollingConfig(
   583→    config: Partial<PollingConfig>,
   584→    depotId?: string
   585→  ): Promise<void> {
   586→    return this.traced('updatePollingConfig', async () => {
   587→      const healthRecord = await this.getOrCreateHealthRecord(depotId);
   588→
   589→      await this.db
   590→        .update(maximoWebhookHealth)
   591→        .set({
   592→          pollingIntervalMinutes: config.intervalMinutes,
   593→          webhookTimeoutMinutes: config.webhookTimeoutMinutes,
   594→          gracePeriodMinutes: config.gracePeriodMinutes,
   595→          deduplicationWindowMs: config.deduplicationWindowMs,
   596→          updatedAt: new Date(),
   597→        })
   598→        .where(eq(maximoWebhookHealth.id, healthRecord.id));
   599→
   600→      // If polling is active, reschedule with new interval
   601→      if (healthRecord.pollingEnabled && config.intervalMinutes) {
   602→        await this.removePollingJob(depotId);
   603→        await this.activatePolling(depotId, 'manual');
   604→      }
   605→    });
   606→  }

   620→  /**
   621→   * Set sync mode
   622→   */
   623→  private async setSyncMode(
   624→    healthRecordId: string,
