     1→/**
     2→ * MMS PM Scheduler Service - Tactical PM Scheduling
     3→ *
     4→ * Schedules preventive maintenance within a 12-week horizon, respecting:
     5→ * - Compliance windows (mileage OR time triggers)
     6→ * - PM hierarchy (C4 satisfies C1/C2/C3)
     7→ * - Track capability requirements
     8→ * - Material constraints (hard 14 days, soft 90 days)
     9→ * - Labor capacity (depot man-hour availability)
    10→ * - Maximo lock status (INPRG work orders are read-only)
    11→ * - Service demand (fleet availability)
    12→ *
    13→ * Key business rules:
    14→ * - Compliance window: ±10% of interval (mileage or months)
    15→ * - PM hierarchy: Higher-level PMs satisfy lower-level requirements
    16→ * - Material hard constraint: BLOCKING parts must be available within 14 days
    17→ * - Material soft constraint: Warning if parts unavailable within 90 days
    18→ * - Labor capacity: PM hours must fit within weekly depot capacity
    19→ * - Track utilization target: 85%
    20→ * - Maximo lock: Cannot modify PM with INPRG work order
    21→ *
    22→ * @see specs/003-database-schema.md Section 3.5 (PM Management)
    23→ * @see specs/004-domain-services.md Section 4.4 (PM Scheduler)
    24→ */

   305→
   306→  /**
   307→   * Calculate compliance window for a PM type and train.
   308→   *
   309→   * The compliance window defines the valid date range for scheduling a PM.
   310→   * It is calculated based on:
   311→   * - Mileage trigger: intervalKm ± tolerance%
   312→   * - Time trigger: intervalMonths ± tolerance%
   313→   * - Whichever comes first determines the due date
   314→   *
   315→   * @param trainId - Train to calculate for
   316→   * @param pmTypeId - PM type to calculate for
   317→   * @returns Compliance window with start/end dates and mileage
   318→   */
   319→  async calculateComplianceWindow(
   320→    trainId: string,
   321→    pmTypeId: string
   322→  ): Promise<PmDueResult> {
   323→    return this.traced('calculateComplianceWindow', async (span) => {
   324→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   325→      span.setAttribute('pm.typeId', pmTypeId);
   326→
   327→      // Get train details
   328→      const train = await this.db.query.trains.findFirst({
   329→        where: eq(trains.id, trainId),
   330→        with: { fleet: true },
   331→      });
   332→
   333→      if (!train) {
   334→        throw new TrainNotFoundError(trainId);
   335→      }
   336→
   337→      span.setAttribute(MmsAttributes.TRAIN_NUMBER, train.trainNumber);
   338→      span.setAttribute(MmsAttributes.FLEET_ID, train.fleetId);
   339→
   340→      // Get PM type details
   341→      const pmType = await this.db.query.pmTypes.findFirst({
   342→        where: and(
   343→          eq(pmTypes.id, pmTypeId),
   344→          eq(pmTypes.active, true)
   345→        ),
   346→      });
   347→
   348→      if (!pmType) {
   349→        throw new NotFoundError(`PM type not found: ${pmTypeId}`, {
   350→          entityType: 'pm_type',
   351→          entityId: pmTypeId,
   352→        });
   353→      }
   354→
   355→      // Get last completed PM of this type for this train
   356→      const lastPm = await this.getLastCompletedPm(trainId, pmTypeId);
   357→
   358→      // Check if a higher-level PM has satisfied this requirement
   359→      const hierarchySatisfied = await this.checkHierarchySatisfaction(
   360→        trainId,
   361→        pmType,
   362→        lastPm?.completedAt ?? null
   363→      );
   364→
   365→      // Use hierarchy satisfaction if more recent
   366→      const effectiveLastPm = hierarchySatisfied?.satisfiedByPm &&
   367→        (!lastPm || new Date(hierarchySatisfied.satisfiedByPm.completedAt ?? 0) > new Date(lastPm.completedAt ?? 0))
   368→        ? hierarchySatisfied.satisfiedByPm
   369→        : lastPm;
   370→
   371→      const lastPmDate = effectiveLastPm?.completedAt
   372→        ? new Date(effectiveLastPm.completedAt).toISOString().split('T')[0]
   373→        : train.commissioningDate ?? null;
   374→      const lastPmMileageKm = effectiveLastPm?.completedMileageKm ?? 0;
   375→
   376→      // Calculate due date and mileage
   377→      const currentMileageKm = train.currentMileageKm ?? 0;
   378→      const today = new Date();
   379→      const todayStr = today.toISOString().split('T')[0];
   380→
   381→      // Get early/late tolerance
   382→      const earlyTolerancePct = Number(pmType.earlyTolerancePct) || DEFAULT_TOLERANCE_PCT;
   383→      const lateTolerancePct = Number(pmType.lateTolerancePct) || DEFAULT_TOLERANCE_PCT;
   384→
   385→      // Calculate mileage trigger
   386→      let dueMileageKm: number | null = null;
   387→      let kmUntilDue: number | null = null;
   388→      let mileageWindowStart: number | null = null;
   389→      let mileageWindowEnd: number | null = null;
   390→
   391→      if (pmType.intervalKm) {
   392→        dueMileageKm = lastPmMileageKm + pmType.intervalKm;
   393→        kmUntilDue = dueMileageKm - currentMileageKm;
   394→
   395→        const toleranceKm = pmType.intervalKm * (earlyTolerancePct / 100);
   396→        mileageWindowStart = dueMileageKm - toleranceKm;
   397→        mileageWindowEnd = dueMileageKm + (pmType.intervalKm * (lateTolerancePct / 100));
   398→      }
   399→
   400→      // Calculate time trigger
   401→      let dueDate: string | null = null;
   402→      let daysUntilDue: number | null = null;
   403→      let dateWindowStart: string | null = null;
   404→      let dateWindowEnd: string | null = null;
   405→
   406→      if (pmType.intervalMonths && lastPmDate) {
   407→        const lastDate = new Date(lastPmDate);
   408→        const dueDateTime = new Date(lastDate);
   409→        dueDateTime.setMonth(dueDateTime.getMonth() + pmType.intervalMonths);
   410→        dueDate = dueDateTime.toISOString().split('T')[0];
   411→        daysUntilDue = Math.ceil((dueDateTime.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   412→
   413→        const toleranceDays = Math.round((pmType.intervalMonths * 30) * (earlyTolerancePct / 100));
   414→        const lateDays = Math.round((pmType.intervalMonths * 30) * (lateTolerancePct / 100));
   415→
   416→        const windowStartDate = new Date(dueDateTime);
   417→        windowStartDate.setDate(windowStartDate.getDate() - toleranceDays);
   418→        dateWindowStart = windowStartDate.toISOString().split('T')[0];
   419→
   420→        const windowEndDate = new Date(dueDateTime);
   421→        windowEndDate.setDate(windowEndDate.getDate() + lateDays);
   422→        dateWindowEnd = windowEndDate.toISOString().split('T')[0];
   423→      }
   424→
   425→      // Determine trigger type (which comes first)
   426→      let triggerType: 'MILEAGE' | 'TIME' | 'BOTH' = 'BOTH';
   427→      let idealDate = todayStr;
   428→      let idealMileageKm = currentMileageKm;
   429→
   430→      if (dueMileageKm !== null && dueDate !== null) {
   431→        // Convert mileage due to date using daily rate
   432→        const dailyKm = await this.getAverageDailyKm(trainId);
   433→        const mileageDueDateMs = today.getTime() + ((kmUntilDue ?? 0) / dailyKm) * 24 * 60 * 60 * 1000;
   434→        const mileageDueDate = new Date(mileageDueDateMs);
   435→
   436→        const timeDueDate = new Date(dueDate);
   437→
   438→        if (mileageDueDate < timeDueDate) {
   439→          triggerType = 'MILEAGE';
   440→          idealDate = mileageDueDate.toISOString().split('T')[0];
   441→          idealMileageKm = dueMileageKm;
   442→        } else if (timeDueDate < mileageDueDate) {
   443→          triggerType = 'TIME';
   444→          idealDate = dueDate;
   445→          // Estimate mileage at time due date
   446→          const daysToTimeDue = Math.ceil((timeDueDate.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   447→          idealMileageKm = currentMileageKm + (daysToTimeDue * dailyKm);
   448→        } else {
   449→          triggerType = 'BOTH';
   450→          idealDate = dueDate;
   451→          idealMileageKm = dueMileageKm;
   452→        }
   453→      } else if (dueMileageKm !== null) {
   454→        triggerType = 'MILEAGE';
   455→        const dailyKm = await this.getAverageDailyKm(trainId);
   456→        const mileageDueDateMs = today.getTime() + ((kmUntilDue ?? 0) / dailyKm) * 24 * 60 * 60 * 1000;
   457→        idealDate = new Date(mileageDueDateMs).toISOString().split('T')[0];
   458→        idealMileageKm = dueMileageKm;
   459→      } else if (dueDate !== null) {
   460→        triggerType = 'TIME';
   461→        idealDate = dueDate;
   462→        const dailyKm = await this.getAverageDailyKm(trainId);
   463→        const daysToTimeDue = Math.ceil((new Date(dueDate).getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   464→        idealMileageKm = currentMileageKm + (daysToTimeDue * dailyKm);
   465→      }
   466→
   467→      // Calculate compliance window dates
   468→      const dailyKm = await this.getAverageDailyKm(trainId);
   469→      let complianceStartDate = dateWindowStart ?? todayStr;
   470→      let complianceEndDate = dateWindowEnd ?? todayStr;
   471→
   472→      if (mileageWindowStart !== null) {
   473→        const kmToStart = mileageWindowStart - currentMileageKm;
   474→        const daysToStart = Math.max(0, Math.ceil(kmToStart / dailyKm));
   475→        const startDateFromMileage = new Date(today);
   476→        startDateFromMileage.setDate(startDateFromMileage.getDate() + daysToStart);
   477→
   478→        if (!dateWindowStart || startDateFromMileage < new Date(dateWindowStart)) {
   479→          complianceStartDate = startDateFromMileage.toISOString().split('T')[0];
   480→        }
   481→      }
   482→
   483→      if (mileageWindowEnd !== null) {
   484→        const kmToEnd = mileageWindowEnd - currentMileageKm;
   485→        const daysToEnd = Math.ceil(kmToEnd / dailyKm);
   486→        const endDateFromMileage = new Date(today);
   487→        endDateFromMileage.setDate(endDateFromMileage.getDate() + daysToEnd);
   488→
   489→        if (!dateWindowEnd || endDateFromMileage < new Date(dateWindowEnd)) {
   490→          complianceEndDate = endDateFromMileage.toISOString().split('T')[0];
   491→        }
   492→      }
   493→
   494→      // Ensure start date is not in the past
   495→      if (complianceStartDate < todayStr) {
   496→        complianceStartDate = todayStr;
   497→      }
   498→
   499→      // Check if overdue
   500→      const isOverdue = complianceEndDate < todayStr;
   501→
   502→      const result: PmDueResult = {
   503→        pmTypeId,
   504→        pmCode: pmType.pmCode,
   505→        pmName: pmType.pmName,
   506→        trainId,
   507→        trainNumber: train.trainNumber,
   508→        fleetId: train.fleetId,
   509→        dueMileageKm,
   510→        dueDate,
   511→        currentMileageKm,
   512→        kmUntilDue,
   513→        daysUntilDue,
   514→        triggerType,
   515→        complianceWindow: {
   516→          startDate: complianceStartDate,
   517→          endDate: complianceEndDate,
   518→          startMileageKm: mileageWindowStart,
   519→          endMileageKm: mileageWindowEnd,
   520→          idealDate,
   521→          idealMileageKm,
   522→        },
   523→        isOverdue,
   524→        lastPmDate,
   525→        lastPmMileageKm,
   526→      };
   527→
   528→      span.setAttribute('pm.triggerType', triggerType);
   529→      span.setAttribute('pm.isOverdue', isOverdue);
   530→      span.setAttribute('pm.kmUntilDue', kmUntilDue ?? -1);
   531→
   532→      return result;
   533→    });
   534→  }

   536→  /**
   537→   * Calculate PM due dates for all PM types for a train.
   538→   */
   539→  async calculateAllPmDue(trainId: string): Promise<PmDueResult[]> {
   540→    return this.traced('calculateAllPmDue', async (span) => {
   541→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   542→
   543→      // Get train with fleet
   544→      const train = await this.db.query.trains.findFirst({
   545→        where: eq(trains.id, trainId),
   546→      });
   547→
   548→      if (!train) {
   549→        throw new TrainNotFoundError(trainId);
   550→      }
   551→
   552→      // Get all active PM types for this fleet
   553→      const pmTypesList = await this.db.query.pmTypes.findMany({
   554→        where: and(
   555→          eq(pmTypes.fleetId, train.fleetId),
   556→          eq(pmTypes.active, true)
   557→        ),
   558→        orderBy: [desc(pmTypes.intervalKm)],
   559→      });
   560→
   561→      const results: PmDueResult[] = [];
   562→
   563→      for (const pmType of pmTypesList) {
   564→        const dueResult = await this.calculateComplianceWindow(trainId, pmType.id);
   565→        results.push(dueResult);
   566→      }
   567→
   568→      // Sort by due date (most urgent first)
   569→      results.sort((a, b) => {
   570→        const aDate = a.complianceWindow.idealDate;
   571→        const bDate = b.complianceWindow.idealDate;
   572→        return aDate.localeCompare(bDate);
   573→      });
   574→
   575→      span.setAttribute('result.pmCount', results.length);
   576→      span.setAttribute('result.overdueCount', results.filter(r => r.isOverdue).length);
   577→
   578→      return results;
   579→    });
   580→  }

   586→  /**
   587→   * Check if a higher-level PM has satisfied this PM type's requirement.
   588→   *
   589→   * PM Hierarchy: C4 satisfies C1/C2/C3, Overhaul satisfies all
   590→   */
   591→  private async checkHierarchySatisfaction(
   592→    trainId: string,
   593→    pmType: PmType,
   594→    sinceDate: Date | null
   595→  ): Promise<{ satisfiedBy: string | null; satisfiedByPm: PmScheduleRecord | null }> {
   596→    // Find PM types that satisfy this one
   597→    const satisfyingTypes = await this.db.query.pmTypes.findMany({
   598→      where: and(
   599→        eq(pmTypes.fleetId, pmType.fleetId),
   600→        eq(pmTypes.active, true),
   601→        sql`${pmType.pmCode} = ANY(${pmTypes.satisfiesCodes})`
   602→      ),
   603→    });
   604→
   605→    if (satisfyingTypes.length === 0) {
   606→      return { satisfiedBy: null, satisfiedByPm: null };
   607→    }
   608→
   609→    // Find the most recent completed PM of a satisfying type
   610→    const satisfyingTypeIds = satisfyingTypes.map(t => t.id);
   611→    const sinceDateFilter = sinceDate
   612→      ? gte(pmSchedule.completedAt, sinceDate)
   613→      : undefined;
   614→
   615→    const satisfyingPm = await this.db.query.pmSchedule.findFirst({
   616→      where: and(
   617→        eq(pmSchedule.trainId, trainId),
   618→        inArray(pmSchedule.pmTypeId, satisfyingTypeIds),
   619→        eq(pmSchedule.status, 'COMPLETED'),
   620→        isNull(pmSchedule.scenarioId),
   621→        sinceDateFilter
   622→      ),
   623→      orderBy: desc(pmSchedule.completedAt),
   624→    });
   625→
   626→    if (satisfyingPm) {
   627→      const satisfyingType = satisfyingTypes.find(t => t.id === satisfyingPm.pmTypeId);
   628→      return {
   629→        satisfiedBy: satisfyingType?.pmCode ?? null,
   630→        satisfiedByPm: satisfyingPm,
   631→      };
   632→    }
   633→
   634→    return { satisfiedBy: null, satisfiedByPm: null };
   635→  }

   669→  /**
   670→   * Find suitable tracks for a PM based on type requirements.
   671→   */
   672→  async findSuitableTracks(
   673→    pmTypeId: string,
   674→    depotId: string | null,
   675→    targetDate: string,
   676→    durationDays: number
   677→  ): Promise<TrackAllocation[]> {
   678→    return this.traced('findSuitableTracks', async (span) => {
   679→      span.setAttribute('pm.typeId', pmTypeId);
   680→      span.setAttribute('pm.targetDate', targetDate);
   681→
   682→      // Get PM type requirements
   683→      const pmType = await this.db.query.pmTypes.findFirst({
   684→        where: eq(pmTypes.id, pmTypeId),
   685→      });
   686→
   687→      if (!pmType) {
   688→        throw new NotFoundError(`PM type not found: ${pmTypeId}`, {
   689→          entityType: 'pm_type',
   690→          entityId: pmTypeId,
   691→        });
   692→      }
   693→
   694→      // Get required capabilities for this PM type
   695→      const requiredCapabilities = await this.db.query.pmCapabilityRequirements.findMany({
   696→        where: eq(pmCapabilityRequirements.pmTypeId, pmTypeId),
   697→      });
   698→
   699→      const mandatoryCapabilities = requiredCapabilities
   700→        .filter(c => c.isMandatory)
   701→        .map(c => c.capability);
   702→
   703→      // Find tracks that match the track type
   704→      const trackConditions: SQL<unknown>[] = [
   705→        eq(tracks.trackType, pmType.trackType),
   706→        eq(tracks.active, true),
   707→      ];
   708→
   709→      if (depotId) {
   710→        trackConditions.push(eq(tracks.depotId, depotId));
   711→      }
   712→
   713→      const candidateTracks = await this.db.query.tracks.findMany({
   714→        where: and(...trackConditions),
   715→        with: {
   716→          depot: true,
   717→        },
   718→      });
   719→
   720→      // Check each track for capabilities and availability
   721→      const allocations: TrackAllocation[] = [];
   722→      const endDate = new Date(targetDate);
   723→      endDate.setDate(endDate.getDate() + durationDays - 1);
   724→      const endDateStr = endDate.toISOString().split('T')[0];
   725→
   726→      for (const track of candidateTracks) {
   727→        // Get track capabilities
   728→        const capabilities = await this.db.query.trackCapabilities.findMany({
   729→          where: and(
   730→            eq(trackCapabilities.trackId, track.id),
   731→            eq(trackCapabilities.isOperational, true)
   732→          ),
   733→        });
   734→
   735→        const trackCapabilitySet = new Set(capabilities.map(c => c.capability));
   736→
   737→        // Check mandatory capabilities
   738→        const missingCapabilities = mandatoryCapabilities.filter(
   739→          cap => !trackCapabilitySet.has(cap)
   740→        );
   741→
   742→        // Check availability (no conflicting PM)
   743→        const conflicts = await this.db.query.pmSchedule.findMany({
   744→          where: and(
   745→            eq(pmSchedule.assignedTrackId, track.id),
   746→            inArray(pmSchedule.status, ['SCHEDULED', 'IN_PROGRESS']),
   747→            isNull(pmSchedule.scenarioId),
   748→            sql`${pmSchedule.targetDate} <= ${endDateStr}`,
   749→            sql`DATE(${pmSchedule.targetDate}) + COALESCE(
   750→              (SELECT min_track_days FROM pm_types WHERE id = ${pmSchedule.pmTypeId}), 1
   751→            ) > ${targetDate}`
   752→          ),
   753→        });
   754→
   755→        if (conflicts.length === 0) {
   756→          allocations.push({
   757→            trackId: track.id,
   758→            trackNumber: track.trackNumber,
   759→            depotId: track.depotId,
   760→            depotCode: track.depot?.code ?? 'unknown',
   761→            trackType: track.trackType as TrackType,
   762→            startDate: targetDate,
   763→            endDate: endDateStr,
   764→            hasAllCapabilities: missingCapabilities.length === 0,
   765→            missingCapabilities,
   766→          });
   767→        }
   768→      }
   769→
   770→      // Sort by capability match (prefer tracks with all capabilities)
   771→      allocations.sort((a, b) => {
   772→        if (a.hasAllCapabilities && !b.hasAllCapabilities) return -1;
   773→        if (!a.hasAllCapabilities && b.hasAllCapabilities) return 1;
   774→        return a.missingCapabilities.length - b.missingCapabilities.length;
   775→      });
   776→
   777→      span.setAttribute('result.trackCount', allocations.length);
   778→      return allocations;
   779→    });
   780→  }

   786→    trackId: string,
   787→    targetDate: string,

   821→  // ===========================================================================
   822→  // Material Constraint Checking
   823→  // ===========================================================================
   824→
   825→  /**
   826→   * Check material availability for a PM.
   827→   *
   828→   * Material constraints:
   829→   * - Hard (14 days): BLOCKING parts must be available
   830→   * - Soft (90 days): Warning if parts unavailable
   831→   */
   832→  async checkMaterialAvailability(
   833→    pmTypeId: string,
   834→    depotId: string,
   835→    targetDate: string
   836→  ): Promise<MaterialConstraintResult> {
   837→    return this.traced('checkMaterialAvailability', async (span) => {
   838→      span.setAttribute('pm.typeId', pmTypeId);
   839→      span.setAttribute('depot.id', depotId);
   840→      span.setAttribute('pm.targetDate', targetDate);
   841→
   842→      // Get required materials for this PM type
   843→      const requirements = await this.db.query.pmMaterialRequirements.findMany({
   844→        where: eq(pmMaterialRequirements.pmTypeId, pmTypeId),
   845→      });
   846→
   847→      if (requirements.length === 0) {
   848→        return {
   849→          isBlocked: false,
   850→          hardConstraintViolations: [],
   851→          softConstraintWarnings: [],
   852→          totalMissingParts: 0,
   853→          reservations: [],
   854→        };
   855→      }
   856→
   857→      const today = new Date();
   858→      const pmDate = new Date(targetDate);
   859→      const daysUntilPm = Math.ceil((pmDate.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   860→
   861→      const hardViolations: MaterialAvailability[] = [];
   862→      const softWarnings: MaterialAvailability[] = [];
   863→      const reservations: Array<{ partNumber: string; quantity: number; type: ReservationType }> = [];
   864→
   865→      for (const req of requirements) {
   866→        // Get inventory at the depot
   867→        const inventory = await this.db.query.inventoryPositions.findFirst({
   868→          where: and(
   869→            eq(inventoryPositions.depotId, depotId),
   870→            eq(inventoryPositions.partNumber, req.partNumber)
   871→          ),
   872→        });
   873→
   874→        const availableQty = inventory
   875→          ? (inventory.quantityOnHand ?? 0) - (inventory.quantityReserved ?? 0)
   876→          : 0;
   877→
   878→        const availability: MaterialAvailability = {
   879→          partNumber: req.partNumber,
   880→          partDescription: req.partDescription,
   881→          criticality: req.criticality as MaterialCriticality,
   882→          requiredQuantity: req.quantityPerPm,
   883→          availableQuantity: availableQty,
   884→          depotId,
   885→          isAvailable: availableQty >= req.quantityPerPm,
   886→          shortfall: Math.max(0, req.quantityPerPm - availableQty),
   887→          leadTimeDays: req.leadTimeDays ?? 14,
   888→          expectedAvailableDate: inventory?.nextDeliveryDate ?? null,
   889→        };
   890→
   891→        if (!availability.isAvailable) {
   892→          // Check if this is a hard or soft constraint violation
   893→          if (req.criticality === 'BLOCKING' && daysUntilPm <= HARD_MATERIAL_WINDOW_DAYS) {
   894→            hardViolations.push(availability);
   895→          } else if (daysUntilPm <= SOFT_MATERIAL_WINDOW_DAYS) {
   896→            softWarnings.push(availability);
   897→          }
   898→        }
   899→
   900→        // Create reservation record
   901→        if (availableQty >= req.quantityPerPm) {
   902→          reservations.push({
   903→            partNumber: req.partNumber,
   904→            quantity: req.quantityPerPm,
   905→            type: daysUntilPm <= HARD_MATERIAL_WINDOW_DAYS ? 'HARD' : 'SOFT',
   906→          });
   907→        }
   908→      }
   909→
   910→      const isBlocked = hardViolations.length > 0;
   911→
   912→      span.setAttribute('material.isBlocked', isBlocked);
   913→      span.setAttribute('material.hardViolations', hardViolations.length);
   914→      span.setAttribute('material.softWarnings', softWarnings.length);
   915→
   916→      return {
   917→        isBlocked,
   918→        hardConstraintViolations: hardViolations,
   919→        softConstraintWarnings: softWarnings,
   920→        totalMissingParts: hardViolations.length + softWarnings.length,
   921→        reservations,
   922→      };
   923→    });
   924→  }

   987→  /**
   988→   * Schedule a new PM.
   989→   */
   990→  async schedulePm(request: SchedulePmRequest): Promise<PmScheduleRecord> {
   991→    return this.inTransaction('schedulePm', async (tx, span) => {
   992→      const {
   993→        trainId,
   994→        pmTypeId,
   995→        targetDate,
   996→        assignedDepotId,
   997→        assignedTrackId,
   998→        assignedShift,
   999→        notes,
  1000→        createdBy,
  1001→        skipMaterialCheck,
  1002→        skipServiceDemandCheck,
  1003→        skipLaborCapacityCheck,
  1004→      } = request;
  1005→
  1006→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
  1007→      span.setAttribute('pm.typeId', pmTypeId);
  1008→      span.setAttribute('pm.targetDate', targetDate);
  1009→
  1010→      // Get train
  1011→      const train = await tx.query.trains.findFirst({
  1012→        where: eq(trains.id, trainId),
  1013→        with: { fleet: true, currentDepot: true },
  1014→      });
  1015→
  1016→      if (!train) {
  1017→        throw new TrainNotFoundError(trainId);
  1018→      }
  1019→
  1020→      span.setAttribute(MmsAttributes.TRAIN_NUMBER, train.trainNumber);
  1021→
  1022→      // Get PM type
  1023→      const pmType = await tx.query.pmTypes.findFirst({
  1024→        where: eq(pmTypes.id, pmTypeId),
  1025→      });
  1026→
  1027→      if (!pmType) {
  1028→        throw new NotFoundError(`PM type not found: ${pmTypeId}`, {
  1029→          entityType: 'pm_type',
  1030→          entityId: pmTypeId,
  1031→        });
  1032→      }
  1033→
  1034→      // Calculate compliance window
  1035→      const dueResult = await this.calculateComplianceWindow(trainId, pmTypeId);
  1036→
  1037→      // Validate target date is within compliance window
  1038→      if (targetDate < dueResult.complianceWindow.startDate ||
  1039→          targetDate > dueResult.complianceWindow.endDate) {
  1040→        throw new ComplianceWindowError(
  1041→          `Target date ${targetDate} is outside compliance window [${dueResult.complianceWindow.startDate}, ${dueResult.complianceWindow.endDate}]`,
  1042→          {
  1043→            windowStart: new Date(dueResult.complianceWindow.startDate),
  1044→            windowEnd: new Date(dueResult.complianceWindow.endDate),
  1045→            targetDate: new Date(targetDate),
  1046→          }
  1047→        );
  1048→      }
  1049→
  1050→      // Check service demand (unless skipped)
  1051→      if (!skipServiceDemandCheck) {
  1052→        const { canSchedule, validation } = await this.serviceDemandService.canScheduleTrainForMaintenance(
  1053→          trainId,
  1054→          targetDate
  1055→        );
  1056→
  1057→        if (!canSchedule) {
  1058→          throw new ServiceDemandViolationError(
  1059→            validation.message,
  1060→            {
  1061→              fleetId: validation.fleetId,
  1062→              requiredTrains: validation.required,
  1063→              availableTrains: validation.availableTrains,
  1064→            }
  1065→          );
  1066→        }
  1067→      }
  1068→
  1069→      // Check unavailability
  1070→      const activeUnavailability = await this.unavailabilityService.getActiveEventsForTrain(trainId);
  1071→      const blockingEvent = activeUnavailability.find(e => e.blocksPmScheduling);
  1072→
  1073→      if (blockingEvent) {
  1074→        throw new BusinessRuleError(
  1075→          `Train ${train.trainNumber} has active unavailability blocking PM scheduling: ${blockingEvent.eventType}`,
  1076→          {
  1077→            rule: 'unavailability_blocks_pm',
  1078→            context: { eventId: blockingEvent.id, eventType: blockingEvent.eventType },
  1079→          }
  1080→        );
  1081→      }
  1082→
  1083→      // Determine depot
  1084→      const depotId = assignedDepotId ?? train.currentDepotId ?? train.homeDepotId;
  1085→      if (!depotId) {
  1086→        throw new ValidationError('No depot assigned and train has no current/home depot');
  1087→      }
  1088→
  1089→      // Check labor capacity (unless skipped)
  1090→      // This validates that the depot has enough man-hours available for the target week
  1091→      let laborCapacityResult: LaborCapacityResult | null = null;
  1092→      if (!skipLaborCapacityCheck && pmType.estimatedHours) {
  1093→        const estimatedHours = parseFloat(pmType.estimatedHours);
  1094→        if (estimatedHours > 0) {
  1095→          laborCapacityResult = await this.manHourLevelingService.checkLaborCapacity(
  1096→            depotId,
  1097→            targetDate,
  1098→            estimatedHours
  1099→          );
  1100→
  1101→          if (!laborCapacityResult.hasCapacity) {
  1102→            throw new LaborCapacityError(
  1103→              laborCapacityResult.message,
  1104→              {
  1105→                depotId,
  1106→                requiredHours: laborCapacityResult.totalRequiredHours,
  1107→                availableHours: laborCapacityResult.availableHours.total,
  1108→                utilizationPct: laborCapacityResult.utilizationPct,
  1109→                weekStart: laborCapacityResult.weekStart,
  1110→              }
  1111→            );
  1112→          }
  1113→        }
  1114→      }
  1115→
  1116→      // Check material availability (unless skipped)
  1117→      let materialResult: MaterialConstraintResult | null = null;
  1118→      if (!skipMaterialCheck) {
  1119→        materialResult = await this.checkMaterialAvailability(pmTypeId, depotId, targetDate);
  1120→
  1121→        if (materialResult.isBlocked) {
  1122→          const blockingParts = materialResult.hardConstraintViolations
  1123→            .map(v => v.partNumber)
  1124→            .join(', ');
  1125→
  1126→          throw new MaterialConstraintError(
  1127→            `Blocking materials not available: ${blockingParts}`,
  1128→            {
  1129→              partNumber: materialResult.hardConstraintViolations[0]?.partNumber,
  1130→              requiredQuantity: materialResult.hardConstraintViolations[0]?.requiredQuantity,
  1131→              availableQuantity: materialResult.hardConstraintViolations[0]?.availableQuantity,
  1132→            }
  1133→          );
  1134→        }
  1135→      }
  1136→
  1137→      // Find or validate track
  1138→      let trackId = assignedTrackId;
  1139→      if (!trackId) {
  1140→        const trackDuration = pmType.minTrackDays ?? 1;
  1141→        const suitableTracks = await this.findSuitableTracks(
  1142→          pmTypeId,
  1143→          depotId,
  1144→          targetDate,
  1145→          trackDuration
  1146→        );
  1147→
  1148→        if (suitableTracks.length > 0) {
  1149→          trackId = suitableTracks[0].trackId;
  1150→        }
  1151→      } else {
  1152→        // Validate provided track
  1153→        const conflict = await this.checkTrackConflict(
  1154→          trackId,
  1155→          targetDate,
  1156→          pmType.minTrackDays ?? 1
  1157→        );
  1158→
  1159→        if (conflict) {
  1160→          throw new TrackConflictError(
  1161→            `Track is already occupied on ${targetDate}`,
  1162→            {
  1163→              trackId,
  1164→              conflictingScheduleId: conflict.id,
  1165→            }
  1166→          );
  1167→        }
  1168→      }
  1169→
  1170→      // Estimate mileage at target date
  1171→      const dailyKm = await this.getAverageDailyKm(trainId);
  1172→      const today = new Date();
  1173→      const daysUntilTarget = Math.ceil(
  1174→        (new Date(targetDate).getTime() - today.getTime()) / (1000 * 60 * 60 * 24)
  1175→      );
  1176→      const targetMileageKm = (train.currentMileageKm ?? 0) + (daysUntilTarget * dailyKm);
  1177→
  1178→      // Create PM schedule record
  1179→      const pmData: NewPmScheduleRecord = {
  1180→        trainId,
  1181→        pmTypeId,
  1182→        targetDate,
  1183→        targetMileageKm: Math.round(targetMileageKm),
  1184→        status: 'SCHEDULED',
  1185→        constraintStart: dueResult.complianceWindow.startDate,
  1186→        constraintEnd: dueResult.complianceWindow.endDate,
  1187→        assignedDepotId: depotId,
  1188→        assignedTrackId: trackId ?? null,
  1189→        assignedHours: pmType.estimatedHours,
  1190→        assignedShift: assignedShift ?? null,
  1191→        createdBy: createdBy ?? null,
  1192→      };
  1193→
  1194→      const [created] = await tx.insert(pmSchedule).values(pmData).returning();
  1195→      span.setAttribute('pm.scheduleId', created.id);
  1196→
  1197→      // Create audit history record
  1198→      if (createdBy) {

  1346→          trackId,
  1347→          newTargetDate,
  1348→          existingPm.pmType?.minTrackDays ?? 1,
  1349→          pmScheduleId
  1350→        );
  1351→

  1363→              trackId = alternatives[0].trackId;

  1365→              trackId = null; // Clear track assignment
  1366→            }

  1369→              `Track is already occupied on ${newTargetDate}`,
  1370→              {
  1371→                trackId,

  1383→        (new Date(newTargetDate).getTime() - today.getTime()) / (1000 * 60 * 60 * 24)
  1384→      );

  1387→

  1390→      const oldTrackId = existingPm.assignedTrackId;

  1394→        .update(pmSchedule)

  1396→          targetDate: newTargetDate,
  1397→          targetMileageKm: Math.round(newTargetMileageKm),
  1398→          assignedTrackId: trackId,
  1399→          updatedAt: new Date(),
  1400→        })
  1401→        .where(eq(pmSchedule.id, pmScheduleId))
  1402→        .returning();
  1403→
  1404→      // Create audit history record
  1405→      const historyRecord: NewPmScheduleHistory = {
  1406→        pmScheduleId,
  1407→        changeType: 'DATE_CHANGED',
  1408→        oldTargetDate,
  1409→        newTargetDate,
  1410→        oldAssignedTrackId: oldTrackId,
  1411→        newAssignedTrackId: trackId,
  1412→        changedBy: movedBy,
  1413→        changeReason: reason,
  1414→        changeSource: 'USER',
  1415→      };
  1416→
  1417→      await tx.insert(pmScheduleHistory).values(historyRecord);
  1418→
  1419→      const trainNumber = existingPm.train?.trainNumber ?? 'unknown';
  1420→      const pmCode = existingPm.pmType?.pmCode ?? 'unknown';
  1421→
  1422→      this.log.info(
  1423→        {
  1424→          pmScheduleId,
  1425→          trainNumber,
  1426→          pmCode,
  1427→          oldTargetDate,
  1428→          newTargetDate,
  1429→          reason,
  1430→          movedBy,
  1431→        },

  1436→    });
  1437→  }

  1564→
  1565→  /**
  1566→   * Run tactical scheduling for the 12-week horizon.
  1567→   *
  1568→   * This method:
  1569→   * 1. Calculates PM due dates for all trains in scope
  1570→   * 2. Prioritizes by overdue status and due date
  1571→   * 3. Attempts to schedule each PM respecting constraints
  1572→   * 4. Returns a detailed result with scheduled, deferred, and blocked PMs
  1573→   */
  1574→  async runTacticalSchedule(request: TacticalScheduleRequest): Promise<SchedulingResult> {
  1575→    return this.traced('runTacticalSchedule', async (span) => {
  1576→      const {
  1577→        fleetId,
  1578→        depotId,
  1579→        horizonWeeks = DEFAULT_HORIZON_WEEKS,
  1580→        dryRun = false,
  1581→        respectLocks = true,
  1582→      } = request;
  1583→
  1584→      span.setAttribute('schedule.horizonWeeks', horizonWeeks);
  1585→      span.setAttribute('schedule.dryRun', dryRun);
  1586→
  1587→      const today = new Date();
  1588→      const horizonStart = today.toISOString().split('T')[0];
  1589→      const horizonEnd = new Date(today.getTime() + horizonWeeks * 7 * 24 * 60 * 60 * 1000)
  1590→        .toISOString()
  1591→        .split('T')[0];
  1592→
  1593→      // Get trains in scope
  1594→      const trainConditions: SQL<unknown>[] = [
  1595→        eq(trains.active, true),
  1596→        eq(trains.fitnessStatus, 'FIT'),
  1597→      ];
  1598→
  1599→      if (fleetId) {
  1600→        trainConditions.push(eq(trains.fleetId, fleetId));
  1601→      }
  1602→      if (depotId) {
  1603→        trainConditions.push(eq(trains.currentDepotId, depotId));
  1604→      }
  1605→
  1606→      const trainsInScope = await this.db.query.trains.findMany({
  1607→        where: and(...trainConditions),
  1608→        with: { fleet: true },
  1609→      });
  1610→
  1611→      span.setAttribute('schedule.trainsInScope', trainsInScope.length);
  1612→
  1613→      // Calculate PM due for all trains
  1614→      const allDueResults: PmDueResult[] = [];
  1615→
  1616→      for (const train of trainsInScope) {
  1617→        const dueResults = await this.calculateAllPmDue(train.id);
  1618→        allDueResults.push(...dueResults);
  1619→      }
  1620→
  1621→      // Filter to PMs due within horizon
  1622→      const pmsDueInHorizon = allDueResults.filter(pm =>
  1623→        pm.complianceWindow.startDate <= horizonEnd &&
  1624→        pm.complianceWindow.endDate >= horizonStart
  1625→      );
  1626→
  1627→      // Sort by priority (overdue first, then by due date)
  1628→      pmsDueInHorizon.sort((a, b) => {
  1629→        if (a.isOverdue && !b.isOverdue) return -1;
  1630→        if (!a.isOverdue && b.isOverdue) return 1;
  1631→        return a.complianceWindow.idealDate.localeCompare(b.complianceWindow.idealDate);
  1632→      });
  1633→
  1634→      span.setAttribute('schedule.pmsDueInHorizon', pmsDueInHorizon.length);
  1635→
  1636→      // Process each PM
  1637→      const scheduled: SchedulingResultItem[] = [];
  1638→      const deferred: SchedulingResultItem[] = [];
  1639→      const blocked: SchedulingResultItem[] = [];
  1640→      const materialWarnings: SchedulingResult['materialWarnings'] = [];
  1641→
  1642→      for (const pmDue of pmsDueInHorizon) {
  1643→        // Check if already scheduled
  1644→        const existingPm = await this.db.query.pmSchedule.findFirst({
  1645→          where: and(
  1646→            eq(pmSchedule.trainId, pmDue.trainId),
  1647→            eq(pmSchedule.pmTypeId, pmDue.pmTypeId),
  1648→            inArray(pmSchedule.status, ['SCHEDULED', 'IN_PROGRESS']),
  1649→            isNull(pmSchedule.scenarioId),
  1650→            gte(pmSchedule.targetDate, horizonStart),
  1651→            lte(pmSchedule.targetDate, horizonEnd)
  1652→          ),
  1653→        });
  1654→
  1655→        if (existingPm) {
  1656→          // Skip if locked and we're respecting locks
  1657→          if (respectLocks && existingPm.isLocked) {
  1658→            continue;
  1659→          }
  1660→        }
  1661→
  1662→        // Try to schedule
  1663→        const result = await this.attemptSchedulePm(pmDue, horizonEnd, dryRun);
  1664→
  1665→        if (result.status === 'SCHEDULED') {
  1666→          scheduled.push(result);
  1667→
  1668→          if (result.materialConstraint?.softConstraintWarnings.length) {
  1669→            materialWarnings.push({
  1670→              trainId: pmDue.trainId,
  1671→              trainNumber: pmDue.trainNumber,
  1672→              pmCode: pmDue.pmCode,
  1673→              targetDate: result.targetDate!,
  1674→              warnings: result.materialConstraint.softConstraintWarnings,
  1675→            });
  1676→          }
  1677→        } else if (result.status === 'DEFERRED') {
  1678→          deferred.push(result);
  1679→        } else {
  1680→          blocked.push(result);
  1681→        }
  1682→      }
  1683→
  1684→      // Calculate track utilization
  1685→      const trackUtilization = await this.calculateTrackUtilization(
  1686→        horizonStart,
  1687→        horizonEnd,
  1688→        depotId ?? null
  1689→      );
  1690→
  1691→      const summary = {
  1692→        totalProcessed: pmsDueInHorizon.length,
  1693→        totalScheduled: scheduled.length,
  1694→        totalDeferred: deferred.length,
  1695→        totalBlocked: blocked.length,
  1696→        trackUtilizationPct: trackUtilization,
  1697→      };
  1698→
  1699→      span.setAttribute('result.scheduled', scheduled.length);
  1700→      span.setAttribute('result.deferred', deferred.length);
  1701→      span.setAttribute('result.blocked', blocked.length);
  1702→      span.setAttribute('result.trackUtilization', trackUtilization);
  1703→
  1704→      this.log.info(
  1705→        {
  1706→          horizonStart,
  1707→          horizonEnd,
  1708→          ...summary,
  1709→          fleetId,
  1710→          depotId,
  1711→        },
  1712→        `Tactical scheduling complete: ${scheduled.length} scheduled, ${deferred.length} deferred, ${blocked.length} blocked`
  1713→      );
  1714→
  1715→      return {
  1716→        horizonStart,
  1717→        horizonEnd,
  1718→        scheduled,
  1719→        deferred,
  1720→        blocked,
  1721→        materialWarnings,
  1722→        summary,
  1723→      };
  1724→    });
  1725→  }

  1727→  /**
  1728→   * Attempt to schedule a single PM within constraints.
  1729→   */
  1730→  private async attemptSchedulePm(
  1731→    pmDue: PmDueResult,
  1732→    horizonEnd: string,
  1733→    dryRun: boolean
  1734→  ): Promise<SchedulingResultItem> {
  1735→    const explanation: SchedulingExplanation = {
  1736→      mileageTrigger: {
  1737→        intervalKm: null,
  1738→        currentMileageKm: pmDue.currentMileageKm,
  1739→        lastPmMileageKm: pmDue.lastPmMileageKm,
  1740→        kmSinceLastPm: pmDue.lastPmMileageKm !== null
  1741→          ? pmDue.currentMileageKm - pmDue.lastPmMileageKm
  1742→          : null,
  1743→        kmUntilDue: pmDue.kmUntilDue,
  1744→      },
  1745→      timeTrigger: {
  1746→        intervalMonths: null,
  1747→        lastPmDate: pmDue.lastPmDate,
  1748→        daysSinceLastPm: pmDue.lastPmDate
  1749→          ? Math.floor((Date.now() - new Date(pmDue.lastPmDate).getTime()) / (1000 * 60 * 60 * 24))
  1750→          : null,
  1751→        daysUntilDue: pmDue.daysUntilDue,
  1752→      },
  1753→      hierarchySatisfied: {
  1754→        satisfiedBy: null,
  1755→        satisfiedOnDate: null,
  1756→        counterResetFrom: null,
  1757→      },
  1758→      trackSelection: {
  1759→        candidateTracks: 0,
  1760→        selectedTrackId: null,
  1761→        selectionReason: '',
  1762→      },
  1763→      materialCheck: {
  1764→        totalParts: 0,
  1765→        availableParts: 0,
  1766→        blockedParts: 0,
  1767→        warningParts: 0,
  1768→      },
  1769→      laborCapacityCheck: {
  1770→        estimatedHours: 0,
  1771→        weeklyCapacity: 0,
  1772→        currentUtilization: 0,
  1773→        utilizationAfterScheduling: 0,
  1774→        hasCapacity: true,
  1775→      },
  1776→      constraints: [],
  1777→    };
  1778→
  1779→    const warnings: string[] = [];
  1780→
  1781→    // Check unavailability
  1782→    const activeUnavailability = await this.unavailabilityService.getActiveEventsForTrain(pmDue.trainId);
  1783→    const blockingEvent = activeUnavailability.find(e => e.blocksPmScheduling);
  1784→
  1785→    if (blockingEvent) {
  1786→      explanation.constraints.push(`Blocked by unavailability: ${blockingEvent.eventType}`);
  1787→
  1788→      return {
  1789→        trainId: pmDue.trainId,
  1790→        trainNumber: pmDue.trainNumber,
  1791→        pmTypeId: pmDue.pmTypeId,
  1792→        pmCode: pmDue.pmCode,
  1793→        status: 'BLOCKED',
  1794→        targetDate: null,
  1795→        targetMileageKm: null,
  1796→        trackAllocation: null,
  1797→        materialConstraint: null,

  2161→  private async calculateTrackUtilization(

  2163→    endDate: string,
  2164→    depotId: string | null

  2189→    // Get scheduled PM track-days

  2203→      where: and(...pmConditions),

  2208→      const duration = pm.pmType?.minTrackDays ?? 1;
