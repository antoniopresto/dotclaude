     1→/**
     2→ * MMS Baseline Service
     3→ *
     4→ * Manages published plan baselines for drift analysis and scenario comparison.
     5→ * Baselines are immutable snapshots of the PM schedule and mileage projections
     6→ * at a point in time.
     7→ *
     8→ * Key responsibilities:
     9→ * - Create and publish plan baselines
    10→ * - Snapshot PM schedule and mileage projections
    11→ * - Version management with semantic versioning
    12→ * - Supersede previous baselines on new publication
    13→ * - Audit logging for compliance
    14→ *
    15→ * Baseline lifecycle:
    16→ * DRAFT -> PUBLISHED -> SUPERSEDED (when new baseline published)
    17→ *
    18→ * @see specs/004-domain-services.md Section 4.7
    19→ */

   161→/**
   162→ * Baseline Service for managing plan baselines
   163→ *
   164→ * @example
   165→ * ```typescript
   166→ * const baselineService = getBaselineService();
   167→ *
   168→ * // Publish a new baseline
   169→ * const baseline = await baselineService.publishBaseline({

   178→ * // Get PM schedule snapshot
   179→ * const pmSchedule = await baselineService.getBaselinePmSchedule(baseline.id);
   180→ * ```
   181→ */
   182→export class BaselineService extends BaseService {
   183→  /** Default projection horizon in years */
   184→  private readonly DEFAULT_HORIZON_YEARS = 10;
   185→
   186→  /** Default daily mileage when no history available (km) */
   187→  private readonly DEFAULT_DAILY_KM = 150;
   188→
   189→  /** Days of mileage history to use for average calculation */

   193→    super({ name: 'baseline-service' });

   199→
   200→  /**
   201→   * Publish a new baseline from the current plan state
   202→   *
   203→   * This operation:
   204→   * 1. Validates the current plan for critical conflicts
   205→   * 2. Generates a version number
   206→   * 3. Creates the baseline record
   207→   * 4. Snapshots the PM schedule (SCHEDULED and IN_PROGRESS items)
   208→   * 5. Generates mileage projections at monthly intervals
   209→   * 6. Supersedes any previous published baseline
   210→   * 7. Records audit log
   211→   *
   212→   * @throws ValidationError if plan has critical conflicts
   213→   * @throws BusinessRuleError if baseline cannot be created
   214→   */
   215→  async publishBaseline(request: PublishBaselineRequest): Promise<PlanBaseline> {
   216→    return this.traced('publishBaseline', async (span) => {
   217→      span.setAttribute('baseline.name', request.name);
   218→      span.setAttribute(MmsAttributes.USER_ID, request.userId);
   219→
   220→      // 1. Validate current plan
   221→      const validation = await this.validatePlanForPublishing();
   222→      if (!validation.valid) {
   223→        this.log.warn(
   224→          { conflicts: validation.conflicts },
   225→          'Cannot publish baseline due to critical conflicts'
   226→        );
   227→        throw new ValidationError(
   228→          'Cannot publish baseline with critical conflicts',
   229→          { context: { conflicts: validation.conflicts } }
   230→        );
   231→      }
   232→
   233→      // Log warnings but continue
   234→      if (validation.warnings.length > 0) {
   235→        this.log.info(
   236→          { warnings: validation.warnings },
   237→          `Publishing baseline with ${validation.warnings.length} warnings`
   238→        );
   239→      }
   240→
   241→      // 2. Get latest baseline for version increment
   242→      const latestBaseline = await this.getLatestBaseline();
   243→      const version = this.incrementVersion(latestBaseline?.version ?? 'v0.0');
   244→      span.setAttribute('baseline.version', version);
   245→
   246→      const horizonYears = request.horizonYears ?? this.DEFAULT_HORIZON_YEARS;
   247→      const baselineDate = new Date();
   248→      const horizonEnd = addYears(baselineDate, horizonYears);
   249→
   250→      // 3. Create baseline in transaction
   251→      const baseline = await this.inTransaction(
   252→        'publishBaseline',
   253→        async (tx, txSpan) => {
   254→          // Get metrics for the baseline
   255→          const fleetCount = await this.getActiveTrainCount();
   256→          const pmCount = await this.getScheduledPMCount();
   257→          const manhours = await this.getTotalProjectedHours();
   258→
   259→          txSpan.setAttribute('baseline.fleetCount', fleetCount);
   260→          txSpan.setAttribute('baseline.pmCount', pmCount);
   261→
   262→          // Create baseline record
   263→          const [baselineRecord] = await tx
   264→            .insert(planBaselines)
   265→            .values({
   266→              name: request.name,
   267→              description: request.description ?? null,
   268→              version,
   269→              baselineDate: format(baselineDate, 'yyyy-MM-dd'),
   270→              horizonEnd: format(horizonEnd, 'yyyy-MM-dd'),
   271→              status: 'PUBLISHED',
   272→              createdBy: request.userId,
   273→              publishedBy: request.userId,
   274→              publishedAt: new Date(),
   275→              fleetCount,
   276→              totalPmScheduled: pmCount,
   277→              totalManhourProjected: manhours.toString(),
   278→            })
   279→            .returning();
   280→
   281→          this.log.info(
   282→            { baselineId: baselineRecord.id, version },
   283→            'Created baseline record'
   284→          );
   285→
   286→          // Snapshot PM schedule
   287→          const pmSnapshotCount = await this.snapshotPmSchedule(tx, baselineRecord.id);
   288→          txSpan.setAttribute('baseline.pmSnapshotCount', pmSnapshotCount);
   289→
   290→          // Generate and store mileage projections
   291→          const projections = await this.generateMileageProjections(horizonEnd);
   292→
   293→          if (projections.length > 0) {
   294→            // Batch insert projections for performance
   295→            const BATCH_SIZE = 500;
   296→            for (let i = 0; i < projections.length; i += BATCH_SIZE) {
   297→              const batch = projections.slice(i, i + BATCH_SIZE);
   298→              await tx.insert(baselineMileageProjection).values(
   299→                batch.map((proj) => ({
   300→                  baselineId: baselineRecord.id,
   301→                  trainId: proj.trainId,
   302→                  projectionDate: format(proj.date, 'yyyy-MM-dd'),
   303→                  projectedCumulativeKm: proj.projectedKm,
   304→                  dailyRateKm: proj.dailyRate,
   305→                }))
   306→              );
   307→            }
   308→          }
   309→
   310→          txSpan.setAttribute('baseline.mileageProjections', projections.length);
   311→
   312→          // Supersede previous baseline
   313→          if (latestBaseline) {
   314→            await tx
   315→              .update(planBaselines)
   316→              .set({
   317→                status: 'SUPERSEDED',
   318→                supersededBy: baselineRecord.id,
   319→                supersededAt: new Date(),
   320→              })
   321→              .where(eq(planBaselines.id, latestBaseline.id));
   322→
   323→            this.log.info(
   324→              { supersededId: latestBaseline.id, newId: baselineRecord.id },
   325→              'Superseded previous baseline'
   326→            );
   327→          }
   328→
   329→          // Record audit log
   330→          await tx.insert(auditLog).values({
   331→            entityType: 'plan_baseline',
   332→            entityId: baselineRecord.id,
   333→            action: 'PUBLISH',
   334→            userId: request.userId,
   335→            newValues: {
   336→              name: request.name,
   337→              version,
   338→              description: request.description,
   339→              fleetCount,
   340→              pmCount,
   341→            },
   342→            complianceFlags: ['PLANNING'],
   343→          });
   344→
   345→          return baselineRecord;
   346→        }
   347→      );
   348→
   349→      this.log.info(
   350→        { baselineId: baseline.id, version: baseline.version },
   351→        'Published new baseline successfully'
   352→      );
   353→
   354→      return baseline;
   355→    });
   356→  }

   407→  /**
   408→   * List baselines with optional filtering and pagination
   409→   */
   410→  async listBaselines(
   411→    filters?: BaselineListFilters,
   412→    pagination?: BaselinePaginationOptions
   413→  ): Promise<PaginatedBaselineResult> {
   414→    return this.traced('listBaselines', async (span) => {
   415→      const page = pagination?.page ?? 1;
   416→      const pageSize = pagination?.pageSize ?? 20;
   417→      const offset = (page - 1) * pageSize;
   418→
   419→      // Build where conditions
   420→      const conditions: ReturnType<typeof eq>[] = [];
   421→
   422→      if (filters?.status) {
   423→        const statuses = Array.isArray(filters.status)
   424→          ? filters.status
   425→          : [filters.status];
   426→        conditions.push(inArray(planBaselines.status, statuses));
   427→      }
   428→
   429→      if (filters?.fromDate) {
   430→        conditions.push(
   431→          gte(planBaselines.baselineDate, format(filters.fromDate, 'yyyy-MM-dd'))
   432→        );
   433→      }
   434→
   435→      if (filters?.createdBy) {
   436→        conditions.push(eq(planBaselines.createdBy, filters.createdBy));
   437→      }
   438→
   439→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   440→
   441→      // Get total count
   442→      const [{ totalCount }] = await this.db
   443→        .select({ totalCount: count() })
   444→        .from(planBaselines)
   445→        .where(whereClause);
   446→
   447→      // Determine sort order
   448→      const sortColumn =
   449→        pagination?.sortBy === 'baselineDate'
   450→          ? planBaselines.baselineDate
   451→          : pagination?.sortBy === 'version'
   452→          ? planBaselines.version
   453→          : planBaselines.createdAt;
   454→
   455→      // Get paginated results
   456→      const data = await this.db
   457→        .select()
   458→        .from(planBaselines)
   459→        .where(whereClause)
   460→        .orderBy(pagination?.sortOrder === 'asc' ? sortColumn : desc(sortColumn))
   461→        .limit(pageSize)
   462→        .offset(offset);
   463→
   464→      span.setAttribute('baseline.resultCount', data.length);
   465→      span.setAttribute('baseline.totalCount', totalCount);
   466→
   467→      return {
   468→        data,
   469→        total: totalCount,
   470→        page,
   471→        pageSize,
   472→        totalPages: Math.ceil(totalCount / pageSize),
   473→      };
   474→    });
   475→  }

   593→  /**
   594→   * Archive a superseded baseline
   595→   *
   596→   * Archived baselines are retained for audit purposes but excluded
   597→   * from normal queries.
   598→   */
   599→  async archiveBaseline(baselineId: string, userId: string): Promise<PlanBaseline> {
   600→    return this.traced('archiveBaseline', async (span) => {
   601→      span.setAttribute(MmsAttributes.BASELINE_ID, baselineId);
   602→      span.setAttribute(MmsAttributes.USER_ID, userId);
   603→
   604→      const baseline = await this.getBaselineById(baselineId);
   605→
   606→      if (baseline.status === 'PUBLISHED') {
   607→        throw new BusinessRuleError(
   608→          'Cannot archive published baseline. Publish a new baseline first to supersede it.'
   609→        );
   610→      }
   611→
   612→      if (baseline.status === 'ARCHIVED') {
   613→        return baseline; // Already archived
   614→      }
   615→
   616→      const [updated] = await this.db
   617→        .update(planBaselines)
   618→        .set({ status: 'ARCHIVED' })
   619→        .where(eq(planBaselines.id, baselineId))
   620→        .returning();
   621→
   622→      // Audit log
   623→      await this.db.insert(auditLog).values({
   624→        entityType: 'plan_baseline',
   625→        entityId: baselineId,
   626→        action: 'UPDATE',
   627→        userId,
   628→        oldValues: { status: baseline.status },
   629→        newValues: { status: 'ARCHIVED' },
   630→      });
   631→
   632→      this.log.info({ baselineId }, 'Archived baseline');
   633→
   634→      return updated;
   635→    });
   636→  }

   642→  /**
   643→   * Validate the current plan for publishing
   644→   *
   645→   * Checks for critical conflicts that prevent baseline publication:
   646→   * - Overdue PMs (past compliance window)
   647→   * - Track conflicts (double-booked tracks)
   648→   * - Material blocks (missing critical materials)
   649→   * - Grounded trains with scheduled PMs
   650→   */
   651→  private async validatePlanForPublishing(): Promise<PlanValidationResult> {
   652→    return this.traced('validatePlanForPublishing', async (span) => {
   653→      const conflicts: PlanConflict[] = [];
   654→      const warnings: PlanWarning[] = [];
   655→      const today = new Date();
   656→      const todayStr = format(today, 'yyyy-MM-dd');
   657→
   658→      // Check for overdue PMs (constraint_end < today and status = SCHEDULED)
   659→      const overduePms = await this.db
   660→        .select({
   661→          id: pmSchedule.id,
   662→          trainId: pmSchedule.trainId,
   663→          constraintEnd: pmSchedule.constraintEnd,
   664→        })
   665→        .from(pmSchedule)
   666→        .where(
   667→          and(
   668→            eq(pmSchedule.status, 'SCHEDULED'),
   669→            sql`${pmSchedule.constraintEnd} < ${todayStr}`,

   732→  /**
   733→   * Increment version number
   734→   *
   735→   * Version format: vMajor.Minor (e.g., v1.0, v1.1, v2.0)
   736→   * Increments minor version by default.
   737→   */
   738→  private incrementVersion(currentVersion: string | null): string {
   739→    if (!currentVersion) {
   740→      return 'v1.0';
   741→    }
   742→
   743→    // Parse current version (vX.Y format)
   744→    const match = currentVersion.match(/^v?(\d+)\.(\d+)$/);
   745→    if (!match) {
   746→      return 'v1.0';
   747→    }
   748→
   749→    const major = parseInt(match[1], 10);
   750→    const minor = parseInt(match[2], 10);
   751→
   752→    // Increment minor, or increment major if minor >= 99
   753→    if (minor >= 99) {
   754→      return `v${major + 1}.0`;
   755→    }
   756→
   757→    return `v${major}.${minor + 1}`;
   758→  }

   853→  /**
   854→   * Generate mileage projections for all active trains
   855→   *
   856→   * Projects mileage forward at monthly intervals using:
   857→   * 1. Average daily mileage from last 30 days (if available)
   858→   * 2. Default daily mileage (150 km) if no history
   859→   */
   860→  private async generateMileageProjections(
   861→    horizonEnd: Date
   862→  ): Promise<MileageProjectionData[]> {
   863→    return this.traced('generateMileageProjections', async (span) => {
   864→      // Get all active trains with their current mileage
   865→      const activeTrains = await this.db
   866→        .select({
   867→          id: trains.id,
   868→          trainNumber: trains.trainNumber,
   869→          currentMileageKm: trains.currentMileageKm,
   870→        })
   871→        .from(trains)
   872→        .where(eq(trains.active, true));
   873→
   874→      span.setAttribute('baseline.trainCount', activeTrains.length);
   875→
   876→      const projections: MileageProjectionData[] = [];
   877→      const cutoffDate = subDays(new Date(), this.MILEAGE_HISTORY_DAYS);
   878→      const cutoffDateStr = format(cutoffDate, 'yyyy-MM-dd');
   879→
   880→      for (const train of activeTrains) {
   881→        // Calculate average daily mileage from recent history
   882→        const recentMileage = await this.db
   883→          .select({
   884→            avgDailyKm: sql<string>`AVG(daily_km)`,
   885→            readingCount: count(),
   886→          })
   887→          .from(mileageReadings)
   888→          .where(
   889→            and(
   890→              eq(mileageReadings.trainId, train.id),
   891→              sql`${mileageReadings.readingDate} >= ${cutoffDateStr}`,
   892→              eq(mileageReadings.validationStatus, 'VALID')
   893→            )
   894→          );
   895→
   896→        const avgDailyKm =
   897→          recentMileage[0]?.avgDailyKm && recentMileage[0].readingCount > 5
   898→            ? parseFloat(recentMileage[0].avgDailyKm)
   899→            : this.DEFAULT_DAILY_KM;
   900→
   901→        // Project forward at monthly intervals
   902→        let currentKm = train.currentMileageKm ?? 0;
   903→        let currentDate = new Date();
   904→
   905→        while (currentDate < horizonEnd) {
   906→          currentDate = addMonths(currentDate, 1);
   907→          currentKm += avgDailyKm * 30;
   908→
   909→          projections.push({
   910→            trainId: train.id,
   911→            date: new Date(currentDate),
   912→            projectedKm: Math.round(currentKm),
   913→            dailyRate: Math.round(avgDailyKm),
   914→          });
   915→        }
   916→      }
   917→
   918→      span.setAttribute('baseline.totalProjections', projections.length);
   919→
   920→      return projections;
   921→    });
   922→  }
