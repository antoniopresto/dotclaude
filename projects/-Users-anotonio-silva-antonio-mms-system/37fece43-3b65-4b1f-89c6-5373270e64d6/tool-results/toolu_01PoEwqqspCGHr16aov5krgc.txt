     1→/**
     2→ * MMS Fleet Service
     3→ *
     4→ * Core domain service for fleet asset management.
     5→ * Provides CRUD operations for depots, fleets, trains, routes, and tracks.
     6→ * Manages train lifecycle states and mileage tracking.
     7→ *
     8→ * Business rules:
     9→ * - Train fitness transitions: FIT <-> RESTRICTED <-> GROUNDED
    10→ * - Mileage accumulation is additive (never decreases except for corrections)
    11→ * - Fleet mileage balancing: target = fleet average, variance = current - target
    12→ * - Active entities can be deactivated but not deleted (audit compliance)
    13→ *
    14→ * @see specs/003-database-schema.md Section 3.2 (Core Fleet Domain)
    15→ */

   263→
   264→      // Check for duplicate code

   277→      span.setAttribute(MmsAttributes.DEPOT_ID, result.id);
   278→
   279→      this.log.info({ depotId: result.id, code: data.code }, 'Depot created');
   280→      return result;

   284→  async updateDepot(id: string, data: Partial<NewDepot>): Promise<Depot> {
   285→    return this.inTransaction('updateDepot', async (tx, span) => {
   286→      span.setAttribute(MmsAttributes.DEPOT_ID, id);
   287→
   288→      const existing = await tx.query.depots.findFirst({
   289→        where: eq(depots.id, id),
   290→      });
   291→
   292→      if (!existing) {
   293→        throw new NotFoundError(`Depot not found: ${id}`, {
   294→          entityType: 'depot',
   295→          entityId: id,
   296→        });
   297→      }
   298→
   299→      // Check for duplicate code if changing
   300→      if (data.code && data.code !== existing.code) {
   301→        const duplicate = await tx.query.depots.findFirst({
   302→          where: eq(depots.code, data.code),
   303→        });
   304→        if (duplicate) {
   305→          throw new DuplicateError(`Depot with code ${data.code} already exists`, {
   306→            field: 'code',
   307→            value: data.code,
   308→          });
   309→        }
   310→      }
   311→
   312→      const [result] = await tx
   313→        .update(depots)
   314→        .set({ ...data, updatedAt: new Date() })
   315→        .where(eq(depots.id, id))
   316→        .returning();
   317→
   318→      return result;
   319→    });
   320→  }

   531→  async listTrains(
   532→    filters?: TrainListFilters,
   533→    pagination?: PaginationOptions
   534→  ): Promise<PaginatedResult<Train>> {
   535→    return this.traced('listTrains', async (span) => {
   536→      const conditions: SQL<unknown>[] = [];
   537→
   538→      if (filters?.fleetId) {
   539→        conditions.push(eq(trains.fleetId, filters.fleetId));
   540→      }
   541→      if (filters?.depotId) {
   542→        conditions.push(eq(trains.currentDepotId, filters.depotId));
   543→      }
   544→      if (filters?.fitnessStatus) {
   545→        if (Array.isArray(filters.fitnessStatus)) {
   546→          conditions.push(inArray(trains.fitnessStatus, filters.fitnessStatus));
   547→        } else {
   548→          conditions.push(eq(trains.fitnessStatus, filters.fitnessStatus));
   549→        }
   550→      }
   551→      if (filters?.active !== undefined) {
   552→        conditions.push(eq(trains.active, filters.active));
   553→      }
   554→      if (filters?.trainNumbers && filters.trainNumbers.length > 0) {
   555→        conditions.push(inArray(trains.trainNumber, filters.trainNumbers));
   556→      }
   557→
   558→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   559→
   560→      // Get total count
   561→      const [{ totalCount }] = await this.db
   562→        .select({ totalCount: count() })
   563→        .from(trains)
   564→        .where(whereClause);
   565→
   566→      // Apply pagination
   567→      const page = pagination?.page ?? 1;
   568→      const pageSize = Math.min(pagination?.pageSize ?? 50, 500);
   569→      const offset = (page - 1) * pageSize;
   570→
   571→      // Determine sort order
   572→      const sortDirection = pagination?.sortOrder === 'desc' ? desc : asc;
   573→      const getOrderBy = () => {
   574→        switch (pagination?.sortBy) {
   575→          case 'mileage':
   576→            return sortDirection(trains.currentMileageKm);
   577→          case 'fitnessStatus':
   578→            return sortDirection(trains.fitnessStatus);
   579→          default:
   580→            return sortDirection(trains.trainNumber);
   581→        }
   582→      };
   583→
   584→      const data = await this.db.query.trains.findMany({
   585→        where: whereClause,
   586→        orderBy: getOrderBy(),
   587→        limit: pageSize,
   588→        offset,
   589→      });
   590→
   591→      span.setAttribute('result.count', data.length);
   592→      span.setAttribute('result.total', totalCount);
   593→
   594→      return {
   595→        data,
   596→        total: totalCount,
   597→        page,
   598→        pageSize,
   599→        totalPages: Math.ceil(totalCount / pageSize),
   600→      };
   601→    });
   602→  }

   605→    return this.inTransaction('createTrain', async (tx, span) => {
   606→      span.setAttribute(MmsAttributes.TRAIN_NUMBER, data.trainNumber);

   617→      }

   619→      // Validate depots if provided
   620→      if (data.currentDepotId) {

   642→      }

   644→      // Check for duplicate train number

   647→      });

   649→      if (existing) {

   710→  /**
   711→   * Update train fitness status with validation.
   712→   *
   713→   * Valid transitions:
   714→   * - FIT -> RESTRICTED, GROUNDED
   715→   * - RESTRICTED -> FIT, GROUNDED
   716→   * - GROUNDED -> RESTRICTED, FIT
   717→   *
   718→   * Note: Transitions from GROUNDED to FIT should typically go through RESTRICTED
   719→   * for inspection, but this is not enforced (operational flexibility).
   720→   */
   721→  async updateFitnessStatus(request: FitnessStatusChangeRequest): Promise<Train> {
   722→    return this.inTransaction('updateFitnessStatus', async (tx, span) => {
   723→      const { trainId, newStatus, notes, changedBy } = request;
   724→
   725→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   726→      span.setAttribute('fitness.newStatus', newStatus);
   727→
   728→      const train = await tx.query.trains.findFirst({
   729→        where: eq(trains.id, trainId),
   730→      });
   731→
   732→      if (!train) {
   733→        throw new TrainNotFoundError(trainId);
   734→      }
   735→
   736→      const oldStatus = train.fitnessStatus;
   737→      span.setAttribute('fitness.oldStatus', oldStatus ?? 'null');
   738→
   739→      // Log transition
   740→      this.log.info(
   741→        {
   742→          trainId,
   743→          trainNumber: train.trainNumber,
   744→          oldStatus,
   745→          newStatus,
   746→          changedBy,
   747→        },
   748→        `Train fitness status change: ${oldStatus} -> ${newStatus}`
   749→      );
   750→
   751→      // Update with notes
   752→      const [result] = await tx
   753→        .update(trains)
   754→        .set({
   755→          fitnessStatus: newStatus,
   756→          fitnessNotes: notes ?? train.fitnessNotes,
   757→          updatedAt: new Date(),
   758→        })
   759→        .where(eq(trains.id, trainId))
   760→        .returning();
   761→
   762→      this.recordEvent('train.fitness_changed', {
   763→        [MmsAttributes.TRAIN_ID]: trainId,
   764→        [MmsAttributes.TRAIN_NUMBER]: train.trainNumber,
   765→        'fitness.old': oldStatus ?? 'unknown',
   766→        'fitness.new': newStatus,
   767→      }, span);
   768→
   769→      return result;
   770→    });
   771→  }

   832→  /**
   833→   * Update train mileage (typically called by ingestion service)
   834→   *
   835→   * Validates:
   836→   * - New mileage must be >= current (no rollback unless meter event)
   837→   * - Maximum daily increase: 2000 km (configurable)
   838→   */
   839→  async updateMileage(request: MileageUpdateRequest): Promise<Train> {
   840→    return this.inTransaction('updateMileage', async (tx, span) => {
   841→      const { trainId, newMileageKm, source } = request;
   842→
   843→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   844→      span.setAttribute(MmsAttributes.MILEAGE_KM, newMileageKm);
   845→
   846→      const train = await tx.query.trains.findFirst({
   847→        where: eq(trains.id, trainId),
   848→      });
   849→
   850→      if (!train) {
   851→        throw new TrainNotFoundError(trainId);
   852→      }
   853→
   854→      const oldMileage = train.currentMileageKm;
   855→      const delta = newMileageKm - oldMileage;
   856→
   857→      span.setAttribute('mileage.old', oldMileage);
   858→      span.setAttribute('mileage.delta', delta);
   859→
   860→      // Validate mileage increase
   861→      if (delta < 0) {
   862→        throw new BusinessRuleError(
   863→          `Mileage cannot decrease (${oldMileage} -> ${newMileageKm}). Use meter event for corrections.`,
   864→          {
   865→            rule: 'mileage_no_decrease',
   866→            context: { trainId, oldMileage, newMileageKm },
   867→          }
   868→        );
   869→      }
   870→
   871→      // Warn on large delta (>2000km in single update)
   872→      if (delta > 2000) {
   873→        this.log.warn(
   874→          {
   875→            trainId,
   876→            trainNumber: train.trainNumber,
   877→            oldMileage,
   878→            newMileageKm,
   879→            delta,
   880→            source,
   881→          },
   882→          'Large mileage delta detected - possible anomaly'
   883→        );
   884→      }
   885→
   886→      const [result] = await tx
   887→        .update(trains)
   888→        .set({
   889→          currentMileageKm: newMileageKm,
   890→          updatedAt: new Date(),
   891→        })
   892→        .where(eq(trains.id, trainId))
   893→        .returning();
   894→
   895→      this.log.info(
   896→        {
   897→          trainId,
   898→          trainNumber: train.trainNumber,
   899→          oldMileage,
   900→          newMileageKm,
   901→          delta,
   902→          source,
   903→        },
   904→        `Train mileage updated: ${oldMileage} -> ${newMileageKm} (+${delta} km)`
   905→      );
   906→
   907→      return result;
   908→    });
   909→  }

  1480→  /**
  1481→   * List components with optional filters
  1482→   */
  1483→  async listComponents(
  1484→    filters?: ComponentListFilters,
  1485→    pagination?: PaginationOptions
  1486→  ): Promise<PaginatedResult<TrainComponent>> {
  1487→    return this.traced('listComponents', async (span) => {
  1488→      const conditions: SQL<unknown>[] = [];
  1489→
  1490→      if (filters?.componentType) {
  1491→        conditions.push(eq(trainComponents.componentType, filters.componentType));
  1492→      }
  1493→      if (filters?.status) {
  1494→        if (Array.isArray(filters.status)) {
  1495→          conditions.push(inArray(trainComponents.status, filters.status));
  1496→        } else {
  1497→          conditions.push(eq(trainComponents.status, filters.status));
  1498→        }
  1499→      }
  1500→      if (filters?.trainId) {
  1501→        conditions.push(eq(trainComponents.currentTrainId, filters.trainId));
  1502→      }
  1503→      if (filters?.serialNumbers && filters.serialNumbers.length > 0) {
  1504→        conditions.push(inArray(trainComponents.serialNumber, filters.serialNumbers));
  1505→      }
  1506→      if (filters?.hasCurrentTrain !== undefined) {
  1507→        if (filters.hasCurrentTrain) {
  1508→          conditions.push(sql`${trainComponents.currentTrainId} IS NOT NULL`);
  1509→        } else {
  1510→          conditions.push(sql`${trainComponents.currentTrainId} IS NULL`);
  1511→        }
  1512→      }
  1513→
  1514→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
  1515→
  1516→      // Get total count
  1517→      const [{ totalCount }] = await this.db
  1518→        .select({ totalCount: count() })
  1519→        .from(trainComponents)
  1520→        .where(whereClause);
  1521→
  1522→      // Apply pagination
  1523→      const page = pagination?.page ?? 1;
  1524→      const pageSize = Math.min(pagination?.pageSize ?? 50, 500);
  1525→      const offset = (page - 1) * pageSize;
  1526→
  1527→      // Determine sort order
  1528→      const sortDirection = pagination?.sortOrder === 'desc' ? desc : asc;
  1529→      const getOrderBy = () => {
  1530→        switch (pagination?.sortBy) {
  1531→          case 'mileage':
  1532→            return sortDirection(trainComponents.componentMileageKm);
  1533→          case 'status':
  1534→            return sortDirection(trainComponents.status);
  1535→          case 'type':
  1536→            return sortDirection(trainComponents.componentType);
  1537→          default:
  1538→            return sortDirection(trainComponents.serialNumber);
  1539→        }
  1540→      };
  1541→
  1542→      const data = await this.db.query.trainComponents.findMany({
  1543→        where: whereClause,
  1544→        orderBy: getOrderBy(),
  1545→        limit: pageSize,
  1546→        offset,
  1547→      });
  1548→
  1549→      span.setAttribute('result.count', data.length);
  1550→      span.setAttribute('result.total', totalCount);
  1551→
  1552→      return {
  1553→        data,
  1554→        total: totalCount,
  1555→        page,
  1556→        pageSize,
  1557→        totalPages: Math.ceil(totalCount / pageSize),
  1558→      };
  1559→    });
  1560→  }

  1651→  /**
  1652→   * Install a component on a train.
  1653→   * Creates an installation record and updates the component's current train.
  1654→   *
  1655→   * Business rules:
  1656→   * - Component must not already be installed on another train
  1657→   * - Train must exist and be active
  1658→   * - Component mileage tracking starts from current train mileage
  1659→   */
  1660→  async installComponent(request: InstallComponentRequest): Promise<ComponentInstallation> {
  1661→    return this.inTransaction('installComponent', async (tx, span) => {
  1662→      const { componentId, trainId, position, workOrderId } = request;
  1663→
  1664→      span.setAttribute('component.id', componentId);
  1665→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
  1666→
  1667→      // Get component
  1668→      const component = await tx.query.trainComponents.findFirst({
  1669→        where: eq(trainComponents.id, componentId),
  1670→      });
  1671→
  1672→      if (!component) {
  1673→        throw new NotFoundError(`Component not found: ${componentId}`, {
  1674→          entityType: 'component',
  1675→          entityId: componentId,
  1676→        });
  1677→      }
  1678→
  1679→      // Check if component is already installed
  1680→      if (component.currentTrainId) {
  1681→        throw new BusinessRuleError(
  1682→          `Component ${component.serialNumber} is already installed on a train. Remove it first.`,
  1683→          {
  1684→            rule: 'component_already_installed',
  1685→            context: { componentId, currentTrainId: component.currentTrainId },
  1686→          }
  1687→        );
  1688→      }
  1689→
  1690→      // Get target train
  1691→      const train = await tx.query.trains.findFirst({
  1692→        where: eq(trains.id, trainId),
  1693→      });
  1694→
  1695→      if (!train) {
  1696→        throw new TrainNotFoundError(trainId);
  1697→      }
  1698→
  1699→      if (!train.active) {
  1700→        throw new BusinessRuleError('Cannot install component on inactive train', {
  1701→          rule: 'train_inactive',
  1702→          context: { trainId },
  1703→        });
  1704→      }
  1705→
  1706→      const today = new Date().toISOString().split('T')[0];
  1707→      const trainMileage = train.currentMileageKm;
  1708→      const componentMileage = component.componentMileageKm;
  1709→
  1710→      // Create installation record
  1711→      const installationData: NewComponentInstallation = {
  1712→        componentId,
  1713→        trainId,
  1714→        position,
  1715→        installedDate: today,
  1716→        installedTrainMileageKm: trainMileage,
  1717→        installedComponentMileageKm: componentMileage,
  1718→        workOrderId,
  1719→      };
  1720→
  1721→      const [installation] = await tx
  1722→        .insert(componentInstallations)
  1723→        .values(installationData)
  1724→        .returning();
  1725→
  1726→      // Update component with current train
  1727→      await tx
  1728→        .update(trainComponents)
  1729→        .set({
  1730→          currentTrainId: trainId,
  1731→          currentPosition: position,
  1732→          installedDate: today,
  1733→          installedMileageKm: trainMileage,
  1734→          status: 'IN_SERVICE',
  1735→          updatedAt: new Date(),
  1736→        })
  1737→        .where(eq(trainComponents.id, componentId));
  1738→
  1739→      span.setAttribute('installation.id', installation.id);
  1740→
  1741→      this.log.info(
  1742→        {
  1743→          componentId,
  1744→          serialNumber: component.serialNumber,
  1745→          trainId,
  1746→          trainNumber: train.trainNumber,
  1747→          position,
  1748→          trainMileageAtInstall: trainMileage,
  1749→          componentMileageAtInstall: componentMileage,
  1750→        },
  1751→        'Component installed on train'
  1752→      );
  1753→
  1754→      this.recordEvent('component.installed', {
  1755→        'component.id': componentId,
  1756→        'component.serialNumber': component.serialNumber,
  1757→        [MmsAttributes.TRAIN_ID]: trainId,
  1758→        [MmsAttributes.TRAIN_NUMBER]: train.trainNumber,
  1759→        'component.position': position ?? 'unspecified',
  1760→      }, span);
  1761→
  1762→      return installation;
  1763→    });
  1764→  }
  1765→
  1766→  /**
  1767→   * Remove a component from a train.
  1768→   * Closes the installation record and updates component mileage.
  1769→   *
  1770→   * Business rules:
  1771→   * - Component must be installed on a train
  1772→   * - Component mileage is updated based on train mileage during installation
  1773→   */
  1774→  async removeComponent(request: RemoveComponentRequest): Promise<ComponentInstallation> {
  1775→    return this.inTransaction('removeComponent', async (tx, span) => {
  1776→      const { componentId, reason, workOrderId } = request;
  1777→
  1778→      span.setAttribute('component.id', componentId);
  1779→      span.setAttribute('removal.reason', reason);
  1780→
  1781→      // Get component with current train
  1782→      const component = await tx.query.trainComponents.findFirst({
  1783→        where: eq(trainComponents.id, componentId),
  1784→        with: {
  1785→          currentTrain: true,
  1786→        },
  1787→      });
  1788→
  1789→      if (!component) {
  1790→        throw new NotFoundError(`Component not found: ${componentId}`, {
  1791→          entityType: 'component',
  1792→          entityId: componentId,
  1793→        });
  1794→      }
  1795→
  1796→      if (!component.currentTrainId) {
  1797→        throw new BusinessRuleError(
  1798→          `Component ${component.serialNumber} is not installed on any train`,
  1799→          {
  1800→            rule: 'component_not_installed',
  1801→            context: { componentId },
  1802→          }
  1803→        );
  1804→      }
  1805→
  1806→      const train = component.currentTrain;
  1807→      if (!train) {
  1808→        throw new NotFoundError(`Train not found for component`, {
  1809→          entityType: 'train',
  1810→          entityId: component.currentTrainId,
  1811→        });
  1812→      }
  1813→
  1814→      // Find the active installation record
  1815→      const activeInstallation = await tx.query.componentInstallations.findFirst({
  1816→        where: and(
  1817→          eq(componentInstallations.componentId, componentId),
  1818→          eq(componentInstallations.trainId, component.currentTrainId),
  1819→          sql`${componentInstallations.removedDate} IS NULL`
  1820→        ),
  1821→        orderBy: desc(componentInstallations.installedDate),
  1822→      });
  1823→
  1824→      if (!activeInstallation) {
  1825→        throw new BusinessRuleError(
  1826→          'No active installation record found for component',
  1827→          {
  1828→            rule: 'no_active_installation',
  1829→            context: { componentId, trainId: component.currentTrainId },
  1830→          }
  1831→        );
  1832→      }
  1833→
  1834→      const today = new Date().toISOString().split('T')[0];
  1835→      const currentTrainMileage = train.currentMileageKm;
  1836→
  1837→      // Calculate mileage accumulated during this installation
  1838→      const mileageDuringInstallation = currentTrainMileage - activeInstallation.installedTrainMileageKm;
  1839→      const newComponentMileage = activeInstallation.installedComponentMileageKm + mileageDuringInstallation;
  1840→
  1841→      // Update installation record with removal data
  1842→      const [updatedInstallation] = await tx
  1843→        .update(componentInstallations)
  1844→        .set({
  1845→          removedDate: today,
  1846→          removedTrainMileageKm: currentTrainMileage,
  1847→          removedComponentMileageKm: newComponentMileage,
  1848→          removalReason: reason,
  1849→          workOrderId: workOrderId ?? activeInstallation.workOrderId,
  1850→        })
  1851→        .where(eq(componentInstallations.id, activeInstallation.id))
  1852→        .returning();
  1853→
  1854→      // Update component - clear current train, update mileage
  1855→      await tx
  1856→        .update(trainComponents)
  1857→        .set({
  1858→          currentTrainId: null,
  1859→          currentPosition: null,
  1860→          installedDate: null,
  1861→          installedMileageKm: null,
  1862→          componentMileageKm: newComponentMileage,
  1863→          status: 'SPARE',
  1864→          updatedAt: new Date(),
  1865→        })
  1866→        .where(eq(trainComponents.id, componentId));
  1867→
  1868→      span.setAttribute('mileage.during_installation', mileageDuringInstallation);
  1869→      span.setAttribute('mileage.new_total', newComponentMileage);
  1870→
  1871→      this.log.info(
  1872→        {
  1873→          componentId,
  1874→          serialNumber: component.serialNumber,
  1875→          trainId: component.currentTrainId,
  1876→          trainNumber: train.trainNumber,
  1877→          reason,
  1878→          mileageDuringInstallation,
  1879→          newComponentMileage,
  1880→        },
  1881→        'Component removed from train'
  1882→      );
  1883→
  1884→      this.recordEvent('component.removed', {
  1885→        'component.id': componentId,
  1886→        'component.serialNumber': component.serialNumber,
  1887→        [MmsAttributes.TRAIN_ID]: component.currentTrainId,
  1888→        [MmsAttributes.TRAIN_NUMBER]: train.trainNumber,

  1897→  /**
  1898→   * Transfer a component from one train to another.
  1899→   * Combines remove and install in a single transaction.
  1900→   *
  1901→   * Business rules:
  1902→   * - Component must be installed on a train
  1903→   * - Component mileage is updated during transfer

  1906→  async transferComponent(request: TransferComponentRequest): Promise<{
  1907→    removal: ComponentInstallation;

  1913→
  1914→      span.setAttribute('component.id', componentId);
  1915→      span.setAttribute('train.new', newTrainId);
  1916→
  1917→      // Get component with current train
  1918→      const component = await tx.query.trainComponents.findFirst({
  1919→        where: eq(trainComponents.id, componentId),
  1920→        with: {
  1921→          currentTrain: true,
  1922→        },
  1923→      });
  1924→
  1925→      if (!component) {
  1926→        throw new NotFoundError(`Component not found: ${componentId}`, {
  1927→          entityType: 'component',
  1928→          entityId: componentId,
  1929→        });
  1930→      }
  1931→
  1932→      if (!component.currentTrainId) {
  1933→        throw new BusinessRuleError(
  1934→          `Component ${component.serialNumber} is not installed. Use installComponent instead.`,
  1935→          {
  1936→            rule: 'component_not_installed',
  1937→            context: { componentId },
  1938→          }
  1939→        );
  1940→      }
  1941→
  1942→      const oldTrain = component.currentTrain;
  1943→      if (!oldTrain) {
  1944→        throw new NotFoundError(`Train not found for component`, {
  1945→          entityType: 'train',
  1946→          entityId: component.currentTrainId,
  1947→        });
  1948→      }
  1949→
  1950→      // Get new train
  1951→      const newTrain = await tx.query.trains.findFirst({
  1952→        where: eq(trains.id, newTrainId),
  1953→      });
  1954→
  1955→      if (!newTrain) {
  1956→        throw new TrainNotFoundError(newTrainId);
  1957→      }
  1958→
  1959→      if (!newTrain.active) {
  1960→        throw new BusinessRuleError('Cannot transfer component to inactive train', {
  1961→          rule: 'train_inactive',
  1962→          context: { trainId: newTrainId },
  1963→        });
  1964→      }
  1965→
  1966→      span.setAttribute('train.old', component.currentTrainId);
  1967→      span.setAttribute(MmsAttributes.TRAIN_NUMBER, `${oldTrain.trainNumber} -> ${newTrain.trainNumber}`);
  1968→
  1969→      // Find active installation
  1970→      const activeInstallation = await tx.query.componentInstallations.findFirst({
  1971→        where: and(
  1972→          eq(componentInstallations.componentId, componentId),
  1973→          eq(componentInstallations.trainId, component.currentTrainId),
  1974→          sql`${componentInstallations.removedDate} IS NULL`
  1975→        ),
  1976→        orderBy: desc(componentInstallations.installedDate),
  1977→      });
  1978→
  1979→      if (!activeInstallation) {
  1980→        throw new BusinessRuleError('No active installation record found', {
  1981→          rule: 'no_active_installation',
  1982→          context: { componentId },
  1983→        });
  1984→      }
  1985→
  1986→      const today = new Date().toISOString().split('T')[0];
  1987→      const oldTrainMileage = oldTrain.currentMileageKm;
  1988→      const newTrainMileage = newTrain.currentMileageKm;
  1989→
  1990→      // Calculate mileage accumulated during previous installation
  1991→      const mileageAccumulated = oldTrainMileage - activeInstallation.installedTrainMileageKm;
  1992→      const updatedComponentMileage = activeInstallation.installedComponentMileageKm + mileageAccumulated;
  1993→

  1998→          removedDate: today,
  1999→          removedTrainMileageKm: oldTrainMileage,

  2001→          removalReason: reason,
  2002→          workOrderId,
  2003→        })
  2004→        .where(eq(componentInstallations.id, activeInstallation.id))
  2005→        .returning();
  2006→
  2007→      // Create new installation record
  2008→      const newInstallationData: NewComponentInstallation = {
  2009→        componentId,
  2010→        trainId: newTrainId,
  2011→        position: newPosition,
  2012→        installedDate: today,
  2013→        installedTrainMileageKm: newTrainMileage,
  2014→        installedComponentMileageKm: updatedComponentMileage,
  2015→        workOrderId,
  2016→      };
  2017→
  2018→      const [installationRecord] = await tx
  2019→        .insert(componentInstallations)
  2020→        .values(newInstallationData)
  2021→        .returning();
  2022→
  2023→      // Update component
  2024→      await tx
  2025→        .update(trainComponents)
  2026→        .set({
  2027→          currentTrainId: newTrainId,
  2028→          currentPosition: newPosition,
  2029→          installedDate: today,
  2030→          installedMileageKm: newTrainMileage,
  2031→          componentMileageKm: updatedComponentMileage,
  2032→          status: 'IN_SERVICE',
  2033→          updatedAt: new Date(),
  2034→        })
  2035→        .where(eq(trainComponents.id, componentId));
  2036→
  2037→      span.setAttribute('mileage.accumulated', mileageAccumulated);
  2038→      span.setAttribute('mileage.component_total', updatedComponentMileage);
  2039→
  2040→      this.log.info(
  2041→        {
  2042→          componentId,
  2043→          serialNumber: component.serialNumber,
  2044→          oldTrainId: component.currentTrainId,
  2045→          oldTrainNumber: oldTrain.trainNumber,
  2046→          newTrainId,
  2047→          newTrainNumber: newTrain.trainNumber,
  2048→          position: newPosition,
  2049→          mileageAccumulated,
  2050→          newComponentMileage: updatedComponentMileage,
  2051→        },
  2052→        'Component transferred between trains'
  2053→      );
  2054→

  2056→        'component.id': componentId,
  2057→        'component.serialNumber': component.serialNumber,
  2058→        'train.old.id': component.currentTrainId,
  2059→        'train.old.number': oldTrain.trainNumber,
  2060→        'train.new.id': newTrainId,
  2061→        'train.new.number': newTrain.trainNumber,
  2062→        'mileage.accumulated': mileageAccumulated,
  2063→      }, span);
  2064→
  2065→      return {
  2066→        removal: removalRecord,
  2067→        installation: installationRecord,
  2068→        mileageAccumulated,
  2069→      };
  2070→    });
  2071→  }

  2073→  /**
  2074→   * Recalculate and update component mileage based on installation history.
  2075→   * Useful for reconciliation or after data corrections.
  2076→   */
  2077→  async recalculateComponentMileage(componentId: string): Promise<ComponentMileageResult> {
  2078→    return this.inTransaction('recalculateComponentMileage', async (tx, span) => {
  2079→      span.setAttribute('component.id', componentId);
  2080→
  2081→      const component = await tx.query.trainComponents.findFirst({
  2082→        where: eq(trainComponents.id, componentId),
  2083→        with: {
  2084→          installations: {
  2085→            orderBy: asc(componentInstallations.installedDate),
  2086→          },
  2087→          currentTrain: true,
  2088→        },
  2089→      });
  2090→
  2091→      if (!component) {
  2092→        throw new NotFoundError(`Component not found: ${componentId}`, {
  2093→          entityType: 'component',
  2094→          entityId: componentId,
  2095→        });
  2096→      }
  2097→
  2098→      let totalMileage = 0;
  2099→      let installationsCalculated = 0;
  2100→
  2101→      for (const installation of component.installations) {
  2102→        if (installation.removedDate && installation.removedTrainMileageKm !== null) {
  2103→          // Completed installation - use recorded values
  2104→          const mileageDelta = installation.removedTrainMileageKm - installation.installedTrainMileageKm;
  2105→          totalMileage += Math.max(0, mileageDelta);
  2106→          installationsCalculated++;
  2107→        } else if (component.currentTrain && component.currentTrainId === installation.trainId) {
  2108→          // Active installation - calculate from current train mileage
  2109→          const mileageDelta = component.currentTrain.currentMileageKm - installation.installedTrainMileageKm;
  2110→          totalMileage += Math.max(0, mileageDelta);
  2111→          installationsCalculated++;
  2112→        }
  2113→      }
  2114→
  2115→      // Update component mileage
  2116→      await tx
  2117→        .update(trainComponents)
  2118→        .set({
  2119→          componentMileageKm: totalMileage,
  2120→          updatedAt: new Date(),
  2121→        })
  2122→        .where(eq(trainComponents.id, componentId));
  2123→
  2124→      span.setAttribute('mileage.calculated', totalMileage);
  2125→      span.setAttribute('installations.count', installationsCalculated);
  2126→
  2127→      this.log.info(
  2128→        {
  2129→          componentId,
  2130→          serialNumber: component.serialNumber,
  2131→          totalMileage,
  2132→          installationsCalculated,
  2133→        },
  2134→        'Component mileage recalculated'
  2135→      );
  2136→
  2137→      return {
  2138→        componentId,
  2139→        serialNumber: component.serialNumber,
  2140→        totalComponentMileageKm: totalMileage,
  2141→        installationsCalculated,
  2142→      };
  2143→    });
