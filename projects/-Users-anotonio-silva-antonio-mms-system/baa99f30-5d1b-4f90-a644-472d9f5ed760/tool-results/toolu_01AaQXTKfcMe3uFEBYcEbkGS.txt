   310→#### Reference: @belt/tools Patterns
   311→```typescript
   312→// Schema validation (replaces Zod)
   313→import { createSchema, createType, InferType } from '@belt/tools';
   314→
   315→const TrainModel = createSchema({
   316→  id: 'string',
   317→  trainNumber: 'string',
   318→  mileageKm: 'number',
   319→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   320→  tags: 'string[]',
   321→  metadata: 'object?',
   322→});
   323→type Train = InferType<typeof TrainModel>;
   324→const train = TrainModel.parse(data);
   325→
   326→// State machine (replaces React hooks + zustand)
   327→import { createSlice, createMachine } from '@belt/tools';
   328→
   329→const fleetSlice = createSlice({
   330→  id: 'fleet',
   331→  initial: { trains: [], selected: null },
   332→  cache: { policy: 'session' },
   333→  reducers: {
   334→    setTrains: (state, trains) => { state.trains = trains; },
   335→    selectTrain: (state, id) => { state.selected = id; },
   336→  },
   337→});
   338→
   339→const machine = createMachine({
   340→  id: 'mms',
   341→  slices: { fleet: fleetSlice },
   342→  trackMetadata: false, // Critical for 60fps
   343→});
   344→
   345→// In React component (NO useState/useEffect!)
   346→const { state, actions } = machine.useMachine('fleet');
   347→```
