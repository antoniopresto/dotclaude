   319→#### Reference: @belt/tools Patterns
   320→```typescript
   321→// Schema validation (replaces Zod)
   322→import { createSchema, createType, InferType } from '@belt/tools';
   323→
   324→const TrainModel = createSchema({
   325→  id: 'string',
   326→  trainNumber: 'string',
   327→  mileageKm: 'number',
   328→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   329→  tags: 'string[]',
   330→  metadata: 'object?',
   331→});
   332→type Train = InferType<typeof TrainModel>;
   333→const train = TrainModel.parse(data);
   334→
   335→// State machine (replaces React hooks + zustand)
   336→import { createSlice, createMachine } from '@belt/tools';
   337→
   338→const fleetSlice = createSlice({
   339→  id: 'fleet',
   340→  initial: { trains: [], selected: null },
   341→  cache: { policy: 'session' },
   342→  reducers: {
   343→    setTrains: (state, trains) => { state.trains = trains; },
   344→    selectTrain: (state, id) => { state.selected = id; },
   345→  },
   346→});
   347→
   348→const machine = createMachine({
   349→  id: 'mms',
   350→  slices: { fleet: fleetSlice },
   351→  trackMetadata: false, // Critical for 60fps
   352→});
   353→
   354→// In React component (NO useState/useEffect!)
   355→const { state, actions } = machine.useMachine('fleet');
   356→```
