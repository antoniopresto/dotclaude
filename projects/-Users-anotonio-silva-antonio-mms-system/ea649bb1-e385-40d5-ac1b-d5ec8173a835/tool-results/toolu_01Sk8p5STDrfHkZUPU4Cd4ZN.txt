   305→#### Reference: @belt/tools Patterns
   306→```typescript
   307→// Schema validation (replaces Zod)
   308→import { createSchema, createType, InferType } from '@belt/tools';
   309→
   310→const TrainModel = createSchema({
   311→  id: 'string',
   312→  trainNumber: 'string',
   313→  mileageKm: 'number',
   314→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   315→  tags: 'string[]',
   316→  metadata: 'object?',
   317→});
   318→type Train = InferType<typeof TrainModel>;
   319→const train = TrainModel.parse(data);
   320→
   321→// State machine (replaces React hooks + zustand)
   322→import { createSlice, createMachine } from '@belt/tools';
   323→
   324→const fleetSlice = createSlice({
   325→  id: 'fleet',
   326→  initial: { trains: [], selected: null },
   327→  cache: { policy: 'session' },
   328→  reducers: {
   329→    setTrains: (state, trains) => { state.trains = trains; },
   330→    selectTrain: (state, id) => { state.selected = id; },
   331→  },
   332→});
   333→
   334→const machine = createMachine({
   335→  id: 'mms',
   336→  slices: { fleet: fleetSlice },
   337→  trackMetadata: false, // Critical for 60fps
   338→});
   339→
   340→// In React component (NO useState/useEffect!)
   341→const { state, actions } = machine.useMachine('fleet');
   342→```
