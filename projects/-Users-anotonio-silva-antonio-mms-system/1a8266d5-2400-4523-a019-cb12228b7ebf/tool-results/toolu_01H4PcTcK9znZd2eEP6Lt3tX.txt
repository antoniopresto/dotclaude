     1→/**
     2→ * MMS Toolbar
     3→ *
     4→ * View controls, navigation, and action buttons.
     5→ * Based on specs/006-user-interface.md Section 6.2.
     6→ */
     7→
     8→import {
     9→  Button,
    10→  ButtonGroup,
    11→  Divider,
    12→  Popover,
    13→  Menu,
    14→  MenuItem,
    15→  Tag,
    16→} from '@blueprintjs/core';
    17→import { format } from 'date-fns';
    18→
    19→import { useCanvas, useScenario, useUi } from '../../store';
    20→import type { CanvasSliceState } from '../../store/slices/canvas';
    21→import type { ScenarioSliceState } from '../../store/slices/scenario';
    22→import type { UiSliceState } from '../../store/slices/ui';
    23→
    24→export function Toolbar() {
    25→  const { state: canvasStateRaw, actions: canvasActions } = useCanvas();
    26→  const { state: scenarioStateRaw } = useScenario();
    27→  const { state: uiStateRaw, actions: uiActions } = useUi();
    28→
    29→  const canvasState = canvasStateRaw as unknown as CanvasSliceState;
    30→  const scenarioState = scenarioStateRaw as unknown as ScenarioSliceState;
    31→  const uiState = uiStateRaw as unknown as UiSliceState;
    32→
    33→  const handleZoomLevelClick = (level: 'STRATEGIC' | 'TACTICAL' | 'OPERATIONAL') => {
    34→    const targetZoom =
    35→      level === 'STRATEGIC' ? 2 :
    36→      level === 'TACTICAL' ? 10 :
    37→      level === 'OPERATIONAL' ? 14 : 10;
    38→
    39→    canvasActions.setZoom(targetZoom);
    40→  };
    41→
    42→  const handleZoomIn = () => {
    43→    canvasActions.setZoom(Math.min(20, canvasState.zoom + 1));
    44→  };
    45→
    46→  const handleZoomOut = () => {
    47→    canvasActions.setZoom(Math.max(0, canvasState.zoom - 1));
    48→  };
    49→
    50→  const handleToday = () => {
    51→    canvasActions.goToToday();
    52→  };
    53→
    54→  const handlePrevPeriod = () => {
    55→    const start = new Date(canvasState.viewport.startDate);
    56→    const periodDays = canvasState.zoomLevel === 'STRATEGIC' ? 365 :
    57→                       canvasState.zoomLevel === 'TACTICAL_WIDE' ? 90 :
    58→                       canvasState.zoomLevel === 'TACTICAL' ? 28 : 7;
    59→    start.setDate(start.getDate() - periodDays);
    60→    canvasActions.setViewport({
    61→      startDate: start.toISOString().split('T')[0],
    62→    });
    63→  };
    64→
    65→  const handleNextPeriod = () => {
    66→    const start = new Date(canvasState.viewport.startDate);
    67→    const periodDays = canvasState.zoomLevel === 'STRATEGIC' ? 365 :
    68→                       canvasState.zoomLevel === 'TACTICAL_WIDE' ? 90 :
    69→                       canvasState.zoomLevel === 'TACTICAL' ? 28 : 7;
    70→    start.setDate(start.getDate() + periodDays);
    71→    canvasActions.setViewport({
    72→      startDate: start.toISOString().split('T')[0],
    73→    });
    74→  };
    75→
    76→  const handleToggleLayer = (layer: 'complianceWindows' | 'manhourOverlay' | 'driftOverlay' | 'baselineShadow') => {
    77→    canvasActions.toggleLayer(layer);
    78→  };
    79→
    80→  const handlePublishBaseline = () => {
    81→    uiActions.openModal({ type: 'PUBLISH_BASELINE' });
    82→  };
    83→
    84→  const handleCreateScenario = () => {
    85→    uiActions.openModal({ type: 'CREATE_SCENARIO' });
    86→  };
    87→
    88→  const handleRunOptimizer = () => {
    89→    uiActions.openModal({ type: 'RUN_OPTIMIZER' });
    90→  };
    91→
    92→  const overlayMenu = (
    93→    <Menu>
    94→      <MenuItem
    95→        icon={canvasState.layers.complianceWindows ? 'tick' : 'blank'}
    96→        text="Compliance Windows"
    97→        onClick={() => handleToggleLayer('complianceWindows')}
    98→      />
    99→      <MenuItem
   100→        icon={canvasState.layers.manhourOverlay ? 'tick' : 'blank'}
   101→        text="Man-hour Utilization"
   102→        onClick={() => handleToggleLayer('manhourOverlay')}
   103→      />
   104→      <MenuItem
   105→        icon={canvasState.layers.driftOverlay ? 'tick' : 'blank'}
   106→        text="Drift Analysis"
   107→        onClick={() => handleToggleLayer('driftOverlay')}
   108→      />
   109→      <MenuItem
   110→        icon={canvasState.layers.baselineShadow ? 'tick' : 'blank'}
   111→        text="Baseline Shadow"
   112→        onClick={() => handleToggleLayer('baselineShadow')}
   113→      />
   114→    </Menu>
   115→  );
   116→
   117→  const exportMenu = (
   118→    <Menu>
   119→      <MenuItem icon="document" text="Export PM Schedule (Excel)" />
   120→      <MenuItem icon="document" text="Generate Report (PDF)" />
   121→    </Menu>
   122→  );
   123→
   124→  const isStrategic = canvasState.zoomLevel === 'STRATEGIC' || canvasState.zoomLevel === 'TACTICAL_WIDE';
   125→  const isTactical = canvasState.zoomLevel === 'TACTICAL';
   126→  const isOperational = canvasState.zoomLevel === 'OPERATIONAL' || canvasState.zoomLevel === 'DETAIL';
   127→
   128→  return (
   129→    <div className="mms-toolbar" data-testid="toolbar" style={{
   130→      display: 'flex',
   131→      justifyContent: 'space-between',
   132→      padding: '8px 16px',
   133→      borderBottom: '1px solid var(--pt-divider-black, #e1e8ed)',
   134→      backgroundColor: 'var(--pt-app-background-color, #f5f8fa)',
   135→    }}>
   136→      <div style={{ display: 'flex', gap: '16px', alignItems: 'center' }}>
   137→        {/* Zoom Level Buttons */}
   138→        <ButtonGroup>
   139→          <Button
   140→            text="Strategic"
   141→            active={isStrategic}
   142→            onClick={() => handleZoomLevelClick('STRATEGIC')}
   143→          />
   144→          <Button
   145→            text="Tactical"
   146→            active={isTactical}
   147→            onClick={() => handleZoomLevelClick('TACTICAL')}
   148→          />
   149→          <Button
   150→            text="Operational"
   151→            active={isOperational}
   152→            onClick={() => handleZoomLevelClick('OPERATIONAL')}
   153→          />
   154→        </ButtonGroup>
   155→
   156→        <Divider />
   157→
   158→        {/* Navigation */}
   159→        <ButtonGroup>
   160→          <Button icon="chevron-left" onClick={handlePrevPeriod} />
   161→          <Button text="Today" onClick={handleToday} />
   162→          <Button icon="chevron-right" onClick={handleNextPeriod} />
   163→        </ButtonGroup>
   164→
   165→        <Tag minimal style={{ fontFamily: 'monospace' }}>
   166→          {format(new Date(canvasState.viewport.startDate), 'MMM yyyy')}
   167→        </Tag>
   168→
   169→        <Divider />
   170→
   171→        {/* Zoom Controls */}
   172→        <ButtonGroup>
   173→          <Button icon="zoom-out" onClick={handleZoomOut} disabled={canvasState.zoom <= 0} />
   174→          <Button icon="zoom-in" onClick={handleZoomIn} disabled={canvasState.zoom >= 20} />
   175→        </ButtonGroup>
   176→
   177→        <Tag minimal intent="primary">
   178→          {canvasState.zoomLevel.replace('_', ' ')}
   179→        </Tag>
   180→      </div>
   181→
   182→      <div style={{ display: 'flex', gap: '8px', alignItems: 'center' }}>
   183→        {/* Scenario Indicator */}
   184→        {scenarioState.activeScenarioId && (
   185→          <Tag intent="warning" icon="annotation">
   186→            Scenario Active
   187→          </Tag>
   188→        )}
   189→
   190→        {/* Actions */}
   191→        <Button
   192→          icon="git-commit"
   193→          text="Publish Baseline"
   194→          onClick={handlePublishBaseline}
   195→        />
   196→
   197→        <Button
   198→          icon="fork"
   199→          text="Create Scenario"
   200→          onClick={handleCreateScenario}
   201→        />
   202→
   203→        <Button
   204→          intent="primary"
   205→          icon="play"
   206→          text="Run Optimizer"
   207→          onClick={handleRunOptimizer}
   208→          loading={uiState.loading.optimizer}
   209→        />
   210→
   211→        <Divider />
   212→
   213→        {/* Overlays */}
   214→        <Popover content={overlayMenu} placement="bottom-end">
   215→          <Button icon="layers" text="Overlays" rightIcon="caret-down" />
   216→        </Popover>
   217→
   218→        {/* Export */}
   219→        <Popover content={exportMenu} placement="bottom-end">
   220→          <Button icon="export" text="Export" rightIcon="caret-down" />
   221→        </Popover>
   222→
   223→        {/* Help */}
   224→        <Button
   225→          minimal
   226→          icon="help"
   227→          onClick={() => uiActions.toggleHelpPanel()}
   228→        />
   229→      </div>
   230→    </div>
   231→  );
   232→}
   233→
