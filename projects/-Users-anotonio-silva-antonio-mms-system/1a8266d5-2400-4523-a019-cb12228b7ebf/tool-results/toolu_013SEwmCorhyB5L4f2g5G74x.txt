     1→/**
     2→ * Drag Confirmation Dialog
     3→ *
     4→ * Modal dialog for confirming PM reschedule operations.
     5→ * Shows impact assessment and allows user to confirm or cancel.
     6→ *
     7→ * @see specs/006-user-interface.md Section 6.3.4
     8→ */
     9→
    10→import {
    11→  Dialog,
    12→  DialogBody,
    13→  DialogFooter,
    14→  Button,
    15→  Intent,
    16→  Callout,
    17→  Spinner,
    18→  Tag,
    19→  Icon,
    20→} from '@blueprintjs/core';
    21→
    22→import type { PmMoveImpact, DragConstraintViolation } from './layers/types';
    23→
    24→// =============================================================================
    25→// Types
    26→// =============================================================================
    27→
    28→export interface DragConfirmationDialogProps {
    29→  isOpen: boolean;
    30→  pmCode: string;
    31→  trainNumber: string;
    32→  sourceDate: string;
    33→  targetDate: string;
    34→  impact: PmMoveImpact | null;
    35→  violations: DragConstraintViolation[];
    36→  isLoading: boolean;
    37→  onConfirm: () => void;
    38→  onCancel: () => void;
    39→}
    40→
    41→// =============================================================================
    42→// Component
    43→// =============================================================================
    44→
    45→export function DragConfirmationDialog({
    46→  isOpen,
    47→  pmCode,
    48→  trainNumber,
    49→  sourceDate,
    50→  targetDate,
    51→  impact,
    52→  violations,
    53→  isLoading,
    54→  onConfirm,
    55→  onCancel,
    56→}: DragConfirmationDialogProps) {
    57→  const hasErrors = violations.some(v => v.severity === 'ERROR');
    58→  const hasWarnings = violations.some(v => v.severity === 'WARNING');
    59→
    60→  const formatDate = (dateStr: string) => {
    61→    const date = new Date(dateStr);
    62→    return date.toLocaleDateString('en-US', {
    63→      weekday: 'short',
    64→      month: 'short',
    65→      day: 'numeric',
    66→      year: 'numeric',
    67→    });
    68→  };
    69→
    70→  const daysDiff = Math.round(
    71→    (new Date(targetDate).getTime() - new Date(sourceDate).getTime()) / (1000 * 60 * 60 * 24)
    72→  );
    73→
    74→  return (
    75→    <Dialog
    76→      isOpen={isOpen}
    77→      onClose={onCancel}
    78→      title="Confirm PM Reschedule"
    79→      icon="calendar"
    80→      canEscapeKeyClose={!isLoading}
    81→      canOutsideClickClose={!isLoading}
    82→      data-testid="drag-confirmation-dialog"
    83→    >
    84→      <DialogBody>
    85→        {/* PM Info Header */}
    86→        <div style={{
    87→          display: 'flex',
    88→          alignItems: 'center',
    89→          gap: '12px',
    90→          marginBottom: '16px',
    91→          padding: '12px',
    92→          backgroundColor: 'var(--pt-app-secondary-background-color, #f5f8fa)',
    93→          borderRadius: '6px',
    94→        }}>
    95→          <Icon icon="wrench" size={24} color="var(--pt-icon-color)" />
    96→          <div>
    97→            <div style={{ fontWeight: 600, fontSize: '16px' }}>
    98→              {pmCode}
    99→            </div>
   100→            <div style={{ color: 'var(--pt-text-muted)', fontSize: '13px' }}>
   101→              Train {trainNumber}
   102→            </div>
   103→          </div>
   104→        </div>
   105→
   106→        {/* Date Change Summary */}
   107→        <div style={{
   108→          display: 'flex',
   109→          alignItems: 'center',
   110→          justifyContent: 'center',
   111→          gap: '16px',
   112→          marginBottom: '20px',
   113→          padding: '16px',
   114→          backgroundColor: 'var(--pt-app-background-color)',
   115→          border: '1px solid var(--pt-divider-black)',
   116→          borderRadius: '6px',
   117→        }}>
   118→          <div style={{ textAlign: 'center' }}>
   119→            <div style={{ fontSize: '11px', color: 'var(--pt-text-muted)', marginBottom: '4px' }}>
   120→              FROM
   121→            </div>
   122→            <div style={{ fontWeight: 600 }}>
   123→              {formatDate(sourceDate)}
   124→            </div>
   125→          </div>
   126→
   127→          <div style={{
   128→            display: 'flex',
   129→            alignItems: 'center',
   130→            gap: '8px',
   131→            padding: '8px 12px',
   132→            backgroundColor: daysDiff > 0 ? '#fff3cd' : '#d4edda',
   133→            borderRadius: '4px',
   134→          }}>
   135→            <Icon
   136→              icon={daysDiff > 0 ? 'arrow-right' : 'arrow-left'}
   137→              color={daysDiff > 0 ? '#856404' : '#155724'}
   138→            />
   139→            <span style={{
   140→              fontWeight: 600,
   141→              color: daysDiff > 0 ? '#856404' : '#155724',
   142→            }}>
   143→              {Math.abs(daysDiff)} day{Math.abs(daysDiff) !== 1 ? 's' : ''} {daysDiff > 0 ? 'later' : 'earlier'}
   144→            </span>
   145→          </div>
   146→
   147→          <div style={{ textAlign: 'center' }}>
   148→            <div style={{ fontSize: '11px', color: 'var(--pt-text-muted)', marginBottom: '4px' }}>
   149→              TO
   150→            </div>
   151→            <div style={{ fontWeight: 600 }}>
   152→              {formatDate(targetDate)}
   153→            </div>
   154→          </div>
   155→        </div>
   156→
   157→        {/* Violations */}
   158→        {violations.length > 0 && (
   159→          <div style={{ marginBottom: '16px' }}>
   160→            {violations.filter(v => v.severity === 'ERROR').map((violation, i) => (
   161→              <Callout
   162→                key={i}
   163→                intent={Intent.DANGER}
   164→                icon="error"
   165→                style={{ marginBottom: '8px' }}
   166→              >
   167→                <strong>{violation.type.replace(/_/g, ' ')}</strong>
   168→                <div style={{ marginTop: '4px' }}>{violation.message}</div>
   169→              </Callout>
   170→            ))}
   171→            {violations.filter(v => v.severity === 'WARNING').map((violation, i) => (
   172→              <Callout
   173→                key={i}
   174→                intent={Intent.WARNING}
   175→                icon="warning-sign"
   176→                style={{ marginBottom: '8px' }}
   177→              >
   178→                <strong>{violation.type.replace(/_/g, ' ')}</strong>
   179→                <div style={{ marginTop: '4px' }}>{violation.message}</div>
   180→              </Callout>
   181→            ))}
   182→          </div>
   183→        )}
   184→
   185→        {/* Impact Assessment */}
   186→        {isLoading ? (
   187→          <div style={{ textAlign: 'center', padding: '24px' }}>
   188→            <Spinner size={24} />
   189→            <div style={{ marginTop: '8px', color: 'var(--pt-text-muted)' }}>
   190→              Calculating impact...
   191→            </div>
   192→          </div>
   193→        ) : impact ? (
   194→          <div style={{
   195→            display: 'grid',
   196→            gridTemplateColumns: 'repeat(2, 1fr)',
   197→            gap: '12px',
   198→            marginTop: '16px',
   199→          }}>
   200→            {/* Man-hour Impact */}
   201→            <div style={{
   202→              padding: '12px',
   203→              backgroundColor: 'var(--pt-app-secondary-background-color)',
   204→              borderRadius: '6px',
   205→            }}>
   206→              <div style={{
   207→                fontSize: '11px',
   208→                fontWeight: 600,
   209→                color: 'var(--pt-text-muted)',
   210→                marginBottom: '8px',
   211→              }}>
   212→                MAN-HOUR IMPACT
   213→              </div>
   214→              <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '4px' }}>
   215→                <span style={{ fontSize: '12px' }}>Source Week:</span>
   216→                <Tag minimal>
   217→                  {impact.manhourChange.sourceWeek.required}h / {impact.manhourChange.sourceWeek.available}h
   218→                </Tag>
   219→              </div>
   220→              <div style={{ display: 'flex', justifyContent: 'space-between' }}>
   221→                <span style={{ fontSize: '12px' }}>Target Week:</span>
   222→                <Tag
   223→                  minimal
   224→                  intent={impact.manhourChange.targetWeek.required > impact.manhourChange.targetWeek.available
   225→                    ? Intent.DANGER
   226→                    : Intent.NONE
   227→                  }
   228→                >
   229→                  {impact.manhourChange.targetWeek.required}h / {impact.manhourChange.targetWeek.available}h
   230→                </Tag>
   231→              </div>
   232→            </div>
   233→
   234→            {/* Cost Impact */}
   235→            <div style={{
   236→              padding: '12px',
   237→              backgroundColor: 'var(--pt-app-secondary-background-color)',
   238→              borderRadius: '6px',
   239→            }}>
   240→              <div style={{
   241→                fontSize: '11px',
   242→                fontWeight: 600,
   243→                color: 'var(--pt-text-muted)',
   244→                marginBottom: '8px',
   245→              }}>
   246→                COST IMPACT
   247→              </div>
   248→              <div style={{
   249→                fontSize: '18px',
   250→                fontWeight: 600,
   251→                color: impact.costChange > 0 ? '#d9534f' : impact.costChange < 0 ? '#5cb85c' : 'inherit',
   252→              }}>
   253→                {impact.costChange > 0 ? '+' : ''}{impact.costChange.toFixed(2)} HKD
   254→              </div>
   255→            </div>
   256→
   257→            {/* Baseline Drift */}
   258→            <div style={{
   259→              padding: '12px',
   260→              backgroundColor: 'var(--pt-app-secondary-background-color)',
   261→              borderRadius: '6px',
   262→            }}>
   263→              <div style={{
   264→                fontSize: '11px',
   265→                fontWeight: 600,
   266→                color: 'var(--pt-text-muted)',
   267→                marginBottom: '8px',
   268→              }}>
   269→                BASELINE DRIFT
   270→              </div>
   271→              <div style={{
   272→                fontSize: '18px',
   273→                fontWeight: 600,
   274→                color: Math.abs(impact.driftFromBaseline) > 7 ? '#d9534f' : 'inherit',
   275→              }}>
   276→                {impact.driftFromBaseline > 0 ? '+' : ''}{impact.driftFromBaseline} days
   277→              </div>
   278→            </div>
   279→
   280→            {/* Conflicts */}
   281→            {impact.conflictsCreated.length > 0 && (
   282→              <div style={{
   283→                padding: '12px',
   284→                backgroundColor: '#fff3cd',
   285→                borderRadius: '6px',
   286→              }}>
   287→                <div style={{
   288→                  fontSize: '11px',
   289→                  fontWeight: 600,
   290→                  color: '#856404',
   291→                  marginBottom: '8px',
   292→                }}>
   293→                  CONFLICTS CREATED
   294→                </div>
   295→                {impact.conflictsCreated.map((conflict, i) => (
   296→                  <Tag key={i} intent={Intent.WARNING} minimal style={{ marginRight: '4px', marginBottom: '4px' }}>
   297→                    {conflict}
   298→                  </Tag>
   299→                ))}
   300→              </div>
   301→            )}
   302→          </div>
   303→        ) : null}
   304→      </DialogBody>
   305→
   306→      <DialogFooter
   307→        actions={
   308→          <>
   309→            <Button
   310→              onClick={onCancel}
   311→              disabled={isLoading}
   312→            >
   313→              Cancel
   314→            </Button>
   315→            <Button
   316→              intent={hasErrors ? Intent.DANGER : hasWarnings ? Intent.WARNING : Intent.PRIMARY}
   317→              onClick={onConfirm}
   318→              disabled={isLoading || hasErrors}
   319→              icon={hasWarnings ? 'warning-sign' : 'tick'}
   320→            >
   321→              {hasErrors ? 'Cannot Reschedule' : hasWarnings ? 'Reschedule Anyway' : 'Confirm Reschedule'}
   322→            </Button>
   323→          </>
   324→        }
   325→      />
   326→    </Dialog>
   327→  );
   328→}
   329→
   330→export default DragConfirmationDialog;
   331→
