    56→    case 'IN_PROGRESS':
    57→      return PM_STATUS_COLORS.inProgress;

    62→    case 'DEFERRED':

   128→function getDays(fromDate: Date, toDate: Date): DayData[] {
   129→  const days: DayData[] = [];
   130→  const current = new Date(fromDate);
   131→  const today = new Date();
   132→  today.setHours(0, 0, 0, 0);
   133→
   134→  const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
   135→
   136→  while (current <= toDate) {
   137→    const dayOfWeek = current.getDay();
   138→    days.push({
   139→      date: new Date(current),
   140→      dateStr: current.toISOString().split('T')[0],
   141→      dayOfWeek,
   142→      isWeekStart: dayOfWeek === 0,
   143→      isToday: current.getTime() === today.getTime(),
   144→      label: `${dayNames[dayOfWeek]} ${current.getDate()}`,
   145→    });
   146→    current.setDate(current.getDate() + 1);
   147→  }
   148→  return days;
   149→}
   150→
   151→function getWeeks(days: DayData[]): WeekData[] {
   152→  const weeks: WeekData[] = [];
   153→  let currentWeek: WeekData | null = null;
   154→  const monthNames = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   155→
   156→  for (const day of days) {
   157→    if (day.isWeekStart || currentWeek === null) {
   158→      if (currentWeek) {
   159→        weeks.push(currentWeek);
   160→      }
   161→      const weekNum = getWeekNumber(day.date);
   162→      currentWeek = {
   163→        weekStart: day.date,
   164→        weekNum,
   165→        month: monthNames[day.date.getMonth()],
   166→        daysInWeek: 1,
   167→      };
   168→    } else {
   169→      currentWeek.daysInWeek++;
   170→    }
   171→  }
   172→  if (currentWeek) {
   173→    weeks.push(currentWeek);
   174→  }
   175→  return weeks;
   176→}
   177→
   178→function getWeekNumber(date: Date): number {
   179→  const d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));
   180→  const dayNum = d.getUTCDay() || 7;
   181→  d.setUTCDate(d.getUTCDate() + 4 - dayNum);
   182→  const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
   183→  return Math.ceil(((d.getTime() - yearStart.getTime()) / 86400000 + 1) / 7);
   184→}
   185→
   186→function dateDiffDays(a: Date, b: Date): number {
   187→  return Math.floor((a.getTime() - b.getTime()) / (1000 * 60 * 60 * 24));
   188→}
   189→
   190→function parseDate(dateStr: string): Date {
   191→  return new Date(dateStr + 'T00:00:00');
   192→}
   193→
   194→function formatDateRange(startDate: string, endDate: string): string {
   195→  const start = new Date(startDate);
   196→  const end = new Date(endDate);
   197→  const monthNames = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   198→
   199→  if (startDate === endDate) {
   200→    return `${monthNames[start.getMonth()]} ${start.getDate()}`;
   201→  }
   202→
   203→  if (start.getMonth() === end.getMonth()) {
   204→    return `${monthNames[start.getMonth()]} ${start.getDate()}-${end.getDate()}`;
   205→  }
   206→
   207→  return `${monthNames[start.getMonth()]} ${start.getDate()} - ${monthNames[end.getMonth()]} ${end.getDate()}`;
   208→}

   210→function buildTrainGroups(pmBlocks: OperationalPmBlock[]): Map<string, OperationalPmBlock[]> {
   211→  const groups = new Map<string, OperationalPmBlock[]>();
   212→
   213→  for (const pm of pmBlocks) {
   214→    if (!groups.has(pm.trainId)) {
   215→      groups.set(pm.trainId, []);
   216→    }
   217→    groups.get(pm.trainId)!.push(pm);
   218→  }
   219→
   220→  return groups;
   221→}

   373→function GridLayer({ days, rowCount, offsetX }: GridLayerProps) {

   387→          strokeWidth={0.5}

   389→      ))}
   390→
   391→      {/* Today highlight column */}
   392→      {todayIndex >= 0 && (
   393→        <Rect
   394→          x={todayIndex * LAYOUT.DAY_WIDTH}
   395→          y={0}
   396→          width={LAYOUT.DAY_WIDTH}
   397→          height={rowCount * LAYOUT.ROW_HEIGHT}
   398→          fill="rgba(37, 99, 235, 0.05)"

   401→

   403→      {days.filter((d) => d.isWeekStart).map((day) => {
   404→        const x = days.indexOf(day) * LAYOUT.DAY_WIDTH;
   405→        return (
   406→          <Line

   408→            points={[x, 0, x, rowCount * LAYOUT.ROW_HEIGHT]}
   409→            stroke="#cbd5e1"
   410→            strokeWidth={1}
   411→          />
   412→        );
   413→      })}
   414→
   415→      {/* Today indicator line */}
   416→      {todayIndex >= 0 && (
   417→        <Line
   418→          points={[
   419→            todayIndex * LAYOUT.DAY_WIDTH + LAYOUT.DAY_WIDTH / 2,
   420→            0,
   421→            todayIndex * LAYOUT.DAY_WIDTH + LAYOUT.DAY_WIDTH / 2,

   424→          stroke="#2563eb"
   425→          strokeWidth={2}

   428→      )}

   445→function PmCard({
   446→  pm,
   447→  x,
   448→  y,
   449→  width,
   450→  isSelected,
   451→  onHover,
   452→  onClick,
   453→}: PmCardProps) {
   454→  const statusColor = pm.displayColor || getPmStatusColor(pm.status);
   455→  const cardWidth = Math.max(width, LAYOUT.MIN_CARD_WIDTH);
   456→  const cardHeight = LAYOUT.CARD_HEIGHT;
   457→
   458→  // Calculate compliance window progress
   459→  const constraintStart = parseDate(pm.constraintStart);
   460→  const constraintEnd = parseDate(pm.constraintEnd);
   461→  const targetDate = parseDate(pm.targetDate);
   462→  const totalWindow = dateDiffDays(constraintEnd, constraintStart) || 1;
   463→  const positionInWindow = dateDiffDays(targetDate, constraintStart);
   464→  const progressPct = Math.min(100, Math.max(0, (positionInWindow / totalWindow) * 100));
   465→
   466→  // Determine compliance status
   467→  let complianceColor: string = CONSTRAINT_COLORS.ok;
   468→  if (progressPct > 90) {
   469→    complianceColor = CONSTRAINT_COLORS.hardViolation;
   470→  } else if (progressPct > 70) {
   471→    complianceColor = CONSTRAINT_COLORS.approaching;
   472→  }
   473→
   474→  return (
   475→    <Group
   476→      x={x}
   477→      y={y}
   478→      onMouseEnter={() => onHover?.(pm)}
   479→      onMouseLeave={() => onHover?.(null)}
   480→      onClick={() => onClick?.(pm.pmId)}
   481→    >
   482→      {/* Selection highlight */}
   483→      {isSelected && (
   484→        <Rect
   485→          x={-3}
   486→          y={-3}
   487→          width={cardWidth + 6}
   488→          height={cardHeight + 6}
   489→          fill="transparent"
   490→          stroke="#3b82f6"
   491→          strokeWidth={2}
   492→          cornerRadius={10}
   493→        />
   494→      )}
   495→
   496→      {/* Card background */}
   497→      <Rect
   498→        width={cardWidth}
   499→        height={cardHeight}
   500→        fill={LAYOUT.CARD_BG}
   501→        stroke={isSelected ? '#2563eb' : '#e2e8f0'}
   502→        strokeWidth={1}
   503→        cornerRadius={8}
   504→        shadowColor="#000"
   505→        shadowBlur={4}
   506→        shadowOpacity={0.08}
   507→        shadowOffsetY={2}
   508→      />
   509→
   510→      {/* Status color bar */}
   511→      <Rect
   512→        x={0}
   513→        y={0}
   514→        width={8}
   515→        height={cardHeight}
   516→        fill={statusColor}
   517→        cornerRadius={[8, 0, 0, 8]}
   518→      />
   519→
   520→      {/* Header section */}
   521→      <Group x={LAYOUT.CARD_PADDING + 4} y={LAYOUT.CARD_PADDING}>
   522→        {/* PM Code */}
   523→        <Text
   524→          text={pm.pmCode}
   525→          fontSize={14}
   526→          fontFamily={LAYOUT.FONT_FAMILY}
   527→          fill="#1e293b"
   528→          fontStyle="bold"
   529→        />
   530→
   531→        {/* Train number */}
   532→        <Text
   533→          y={18}
   534→          text={pm.trainNumber}
   535→          fontSize={11}
   536→          fontFamily={LAYOUT.FONT_FAMILY}
   537→          fill="#64748b"
   538→        />
   539→
   540→        {/* Lock indicator */}
   541→        {pm.isLocked && (
   542→          <Text
   543→            x={cardWidth - 40}
   544→            y={0}
   545→            text="🔒"
   546→            fontSize={14}
   547→          />
   548→        )}
   549→
   550→        {/* Status badge */}
   551→        <Group x={cardWidth - 90} y={0}>
   552→          <Rect
   553→            width={60}
   554→            height={20}
   555→            fill={statusColor}
   556→            cornerRadius={4}
   557→            opacity={0.2}
   558→          />
   559→          <Text
   560→            x={4}
   561→            y={4}
   562→            text={pm.status.replace('_', ' ')}
   563→            fontSize={9}
   564→            fontFamily={LAYOUT.FONT_FAMILY}
   565→            fill={statusColor}
   566→            fontStyle="bold"
   567→          />
   568→        </Group>
   569→      </Group>
   570→
   571→      {/* Date and details section */}
   572→      <Group x={LAYOUT.CARD_PADDING + 4} y={48}>
   573→        {/* Date range */}
   574→        <Text
   575→          text={formatDateRange(pm.targetDate, pm.endDate)}
   576→          fontSize={11}
   577→          fontFamily={LAYOUT.FONT_FAMILY}
   578→          fill="#475569"
   579→        />
   580→
   581→        {/* Duration */}
   582→        <Text
   583→          x={100}
   584→          text={`${pm.durationDays}d`}
   585→          fontSize={11}
   586→          fontFamily={LAYOUT.FONT_FAMILY}
   587→          fill="#64748b"
   588→        />
   589→
   590→        {/* Hours */}
   591→        {pm.estimatedHours && (
   592→          <Text
   593→            x={130}
   594→            text={`${pm.estimatedHours}h`}
   595→            fontSize={11}
   596→            fontFamily={LAYOUT.FONT_FAMILY}
   597→            fill="#64748b"
   598→          />

   617→            fill="#94a3b8"
   618→          />

   622→      {/* Compliance window visualization */}

   624→        <Text
   625→          text="Compliance Window"
   626→          fontSize={9}
   627→          fontFamily={LAYOUT.FONT_FAMILY}
   628→          fill="#94a3b8"
   629→        />
   630→
   631→        {/* Progress bar background */}
   632→        <Rect
   633→          y={14}
   634→          width={cardWidth - LAYOUT.CARD_PADDING * 2 - 8}
   635→          height={8}
   636→          fill="#f1f5f9"
   637→          cornerRadius={4}
   638→        />
   639→
   640→        {/* Progress bar fill */}

   731→export function OperationalDetailLayer({
   732→  data,
   733→  fromDate,
   734→  toDate,
   735→  width,
   736→  height,
   737→  selectedPmIds = [],
   738→  onPmClick,
   739→  onPmHover,
   740→  showGrid = true,
   741→}: OperationalDetailLayerProps) {
   742→  const stageRef = useRef<StageType>(null);
   743→
   744→  if (!data) {
   745→    return null;
   746→  }
   747→
   748→  // Compute derived data
   749→  const days = getDays(fromDate, toDate);
   750→  const weeks = getWeeks(days);
   751→  const trainGroups = buildTrainGroups(data.pmBlocks);
   752→  const trainOrder = Array.from(trainGroups.keys());
   753→  const selectedSet = new Set(selectedPmIds);
   754→
   755→  const gridOffsetX = LAYOUT.TRAIN_LABEL_WIDTH;
   756→  const contentWidth = days.length * LAYOUT.DAY_WIDTH + gridOffsetX;
   757→  const contentHeight = trainOrder.length * LAYOUT.ROW_HEIGHT + LAYOUT.HEADER_HEIGHT;
   758→
   759→  return (
   760→    <Stage
   761→      ref={stageRef}
   762→      width={width}
   763→      height={height}
   764→      style={{ backgroundColor: '#f8fafc' }}
   765→    >
   766→      {/* Background layer */}
   767→      <Layer>
   768→        <Rect width={contentWidth} height={contentHeight} fill="#ffffff" />
   769→      </Layer>
   770→
   771→      {/* Grid layer */}
   772→      {showGrid && (
   773→        <Layer>
   774→          <GridLayer days={days} rowCount={trainOrder.length} offsetX={gridOffsetX} />
   775→        </Layer>
   776→      )}
   777→
   778→      {/* PM Cards layer */}
   779→      <Layer>
   780→        <Group x={gridOffsetX} y={LAYOUT.HEADER_HEIGHT}>
   781→          {trainOrder.map((trainId, trainIndex) => {
   782→            const pms = trainGroups.get(trainId) || [];
   783→
   784→            return pms.map((pm) => {
   785→              const pmDate = parseDate(pm.targetDate);
   786→              const dayOffset = dateDiffDays(pmDate, fromDate);
   787→              if (dayOffset < 0 || dayOffset >= days.length) return null;
   788→
   789→              const x = dayOffset * LAYOUT.DAY_WIDTH + LAYOUT.CARD_MARGIN;
   790→              const y = trainIndex * LAYOUT.ROW_HEIGHT + LAYOUT.CARD_MARGIN;
   791→              const cardWidth = pm.durationDays * LAYOUT.DAY_WIDTH - LAYOUT.CARD_MARGIN * 2;
   792→
   793→              return (
   794→                <PmCard
   795→                  key={pm.pmId}
   796→                  pm={pm}
   797→                  x={x}
   798→                  y={y}
   799→                  width={cardWidth}
   800→                  isSelected={selectedSet.has(pm.pmId)}
   801→                  onHover={onPmHover}
   802→                  onClick={onPmClick}
   803→                />
   804→              );
   805→            });
   806→          })}
   807→        </Group>
   808→      </Layer>
   809→
   810→      {/* Timeline header layer (on top) */}
   811→      <Layer>
   812→        <TimelineHeader days={days} weeks={weeks} offsetX={gridOffsetX} />
   813→      </Layer>
   814→
   815→      {/* Train labels layer (fixed left) */}
   816→      <Layer>
   817→        <TrainLabelsPanel trainGroups={trainGroups} trainOrder={trainOrder} />
   818→      </Layer>
   819→    </Stage>
   820→  );
   821→}
   822→
   823→export default OperationalDetailLayer;
