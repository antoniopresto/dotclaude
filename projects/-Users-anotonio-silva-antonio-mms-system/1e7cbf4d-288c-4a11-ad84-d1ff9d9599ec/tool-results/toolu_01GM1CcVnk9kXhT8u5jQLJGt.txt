    13→// =============================================================================

    17→export const ZOOM_THRESHOLDS = {
    18→  STRATEGIC: { min: 0, max: 4, granularity: 'year' as const },
    19→  TACTICAL_WIDE: { min: 4, max: 8, granularity: 'month' as const },
    20→  TACTICAL: { min: 8, max: 12, granularity: 'week' as const },
    21→  OPERATIONAL: { min: 12, max: 16, granularity: 'day' as const },
    22→  DETAIL: { min: 16, max: 20, granularity: 'hour' as const },
    23→};
    24→
    25→export function getZoomLevelName(zoom: number): ZoomLevel {
    26→  if (zoom < ZOOM_THRESHOLDS.TACTICAL_WIDE.min) return 'STRATEGIC';
    27→  if (zoom < ZOOM_THRESHOLDS.TACTICAL.min) return 'TACTICAL_WIDE';
    28→  if (zoom < ZOOM_THRESHOLDS.OPERATIONAL.min) return 'TACTICAL';
    29→  if (zoom < ZOOM_THRESHOLDS.DETAIL.min) return 'OPERATIONAL';
    30→  return 'DETAIL';
    31→}

    98→export const canvasSlice = createSlice({
    99→  id: 'canvas',
   100→  initial: initialCanvasState,
   101→  cache: { policy: 'session' },
   102→  reducers: {
   103→    // Viewport Control
   104→    setZoom: (s, zoom: number) => {
   105→      const state = s as CanvasSliceState;
   106→      const clampedZoom = Math.max(0, Math.min(20, zoom));
   107→      state.zoom = clampedZoom;
   108→      state.zoomLevel = getZoomLevelName(clampedZoom);
   109→    },
   110→
   111→    zoomIn: (s, factor = 1.2) => {
   112→      const state = s as CanvasSliceState;
   113→      const newZoom = Math.min(20, state.zoom * factor);
   114→      state.zoom = newZoom;
   115→      state.zoomLevel = getZoomLevelName(newZoom);
   116→    },
   117→
   118→    zoomOut: (s, factor = 1.2) => {
   119→      const state = s as CanvasSliceState;
   120→      const newZoom = Math.max(0, state.zoom / factor);
   121→      state.zoom = newZoom;
   122→      state.zoomLevel = getZoomLevelName(newZoom);
   123→    },
   124→
   125→    setZoomLevel: (s, level: ZoomLevel) => {
   126→      const state = s as CanvasSliceState;
   127→      const thresholds = ZOOM_THRESHOLDS[level];
   128→      const targetZoom = (thresholds.min + thresholds.max) / 2;
   129→      state.zoom = targetZoom;
   130→      state.zoomLevel = level;
   131→    },
   132→
   133→    setPan: (s, payload: { x: number; y: number }) => {
   134→      const state = s as CanvasSliceState;
   135→      state.panX = payload.x;
   136→      state.panY = payload.y;
   137→    },
   138→
   139→    pan: (s, delta: { dx: number; dy: number }) => {
   140→      const state = s as CanvasSliceState;
   141→      state.panX += delta.dx;
   142→      state.panY += delta.dy;
   143→    },
   144→
   145→    setViewport: (s, viewport: Partial<ViewportBounds>) => {
   146→      const state = s as CanvasSliceState;
   147→      state.viewport = { ...state.viewport, ...viewport };
   148→    },
   149→
   150→    // Layer Control
   151→    setLayerVisibility: (s, layers: Partial<LayerVisibility>) => {
   152→      const state = s as CanvasSliceState;
   153→      state.layers = { ...state.layers, ...layers };
   154→    },
   155→
   156→    toggleLayer: (s, layer: keyof LayerVisibility) => {
   157→      const state = s as CanvasSliceState;
   158→      state.layers[layer] = !state.layers[layer];
   159→    },
   160→
   161→    showAllLayers: (s) => {
   162→      const state = s as CanvasSliceState;
   163→      state.layers = {
   164→        pmBlocks: true,
   165→        complianceWindows: true,
   166→        manhourOverlay: true,
   167→        driftOverlay: true,
   168→        baselineShadow: true,
   169→      };
   170→    },
   171→
   172→    hideAllOverlays: (s) => {
   173→      const state = s as CanvasSliceState;
   174→      state.layers.manhourOverlay = false;
   175→      state.layers.driftOverlay = false;
   176→      state.layers.baselineShadow = false;
   177→    },
   178→
   179→    // Selection
   180→    selectPm: (s, pmId: string) => {
   181→      const state = s as CanvasSliceState;
   182→      if (state.selection.isMultiSelect) {
   183→        if (!state.selection.selectedPmIds.includes(pmId)) {
   184→          state.selection.selectedPmIds.push(pmId);
   185→        }
   186→      } else {
   187→        state.selection.selectedPmIds = [pmId];
   188→      }
   189→    },
   190→
   191→    deselectPm: (s, pmId: string) => {
   192→      const state = s as CanvasSliceState;
   193→      state.selection.selectedPmIds = state.selection.selectedPmIds.filter((id) => id !== pmId);
   194→    },
   195→
   196→    selectTrain: (s, trainId: string) => {
   197→      const state = s as CanvasSliceState;
   198→      if (state.selection.isMultiSelect) {
   199→        if (!state.selection.selectedTrainIds.includes(trainId)) {
   200→          state.selection.selectedTrainIds.push(trainId);
   201→        }
   202→      } else {
   203→        state.selection.selectedTrainIds = [trainId];
   204→      }
   205→    },
   206→
   207→    deselectTrain: (s, trainId: string) => {
   208→      const state = s as CanvasSliceState;
   209→      state.selection.selectedTrainIds = state.selection.selectedTrainIds.filter(
   210→        (id) => id !== trainId
   211→      );
   212→    },
   213→
   214→    setMultiSelect: (s, enabled: boolean) => {
   215→      const state = s as CanvasSliceState;
   216→      state.selection.isMultiSelect = enabled;
   217→    },
   218→
   219→    clearSelection: (s) => {
   220→      const state = s as CanvasSliceState;
   221→      state.selection.selectedPmIds = [];
   222→      state.selection.selectedTrainIds = [];
   223→    },
   224→
   225→    selectAll: (s, payload: { pmIds?: string[]; trainIds?: string[] }) => {
   226→      const state = s as CanvasSliceState;
   227→      if (payload.pmIds) {
   228→        state.selection.selectedPmIds = payload.pmIds;
   229→      }
   230→      if (payload.trainIds) {
   231→        state.selection.selectedTrainIds = payload.trainIds;
   232→      }
   233→    },
   234→
   235→    // Navigation
   236→    goToToday: (s) => {
   237→      const state = s as CanvasSliceState;
   238→      state.todayDate = new Date().toISOString().split('T')[0];
   239→      state.panX = 0;
   240→    },
   241→
   242→    goToDate: (s, date: string) => {
   243→      const state = s as CanvasSliceState;
   244→      state.viewport.startDate = date;
   245→    },
   246→
   247→    // UI Controls
   248→    toggleMinimap: (s) => {
   249→      const state = s as CanvasSliceState;
   250→      state.showMinimap = !state.showMinimap;
   251→    },
   252→
   253→    toggleGrid: (s) => {
   254→      const state = s as CanvasSliceState;
   255→      state.showGrid = !state.showGrid;
   256→    },
   257→
   258→    setPerformanceMode: (s, enabled: boolean) => {
   259→      const state = s as CanvasSliceState;
   260→      state.isPerformanceMode = enabled;
   261→    },
   262→
   263→    lockViewport: (s) => {
   264→      const state = s as CanvasSliceState;
   265→      state.isViewportLocked = true;
   266→    },
   267→
   268→    unlockViewport: (s) => {
   269→      const state = s as CanvasSliceState;
   270→      state.isViewportLocked = false;
   271→    },
   272→
   273→    // Interaction State
   274→    startPanning: (s) => {
   275→      const state = s as CanvasSliceState;
   276→      state.isPanning = true;
   277→    },
   278→
   279→    stopPanning: (s) => {
   280→      const state = s as CanvasSliceState;
   281→      state.isPanning = false;
   282→    },
   283→
   284→    startZooming: (s) => {
   285→      const state = s as CanvasSliceState;
   286→      state.isZooming = true;
   287→    },
   288→
   289→    stopZooming: (s) => {
   290→      const state = s as CanvasSliceState;
   291→      state.isZooming = false;
   292→    },
   293→
   294→    // Animation
   295→    setAnimationTarget: (
   296→      s,
   297→      target: { zoom?: number | null; panX?: number | null; panY?: number | null }
   298→    ) => {
   299→      const state = s as CanvasSliceState;
   300→      state.animationTarget = {
   301→        zoom: target.zoom ?? state.animationTarget.zoom,
   302→        panX: target.panX ?? state.animationTarget.panX,
   303→        panY: target.panY ?? state.animationTarget.panY,
   304→      };
   305→    },
   306→
   307→    clearAnimationTarget: (s) => {
   308→      const state = s as CanvasSliceState;
   309→      state.animationTarget = { zoom: null, panX: null, panY: null };
   310→    },
   311→
   312→    // Reset
   313→    reset: (s) => {
   314→      const state = s as CanvasSliceState;
   315→      Object.assign(state, initialCanvasState);
   316→    },
   317→
   318→    resetViewport: (s) => {
   319→      const state = s as CanvasSliceState;
   320→      state.zoom = 10;
   321→      state.panX = 0;
   322→      state.panY = 0;
   323→      state.zoomLevel = 'TACTICAL';
   324→    },
   325→  },
   326→});

   529→function getDefaultTacticalDates(): { from: string; to: string } {
   530→  const today = new Date();
   531→  const fromDate = new Date(today);
   532→  fromDate.setDate(today.getDate() - today.getDay()); // Start of current week
   533→  const toDate = new Date(fromDate);
   534→  toDate.setDate(toDate.getDate() + 12 * 7); // 12 weeks ahead
   535→  return {
   536→    from: fromDate.toISOString().split('T')[0],
   537→    to: toDate.toISOString().split('T')[0],
   538→  };
   539→}
   540→
   541→function getDefaultTacticalWideDates(): { from: string; to: string } {
   542→  const today = new Date();
   543→  const fromDate = new Date(today.getFullYear(), today.getMonth(), 1); // Start of current month
   544→  const toDate = new Date(fromDate);
   545→  toDate.setMonth(toDate.getMonth() + 12); // 12 months ahead
   546→  return {
   547→    from: fromDate.toISOString().split('T')[0],
   548→    to: toDate.toISOString().split('T')[0],
   549→  };
   550→}
   551→
   552→function getDefaultOperationalDates(): { from: string; to: string } {
   553→  const today = new Date();
   554→  const fromDate = new Date(today);
   555→  fromDate.setDate(today.getDate() - today.getDay()); // Start of current week
   556→  const toDate = new Date(fromDate);
   557→  toDate.setDate(toDate.getDate() + 4 * 7); // 4 weeks ahead
   558→  return {
   559→    from: fromDate.toISOString().split('T')[0],
   560→    to: toDate.toISOString().split('T')[0],
   561→  };
   562→}
