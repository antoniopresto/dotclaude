    63вҶ’## 7.3 Mileage Data Ingestion
    64вҶ’
    65вҶ’### 7.3.1 CSV File Format
    66вҶ’
    67вҶ’**Filename Pattern:** `mileage_YYYYMMDD.csv`  
    68вҶ’**Encoding:** UTF-8  
    69вҶ’**Delimiter:** Comma  
    70вҶ’**Header:** Required
    71вҶ’
    72вҶ’```csv
    73вҶ’train_number,reading_date,odometer_km,source
    74вҶ’1001,2026-01-14,245152,ATS
    75вҶ’1002,2026-01-14,248523,ATS
    76вҶ’1003,2026-01-14,242087,ATS
    77вҶ’```
    78вҶ’
    79вҶ’### 7.3.2 Field Specifications
    80вҶ’
    81вҶ’| Field | Type | Required | Validation |
    82вҶ’|-------|------|----------|------------|
    83вҶ’| `train_number` | string | Yes | Must exist in trains table |
    84вҶ’| `reading_date` | date | Yes | Format: YYYY-MM-DD, not future |
    85вҶ’| `odometer_km` | integer | Yes | Positive, вүӨ previous + 2000 |
    86вҶ’| `source` | string | No | Default: 'CSV' |
    87вҶ’
    88вҶ’### 7.3.3 Validation Rules
    89вҶ’
    90вҶ’```typescript
    91вҶ’interface MileageValidationRules {
    92вҶ’  // Format validation
    93вҶ’  maxFileSize: '10MB';
    94вҶ’  requiredColumns: ['train_number', 'reading_date', 'odometer_km'];
    95вҶ’  dateFormat: 'YYYY-MM-DD';
    96вҶ’  
    97вҶ’  // Business validation
    98вҶ’  maxDailyIncrease: 2000;           // km - anything more is anomaly
    99вҶ’  maxDailyDecrease: 0;              // Negative not allowed without meter event
   100вҶ’  duplicateHandling: 'UPDATE';       // UPDATE or REJECT
   101вҶ’  unknownTrainHandling: 'QUARANTINE'; // QUARANTINE or REJECT_FILE
   102вҶ’  
   103вҶ’  // Timing
   104вҶ’  expectedArrivalWindow: {
   105вҶ’    start: '05:00',                 // Local time
   106вҶ’    end: '07:00'
   107вҶ’  };
   108вҶ’  alertIfMissingAfter: '08:00';
   109вҶ’}
   110вҶ’```

   165вҶ’### 7.4.2 Shadow Record Pattern
   166вҶ’
   167вҶ’MMS maintains "shadow records" that mirror Maximo work orders, allowing operations to continue if Maximo is unavailable.
   168вҶ’
   169вҶ’```typescript
   170вҶ’interface MaximoShadowRecord {
   171вҶ’  mmsId: string;                    // MMS primary key
   172вҶ’  maximoWoId: string | null;        // Maximo work order ID (null if not synced)
   173вҶ’  syncStatus: 'PENDING' | 'SYNCED' | 'CONFLICT' | 'ERROR';
   174вҶ’  lastSyncAt: Date | null;
   175вҶ’  lastSyncError: string | null;
   176вҶ’  
   177вҶ’  // MMS data (source of truth for scheduling)
   178вҶ’  mmsData: {
   179вҶ’    trainId: string;
   180вҶ’    pmTypeCode: string;
   181вҶ’    targetDate: Date;
   182вҶ’    status: string;
   183вҶ’  };
   184вҶ’  
   185вҶ’  // Maximo data (for reconciliation)
   186вҶ’  maximoData: {
   187вҶ’    status: string;
   188вҶ’    actualStart: Date | null;
   189вҶ’    actualFinish: Date | null;
   190вҶ’    actualLabor: number | null;
   191вҶ’  } | null;
   192вҶ’}
   193вҶ’```

   208вҶ’### 7.4.4 Work Order Creation
   209вҶ’
   210вҶ’When MMS schedules a PM, it creates a shadow record. On sync:
   211вҶ’
   212вҶ’```typescript
   213вҶ’interface MaximoWorkOrderCreate {
   214вҶ’  // Required fields
   215вҶ’  siteid: string;                  // 'METRO'
   216вҶ’  woclass: string;                 // 'WORKORDER'
   217вҶ’  worktype: string;                // 'PM'
   218вҶ’  assetnum: string;                // Train number
   219вҶ’  description: string;             // PM description
   220вҶ’  schedstart: string;              // ISO 8601 date
   221вҶ’  schedfinish: string;             // ISO 8601 date
   222вҶ’  
   223вҶ’  // Optional fields
   224вҶ’  location: string;                // Depot code
   225вҶ’  estlabhrs: number;               // Estimated hours
   226вҶ’  pmnum: string;                   // PM type reference

   258вҶ’
   259вҶ’| Maximo Status | MMS Status | Sync Action | MMS Edit Lock |
   260вҶ’|---------------|------------|-------------|---------------|
   261вҶ’| WAPPR | SCHEDULED | None (initial state) | вқҢ Unlocked |
   262вҶ’| APPR | SCHEDULED | None | вқҢ Unlocked |

   265вҶ’| CLOSE | COMPLETED | Final state | вң… **LOCKED** |
   266вҶ’| CAN | CANCELLED | Update MMS status | вң… **LOCKED** |

   268вҶ’### 7.4.6 Status-Based Edit Locking
   269вҶ’
   270вҶ’**CRITICAL:** When a Maximo work order is in progress (INPRG), the corresponding PM record in MMS must be **read-only**. This prevents race conditions where a planner reschedules a PM that technicians are actively working on.
   271вҶ’
   272вҶ’```typescript
   273вҶ’/**
   274вҶ’ * Validate PM edit request against Maximo status.
   275вҶ’ * MUST be called before any PM schedule modification.
   276вҶ’ * 
   277вҶ’ * This addresses the race condition where:
   278вҶ’ * 1. Technician opens work order on tablet (Maximo status = INPRG)
   279вҶ’ * 2. Planner reschedules PM in MMS (unaware of in-progress work)
   280вҶ’ * 3. Maximo sync fails or overwrites technician's work
   281вҶ’ * 
   282вҶ’ * Operational reality MUST override planning data.
   283вҶ’ */
   284вҶ’class PMEditValidator {
   285вҶ’  async validateEdit(pmId: string): Promise<EditValidationResult> {
   286вҶ’    const pm = await this.getPMWithShadow(pmId);
   287вҶ’    
   288вҶ’    // Check 1: Maximo status lock
   289вҶ’    if (pm.shadowRecord?.maximoData?.status) {
   290вҶ’      const lockedStatuses = ['INPRG', 'COMP', 'CLOSE', 'CAN'];
   291вҶ’      
   292вҶ’      if (lockedStatuses.includes(pm.shadowRecord.maximoData.status)) {
   293вҶ’        return {
   294вҶ’          valid: false,
   295вҶ’          reason: 'MAXIMO_STATUS_LOCK',
   296вҶ’          message: `Cannot modify: Work order is ${this.formatStatus(pm.shadowRecord.maximoData.status)} in Maximo`,
   297вҶ’          maximoStatus: pm.shadowRecord.maximoData.status,
   298вҶ’          maximoWonum: pm.maximoWoId,
   299вҶ’          suggestion: pm.shadowRecord.maximoData.status === 'INPRG' 
   300вҶ’            ? 'Wait for technicians to complete work, or contact supervisor to cancel in Maximo first'
   301вҶ’            : 'This PM has been completed/cancelled. Create a new PM if rescheduling is needed.'
   302вҶ’        };
   303вҶ’      }
   304вҶ’    }
   305вҶ’    
   306вҶ’    // Check 2: Pending sync conflict
   307вҶ’    if (pm.shadowRecord?.syncStatus === 'CONFLICT') {
   308вҶ’      return {
   309вҶ’        valid: false,
   310вҶ’        reason: 'SYNC_CONFLICT',
   311вҶ’        message: 'Unresolved sync conflict exists. Resolve before editing.',
   312вҶ’        suggestion: 'Go to Integration > Conflicts to resolve'
   313вҶ’      };
   314вҶ’    }
   315вҶ’    
   316вҶ’    // Check 3: Baseline lock (optional, if locked in current baseline)
   317вҶ’    if (pm.lockedInBaseline) {
   318вҶ’      return {
   319вҶ’        valid: false,
   320вҶ’        reason: 'BASELINE_LOCK',
   321вҶ’        message: 'PM is locked in published baseline',
   322вҶ’        suggestion: 'Unlock in baseline editor or create scenario to test changes'
   323вҶ’      };
   324вҶ’    }
   325вҶ’    
   326вҶ’    return { valid: true };
   327вҶ’  }
   328вҶ’  
   329вҶ’  private formatStatus(status: string): string {
   330вҶ’    const statusLabels: Record<string, string> = {
   331вҶ’      'INPRG': 'In Progress',
   332вҶ’      'COMP': 'Completed',
   333вҶ’      'CLOSE': 'Closed',
   334вҶ’      'CAN': 'Cancelled'
   335вҶ’    };
   336вҶ’    return statusLabels[status] || status;
   337вҶ’  }
   338вҶ’}
   339вҶ’
   340вҶ’interface EditValidationResult {
   341вҶ’  valid: boolean;
   342вҶ’  reason?: 'MAXIMO_STATUS_LOCK' | 'SYNC_CONFLICT' | 'BASELINE_LOCK';
   343вҶ’  message?: string;
   344вҶ’  maximoStatus?: string;
   345вҶ’  maximoWonum?: string;
   346вҶ’  suggestion?: string;
   347вҶ’}
   348вҶ’```
   349вҶ’

   392вҶ’### 7.4.7 Bidirectional Sync Logic
   393вҶ’
   394вҶ’```typescript
   395вҶ’class MaximoSyncService {
   396вҶ’  /**
   397вҶ’   * Push MMS changes to Maximo
   398вҶ’   */
   399вҶ’  async pushToMaximo(pmId: string): Promise<SyncResult> {
   400вҶ’    const pm = await this.getPMWithShadow(pmId);
   401вҶ’    
   402вҶ’    if (!pm.maximoWoId) {
   403вҶ’      // Create new work order
   404вҶ’      const wonum = await this.createWorkOrder(pm);
   405вҶ’      await this.updateShadowRecord(pm.id, { 
   406вҶ’        maximoWoId: wonum, 
   407вҶ’        syncStatus: 'SYNCED' 
   408вҶ’      });
   409вҶ’      return { action: 'CREATED', wonum };
   410вҶ’    }
   411вҶ’    
   412вҶ’    // Update existing work order
   413вҶ’    const maximoWO = await this.getMaximoWorkOrder(pm.maximoWoId);
   414вҶ’    
   415вҶ’    if (maximoWO.status === 'INPRG' || maximoWO.status === 'COMP') {
   416вҶ’      // Cannot update in-progress or completed WOs
   417вҶ’      return { action: 'SKIPPED', reason: 'WO already in progress' };
   418вҶ’    }
   419вҶ’    
   420вҶ’    // Check for conflicts
   421вҶ’    if (new Date(maximoWO.changedate) > pm.shadowRecord.lastSyncAt) {
   422вҶ’      await this.updateShadowRecord(pm.id, { syncStatus: 'CONFLICT' });
   423вҶ’      return { action: 'CONFLICT', maximoData: maximoWO };
   424вҶ’    }
   425вҶ’    
   426вҶ’    // Push update
   427вҶ’    await this.updateWorkOrder(pm.maximoWoId, {
   428вҶ’      schedstart: pm.targetDate.toISOString(),
   429вҶ’      schedfinish: addDays(pm.targetDate, pm.pmType.minTrackDays).toISOString(),
   430вҶ’      location: pm.assignedDepot.code
   431вҶ’    });
   432вҶ’    
   433вҶ’    await this.updateShadowRecord(pm.id, { 
   434вҶ’      syncStatus: 'SYNCED',
   435вҶ’      lastSyncAt: new Date()
   436вҶ’    });
   437вҶ’    
   438вҶ’    return { action: 'UPDATED', wonum: pm.maximoWoId };
   439вҶ’  }
   440вҶ’  
   441вҶ’  /**
   442вҶ’   * Pull Maximo changes to MMS
   443вҶ’   */
   444вҶ’  async pullFromMaximo(): Promise<PullResult> {
   445вҶ’    // Get recently modified work orders
   446вҶ’    const modifiedSince = await this.getLastPullTimestamp();
   447вҶ’    const workOrders = await this.queryMaximoWorkOrders({
   448вҶ’      worktype: 'PM',
   449вҶ’      changedate: `>${modifiedSince.toISOString()}`
   450вҶ’    });
   451вҶ’    
   452вҶ’    const results: PullResult = {
   453вҶ’      processed: 0,
   454вҶ’      updated: 0,
   455вҶ’      conflicts: 0,
   456вҶ’      errors: 0
   457вҶ’    };
   458вҶ’    
   459вҶ’    for (const wo of workOrders) {
   460вҶ’      results.processed++;
   461вҶ’      
   462вҶ’      try {
   463вҶ’        const pm = await this.findPMByMaximoId(wo.wonum);
   464вҶ’        
   465вҶ’        if (!pm) {
   466вҶ’          // Work order not managed by MMS
   467вҶ’          continue;
   468вҶ’        }
   469вҶ’        
   470вҶ’        // Status change
   471вҶ’        if (wo.status !== pm.shadowRecord.maximoData?.status) {
   472вҶ’          await this.handleStatusChange(pm, wo);
   473вҶ’          results.updated++;
   474вҶ’        }
   475вҶ’        
   476вҶ’        // Actuals recorded
   477вҶ’        if (wo.actlabhrs && !pm.actualHours) {
   478вҶ’          await this.recordActuals(pm, wo);
   479вҶ’          results.updated++;
   480вҶ’        }
   481вҶ’        
   482вҶ’      } catch (error) {
   483вҶ’        results.errors++;
   484вҶ’        this.logger.error({ wo, error }, 'Pull processing failed');
   485вҶ’      }
   486вҶ’    }
   487вҶ’    
   488вҶ’    await this.setLastPullTimestamp(new Date());
   489вҶ’    return results;
   490вҶ’  }
   491вҶ’  
   492вҶ’  /**
   493вҶ’   * Resolve sync conflict
   494вҶ’   */
   495вҶ’  async resolveConflict(
   496вҶ’    pmId: string, 
   497вҶ’    resolution: 'USE_MMS' | 'USE_MAXIMO'
   498вҶ’  ): Promise<void> {
   499вҶ’    const pm = await this.getPMWithShadow(pmId);
   500вҶ’    
   501вҶ’    if (resolution === 'USE_MMS') {
   502вҶ’      // Force push MMS data to Maximo
   503вҶ’      await this.updateWorkOrder(pm.maximoWoId, {
   504вҶ’        schedstart: pm.targetDate.toISOString(),
   505вҶ’        schedfinish: addDays(pm.targetDate, pm.pmType.minTrackDays).toISOString()
   506вҶ’      }, { force: true });
   507вҶ’    } else {
   508вҶ’      // Apply Maximo data to MMS
   509вҶ’      const maximoWO = await this.getMaximoWorkOrder(pm.maximoWoId);
   510вҶ’      await this.updatePMFromMaximo(pm.id, maximoWO);
   511вҶ’    }
   512вҶ’    
   513вҶ’    await this.updateShadowRecord(pm.id, { 
   514вҶ’      syncStatus: 'SYNCED',
   515вҶ’      lastSyncAt: new Date()
   516вҶ’    });
   517вҶ’  }
   518вҶ’}
   519вҶ’```

   731вҶ’        return this.generatePDF(template, data);
   732вҶ’      case 'XLSX':

   734вҶ’      default:

   736вҶ’    }
   737вҶ’  }
   738вҶ’  
   739вҶ’  private async generatePDF(
   740вҶ’    template: ReportTemplate,
   741вҶ’    data: any
   742вҶ’  ): Promise<GeneratedReport> {
   743вҶ’    // Use Puppeteer for PDF generation
   744вҶ’    const html = await this.renderTemplate(template.htmlTemplate, data);
   745вҶ’    
   746вҶ’    const browser = await puppeteer.launch({ headless: true });
   747вҶ’    const page = await browser.newPage();
   748вҶ’    await page.setContent(html);
   749вҶ’    
   750вҶ’    const pdfBuffer = await page.pdf({
   751вҶ’      format: 'A4',

   753вҶ’      printBackground: true

   755вҶ’    
   756вҶ’    await browser.close();
   757вҶ’    
   758вҶ’    // Store in MinIO
   759вҶ’    const filename = `${template.name}_${format(new Date(), 'yyyyMMdd_HHmmss')}.pdf`;
   760вҶ’    const url = await this.storage.upload(filename, pdfBuffer);
   761вҶ’    
   762вҶ’    return {
   763вҶ’      filename,
   764вҶ’      url,
   765вҶ’      expiresAt: addDays(new Date(), 7),
   766вҶ’      sizeBytes: pdfBuffer.length
   767вҶ’    };
   768вҶ’  }
   769вҶ’  
   770вҶ’  private async generateExcel(
   771вҶ’    template: ReportTemplate,
   772вҶ’    data: any
   773вҶ’  ): Promise<GeneratedReport> {
   774вҶ’    // Use ExcelJS for Excel generation
   775вҶ’    const workbook = new ExcelJS.Workbook();
   776вҶ’    
   777вҶ’    for (const sheetDef of template.sheets) {
   778вҶ’      const sheet = workbook.addWorksheet(sheetDef.name);
   779вҶ’      
   780вҶ’      // Add headers
   781вҶ’      sheet.addRow(sheetDef.columns.map(c => c.header));
   782вҶ’      
   783вҶ’      // Add data rows
   784вҶ’      const sheetData = data[sheetDef.dataKey];
   785вҶ’      for (const row of sheetData) {
   786вҶ’        sheet.addRow(sheetDef.columns.map(c => row[c.key]));
   787вҶ’      }
   788вҶ’      
   789вҶ’      // Apply formatting
   790вҶ’      this.applyExcelFormatting(sheet, sheetDef);
   791вҶ’    }
   792вҶ’    
   793вҶ’    const buffer = await workbook.xlsx.writeBuffer();
   794вҶ’    
   795вҶ’    const filename = `${template.name}_${format(new Date(), 'yyyyMMdd_HHmmss')}.xlsx`;
   796вҶ’    const url = await this.storage.upload(filename, buffer);
   797вҶ’    
   798вҶ’    return {
   799вҶ’      filename,
   800вҶ’      url,
   801вҶ’      expiresAt: addDays(new Date(), 7),
   802вҶ’      sizeBytes: buffer.length
   803вҶ’    };
   804вҶ’  }
   805вҶ’}

   981вҶ’
   982вҶ’```typescript
   983вҶ’interface CircuitBreakerConfig {
   984вҶ’  failureThreshold: number;    // Failures before opening
   985вҶ’  successThreshold: number;    // Successes before closing
   986вҶ’  timeout: number;             // Time in open state before half-open (ms)
   987вҶ’}
   988вҶ’
   989вҶ’class CircuitBreaker {
   990вҶ’  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED';
   991вҶ’  private failures = 0;
   992вҶ’  private successes = 0;
   993вҶ’  private lastFailureTime: Date | null = null;
   994вҶ’  
   995вҶ’  async execute<T>(fn: () => Promise<T>): Promise<T> {
   996вҶ’    if (this.state === 'OPEN') {
   997вҶ’      if (Date.now() - this.lastFailureTime!.getTime() > this.config.timeout) {
   998вҶ’        this.state = 'HALF_OPEN';
   999вҶ’      } else {
  1000вҶ’        throw new Error('Circuit breaker is OPEN');
  1001вҶ’      }
  1002вҶ’    }
  1003вҶ’    
  1004вҶ’    try {
  1005вҶ’      const result = await fn();
  1006вҶ’      this.onSuccess();
  1007вҶ’      return result;
  1008вҶ’    } catch (error) {
  1009вҶ’      this.onFailure();
  1010вҶ’      throw error;
  1011вҶ’    }
  1012вҶ’  }
  1013вҶ’  
  1014вҶ’  private onSuccess() {
  1015вҶ’    this.failures = 0;
  1016вҶ’    if (this.state === 'HALF_OPEN') {
  1017вҶ’      this.successes++;
  1018вҶ’      if (this.successes >= this.config.successThreshold) {
  1019вҶ’        this.state = 'CLOSED';
  1020вҶ’        this.successes = 0;
  1021вҶ’      }
  1022вҶ’    }
  1023вҶ’  }
  1024вҶ’  
  1025вҶ’  private onFailure() {
  1026вҶ’    this.failures++;
  1027вҶ’    this.lastFailureTime = new Date();
  1028вҶ’    if (this.failures >= this.config.failureThreshold) {
  1029вҶ’      this.state = 'OPEN';
  1030вҶ’    }
  1031вҶ’  }
  1032вҶ’}
