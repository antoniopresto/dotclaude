    38→/**
    39→ * Plan Baselines - Immutable schedule snapshots
    40→ *
    41→ * A baseline captures the state of PM scheduling and mileage projections
    42→ * at a point in time. Published baselines become the reference for drift analysis.
    43→ *
    44→ * Status lifecycle:
    45→ * DRAFT -> PUBLISHED -> SUPERSEDED (when new baseline published)
    46→ */
    47→export const planBaselines = pgTable(
    48→  'plan_baselines',
    49→  {
    50→    id: uuid('id').primaryKey().defaultRandom(),
    51→    name: varchar('name', { length: 100 }).notNull(),
    52→    description: text('description'),
    53→    version: varchar('version', { length: 20 }).notNull(), // 'v3.2'
    54→
    55→    baselineDate: date('baseline_date').notNull(), // "As of" date
    56→    horizonEnd: date('horizon_end').notNull(), // How far into future
    57→
    58→    // Workflow
    59→    status: baselineStatusEnum('status').default('DRAFT'),
    60→    createdBy: uuid('created_by')
    61→      .notNull()
    62→      .references(() => users.id),
    63→    createdAt: timestamp('created_at', { withTimezone: true }).defaultNow(),
    64→
    65→    publishedBy: uuid('published_by').references(() => users.id),
    66→    publishedAt: timestamp('published_at', { withTimezone: true }),
    67→
    68→    supersededBy: uuid('superseded_by'),
    69→    supersededAt: timestamp('superseded_at', { withTimezone: true }),
    70→
    71→    // Metrics at baseline time
    72→    fleetCount: integer('fleet_count'),
    73→    totalPmScheduled: integer('total_pm_scheduled'),
    74→    totalManhourProjected: decimal('total_manhour_projected', { precision: 12, scale: 2 }),
    75→  },
    76→  (table) => [
    77→    index('idx_baselines_status').on(table.status),
    78→    index('idx_baselines_date').on(table.baselineDate),
    79→  ]
    80→);

   151→/**
   152→ * Drift Analysis - Comparison results

   154→ * Generated by drift monitor (nightly job).

   157→export const driftAnalysis = pgTable(
   158→  'drift_analysis',
   159→  {
   160→    id: uuid('id').primaryKey().defaultRandom(),
   161→    baselineId: uuid('baseline_id')
   162→      .notNull()
   163→      .references(() => planBaselines.id),
   164→    analysisDate: date('analysis_date').notNull(),
   165→
   166→    trainId: uuid('train_id')

   169→
   170→    // Mileage drift
   171→    baselineProjectedKm: integer('baseline_projected_km').notNull(),
   172→    actualKm: integer('actual_km').notNull(),
   173→    // Generated column: mileage drift (positive = traveled more than projected)
   174→    mileageDriftKm: integer('mileage_drift_km').generatedAlwaysAs(

   178→
   179→    // Schedule drift (days)

   181→    maxPmDriftDays: integer('max_pm_drift_days'),
   182→
   183→    // Threshold breach
   184→    requiresReplan: boolean('requires_replan').default(false),

   187→    createdAt: timestamp('created_at', { withTimezone: true }).defaultNow(),
   188→  },

   192→    index('idx_drift_replan').on(table.requiresReplan, table.analysisDate),
