    22→import { depots } from './core-fleet';

    24→/**
    25→ * Users - System users
    26→ *
    27→ * Includes planners, depot managers, system admins, and technicians.
    28→ * Each user can have certifications (for technician constraint matching).
    29→ */
    30→export const users = pgTable(
    31→  'users',
    32→  {
    33→    id: uuid('id').primaryKey().defaultRandom(),
    34→    email: varchar('email', { length: 255 }).notNull().unique(),
    35→    passwordHash: varchar('password_hash', { length: 255 }).notNull(),
    36→
    37→    firstName: varchar('first_name', { length: 100 }),
    38→    lastName: varchar('last_name', { length: 100 }),
    39→    displayName: varchar('display_name', { length: 200 }),
    40→
    41→    // Assignment
    42→    primaryDepotId: uuid('primary_depot_id').references(() => depots.id),
    43→    roleIds: text('role_ids').array().notNull().default([]),
    44→
    45→    // Certifications (for technician constraint matching)
    46→    certifications: text('certifications').array(),
    47→
    48→    // Status
    49→    isActive: boolean('is_active').default(true),
    50→    lastLoginAt: timestamp('last_login_at', { withTimezone: true }),
    51→    failedLoginAttempts: integer('failed_login_attempts').default(0),
    52→    lockedUntil: timestamp('locked_until', { withTimezone: true }),
    53→
    54→    createdAt: timestamp('created_at', { withTimezone: true }).defaultNow(),
    55→    updatedAt: timestamp('updated_at', { withTimezone: true }).defaultNow(),
    56→  },
    57→  (table) => [
    58→    index('idx_users_email').on(table.email),
    59→    index('idx_users_depot').on(table.primaryDepotId),
    60→  ]
    61→);
    62→
    63→/**
    64→ * Roles - Permission groups
    65→ *
    66→ * Standard roles include:
    67→ * - PLANNER: Can create/modify PM schedules
    68→ * - DEPOT_MANAGER: Manages depot-specific operations
    69→ * - SYSTEM_ADMIN: Full system access
    70→ * - VIEWER: Read-only access
    71→ *
    72→ * Permissions are stored as JSONB for flexibility:
    73→ * { "pm_schedule": ["read", "write"], "baseline": ["read"] }

    75→export const roles = pgTable('roles', {

    77→  name: varchar('name', { length: 50 }).notNull().unique(),
    78→  description: text('description'),
    79→  permissions: jsonb('permissions').notNull(), // e.g., {"pm_schedule": ["read", "write"]}
    80→  createdAt: timestamp('created_at', { withTimezone: true }).defaultNow(),
    81→});

    83→/**

    98→    createdAt: timestamp('created_at', { withTimezone: true }).defaultNow(),
    99→    expiresAt: timestamp('expires_at', { withTimezone: true }).notNull(),
   100→    lastActivityAt: timestamp('last_activity_at', { withTimezone: true }).defaultNow(),
