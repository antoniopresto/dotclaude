    28→// Singleton service instance

    31→function getService(): ReturnType<typeof getBaselineService> {

    40→/**
    41→ * GET /baselines
    42→ *
    43→ * List plan baselines with filtering and pagination.
    44→ *
    45→ * Query Parameters:
    46→ * - status: Filter by status (DRAFT, PUBLISHED, SUPERSEDED, ARCHIVED)
    47→ * - fromDate: Filter by baseline date (ISO date)
    48→ * - toDate: Filter by baseline date (ISO date)
    49→ * - page: Page number (default: 1)
    50→ * - pageSize: Items per page (default: 20, max: 100)
    51→ * - sort: Sort field (baselineDate:asc, createdAt:desc, etc.)
    52→ */
    53→baselineRoutes.get('/baselines', async (c) => {
    54→  const service = getService();
    55→
    56→  // Parse query filters
    57→  const filters: BaselineListFilters = {};
    58→  const status = c.req.query('status');
    59→  const fromDate = c.req.query('fromDate');
    60→  const toDate = c.req.query('toDate');
    61→
    62→  if (status) {
    63→    const validStatuses: BaselineStatus[] = ['DRAFT', 'PUBLISHED', 'SUPERSEDED', 'ARCHIVED'];
    64→    const statusValues = status.split(',').filter((s) => validStatuses.includes(s as BaselineStatus));
    65→    if (statusValues.length === 1) {
    66→      filters.status = statusValues[0] as BaselineStatus;
    67→    } else if (statusValues.length > 1) {
    68→      filters.status = statusValues as BaselineStatus[];
    69→    }
    70→  }
    71→
    72→  if (fromDate) {
    73→    const parsed = new Date(fromDate);
    74→    if (!isNaN(parsed.getTime())) {
    75→      filters.fromDate = parsed;
    76→    }
    77→  }
    78→
    79→  if (toDate) {
    80→    const parsed = new Date(toDate);
    81→    if (!isNaN(parsed.getTime())) {
    82→      filters.toDate = parsed;
    83→    }
    84→  }
    85→
    86→  // Parse pagination
    87→  const pagination = parsePaginationQuery({
    88→    page: c.req.query('page') ? Number(c.req.query('page')) : undefined,
    89→    pageSize: c.req.query('pageSize') ? Number(c.req.query('pageSize')) : undefined,
    90→    sort: c.req.query('sort'),
    91→  });
    92→
    93→  const result = await service.listBaselines(filters, {
    94→    page: pagination.page,
    95→    pageSize: Math.min(pagination.pageSize, 100), // Cap at 100 for baselines
    96→    sortBy: pagination.sort?.field as 'baselineDate' | 'createdAt' | 'version' | undefined,
    97→    sortOrder: pagination.sort?.direction,
    98→  });
    99→
   100→  // Map to API response format per spec 005.8
   101→  const baselines = result.data.map((baseline) => ({
   102→    id: baseline.id,
   103→    name: baseline.name,
   104→    version: baseline.version,
   105→    description: baseline.description,
   106→    baselineDate: baseline.baselineDate,
   107→    horizonEnd: baseline.horizonEnd,
   108→    status: baseline.status,
   109→    publishedAt: baseline.publishedAt,
   110→    publishedBy: baseline.publishedBy
   111→      ? { id: baseline.publishedBy, displayName: null } // Would need user lookup for displayName
   112→      : null,
   113→    metrics: {
   114→      fleetCount: baseline.fleetCount,
   115→      totalPMScheduled: baseline.totalPmScheduled,
   116→      totalManhoursProjected: baseline.totalManhourProjected
   117→        ? parseFloat(baseline.totalManhourProjected)
   118→        : null,
   119→    },
   120→    createdAt: baseline.createdAt,
   121→  }));
   122→
   123→  return c.json(
   124→    paginatedSuccess(
   125→      baselines,
   126→      {
   127→        page: result.page,
   128→        pageSize: result.pageSize,
   129→        totalItems: result.total,
   130→      },
   131→      { durationMs: getRequestDuration(c) }
   132→    )
   133→  );
   134→});

   223→/**
   224→ * POST /baselines
   225→ *
   226→ * Publish a new baseline from the current plan state.
   227→ *
   228→ * The baseline will:
   229→ * 1. Validate the current plan for critical conflicts
   230→ * 2. Snapshot the current PM schedule
   231→ * 3. Generate mileage projections
   232→ * 4. Supersede any previous published baseline
   233→ *
   234→ * Request body:
   235→ * - name: Baseline name (required)
   236→ * - description: Optional description
   237→ * - horizonYears: Projection horizon in years (default: 10)
   238→ */
   239→baselineRoutes.post('/baselines', async (c) => {
   240→  const requestId = getRequestId(c);
   241→  const body = await c.req.json();
   242→
   243→  // Validate required fields
   244→  if (!body.name || typeof body.name !== 'string' || body.name.trim().length === 0) {
   245→    return c.json(
   246→      error(ErrorCodes.VALIDATION_ERROR, 'Baseline name is required', requestId),
   247→      400
   248→    );
   249→  }
   250→
   251→  // Get user from context (or fallback for development)
   252→  const userId = c.get('user')?.id ?? 'system';
   253→
   254→  const service = getService();
   255→  const baseline = await service.publishBaseline({
   256→    name: body.name.trim(),
   257→    description: body.description?.trim(),
   258→    userId,
   259→    horizonYears: body.horizonYears ? Number(body.horizonYears) : undefined,
   260→  });
   261→
   262→  return c.json(
   263→    success(
   264→      {
   265→        id: baseline.id,
   266→        name: baseline.name,
   267→        version: baseline.version,
   268→        description: baseline.description,
   269→        baselineDate: baseline.baselineDate,
   270→        horizonEnd: baseline.horizonEnd,
   271→        status: baseline.status,
   272→        publishedAt: baseline.publishedAt,
   273→        metrics: {
   274→          fleetCount: baseline.fleetCount,
   275→          totalPMScheduled: baseline.totalPmScheduled,
   276→          totalManhoursProjected: baseline.totalManhourProjected
   277→            ? parseFloat(baseline.totalManhourProjected)
   278→            : null,
   279→        },
   280→      },
   281→      { timing: { durationMs: getRequestDuration(c) } }
   282→    ),
   283→    201
   284→  );
