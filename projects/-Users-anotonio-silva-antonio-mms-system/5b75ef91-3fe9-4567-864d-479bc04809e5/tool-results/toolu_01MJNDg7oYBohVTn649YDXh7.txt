     1→/**

    14→ * @see specs/005-api.md Section 5.13

    43→/**
    44→ * GET /canvas/strategic
    45→ *
    46→ * Get strategic-level canvas data with yearly PM aggregates.
    47→ * Used for the heatmap view showing PM density over multi-year horizons.
    48→ *
    49→ * Query Parameters:
    50→ * - fleetId: Filter by fleet (UUID, optional)
    51→ * - fromYear: Start year (number, defaults to current year)
    52→ * - toYear: End year (number, defaults to current year + 10)
    53→ */
    54→canvasRoutes.get('/canvas/strategic', async (c) => {
    55→  const requestId = getRequestId(c);
    56→  
    57→  const fleetIdParam = c.req.query('fleetId');
    58→  const fromYearParam = c.req.query('fromYear');
    59→  const toYearParam = c.req.query('toYear');
    60→
    61→  if (fleetIdParam && !isValidUuid(fleetIdParam)) {
    62→    return c.json(
    63→      error(ErrorCodes.VALIDATION_ERROR, 'Invalid fleet ID format', requestId),
    64→      400
    65→    );
    66→  }
    67→
    68→  const currentYear = new Date().getFullYear();
    69→  const fromYear = fromYearParam ? parseInt(fromYearParam, 10) : currentYear;
    70→  const toYear = toYearParam ? parseInt(toYearParam, 10) : currentYear + 10;
    71→
    72→  if (isNaN(fromYear) || isNaN(toYear) || fromYear > toYear) {
    73→    return c.json(
    74→      error(ErrorCodes.VALIDATION_ERROR, 'Invalid year range', requestId),
    75→      400
    76→    );
    77→  }
    78→
    79→  const fromDate = `${fromYear}-01-01`;
    80→  const toDate = `${toYear}-12-31`;
    81→
    82→  const conditions = [
    83→    gte(pmSchedule.targetDate, fromDate),
    84→    lte(pmSchedule.targetDate, toDate),
    85→    isNull(pmSchedule.scenarioId),
    86→  ];
    87→
    88→  if (fleetIdParam) {
    89→    conditions.push(eq(trains.fleetId, fleetIdParam));
    90→  }
    91→
    92→  const results = await db
    93→    .select({
    94→      trainId: trains.id,
    95→      trainNumber: trains.trainNumber,
    96→      fleetId: trains.fleetId,
    97→      fleetCode: fleets.code,
    98→      year: sql<number>`EXTRACT(YEAR FROM ${pmSchedule.targetDate})::integer`.as('year'),
    99→      pmCount: sql<number>`COUNT(${pmSchedule.id})::integer`.as('pm_count'),
   100→      totalHours: sql<string>`COALESCE(SUM(${pmSchedule.assignedHours}), 0)`.as('total_hours'),
   101→      completedCount: sql<number>`COUNT(CASE WHEN ${pmSchedule.status} = 'COMPLETED' THEN 1 END)::integer`.as('completed_count'),
   102→      lockedCount: sql<number>`COUNT(CASE WHEN ${pmSchedule.isLocked} = true THEN 1 END)::integer`.as('locked_count'),
   103→    })
   104→    .from(pmSchedule)
   105→    .innerJoin(trains, eq(pmSchedule.trainId, trains.id))
   106→    .innerJoin(fleets, eq(trains.fleetId, fleets.id))
   107→    .where(and(...conditions))
   108→    .groupBy(
   109→      trains.id,
   110→      trains.trainNumber,
   111→      trains.fleetId,
   112→      fleets.code,
   113→      sql`EXTRACT(YEAR FROM ${pmSchedule.targetDate})`
   114→    )
   115→    .orderBy(trains.trainNumber, sql`year`);
   116→
   117→  const trainMap = new Map<string, {
   118→    trainId: string;
   119→    trainNumber: string;
   120→    fleetCode: string;
   121→    years: Array<{
   122→      year: number;
   123→      pmCount: number;
   124→      totalHours: number;
   125→      completedCount: number;
   126→      lockedCount: number;
   127→    }>;
   128→  }>();
   129→
   130→  let maxPmCount = 0;
   131→  let maxHours = 0;
   132→
   133→  for (const row of results) {
   134→    const hours = parseFloat(row.totalHours) || 0;
   135→    maxPmCount = Math.max(maxPmCount, row.pmCount);
   136→    maxHours = Math.max(maxHours, hours);
   137→
   138→    if (!trainMap.has(row.trainId)) {
   139→      trainMap.set(row.trainId, {
   140→        trainId: row.trainId,
   141→        trainNumber: row.trainNumber,
   142→        fleetCode: row.fleetCode,
   143→        years: [],
   144→      });
   145→    }
   146→
   147→    trainMap.get(row.trainId)!.years.push({
   148→      year: row.year,
   149→      pmCount: row.pmCount,
   150→      totalHours: Math.round(hours * 10) / 10,
   151→      completedCount: row.completedCount,
   152→      lockedCount: row.lockedCount,
   153→    });
   154→  }
   155→
   156→  return c.json(
   157→    success(
   158→      {
   159→        trains: Array.from(trainMap.values()),
   160→        heatmapData: {
   161→          maxPmCount,
   162→          maxHours: Math.round(maxHours * 10) / 10,
   163→        },
   164→      },
   165→      { timing: { durationMs: getRequestDuration(c) } }
   166→    )
   167→  );
   168→});

   170→/**
   171→ * GET /canvas/tactical
   172→ *
   173→ * Get tactical-level canvas data with weekly PM blocks.
   174→ * Used for the detailed scheduling view with drag-and-drop PM movement.
   175→ *
   176→ * Query Parameters:
   177→ * - trainIds: Comma-separated train IDs (UUID[], optional - defaults to all)
   178→ * - fromWeek: Start date of first week (ISO date, defaults to today)
   179→ * - toWeek: End date of last week (ISO date, defaults to +12 weeks)
   180→ */
   181→canvasRoutes.get('/canvas/tactical', async (c) => {
   182→  const requestId = getRequestId(c);
   183→  
   184→  const trainIdsParam = c.req.query('trainIds');
   185→  const fromWeekParam = c.req.query('fromWeek');
   186→  const toWeekParam = c.req.query('toWeek');
   187→
   188→  const trainIds = trainIdsParam ? trainIdsParam.split(',').map((id) => id.trim()) : [];
   189→
   190→  for (const id of trainIds) {
   191→    if (!isValidUuid(id)) {
   192→      return c.json(
   193→        error(ErrorCodes.VALIDATION_ERROR, `Invalid train ID format: ${id}`, requestId),
   194→        400
   195→      );
   196→    }
   197→  }
   198→
   199→  const today = new Date();
   200→  const defaultFromWeek = new Date(today);
   201→  defaultFromWeek.setDate(today.getDate() - today.getDay());
   202→
   203→  const defaultToWeek = new Date(defaultFromWeek);
   204→  defaultToWeek.setDate(defaultToWeek.getDate() + 12 * 7);
   205→
   206→  let fromWeek: Date;
   207→  let toWeek: Date;
   208→
   209→  if (fromWeekParam) {
   210→    fromWeek = new Date(fromWeekParam);
   211→    if (isNaN(fromWeek.getTime())) {
   212→      return c.json(
   213→        error(ErrorCodes.VALIDATION_ERROR, 'Invalid fromWeek date format', requestId),
   214→        400
   215→      );
   216→    }
   217→  } else {
   218→    fromWeek = defaultFromWeek;
   219→  }
   220→
   221→  if (toWeekParam) {
   222→    toWeek = new Date(toWeekParam);
   223→    if (isNaN(toWeek.getTime())) {
   224→      return c.json(
   225→        error(ErrorCodes.VALIDATION_ERROR, 'Invalid toWeek date format', requestId),
   226→        400
   227→      );
   228→    }
   229→  } else {
   230→    toWeek = defaultToWeek;
   231→  }
   232→
   233→  const fromDate = fromWeek.toISOString().split('T')[0];
   234→  const toDate = toWeek.toISOString().split('T')[0];
   235→
   236→  const conditions = [
   237→    gte(pmSchedule.targetDate, fromDate),
   238→    lte(pmSchedule.targetDate, toDate),
   239→    isNull(pmSchedule.scenarioId),
   240→  ];
   241→
   242→  if (trainIds.length > 0) {
   243→    conditions.push(inArray(pmSchedule.trainId, trainIds));
   244→  }
   245→
   246→  const results = await db
   247→    .select({
   248→      pmId: pmSchedule.id,
   249→      trainId: pmSchedule.trainId,
   250→      trainNumber: trains.trainNumber,
   251→      pmCode: pmTypes.pmCode,
   252→      targetDate: pmSchedule.targetDate,
   253→      status: pmSchedule.status,
   254→      constraintStart: pmSchedule.constraintStart,
   255→      constraintEnd: pmSchedule.constraintEnd,
   256→      isLocked: pmSchedule.isLocked,
   257→      trackNumber: tracks.trackNumber,
   258→      depotCode: depots.code,
   259→      displayColor: pmTypes.displayColor,
   260→      estimatedHours: pmTypes.estimatedHours,
   261→      minTrackDays: pmTypes.minTrackDays,
   262→    })
   263→    .from(pmSchedule)
   264→    .innerJoin(trains, eq(pmSchedule.trainId, trains.id))
   265→    .innerJoin(pmTypes, eq(pmSchedule.pmTypeId, pmTypes.id))
   266→    .leftJoin(tracks, eq(pmSchedule.assignedTrackId, tracks.id))
   267→    .leftJoin(depots, eq(pmSchedule.assignedDepotId, depots.id))
   268→    .where(and(...conditions))
   269→    .orderBy(pmSchedule.targetDate);
   270→
   271→  const pmBlocks = results.map((row) => ({
   272→    pmId: row.pmId,
   273→    trainId: row.trainId,
   274→    trainNumber: row.trainNumber,
   275→    pmCode: row.pmCode,
   276→    targetDate: row.targetDate,
   277→    durationDays: row.minTrackDays ?? 1,
   278→    status: row.status,
   279→    constraintStart: row.constraintStart,
   280→    constraintEnd: row.constraintEnd,
   281→    isLocked: row.isLocked ?? false,
   282→    trackNumber: row.trackNumber,
   283→    depotCode: row.depotCode,
   284→    displayColor: row.displayColor ?? '#3B82F6',
   285→    estimatedHours: row.estimatedHours ? parseFloat(row.estimatedHours) : null,
   286→  }));
   287→
   288→  const constraintWalls: Array<{
   289→    pmId: string;
   290→    type: 'COMPLIANCE_START' | 'COMPLIANCE_END';
   291→    date: string;
   292→  }> = [];
   293→
   294→  for (const pm of results) {
   295→    constraintWalls.push(
   296→      { pmId: pm.pmId, type: 'COMPLIANCE_START', date: pm.constraintStart },
   297→      { pmId: pm.pmId, type: 'COMPLIANCE_END', date: pm.constraintEnd }
   298→    );
   299→  }
   300→
   301→  return c.json(
   302→    success(
   303→      {
   304→        pmBlocks,
   305→        constraintWalls,
   306→      },
   307→      { timing: { durationMs: getRequestDuration(c) } }
   308→    )
   309→  );
   310→});

   312→/**
   313→ * GET /canvas/tactical-wide
   314→ *
   315→ * Get tactical-wide canvas data with monthly PM blocks.
   316→ * Used for the compressed Gantt view bridging strategic and tactical views.
   317→ * Shows 1-year of PM data at monthly granularity.
   318→ *
   319→ * Query Parameters:
   320→ * - trainIds: Comma-separated train IDs (UUID[], optional - defaults to all)
   321→ * - fromDate: Start date (ISO date, defaults to start of current month)

   324→ * @see specs/006-user-interface.md Section 6.3.1 (Tactical-Wide zoom level 4-8)

   326→canvasRoutes.get('/canvas/tactical-wide', async (c) => {
   327→  const requestId = getRequestId(c);
   328→
   329→  const trainIdsParam = c.req.query('trainIds');
   330→  const fromDateParam = c.req.query('fromDate');
   331→  const toDateParam = c.req.query('toDate');
   332→
   333→  const trainIds = trainIdsParam ? trainIdsParam.split(',').map((id) => id.trim()) : [];
   334→
   335→  for (const id of trainIds) {
   336→    if (!isValidUuid(id)) {
   337→      return c.json(
   338→        error(ErrorCodes.VALIDATION_ERROR, `Invalid train ID format: ${id}`, requestId),
   339→        400
   340→      );
   341→    }
   342→  }
   343→
   344→  // Default to 12 months starting from current month
   345→  const today = new Date();
   346→  const defaultFromDate = new Date(today.getFullYear(), today.getMonth(), 1);
   347→  const defaultToDate = new Date(defaultFromDate);
   348→  defaultToDate.setMonth(defaultToDate.getMonth() + 12);
   349→
   350→  let fromDate: Date;
   351→  let toDate: Date;
   352→
   353→  if (fromDateParam) {
   354→    fromDate = new Date(fromDateParam);
   355→    if (isNaN(fromDate.getTime())) {
   356→      return c.json(
   357→        error(ErrorCodes.VALIDATION_ERROR, 'Invalid fromDate format', requestId),
   358→        400
   359→      );
   360→    }
   361→  } else {
   362→    fromDate = defaultFromDate;
   363→  }
   364→
   365→  if (toDateParam) {
   366→    toDate = new Date(toDateParam);
   367→    if (isNaN(toDate.getTime())) {
   368→      return c.json(
   369→        error(ErrorCodes.VALIDATION_ERROR, 'Invalid toDate format', requestId),
   370→        400
   371→      );
   372→    }
   373→  } else {
   374→    toDate = defaultToDate;
   375→  }
   376→
   377→  const fromDateStr = fromDate.toISOString().split('T')[0];
   378→  const toDateStr = toDate.toISOString().split('T')[0];
   379→
   380→  // Fetch trains for the response
   381→  const trainConditions = trainIds.length > 0
   382→    ? [inArray(trains.id, trainIds)]
   383→    : [];
   384→
   385→  const trainsResult = await db
   386→    .select({
   387→      trainId: trains.id,
   388→      trainNumber: trains.trainNumber,
   389→      fleetCode: fleets.code,
   390→    })
   391→    .from(trains)
   392→    .innerJoin(fleets, eq(trains.fleetId, fleets.id))
   393→    .where(trainConditions.length > 0 ? and(...trainConditions) : undefined)
   394→    .orderBy(trains.trainNumber);
   395→
   396→  // Fetch PM blocks for the date range
   397→  const pmConditions = [
   398→    gte(pmSchedule.targetDate, fromDateStr),
   399→    lte(pmSchedule.targetDate, toDateStr),
   400→    isNull(pmSchedule.scenarioId),
   401→  ];
   402→
   403→  if (trainIds.length > 0) {
   404→    pmConditions.push(inArray(pmSchedule.trainId, trainIds));
   405→  }
   406→
   407→  const pmResults = await db
   408→    .select({
   409→      pmId: pmSchedule.id,
   410→      trainId: pmSchedule.trainId,
   411→      pmCode: pmTypes.pmCode,
   412→      targetDate: pmSchedule.targetDate,
   413→      status: pmSchedule.status,
   414→      isLocked: pmSchedule.isLocked,
   415→      displayColor: pmTypes.displayColor,
   416→      minTrackDays: pmTypes.minTrackDays,
   417→    })
   418→    .from(pmSchedule)
   419→    .innerJoin(pmTypes, eq(pmSchedule.pmTypeId, pmTypes.id))
   420→    .where(and(...pmConditions))
   421→    .orderBy(pmSchedule.targetDate);
   422→
   423→  // Transform trains response
   424→  const trainsList = trainsResult.map((row) => ({
   425→    trainId: row.trainId,
   426→    trainNumber: row.trainNumber,
   427→    fleetCode: row.fleetCode,
   428→  }));
   429→
   430→  // Transform PM blocks (simplified for tactical-wide view)
   431→  const pmBlocks = pmResults.map((row) => ({
   432→    pmId: row.pmId,
   433→    trainId: row.trainId,
   434→    pmCode: row.pmCode,
   435→    targetDate: row.targetDate,
   436→    durationDays: row.minTrackDays ?? 1,
   437→    status: row.status,
   438→    isLocked: row.isLocked ?? false,
   439→    displayColor: row.displayColor ?? '#3B82F6',
   440→  }));
   441→
   442→  return c.json(
   443→    success(
   444→      {
   445→        trains: trainsList,
   446→        pmBlocks,
   447→      },
   448→      { timing: { durationMs: getRequestDuration(c) } }
   449→    )
   450→  );
   451→});

   453→/**

   494→
