     1→/**
     2→ * PM Schedule Management Routes
     3→ *
     4→ * REST endpoints for PM schedule CRUD operations and history.
     5→ * Wraps PMSchedulerService with HTTP interface per spec 005.5.
     6→ *
     7→ * Endpoints:
     8→ * - GET /pm/schedule - List schedules with filtering
     9→ * - GET /pm/schedule/:id - Get PM by ID
    10→ * - POST /pm/schedule - Create new PM schedule
    11→ * - PATCH /pm/schedule/:id - Update PM (reschedule, reassign)
    12→ * - POST /pm/schedule/:id/lock - Lock PM to prevent optimizer changes
    13→ * - POST /pm/schedule/:id/unlock - Unlock PM to allow optimizer changes
    14→ * - GET /pm/schedule/:id/history - Get change history
    15→ *
    16→ * @see specs/005-api.md Section 5.5
    17→ */

    61→/**
    62→ * GET /pm/schedule
    63→ *
    64→ * List PM schedules with filtering and pagination.
    65→ *
    66→ * Query Parameters:
    67→ * - trainId: Filter by train
    68→ * - depotId: Filter by assigned depot
    69→ * - pmTypeId: Filter by PM type
    70→ * - status: Filter by status (SCHEDULED, IN_PROGRESS, COMPLETED, DEFERRED, CANCELLED)
    71→ * - fromDate: Target date from (inclusive)
    72→ * - toDate: Target date to (inclusive)
    73→ * - scenarioId: Get scenario schedule (null/omitted = production)
    74→ */
    75→pmScheduleRoutes.get('/pm/schedule', async (c) => {
    76→  // Parse query filters
    77→  const trainId = c.req.query('trainId');
    78→  const depotId = c.req.query('depotId');
    79→  const pmTypeId = c.req.query('pmTypeId');
    80→  const status = c.req.query('status');
    81→  const fromDate = c.req.query('fromDate');
    82→  const toDate = c.req.query('toDate');
    83→  const scenarioId = c.req.query('scenarioId');
    84→
    85→  // Build filter conditions
    86→  const conditions: SQL<unknown>[] = [];
    87→
    88→  if (trainId && isValidUuid(trainId)) {
    89→    conditions.push(eq(pmSchedule.trainId, trainId));
    90→  }
    91→  if (depotId && isValidUuid(depotId)) {
    92→    conditions.push(eq(pmSchedule.assignedDepotId, depotId));
    93→  }
    94→  if (pmTypeId && isValidUuid(pmTypeId)) {
    95→    conditions.push(eq(pmSchedule.pmTypeId, pmTypeId));
    96→  }
    97→  if (status) {
    98→    const validStatuses = ['SCHEDULED', 'IN_PROGRESS', 'COMPLETED', 'DEFERRED', 'CANCELLED'] as const;
    99→    if (validStatuses.includes(status as typeof validStatuses[number])) {
   100→      conditions.push(eq(pmSchedule.status, status as typeof validStatuses[number]));
   101→    }
   102→  }
   103→  if (fromDate) {
   104→    conditions.push(gte(pmSchedule.targetDate, fromDate));
   105→  }
   106→  if (toDate) {
   107→    conditions.push(lte(pmSchedule.targetDate, toDate));
   108→  }
   109→
   110→  // Handle scenario filter
   111→  if (scenarioId && isValidUuid(scenarioId)) {
   112→    conditions.push(eq(pmSchedule.scenarioId, scenarioId));
   113→  } else {
   114→    // Default to production schedule (scenarioId is NULL)
   115→    conditions.push(isNull(pmSchedule.scenarioId));
   116→  }
   117→
   118→  // Parse pagination
   119→  const pagination = parsePaginationQuery({
   120→    page: c.req.query('page') ? Number(c.req.query('page')) : undefined,
   121→    pageSize: c.req.query('pageSize') ? Number(c.req.query('pageSize')) : undefined,
   122→    sort: c.req.query('sort'),
   123→  });
   124→
   125→  // Count total items
   126→  const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   127→  const totalResult = await db
   128→    .select({ count: count() })
   129→    .from(pmSchedule)
   130→    .where(whereClause);
   131→  const totalItems = totalResult[0]?.count ?? 0;
   132→
   133→  // Query with relations
   134→  const schedules = await db.query.pmSchedule.findMany({
   135→    where: whereClause,
   136→    with: {
   137→      train: true,
   138→      pmType: true,
   139→      assignedDepot: true,
   140→      assignedTrack: true,
   141→    },
   142→    orderBy: desc(pmSchedule.targetDate),
   143→    limit: pagination.pageSize,
   144→    offset: pagination.offset,
   145→  });
   146→
   147→  // Map to API response format
   148→  const data = schedules.map((pm) => ({
   149→    id: pm.id,
   150→    train: pm.train
   151→      ? {
   152→          id: pm.train.id,
   153→          trainNumber: pm.train.trainNumber,
   154→          fleetCode: pm.train.fleetId, // TODO: Get fleet code from relation
   155→        }
   156→      : null,
   157→    pmType: pm.pmType
   158→      ? {
   159→          id: pm.pmType.id,
   160→          code: pm.pmType.pmCode,
   161→          name: pm.pmType.pmName,
   162→          estimatedHours: pm.pmType.estimatedHours,
   163→        }
   164→      : null,
   165→    targetDate: pm.targetDate,
   166→    targetMileageKm: pm.targetMileageKm,
   167→    status: pm.status,
   168→    constraintStart: pm.constraintStart,
   169→    constraintEnd: pm.constraintEnd,
   170→    assignedDepot: pm.assignedDepot
   171→      ? {
   172→          id: pm.assignedDepot.id,
   173→          code: pm.assignedDepot.code,
   174→        }
   175→      : null,
   176→    assignedTrack: pm.assignedTrack
   177→      ? {
   178→          id: pm.assignedTrack.id,
   179→          trackNumber: pm.assignedTrack.trackNumber,
   180→          trackType: pm.assignedTrack.trackType,
   181→        }
   182→      : null,
   183→    assignedHours: pm.assignedHours ? Number(pm.assignedHours) : null,
   184→    isLocked: pm.isLocked ?? false,
   185→    maximoWoId: pm.maximoWoId,
   186→  }));
   187→
   188→  return c.json(
   189→    paginatedSuccess(
   190→      data,
   191→      {
   192→        page: pagination.page,
   193→        pageSize: pagination.pageSize,
   194→        totalItems,
   195→      },
   196→      { durationMs: getRequestDuration(c) }
   197→    )
   198→  );
   199→});

   202→ * GET /pm/schedule/:id
   203→ *
   204→ * Get PM schedule details by ID.

   206→pmScheduleRoutes.get('/pm/schedule/:id', async (c) => {
   207→  const requestId = getRequestId(c);
   208→  const pmId = c.req.param('id');
   209→
   210→  if (!isValidUuid(pmId)) {
   211→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'Invalid PM schedule ID format', requestId), 400);
   212→  }
   213→
   214→  const service = getService();
   215→  const pmResult = await service.getPm(pmId);
   216→
   217→  // Cast to include all query relations (service returns more than type declares)
   218→  const pm = pmResult as typeof pmResult & {
   219→    assignedDepot: { id: string; code: string; name: string } | null;
   220→  };
   221→
   222→  // Check material availability for upcoming PMs
   223→  let materialStatus = { allAvailable: true, warnings: [] as string[] };
   224→  if (pm.status === 'SCHEDULED' && pm.assignedDepotId) {
   225→    try {
   226→      const materialResult = await service.checkMaterialAvailability(
   227→        pm.pmTypeId,
   228→        pm.assignedDepotId,
   229→        pm.targetDate
   230→      );
   231→      materialStatus = {
   232→        allAvailable: !materialResult.isBlocked && materialResult.softConstraintWarnings.length === 0,
   233→        warnings: [
   234→          ...materialResult.hardConstraintViolations.map((v) => `Blocked: ${v.partNumber} (need ${v.requiredQuantity}, have ${v.availableQuantity})`),
   235→          ...materialResult.softConstraintWarnings.map((v) => `Warning: ${v.partNumber} may not be available`),
   236→        ],
   237→      };
   238→    } catch {
   239→      // Material check failed, assume available
   240→    }
   241→  }
   242→
   243→  return c.json(
   244→    success({
   245→      id: pm.id,
   246→      train: pm.train
   247→        ? {
   248→            id: pm.train.id,
   249→            trainNumber: pm.train.trainNumber,
   250→            fleetId: pm.train.fleetId,
   251→          }
   252→        : null,
   253→      pmType: pm.pmType
   254→        ? {
   255→            id: pm.pmType.id,
   256→            code: pm.pmType.pmCode,
   257→            name: pm.pmType.pmName,
   258→            estimatedHours: pm.pmType.estimatedHours,
   259→          }
   260→        : null,
   261→      targetDate: pm.targetDate,
   262→      targetMileageKm: pm.targetMileageKm,
   263→      status: pm.status,
   264→      constraintStart: pm.constraintStart,
   265→      constraintEnd: pm.constraintEnd,
   266→      assignedDepot: pm.assignedDepot
   267→        ? {
   268→            id: pm.assignedDepot.id,
   269→            code: pm.assignedDepot.code,
   270→            name: pm.assignedDepot.name,
   271→          }
   272→        : null,
   273→      assignedTrack: pm.assignedTrack
   274→        ? {
   275→            id: pm.assignedTrack.id,
   276→            trackNumber: pm.assignedTrack.trackNumber,
   277→            trackType: pm.assignedTrack.trackType,
   278→          }
   279→        : null,
   280→      assignedHours: pm.assignedHours ? Number(pm.assignedHours) : null,
   281→      assignedShift: pm.assignedShift,
   282→      isLocked: pm.isLocked ?? false,
   283→      lockedReason: pm.lockedReason,
   284→      maximoWoId: pm.maximoWoId,
   285→      materialStatus,
   286→      actualStartDate: pm.actualStartDate,
   287→      actualEndDate: pm.actualEndDate,
   288→      actualHours: pm.actualHours ? Number(pm.actualHours) : null,
   289→      completedAt: pm.completedAt,
   290→      completedMileageKm: pm.completedMileageKm,
   291→      createdAt: pm.createdAt,
   292→      updatedAt: pm.updatedAt,
   293→    })
   294→  );

   297→/**
   298→ * POST /pm/schedule
   299→ *
   300→ * Create a new PM schedule entry.
   301→ *
   302→ * Request body:
   303→ * - trainId: UUID of train
   304→ * - pmTypeId: UUID of PM type
   305→ * - targetDate: Target date (YYYY-MM-DD)
   306→ * - assignedDepotId?: UUID of assigned depot
   307→ * - assignedTrackId?: UUID of assigned track
   308→ * - scenarioId?: UUID of scenario (null for production)
   309→ * - reason?: Reason for manual scheduling
   310→ */
   311→pmScheduleRoutes.post('/pm/schedule', async (c) => {
   312→  const requestId = getRequestId(c);
   313→  const body = await c.req.json();
   314→
   315→  // Validate required fields
   316→  if (!body.trainId || !isValidUuid(body.trainId)) {
   317→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'trainId is required and must be a valid UUID', requestId), 400);
   318→  }
   319→  if (!body.pmTypeId || !isValidUuid(body.pmTypeId)) {
   320→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'pmTypeId is required and must be a valid UUID', requestId), 400);
   321→  }
   322→  if (!body.targetDate || !/^\d{4}-\d{2}-\d{2}$/.test(body.targetDate)) {
   323→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'targetDate is required and must be in YYYY-MM-DD format', requestId), 400);
   324→  }
   325→
   326→  // Validate optional UUIDs
   327→  if (body.assignedDepotId && !isValidUuid(body.assignedDepotId)) {
   328→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'assignedDepotId must be a valid UUID', requestId), 400);
   329→  }
   330→  if (body.assignedTrackId && !isValidUuid(body.assignedTrackId)) {
   331→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'assignedTrackId must be a valid UUID', requestId), 400);
   332→  }
   333→
   334→  const service = getService();
   335→  const user = c.get('user');
   336→
   337→  const request: SchedulePmRequest = {
   338→    trainId: body.trainId,
   339→    pmTypeId: body.pmTypeId,
   340→    targetDate: body.targetDate,
   341→    assignedDepotId: body.assignedDepotId,
   342→    assignedTrackId: body.assignedTrackId,
   343→    assignedShift: body.assignedShift,
   344→    notes: body.reason,
   345→    createdBy: user?.id,
   346→  };
   347→
   348→  const created = await service.schedulePm(request);
   349→
   350→  return c.json(
   351→    success({
   352→      id: created.id,
   353→      trainId: created.trainId,
   354→      pmTypeId: created.pmTypeId,
   355→      targetDate: created.targetDate,
   356→      targetMileageKm: created.targetMileageKm,
   357→      status: created.status,
   358→      constraintStart: created.constraintStart,
   359→      constraintEnd: created.constraintEnd,
   360→      assignedDepotId: created.assignedDepotId,
   361→      assignedTrackId: created.assignedTrackId,
   362→      assignedHours: created.assignedHours ? Number(created.assignedHours) : null,
   363→      createdAt: created.createdAt,
   364→    }),
   365→    201
   366→  );

   369→/**
   370→ * PATCH /pm/schedule/:id
   371→ *
   372→ * Update a PM schedule entry (move, reassign).
   373→ *
   374→ * Request body:
   375→ * - targetDate?: New target date
   376→ * - assignedTrackId?: New track assignment
   377→ * - reason: Reason for change (required)
   378→ */
   379→pmScheduleRoutes.patch('/pm/schedule/:id', async (c) => {
   380→  const requestId = getRequestId(c);
   381→  const pmId = c.req.param('id');
   382→
   383→  if (!isValidUuid(pmId)) {
   384→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'Invalid PM schedule ID format', requestId), 400);
   385→  }
   386→
   387→  const body = await c.req.json();
   388→
   389→  // Validate reason is provided
   390→  if (!body.reason || typeof body.reason !== 'string' || body.reason.trim().length === 0) {
   391→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'reason is required for PM changes', requestId), 400);
   392→  }
   393→
   394→  // Validate targetDate if provided
   395→  if (body.targetDate && !/^\d{4}-\d{2}-\d{2}$/.test(body.targetDate)) {
   396→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'targetDate must be in YYYY-MM-DD format', requestId), 400);
   397→  }
   398→
   399→  // Validate assignedTrackId if provided
   400→  if (body.assignedTrackId && !isValidUuid(body.assignedTrackId)) {
   401→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'assignedTrackId must be a valid UUID', requestId), 400);
   402→  }
   403→
   404→  const service = getService();
   405→  const user = c.get('user');
   406→
   407→  // Get current PM to capture old values
   408→  const currentPm = await service.getPm(pmId);
   409→
   410→  // Only proceed if targetDate is being changed
   411→  if (!body.targetDate) {
   412→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'targetDate is required for PM move', requestId), 400);
   413→  }
   414→
   415→  const request: MovePmRequest = {
   416→    pmScheduleId: pmId,
   417→    newTargetDate: body.targetDate,
   418→    newTrackId: body.assignedTrackId,
   419→    reason: body.reason,
   420→    movedBy: user?.id ?? 'system',
   421→  };
   422→
   423→  const updated = await service.movePm(request);
   424→
   425→  // Calculate impact assessment
   426→  let manhourImpact = 0;
   427→  let conflictsCreated = 0;
   428→
   429→  // Calculate man-hour impact if depot is assigned
   430→  const depotId = updated.assignedDepotId ?? currentPm.assignedDepotId;
   431→  if (depotId) {
   432→    const estimatedHours = parseFloat(currentPm.pmType?.estimatedHours ?? '0');
   433→    const manHourSvc = getManHourService();
   434→
   435→    // Get old and new week utilization
   436→    // When PM moves, it removes hours from old week and adds to new week
   437→    const [oldWeekCapacity, newWeekCapacity] = await Promise.all([
   438→      manHourSvc.checkLaborCapacity(depotId, currentPm.targetDate, 0),
   439→      manHourSvc.checkLaborCapacity(depotId, updated.targetDate, estimatedHours),
   440→    ]);
   441→
   442→    // Impact is the change in new week's utilization
   443→    // Positive = increased load, Negative = decreased load
   444→    manhourImpact = Math.round(
   445→      (newWeekCapacity.utilizationPct - oldWeekCapacity.utilizationPct) * 100
   446→    ) / 100;
   447→  }
   448→
   449→  // Conflicts check: movePm already validates and throws on track conflicts
   450→  // If we reach here, no new conflicts were created
   451→  // The value 0 is correct - if there were conflicts, movePm would have thrown
   452→  conflictsCreated = 0;
   453→
   454→  const impactAssessment = {
   455→    complianceStatus: updated.targetDate >= updated.constraintStart && updated.targetDate <= updated.constraintEnd
   456→      ? 'WITHIN_WINDOW'
   457→      : 'OUTSIDE_WINDOW',
   458→    manhourImpact,
   459→    conflictsCreated,
   460→  };
   461→
   462→  // Broadcast PM update to all connected canvas subscribers
   463→  const hub = getWebSocketHub();
   464→  hub.broadcastPmUpdated({
   465→    pmId: updated.id,
   466→    trainId: updated.trainId,
   467→    trainNumber: currentPm.train?.trainNumber ?? 'Unknown',
   468→    changes: [
   469→      { field: 'targetDate', oldValue: currentPm.targetDate, newValue: updated.targetDate },
   470→      ...(body.assignedTrackId && body.assignedTrackId !== currentPm.assignedTrackId
   471→        ? [{ field: 'assignedTrackId', oldValue: currentPm.assignedTrackId, newValue: body.assignedTrackId }]
   472→        : []),
   473→    ],
   474→    changedBy: {
   475→      userId: user?.id ?? 'system',
   476→      displayName: user?.email ?? 'System',
   477→    },
   478→  });
   479→
   480→  return c.json(
   481→    success({
   482→      id: updated.id,
   483→      targetDate: updated.targetDate,
   484→      previousTargetDate: currentPm.targetDate,
   485→      assignedTrackId: updated.assignedTrackId,
   486→      previousTrackId: currentPm.assignedTrackId,
   487→      updatedAt: updated.updatedAt,
   488→      updatedBy: user?.id,
   489→      explanation: {
   490→        changeType: 'DATE_CHANGED',
   491→        reason: body.reason,
   492→        impactAssessment,
   493→      },
   494→    })
   495→  );
   496→});
