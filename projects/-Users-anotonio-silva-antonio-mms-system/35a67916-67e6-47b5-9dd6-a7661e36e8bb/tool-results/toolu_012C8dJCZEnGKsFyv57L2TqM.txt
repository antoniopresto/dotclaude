  3166→## 4.8 Drift Monitor Service
  3167→
  3168→### 4.8.1 Purpose
  3169→
  3170→Compare actual state against published baseline, calculate drift metrics, and alert when thresholds are exceeded.

  3172→### 4.8.2 Implementation
  3173→
  3174→```typescript
  3175→interface DriftThresholds {
  3176→  mileageDriftPct: number;        // 5.0
  3177→  scheduleDriftDays: number;      // 14
  3178→  fleetDriftTrainCount: number;   // 5
  3179→}
  3180→
  3181→class DriftMonitorService {
  3182→  private readonly thresholds: DriftThresholds = {
  3183→    mileageDriftPct: 5.0,
  3184→    scheduleDriftDays: 14,
  3185→    fleetDriftTrainCount: 5
  3186→  };
  3187→  
  3188→  /**
  3189→   * Run nightly drift analysis
  3190→   */
  3191→  async runDriftAnalysis(): Promise<DriftAnalysisResult> {
  3192→    const baseline = await this.getPublishedBaseline();
  3193→    if (!baseline) {
  3194→      this.logger.info('No published baseline, skipping drift analysis');
  3195→      return { skipped: true, reason: 'No published baseline' };
  3196→    }
  3197→    
  3198→    const analysisDate = new Date();
  3199→    const trainAnalyses: TrainDriftAnalysis[] = [];
  3200→    const alerts: DriftAlert[] = [];
  3201→    
  3202→    // Analyze each train
  3203→    const trains = await this.db.query.trains.findMany({
  3204→      where: eq(trains.active, true)
  3205→    });
  3206→    
  3207→    for (const train of trains) {
  3208→      const analysis = await this.analyzeTrainDrift(train, baseline, analysisDate);
  3209→      trainAnalyses.push(analysis);
  3210→      
  3211→      // Check thresholds
  3212→      if (analysis.requiresReplan) {
  3213→        alerts.push({
  3214→          trainId: train.id,
  3215→          trainNumber: train.trainNumber,
  3216→          severity: 'HIGH',
  3217→          type: 'REPLAN_REQUIRED',
  3218→          message: analysis.replanReason,

  3220→            mileageDriftPct: analysis.mileageDriftPct,

  3282→  private async analyzeTrainDrift(
  3283→    train: Train,
  3284→    baseline: PlanBaseline,
  3285→    analysisDate: Date
  3286→  ): Promise<TrainDriftAnalysis> {
  3287→    // Get baseline projection for today
  3288→    const baselineProjection = await this.db.query.baselineMileageProjection.findFirst({
  3289→      where: and(
  3290→        eq(baselineMileageProjection.baselineId, baseline.id),
  3291→        eq(baselineMileageProjection.trainId, train.id),
  3292→        lte(baselineMileageProjection.projectionDate, analysisDate)
  3293→      ),
  3294→      orderBy: desc(baselineMileageProjection.projectionDate)
  3295→    });
  3296→    
  3297→    // Calculate mileage drift
  3298→    const baselineKm = baselineProjection?.projectedCumulativeKm || train.currentMileageKm;
  3299→    const actualKm = train.currentMileageKm;
  3300→    const mileageDriftKm = actualKm - baselineKm;
  3301→    const mileageDriftPct = (mileageDriftKm / baselineKm) * 100;
  3302→    
  3303→    // Get baseline PM schedule
  3304→    const baselinePMs = await this.db.query.baselinePmSchedule.findMany({
  3305→      where: and(
  3306→        eq(baselinePmSchedule.baselineId, baseline.id),
  3307→        eq(baselinePmSchedule.trainId, train.id),
  3308→        gte(baselinePmSchedule.projectedDate, analysisDate)
  3309→      )
  3310→    });
  3311→    
  3312→    // Get current PM schedule
  3313→    const currentPMs = await this.db.query.pmSchedule.findMany({
  3314→      where: and(
  3315→        eq(pmSchedule.trainId, train.id),
  3316→        isNull(pmSchedule.scenarioId),
  3317→        inArray(pmSchedule.status, ['SCHEDULED', 'IN_PROGRESS']),
  3318→        gte(pmSchedule.targetDate, analysisDate)
  3319→      )
  3320→    });
  3321→    
  3322→    // Calculate PM drift
  3323→    const pmDrifts: PMDrift[] = [];
  3324→    
  3325→    for (const baselinePM of baselinePMs) {
  3326→      const currentPM = currentPMs.find(
  3327→        p => p.pmTypeId === baselinePM.pmTypeId
  3328→      );
  3329→      
  3330→      if (currentPM) {
  3331→        const driftDays = differenceInDays(
  3332→          new Date(currentPM.targetDate),
  3333→          new Date(baselinePM.projectedDate)
  3334→        );
  3335→        
  3336→        pmDrifts.push({
  3337→          pmTypeId: baselinePM.pmTypeId,
  3338→          baselineDate: baselinePM.projectedDate,
  3339→          currentDate: currentPM.targetDate,
  3340→          driftDays,
  3341→          direction: driftDays > 0 ? 'LATER' : driftDays < 0 ? 'EARLIER' : 'ON_TRACK'
  3342→        });
  3343→      }
  3344→    }
  3345→    
  3346→    const avgPmDriftDays = pmDrifts.length > 0
  3347→      ? pmDrifts.reduce((sum, p) => sum + Math.abs(p.driftDays), 0) / pmDrifts.length
  3348→      : 0;
  3349→    
  3350→    const maxPmDriftDays = pmDrifts.length > 0
  3351→      ? Math.max(...pmDrifts.map(p => Math.abs(p.driftDays)))
  3352→      : 0;
  3353→    
  3354→    // Determine if replan required
  3355→    const requiresReplan = 
  3356→      Math.abs(mileageDriftPct) > this.thresholds.mileageDriftPct ||
  3357→      maxPmDriftDays > this.thresholds.scheduleDriftDays;
  3358→    
  3359→    let replanReason: string | undefined;
  3360→    if (requiresReplan) {
  3361→      const reasons: string[] = [];
  3362→      if (Math.abs(mileageDriftPct) > this.thresholds.mileageDriftPct) {
  3363→        reasons.push(`Mileage drift ${mileageDriftPct.toFixed(1)}% exceeds ${this.thresholds.mileageDriftPct}% threshold`);
  3364→      }
  3365→      if (maxPmDriftDays > this.thresholds.scheduleDriftDays) {
  3366→        reasons.push(`PM schedule drift ${maxPmDriftDays} days exceeds ${this.thresholds.scheduleDriftDays} day threshold`);
  3367→      }
  3368→      replanReason = reasons.join('; ');
  3369→    }
  3370→    
  3371→    return {
  3372→      trainId: train.id,
  3373→      trainNumber: train.trainNumber,
  3374→      baselineProjectedKm: baselineKm,
  3375→      actualKm,
  3376→      mileageDriftKm,
  3377→      mileageDriftPct,
  3378→      pmDrifts,
  3379→      avgPmDriftDays,
  3380→      maxPmDriftDays,
  3381→      requiresReplan,
  3382→      replanReason
  3383→    };
  3384→  }
  3385→}
