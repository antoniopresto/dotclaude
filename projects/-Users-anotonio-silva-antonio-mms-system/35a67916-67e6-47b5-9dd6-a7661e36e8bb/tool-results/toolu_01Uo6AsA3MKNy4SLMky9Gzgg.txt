   119→/**
   120→ * Start a new span for an operation
   121→ */
   122→export function startSpan(
   123→  name: string,
   124→  options?: SpanOptions & { tracer?: Tracer }
   125→): Span {
   126→  const tracer = options?.tracer ?? getTracer();
   127→  return tracer.startSpan(name, {
   128→    ...options,
   129→    attributes: {
   130→      ...getServiceAttributes(),
   131→      ...options?.attributes,
   132→    },
   133→  });
   134→}

   136→/**
   137→ * Execute a function within a traced span
   138→ *
   139→ * Automatically records errors and sets span status.
   140→ *
   141→ * @example
   142→ * const result = await withSpan('processRecord', async (span) => {
   143→ *   span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   144→ *   return processRecord(record);
   145→ * });
   146→ */
   147→export async function withSpan<T>(
   148→  name: string,
   149→  fn: (span: Span) => Promise<T>,
   150→  options?: SpanOptions & { tracer?: Tracer }
   151→): Promise<T> {
   152→  const tracer = options?.tracer ?? getTracer();
   153→  const span = tracer.startSpan(name, {
   154→    kind: SpanKind.INTERNAL,
   155→    ...options,
   156→    attributes: {
   157→      ...getServiceAttributes(),
   158→      ...options?.attributes,
   159→    },
   160→  });
   161→
   162→  const ctx = trace.setSpan(context.active(), span);
   163→
   164→  try {
   165→    const result = await context.with(ctx, () => fn(span));
   166→    span.setStatus({ code: SpanStatusCode.OK });
   167→    return result;
   168→  } catch (error) {
   169→    recordSpanError(span, error);
   170→    throw error;
   171→  } finally {
   172→    span.end();
   173→  }
   174→}

   208→/**
   209→ * Record an error on a span
   210→ */
   211→export function recordSpanError(span: Span, error: unknown): void {
   212→  span.setStatus({
   213→    code: SpanStatusCode.ERROR,
   214→    message: error instanceof Error ? error.message : String(error),
   215→  });
   216→
   217→  if (error instanceof Error) {
   218→    span.recordException(error);
   219→
   220→    // Add MMS error attributes if available
   221→    const mmsError = error as Error & { code?: string; statusCode?: number };
   222→    if (mmsError.code) {
   223→      span.setAttribute(MmsAttributes.ERROR_CODE, mmsError.code);
   224→    }
   225→    if (mmsError.statusCode) {
   226→      span.setAttribute('http.status_code', mmsError.statusCode);
   227→    }
   228→    span.setAttribute(MmsAttributes.ERROR_TYPE, error.name);
   229→  }
   230→}

   232→/**
   233→ * Add an event to the current span
   234→ */
   235→export function addSpanEvent(
   236→  name: string,
   237→  attributes?: Attributes,
   238→  span?: Span
   239→): void {
   240→  const currentSpan = span ?? trace.getActiveSpan();
   241→  currentSpan?.addEvent(name, attributes);
   242→}

   261→export function getSpanId(): string | undefined {
   262→  return getActiveSpan()?.spanContext().spanId;

   265→/**
   266→ * Create a trace context object for logging
   267→ */
   268→export function getTraceContext(): { traceId?: string; spanId?: string } {
   269→  const span = getActiveSpan();
   270→  if (!span) return {};
   271→
   272→  const ctx = span.spanContext();
   273→  return {
   274→    traceId: ctx.traceId,
   275→    spanId: ctx.spanId,
   276→  };
   277→}

   279→/**
   280→ * Execute a function within the context of an existing span
   281→ */
   282→export async function withContext<T>(
   283→  ctx: Context,
   284→  fn: () => Promise<T>
   285→): Promise<T> {
   286→  return context.with(ctx, fn);
   287→}

   289→/**
   290→ * Create a linked span (for async operations that continue from another trace)
   291→ */
   292→export function createLinkedSpan(
   293→  name: string,
   294→  parentSpan: Span,
   295→  options?: SpanOptions
   296→): Span {
   297→  const tracer = getTracer();
   298→  return tracer.startSpan(name, {
   299→    ...options,
   300→    links: [
   301→      ...(options?.links ?? []),
   302→      { context: parentSpan.spanContext() },
   303→    ],
   304→    attributes: {
   305→      ...getServiceAttributes(),
   306→      ...options?.attributes,
   307→    },
   308→  });
   309→}
