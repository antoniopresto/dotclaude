     1→/**
     2→ * MMS Application Header
     3→ *
     4→ * Top navigation bar with logo, fleet/depot filters, search, and user menu.
     5→ * Based on specs/006-user-interface.md Section 6.2.
     6→ */
     7→
     8→import {
     9→  Navbar,
    10→  NavbarGroup,
    11→  NavbarHeading,
    12→  NavbarDivider,
    13→  Button,
    14→  HTMLSelect,
    15→  InputGroup,
    16→  Menu,
    17→  MenuItem,
    18→  MenuDivider,
    19→  Popover,
    20→  Classes,
    21→} from '@blueprintjs/core';
    22→
    23→import { useFleet, useUi } from '../../store';
    24→import type { FleetSliceState } from '../../store/slices/fleet';
    25→import type { UiSliceState } from '../../store/slices/ui';
    26→
    27→export function Header() {
    28→  const { state: fleetStateRaw, actions: fleetActions } = useFleet();
    29→  const { state: uiStateRaw, actions: uiActions } = useUi();
    30→
    31→  const fleetState = fleetStateRaw as unknown as FleetSliceState;
    32→  const uiState = uiStateRaw as unknown as UiSliceState;
    33→
    34→  const fleetOptions = [
    35→    { value: '', label: 'All Fleets' },
    36→    ...fleetState.fleets.map((fleet) => ({
    37→      value: fleet.id,
    38→      label: `${fleet.code} - ${fleet.name}`,
    39→    })),
    40→  ];
    41→
    42→  const depotOptions = [
    43→    { value: '', label: 'All Depots' },
    44→    ...fleetState.depots.map((depot) => ({
    45→      value: depot.id,
    46→      label: `${depot.code} - ${depot.name}`,
    47→    })),
    48→  ];
    49→
    50→  const handleFleetChange = (event: React.ChangeEvent<HTMLSelectElement>) => {
    51→    const fleetId = event.target.value;
    52→    fleetActions.setFleetFilter(fleetId ? [fleetId] : []);
    53→  };
    54→
    55→  const handleDepotChange = (event: React.ChangeEvent<HTMLSelectElement>) => {
    56→    const depotId = event.target.value;
    57→    fleetActions.setDepotFilter(depotId ? [depotId] : []);
    58→  };
    59→
    60→  const handleSearchChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    61→    fleetActions.setSearchTerm(event.target.value);
    62→  };
    63→
    64→  const handleThemeToggle = () => {
    65→    const nextTheme = uiState.theme === 'dark' ? 'light' : 'dark';
    66→    uiActions.setTheme(nextTheme);
    67→  };
    68→
    69→  const selectedFleetId = fleetState.filters.fleetIds[0] || '';
    70→  const selectedDepotId = fleetState.filters.depotIds[0] || '';
    71→
    72→  const userMenu = (
    73→    <Menu>
    74→      <MenuItem icon="user" text="Profile" />
    75→      <MenuItem icon="cog" text="Settings" />
    76→      <MenuDivider />
    77→      <MenuItem icon="log-out" text="Logout" intent="danger" />
    78→    </Menu>
    79→  );
    80→
    81→  return (
    82→    <Navbar className="mms-header" fixedToTop>
    83→      <NavbarGroup>
    84→        <NavbarHeading>
    85→          <span className="mms-logo" style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
    86→            <svg width="24" height="24" viewBox="0 0 24 24" fill="currentColor">
    87→              <path d="M12 2L2 7l10 5 10-5-10-5zM2 17l10 5 10-5M2 12l10 5 10-5" />
    88→            </svg>
    89→            <strong>MMS</strong>
    90→          </span>
    91→        </NavbarHeading>
    92→
    93→        <NavbarDivider />
    94→
    95→        <div style={{ display: 'flex', gap: '12px', alignItems: 'center' }}>
    96→          <label className={Classes.TEXT_MUTED} style={{ fontSize: '12px' }}>
    97→            Fleet:
    98→          </label>
    99→          <HTMLSelect
   100→            minimal
   101→            options={fleetOptions}
   102→            value={selectedFleetId}
   103→            onChange={handleFleetChange}
   104→          />
   105→
   106→          <label className={Classes.TEXT_MUTED} style={{ fontSize: '12px' }}>
   107→            Depot:
   108→          </label>
   109→          <HTMLSelect
   110→            minimal
   111→            options={depotOptions}
   112→            value={selectedDepotId}
   113→            onChange={handleDepotChange}
   114→          />
   115→        </div>
   116→      </NavbarGroup>
   117→
   118→      <NavbarGroup align="right">
   119→        <InputGroup
   120→          leftIcon="search"
   121→          placeholder="Search trains..."
   122→          value={fleetState.filters.searchTerm}
   123→          onChange={handleSearchChange}
   124→          style={{ width: '200px' }}
   125→        />
   126→
   127→        <NavbarDivider />
   128→
   129→        <Button
   130→          minimal
   131→          icon={uiState.theme === 'dark' ? 'flash' : 'moon'}
   132→          title="Toggle theme"
   133→          onClick={handleThemeToggle}
   134→        />
   135→
   136→        <Button
   137→          minimal
   138→          icon="notifications"
   139→          title="Notifications"
   140→        />
   141→
   142→        <NavbarDivider />
   143→
   144→        <Popover content={userMenu} placement="bottom-end">
   145→          <Button minimal icon="user" rightIcon="caret-down" text="User" />
   146→        </Popover>
   147→      </NavbarGroup>
   148→    </Navbar>
   149→  );
   150→}
