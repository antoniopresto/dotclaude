     1→/**

     4→ * REST endpoints for fleet, train, depot, and route management.

    10→import { Hono } from 'hono';
    11→import {
    12→  success,
    13→  paginatedSuccess,
    14→  error,
    15→  parsePaginationQuery,
    16→  ErrorCodes,
    17→  type AppVariables,
    18→} from '../types';
    19→import { getRequestId, getRequestDuration } from '../middleware/request-context';
    20→import { getFleetService, type TrainListFilters, type FleetListFilters } from '../../services';
    21→import { isValidUuid } from '../../lib/schemas';
    22→
    23→// Singleton service instance
    24→let fleetService: ReturnType<typeof getFleetService> | null = null;
    25→
    26→function getService(): ReturnType<typeof getFleetService> {
    27→  if (!fleetService) {
    28→    fleetService = getFleetService();
    29→  }
    30→  return fleetService;
    31→}
    32→
    33→export const fleetRoutes = new Hono<{ Variables: AppVariables }>();
    34→
    35→/**
    36→ * GET /fleets
    37→ *
    38→ * List all fleet types with train counts.
    39→ */
    40→fleetRoutes.get('/fleets', async (c) => {
    41→  const service = getService();
    42→
    43→  // Parse query filters
    44→  const filters: FleetListFilters = {};
    45→  const baseDepotId = c.req.query('baseDepotId');
    46→  const manufacturer = c.req.query('manufacturer');
    47→  const active = c.req.query('active');
    48→
    49→  if (baseDepotId && isValidUuid(baseDepotId)) {
    50→    filters.baseDepotId = baseDepotId;
    51→  }
    52→  if (manufacturer) {
    53→    filters.manufacturer = manufacturer;
    54→  }
    55→  if (active !== undefined) {
    56→    filters.active = active === 'true';
    57→  }
    58→
    59→  const fleets = await service.listFleets(filters);
    60→
    61→  // Enrich with train counts using fleet statistics
    62→  const enrichedFleets = await Promise.all(
    63→    fleets.map(async (fleet) => {
    64→      try {
    65→        const stats = await service.getFleetStatistics(fleet.id);
    66→        return {
    67→          id: fleet.id,
    68→          code: fleet.code,
    69→          name: fleet.name,
    70→          manufacturer: fleet.manufacturer,
    71→          model: fleet.model,
    72→          carsPerTrain: fleet.carsPerTrain,
    73→          trainCount: stats.totalTrains,
    74→          activeTrainCount: stats.activeTrains,
    75→        };
    76→      } catch {
    77→        // If statistics fail, return basic fleet data
    78→        return {
    79→          id: fleet.id,
    80→          code: fleet.code,
    81→          name: fleet.name,
    82→          manufacturer: fleet.manufacturer,
    83→          model: fleet.model,
    84→          carsPerTrain: fleet.carsPerTrain,
    85→          trainCount: 0,
    86→          activeTrainCount: 0,
    87→        };
    88→      }
    89→    })
    90→  );
    91→
    92→  return c.json(
    93→    success(enrichedFleets, {
    94→      timing: { durationMs: getRequestDuration(c) },
    95→    })
    96→  );
    97→});

   114→  const stats = await service.getFleetStatistics(fleetId);
   115→
   116→  return c.json(
   117→    success({
   118→      id: fleet.id,
   119→      code: fleet.code,
   120→      name: fleet.name,
   121→      manufacturer: fleet.manufacturer,
   122→      model: fleet.model,
   123→      carsPerTrain: fleet.carsPerTrain,
   124→      trainCount: stats.totalTrains,
   125→      activeTrainCount: stats.activeTrains,
   126→      fitTrains: stats.fitTrains,
   127→      restrictedTrains: stats.restrictedTrains,
   128→      groundedTrains: stats.groundedTrains,
   129→      statistics: {
   130→        totalMileageKm: stats.totalMileageKm,
   131→        averageMileageKm: stats.averageMileageKm,
   132→        minMileageKm: stats.minMileageKm,
   133→        maxMileageKm: stats.maxMileageKm,
   134→      },
   135→    })
   136→  );

   139→/**
   140→ * GET /trains
   141→ *
   142→ * List trains with filtering and pagination.
   143→ */
   144→fleetRoutes.get('/trains', async (c) => {
   145→  const service = getService();
   146→
   147→  // Parse query filters
   148→  const filters: TrainListFilters = {};
   149→  const fleetId = c.req.query('fleetId');
   150→  const depotId = c.req.query('depotId');
   151→  const fitnessStatus = c.req.query('fitnessStatus');
   152→  const active = c.req.query('active');
   153→
   154→  if (fleetId && isValidUuid(fleetId)) {
   155→    filters.fleetId = fleetId;
   156→  }
   157→  if (depotId && isValidUuid(depotId)) {
   158→    filters.depotId = depotId;
   159→  }
   160→  if (fitnessStatus) {
   161→    const validStatuses = ['FIT', 'RESTRICTED', 'GROUNDED'] as const;
   162→    if (validStatuses.includes(fitnessStatus as (typeof validStatuses)[number])) {
   163→      filters.fitnessStatus = fitnessStatus as 'FIT' | 'RESTRICTED' | 'GROUNDED';
   164→    }
   165→  }
   166→  if (active !== undefined) {
   167→    filters.active = active === 'true';
   168→  }
   169→
   170→  // Parse pagination
   171→  const pagination = parsePaginationQuery({
   172→    page: c.req.query('page') ? Number(c.req.query('page')) : undefined,
   173→    pageSize: c.req.query('pageSize') ? Number(c.req.query('pageSize')) : undefined,
   174→    sort: c.req.query('sort'),
   175→  });
   176→
   177→  const result = await service.listTrains(filters, {
   178→    page: pagination.page,
   179→    pageSize: pagination.pageSize,
   180→    sortBy: pagination.sort?.field,
   181→    sortOrder: pagination.sort?.direction,
   182→  });
   183→
   184→  // Map to API response format
   185→  // Calculate mileage variance as current - target
   186→  const trains = result.data.map((train) => ({
   187→    id: train.id,
   188→    trainNumber: train.trainNumber,
   189→    fleetId: train.fleetId,
   190→    currentMileageKm: train.currentMileageKm,
   191→    targetMileageKm: train.targetMileageKm,
   192→    mileageVarianceKm: train.targetMileageKm
   193→      ? train.currentMileageKm - train.targetMileageKm
   194→      : null,
   195→    fitnessStatus: train.fitnessStatus,
   196→    currentDepotId: train.currentDepotId,
   197→    homeDepotId: train.homeDepotId,
   198→    commissioningDate: train.commissioningDate,
   199→    isActive: train.active,
   200→  }));
   201→
   202→  return c.json(
   203→    paginatedSuccess(
   204→      trains,
   205→      {
   206→        page: result.page,
   207→        pageSize: result.pageSize,
   208→        totalItems: result.total,
   209→      },
   210→      { durationMs: getRequestDuration(c) }
   211→    )
   212→  );
   213→});

   218→ * Get detailed train information.

   220→fleetRoutes.get('/trains/:id', async (c) => {
   221→  const requestId = getRequestId(c);
   222→  const trainId = c.req.param('id');
   223→
   224→  if (!isValidUuid(trainId)) {
   225→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'Invalid train ID format', requestId), 400);

   228→  const service = getService();
   229→  const trainWithDetails = await service.getTrainWithDetails(trainId);
   230→
   231→  // Calculate mileage variance

   233→    ? trainWithDetails.currentMileageKm - trainWithDetails.targetMileageKm
   234→    : null;
   235→
   236→  return c.json(
   237→    success({
   238→      id: trainWithDetails.id,
   239→      trainNumber: trainWithDetails.trainNumber,
   240→      fleet: trainWithDetails.fleet
   241→        ? {
   242→            id: trainWithDetails.fleet.id,

   244→            name: trainWithDetails.fleet.name,

   246→        : null,
   247→      currentMileageKm: trainWithDetails.currentMileageKm,
   248→      targetMileageKm: trainWithDetails.targetMileageKm,
   249→      mileageVarianceKm,
   250→      fitnessStatus: trainWithDetails.fitnessStatus,
   251→      fitnessNotes: trainWithDetails.fitnessNotes,
   252→      currentDepot: trainWithDetails.currentDepot
   253→        ? {
   254→            id: trainWithDetails.currentDepot.id,

   258→        : null,

   272→/**
   273→ * PATCH /trains/:id
   274→ *
   275→ * Update train properties (fitness status, notes).
   276→ */
   277→fleetRoutes.patch('/trains/:id', async (c) => {
   278→  const requestId = getRequestId(c);
   279→  const trainId = c.req.param('id');
   280→
   281→  if (!isValidUuid(trainId)) {
   282→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'Invalid train ID format', requestId), 400);
   283→  }
   284→
   285→  const body = await c.req.json();
   286→  const service = getService();
   287→
   288→  // Handle fitness status change
   289→  if (body.fitnessStatus) {
   290→    const validStatuses = ['FIT', 'RESTRICTED', 'GROUNDED'];
   291→    if (!validStatuses.includes(body.fitnessStatus)) {
   292→      return c.json(
   293→        error(
   294→          ErrorCodes.VALIDATION_ERROR,
   295→          `Invalid fitness status. Must be one of: ${validStatuses.join(', ')}`,
   296→          requestId
   297→        ),
   298→        400
   299→      );
   300→    }
   301→
   302→    await service.updateFitnessStatus({
   303→      trainId,
   304→      newStatus: body.fitnessStatus,
   305→      notes: body.fitnessNotes,
   306→      changedBy: c.get('user')?.id,
   307→    });
   308→  }
   309→
   310→  // Get updated train
   311→  const updatedTrain = await service.getTrainById(trainId);
   312→
   313→  return c.json(
   314→    success({
   315→      id: updatedTrain.id,
   316→      fitnessStatus: updatedTrain.fitnessStatus,
   317→      fitnessNotes: updatedTrain.fitnessNotes,
   318→      updatedAt: updatedTrain.updatedAt,
   319→    })
   320→  );
   321→});
