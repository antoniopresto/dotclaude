     1→/**
     2→ * MMS Ingestion Service
     3→ *
     4→ * Processes daily CSV files containing mileage data.
     5→ * Validates records, handles meter events, and updates the database.
     6→ *
     7→ * Data flow:
     8→ * 1. CSV file arrives via file watcher or API upload
     9→ * 2. File is validated (format, size, structure)
    10→ * 3. Records are parsed and validated in batches
    11→ * 4. Anomalies are detected (NEGATIVE, SPIKE, DUPLICATE)
    12→ * 5. Mileage readings are inserted/updated
    13→ * 6. Train current_mileage_km is updated
    14→ * 7. Fleet mileage targets are recalculated
    15→ *
    16→ * Business rules:
    17→ * - Max file size: 10MB
    18→ * - Expected file pattern: mileage_YYYYMMDD.csv
    19→ * - Batch size: 100 records per transaction
    20→ * - Anomaly thresholds: NEGATIVE < 0, SPIKE > 1000 km/day
    21→ * - Meter adjustments are applied to maintain mileage continuity
    22→ *
    23→ * @see specs/004-domain-services.md Section 4.2
    24→ */

   193→  /**
   194→   * Validate a mileage CSV file before processing
   195→   *
   196→   * Checks:
   197→   * - File exists and is readable
   198→   * - File size within limit
   199→   * - Filename matches expected pattern
   200→   * - CSV structure is valid
   201→   * - Required columns present
   202→   */
   203→  async validateFile(filePath: string): Promise<FileValidationResult> {
   204→    return this.traced('validateFile', async (span) => {
   205→      span.setAttribute('file.path', filePath);
   206→
   207→      const errors: string[] = [];
   208→      const warnings: string[] = [];
   209→      const fileName = path.basename(filePath);
   210→
   211→      // Check file exists
   212→      if (!fs.existsSync(filePath)) {
   213→        return {
   214→          valid: false,
   215→          errors: [`File not found: ${filePath}`],
   216→          warnings: [],
   217→          fileSize: 0,
   218→        };
   219→      }
   220→
   221→      // Check file size
   222→      const stats = fs.statSync(filePath);
   223→      const fileSize = stats.size;
   224→      span.setAttribute('file.size', fileSize);
   225→
   226→      if (fileSize === 0) {
   227→        errors.push('File is empty');
   228→      }
   229→
   230→      if (fileSize > MAX_FILE_SIZE_BYTES) {
   231→        errors.push(`File size ${fileSize} exceeds maximum ${MAX_FILE_SIZE_BYTES} bytes`);
   232→      }
   233→
   234→      // Check filename pattern
   235→      if (!EXPECTED_FILE_PATTERN.test(fileName)) {
   236→        warnings.push(`Filename '${fileName}' does not match expected pattern mileage_YYYYMMDD.csv`);
   237→      }
   238→
   239→      // Validate CSV structure (read first 10 lines)
   240→      if (fileSize > 0) {
   241→        try {
   242→          const sample = this.readFirstLines(filePath, 10);
   243→          const structureErrors = this.validateCsvStructure(sample);
   244→          errors.push(...structureErrors);
   245→        } catch (error) {
   246→          errors.push(`Failed to read CSV: ${error instanceof Error ? error.message : 'Unknown error'}`);
   247→        }
   248→      }
   249→
   250→      const valid = errors.length === 0;
   251→      span.setAttribute('validation.valid', valid);
   252→      span.setAttribute('validation.errorCount', errors.length);
   253→
   254→      return {
   255→        valid,
   256→        errors,
   257→        warnings,
   258→        fileSize,
   259→      };
   260→    });
   261→  }
   262→
   263→  /**
   264→   * Read first N lines from a file
   265→   */
   266→  private readFirstLines(filePath: string, lineCount: number): string {
   267→    const content = fs.readFileSync(filePath, 'utf-8');
   268→    const lines = content.split('\n').slice(0, lineCount);
   269→    return lines.join('\n');
   270→  }

   272→  /**
   273→   * Validate CSV structure (headers and format)
   274→   */
   275→  private validateCsvStructure(csvSample: string): string[] {
   276→    const errors: string[] = [];
   277→
   278→    const parseResult = Papa.parse(csvSample, {
   279→      header: true,
   280→      skipEmptyLines: true,
   281→    });
   282→
   283→    if (parseResult.errors.length > 0) {
   284→      for (const error of parseResult.errors.slice(0, 5)) {
   285→        errors.push(`CSV parse error at row ${error.row ?? 'unknown'}: ${error.message}`);
   286→      }
   287→      return errors;
   288→    }
   289→
   290→    // Check required columns
   291→    const headers = parseResult.meta.fields ?? [];
   292→    const requiredColumns = ['train_number', 'reading_date', 'odometer_km'];
   293→
   294→    for (const col of requiredColumns) {
   295→      if (!headers.includes(col)) {
   296→        errors.push(`Missing required column: ${col}`);
   297→      }
   298→    }
   299→
   300→    return errors;
   301→  }

   307→  /**
   308→   * Queue a file for async processing
   309→   */
   310→  async queueFileForProcessing(
   311→    filePath: string,
   312→    uploadedBy?: string
   313→  ): Promise<{ jobId: string }> {
   314→    return this.traced('queueFileForProcessing', async (span) => {
   315→      span.setAttribute('file.path', filePath);
   316→
   317→      // Validate file first
   318→      const validation = await this.validateFile(filePath);
   319→
   320→      if (!validation.valid) {
   321→        throw new ValidationError('File validation failed', {
   322→          field: 'file',
   323→          context: {
   324→            errors: validation.errors,
   325→            warnings: validation.warnings,
   326→          },
   327→        });
   328→      }
   329→
   330→      const fileName = path.basename(filePath);
   331→      const jobData: IngestionJobData = {
   332→        filePath,
   333→        fileName,
   334→        fileSize: validation.fileSize,
   335→        uploadedBy,
   336→      };
   337→
   338→      const { id } = await addJob(
   339→        QUEUE_NAMES.INGESTION,
   340→        'process-mileage-csv',
   341→        jobData,
   342→        {
   343→          attempts: 3,
   344→          backoff: { type: 'exponential', delay: 5000 },
   345→        }
   346→      );
   347→
   348→      this.lastFileReceivedAt = new Date();
   349→
   350→      this.log.info(
   351→        { jobId: id, fileName, fileSize: validation.fileSize, uploadedBy },
   352→        'File queued for ingestion'
   353→      );
   354→
   355→      return { jobId: id };
   356→    });
   357→  }

   359→  /**
   360→   * Process a mileage CSV file (main ingestion entry point)
   361→   *
   362→   * This is the core method called by the ingestion worker.
   363→   */
   364→  async processFile(
   365→    filePath: string,
   366→    jobId: string = crypto.randomUUID()
   367→  ): Promise<IngestionResult> {
   368→    return this.traced('processFile', async (span) => {
   369→      const startTime = Date.now();
   370→      const fileName = path.basename(filePath);
   371→
   372→      span.setAttribute(MmsAttributes.INGESTION_FILE, fileName);
   373→      span.setAttribute(MmsAttributes.INGESTION_JOB_ID, jobId);
   374→
   375→      ingestionMetrics.filesReceived.inc();
   376→
   377→      const result: IngestionResult = {
   378→        jobId,
   379→        fileName,
   380→        totalRecords: 0,
   381→        validRecords: 0,
   382→        invalidRecords: 0,
   383→        anomalies: [],
   384→        meterEvents: [],
   385→        processingTimeMs: 0,
   386→        trainsUpdated: 0,
   387→        status: 'SUCCESS',
   388→      };
   389→
   390→      try {
   391→        // Parse CSV
   392→        const records = await this.parseCsv(filePath);
   393→        result.totalRecords = records.length;
   394→        span.setAttribute(MmsAttributes.INGESTION_RECORD_COUNT, records.length);
   395→
   396→        ingestionMetrics.recordsProcessed.inc(records.length);
   397→
   398→        if (records.length === 0) {
   399→          result.status = 'PARTIAL';
   400→          result.errorMessage = 'No valid records found in CSV';
   401→          this.log.warn({ fileName, jobId }, 'CSV file contains no valid records');
   402→          return result;
   403→        }
   404→
   405→        // Process in batches
   406→        const batches = this.chunk(records, BATCH_SIZE);
   407→        const trainIdsUpdated = new Set<string>();
   408→
   409→        for (const batch of batches) {
   410→          const batchResult = await this.processBatch(batch, fileName, result);
   411→          for (const trainId of batchResult.trainIdsUpdated) {
   412→            trainIdsUpdated.add(trainId);
   413→          }
   414→        }
   415→
   416→        result.trainsUpdated = trainIdsUpdated.size;
   417→
   418→        // Update train current mileage for all affected trains
   419→        await this.updateTrainMileage(records, trainIdsUpdated);
   420→
   421→        // Determine final status
   422→        if (result.invalidRecords > 0 && result.validRecords > 0) {
   423→          result.status = 'PARTIAL';
   424→        } else if (result.validRecords === 0) {
   425→          result.status = 'FAILED';
   426→          result.errorMessage = 'All records failed validation';
   427→        }
   428→
   429→        result.processingTimeMs = Date.now() - startTime;
   430→
   431→        span.setAttribute('result.validRecords', result.validRecords);
   432→        span.setAttribute('result.invalidRecords', result.invalidRecords);
   433→        span.setAttribute('result.anomalyCount', result.anomalies.length);
   434→        span.setAttribute('result.status', result.status);
   435→
   436→        recordIngestionComplete(result.status, result.processingTimeMs);
   437→
   438→        this.log.info(
   439→          {
   440→            jobId,
   441→            fileName,
   442→            totalRecords: result.totalRecords,
   443→            validRecords: result.validRecords,
   444→            invalidRecords: result.invalidRecords,
   445→            anomalies: result.anomalies.length,
   446→            trainsUpdated: result.trainsUpdated,
   447→            processingTimeMs: result.processingTimeMs,
   448→            status: result.status,
   449→          },
   450→          `Ingestion complete: ${result.validRecords}/${result.totalRecords} records processed`
   451→        );
   452→
   453→        return result;
   454→      } catch (error) {
   455→        result.processingTimeMs = Date.now() - startTime;
   456→        result.status = 'FAILED';
   457→        result.errorMessage = error instanceof Error ? error.message : 'Unknown error';
   458→
   459→        recordIngestionComplete('FAILED', result.processingTimeMs);
   460→
   461→        this.log.error(
   462→          { error, jobId, fileName },
   463→          `Ingestion failed: ${result.errorMessage}`
   464→        );
   465→
   466→        throw error;
   467→      }
   468→    });
   469→  }

   471→  /**
   472→   * Parse a CSV file and validate each row
   473→   */
   474→  private async parseCsv(filePath: string): Promise<ParsedMileageRecord[]> {
   475→    return this.traced('parseCsv', async (span) => {
   476→      span.setAttribute('file.path', filePath);
   477→
   478→      const content = fs.readFileSync(filePath, 'utf-8');
   479→
   480→      return new Promise<ParsedMileageRecord[]>((resolve, reject) => {
   481→        const records: ParsedMileageRecord[] = [];
   482→        let rowIndex = 0;
   483→
   484→        Papa.parse(content, {
   485→          header: true,
   486→          skipEmptyLines: true,
   487→          complete: (results) => {
   488→            span.setAttribute('parse.rowCount', results.data.length);
   489→
   490→            for (const row of results.data) {
   491→              rowIndex++;
   492→              const validation = this.validateRow(row, rowIndex);
   493→
   494→              if (validation.valid && validation.record) {
   495→                records.push(validation.record);
   496→              }
   497→            }
   498→
   499→            this.log.debug(
   500→              { filePath, totalRows: results.data.length, validRows: records.length },
   501→              'CSV parsing complete'
   502→            );
   503→
   504→            resolve(records);
   505→          },
   506→          error: (error: Error) => {
   507→            reject(new FileError(`CSV parse error: ${error.message}`, {
   508→              filePath,
   509→              operation: 'parse',
   510→            }));
   511→          },
   512→        });
   513→      });
   514→    });
   515→  }

   517→  /**
   518→   * Validate a single CSV row
   519→   */
   520→  private validateRow(
   521→    row: unknown,
   522→    rowIndex: number
   523→  ): RowValidationResult {
   524→    const parseResult = parseMileageCsvRow(row);
   525→
   526→    if (!parseResult.success) {
   527→      const errorMessages = parseResult.errors.map(e => e.message).join(', ');
   528→      return {
   529→        valid: false,
   530→        error: `Row ${rowIndex}: ${errorMessages}`,
   531→      };
   532→    }
   533→
   534→    const data = parseResult.data;
   535→
   536→    // Check for future dates
   537→    const readingDate = new Date(data.readingDate);
   538→    const today = new Date();
   539→    today.setHours(23, 59, 59, 999);
   540→
   541→    if (readingDate > today) {
   542→      return {
   543→        valid: false,
   544→        error: `Row ${rowIndex}: Reading date ${data.readingDate} is in the future`,
   545→        anomaly: {
   546→          trainNumber: data.trainNumber,
   547→          readingDate: data.readingDate,
   548→          type: 'FUTURE_DATE',
   549→          value: null,
   550→          message: `Reading date ${data.readingDate} is in the future`,
   551→          rawReading: data.odometerKm,
   552→        },
   553→      };
   554→    }
   555→
   556→    return {
   557→      valid: true,
   558→      record: {
   559→        trainNumber: data.trainNumber,
   560→        readingDate: data.readingDate,
   561→        odometerKm: data.odometerKm,
   562→        rowIndex,
   563→      },
   564→    };
   565→  }

   567→  /**
   568→   * Process a batch of records within a transaction
   569→   */
   570→  private async processBatch(
   571→    batch: ParsedMileageRecord[],
   572→    fileName: string,
   573→    result: IngestionResult
   574→  ): Promise<{ trainIdsUpdated: Set<string> }> {
   575→    return this.inTransaction('processBatch', async (tx, span) => {
   576→      span.setAttribute('batch.size', batch.length);
   577→
   578→      const trainIdsUpdated = new Set<string>();
   579→
   580→      for (const record of batch) {
   581→        try {
   582→          // Find train
   583→          const train = await tx.query.trains.findFirst({
   584→            where: eq(trains.trainNumber, record.trainNumber),
   585→          });
   586→
   587→          if (!train) {
   588→            result.invalidRecords++;
   589→            const anomaly: AnomalyRecord = {
   590→              trainNumber: record.trainNumber,
   591→              readingDate: record.readingDate,
   592→              type: 'UNKNOWN_TRAIN',
   593→              value: null,
   594→              message: `Train number '${record.trainNumber}' not found in database`,
   595→              rawReading: record.odometerKm,
   596→            };
   597→            result.anomalies.push(anomaly);
   598→            recordIngestionAnomaly('UNKNOWN_TRAIN');
   599→            continue;
   600→          }
   601→
   602→          // Get previous reading
   603→          const previousReading = await tx.query.mileageReadings.findFirst({
   604→            where: and(
   605→              eq(mileageReadings.trainId, train.id),
   606→              sql`${mileageReadings.readingDate} < ${record.readingDate}`
   607→            ),
   608→            orderBy: desc(mileageReadings.readingDate),
   609→          });
   610→
   611→          // Calculate adjusted value (handle meter events)
   612→          const adjustment = await this.getMeterAdjustment(tx, train.id, record.readingDate);
   613→          const adjustedReading = record.odometerKm + adjustment;
   614→
   615→          // Calculate daily km
   616→          let dailyKm: number | null = null;
   617→          let anomalyType: AnomalyType | null = null;
   618→          let anomalyMessage: string | null = null;
   619→
   620→          if (previousReading) {
   621→            dailyKm = adjustedReading - previousReading.adjustedReading;
   622→
   623→            // Anomaly detection
   624→            if (dailyKm < 0) {
   625→              anomalyType = 'NEGATIVE';
   626→              anomalyMessage = `Negative daily mileage: ${dailyKm} km (previous: ${previousReading.adjustedReading}, current: ${adjustedReading})`;
   627→
   628→              result.anomalies.push({
   629→                trainNumber: record.trainNumber,
   630→                readingDate: record.readingDate,
   631→                type: 'NEGATIVE',
   632→                value: dailyKm,
   633→                message: anomalyMessage,
   634→                rawReading: record.odometerKm,
   635→                previousReading: previousReading.adjustedReading,
   636→              });
   637→              recordIngestionAnomaly('NEGATIVE');
   638→            } else if (dailyKm > SPIKE_THRESHOLD_KM) {
   639→              anomalyType = 'SPIKE';
   640→              anomalyMessage = `Unusually high daily mileage: ${dailyKm} km (threshold: ${SPIKE_THRESHOLD_KM})`;
   641→
   642→              result.anomalies.push({
   643→                trainNumber: record.trainNumber,
   644→                readingDate: record.readingDate,
   645→                type: 'SPIKE',
   646→                value: dailyKm,
   647→                message: anomalyMessage,
   648→                rawReading: record.odometerKm,
   649→                previousReading: previousReading.adjustedReading,
   650→              });
   651→              recordIngestionAnomaly('SPIKE');
   652→            }
   653→          }
   654→
   655→          // Insert or update reading
   656→          await tx
   657→            .insert(mileageReadings)
   658→            .values({
   659→              trainId: train.id,
   660→              readingDate: record.readingDate,
   661→              rawReading: record.odometerKm,
   662→              adjustedReading,
   663→              dailyKm,
   664→              validationStatus: anomalyType ? 'ANOMALY' : 'VALID',
   665→              anomalyType: anomalyType ?? undefined,

   667→              sourceFile: fileName,

   683→          result.validRecords++;

   687→          this.log.warn(
   688→            { record, error: error instanceof Error ? error.message : 'Unknown error' },

   914→    anomaliesByType: Record<string, number>;
