  1020→    const projections: { date: Date; projectedKm: number; cumulativeKm: number }[] = [];

  1029→        projectedKm: dailyKm,

  1142→  private async runCppOptimizer(input: OptimizerInput): Promise<OptimizerResult> {
  1143→    return new Promise((resolve, reject) => {
  1144→      const timeout = setTimeout(() => {
  1145→        this.process?.kill();

  1173→          buffer = lines.pop() || '';

  1209→  private async runFallbackOptimizer(input: OptimizerInput): Promise<OptimizerResult> {
  1210→    // TypeScript greedy fallback
  1211→    const assignments: Assignment[] = [];
  1212→    const unassigned: UnassignedTrain[] = [];
  1213→    const assignedTrainIds = new Set<string>();
  1214→    
  1215→    // Sort routes by priority (required trains desc)
  1216→    const sortedRoutes = [...input.routes].sort(
  1217→      (a, b) => b.required_trains - a.required_trains
  1218→    );
  1219→    
  1220→    // Sort trains by fitness and mileage balance
  1221→    const sortedTrains = [...input.trains]
  1222→      .filter(t => t.fitness_status === 'FIT')
  1223→      .sort((a, b) => {
  1224→        // Prioritize under-utilized trains
  1225→        const aBalance = a.current_mileage - a.target_mileage;
  1226→        const bBalance = b.current_mileage - b.target_mileage;
  1227→        return aBalance - bBalance;
  1228→      });
  1229→    
  1230→    for (const route of sortedRoutes) {
  1231→      let assigned = 0;
  1232→      
  1233→      for (const train of sortedTrains) {
  1234→        if (assigned >= route.required_trains) break;
  1235→        if (assignedTrainIds.has(train.id)) continue;
  1236→        
  1237→        // Check hard constraints
  1238→        const hardViolation = this.checkHardConstraints(train, route, input);
  1239→        if (hardViolation) {
  1240→          continue;
  1241→        }

  1243→        // Check fleet compatibility
  1244→        if (route.allowed_fleets.length > 0 &&

  1246→          continue;
  1247→        }
  1248→

  1250→        assignments.push({
  1251→          train_id: train.id,
  1252→          route_id: route.id,
  1253→          score: this.calculateScore(train, route),
  1254→          explanation: [
  1255→            {
  1256→              constraint_type: 'SOFT',
  1257→              constraint_name: 'MILEAGE_BALANCE',
  1258→              impact: train.current_mileage - train.target_mileage,
  1259→              description: `Train is ${Math.abs(train.current_mileage - train.target_mileage)}km ${
  1260→                train.current_mileage < train.target_mileage ? 'under' : 'over'
  1261→              } target`
  1262→            }
  1263→          ]
  1264→        });
  1265→        
  1266→        assignedTrainIds.add(train.id);
  1267→        assigned++;
  1268→      }
  1269→      
  1270→      // Record shortfall
  1271→      if (assigned < route.required_trains) {
  1272→        unassigned.push({
  1273→          route_id: route.id,
  1274→          shortfall: route.required_trains - assigned,

  1278→    }
  1279→

  1281→    for (const train of input.trains) {

  1283→

  1289→        train_id: train.id,
  1290→        reason,
  1291→        blocking_constraints: this.getBlockingConstraints(train, input)
  1292→      });
  1293→    }
  1294→    
  1295→    return {

  1297→      request_id: input.request_id,
  1298→      status: 'FALLBACK',

  1302→      material_warnings: [],

  1306→        trains_unassigned: unassigned.length,
  1307→        soft_constraint_breakdown: {},
  1308→        material_blocks: 0
  1309→      }

  1313→  private checkHardConstraints(
  1314→    train: Train,
  1315→    route: Route,

  1328→    
  1329→    // Check explicit hard constraints
  1330→    for (const constraint of input.hard_constraints) {
  1331→      if (constraint.train_id === train.id) {
  1332→        return constraint.description;
  1333→      }
  1334→    }
  1335→    
  1336→    // NEW v2.1: Check route compatibility
  1337→    if (input.route_compatibility) {
  1338→      const incompatible = input.route_compatibility.find(
  1339→        c => c.train_id === train.id && 
  1340→             c.route_id === route.id &&

  1342→      );

  1345→      }
  1346→    }
  1347→    
  1348→    // Check pending PMs that block assignment
  1349→    for (const pm of input.pending_pms) {

  1351→          pm.target_date <= new Date().toISOString().split('T')[0]) {

  1355→

  1360→

  1363→

  1366→        }
  1367→      }
  1368→    }
  1369→    
  1370→    return null;
  1371→  }
