     1→/**
     2→ * MMS Prometheus Metrics
     3→ *
     4→ * Application metrics for monitoring and alerting.
     5→ * Exposed via /metrics endpoint for Prometheus scraping.
     6→ *
     7→ * Metric types:
     8→ * - Counter: Monotonically increasing values (requests, errors)
     9→ * - Gauge: Point-in-time values (active jobs, queue depth)
    10→ * - Histogram: Distribution of values (request duration, solve time)
    11→ *
    12→ * @see specs/004-backend-specification.md Section 4.14
    13→ */

   330→/**
   331→ * HTTP middleware helper - measure request duration
   332→ */
   333→export function measureHttpRequest(
   334→  method: string,
   335→  path: string,
   336→  status: number,
   337→  durationMs: number
   338→): void {
   339→  const labels = { method, path, status: String(status) };
   340→  httpRequestsTotal.inc(labels);
   341→  httpRequestDuration.observe(labels, durationMs / 1000);
   342→}
   343→
   344→/**
   345→ * Queue job helper - record job completion
   346→ */
   347→export function recordQueueJob(
   348→  queue: string,
   349→  jobName: string,
   350→  status: 'completed' | 'failed',
   351→  durationMs: number
   352→): void {
   353→  queueJobsTotal.inc({ queue, status });
   354→  queueJobDuration.observe({ queue, job_name: jobName }, durationMs / 1000);
   355→}
   356→
   357→/**
   358→ * Optimizer helper - record solve completion
   359→ */
   360→export function recordOptimizerRun(
   361→  solverType: 'cpsat' | 'fallback',
   362→  status: 'optimal' | 'feasible' | 'infeasible' | 'timeout' | 'error',
   363→  durationMs: number,
   364→  objective?: number
   365→): void {
   366→  optimizerRunsTotal.inc({ solver_type: solverType, status });
   367→  optimizerSolveSeconds.observe({ solver_type: solverType, status }, durationMs / 1000);
   368→  if (objective !== undefined) {
   369→    optimizerObjectiveValue.set({ solver_type: solverType }, objective);
   370→  }
   371→}
