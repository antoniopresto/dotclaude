     1→/**
     2→ * MMS Strategic Planner Service
     3→ *
     4→ * Generates and maintains 1-10 year strategic plans using genetic algorithms,
     5→ * constrained by capacity envelopes from the tactical model.
     6→ *
     7→ * Key responsibilities:
     8→ * - Project PM requirements for all trains over 10-year horizon
     9→ * - Generate quarterly constraint envelopes (track, labor, material)
    10→ * - Run genetic algorithm to optimize PM allocations
    11→ * - Validate plans against capacity constraints
    12→ * - Calculate plan metrics and fitness scores
    13→ *
    14→ * The strategic planner creates long-term maintenance schedules that:
    15→ * - Respect PM hierarchy (C4 satisfies C1/C2/C3)
    16→ * - Handle dual triggers (mileage OR time)
    17→ * - Balance workload across quarters
    18→ * - Avoid bow-wave accumulation
    19→ *
    20→ * @see specs/004-domain-services.md Section 4.5
    21→ */

   237→  /**
   238→   * Generate a strategic plan for the specified horizon
   239→   */
   240→  async generatePlan(request: GeneratePlanRequest): Promise<StrategicPlanResult> {
   241→    return this.traced(
   242→      'generatePlan',
   243→      async () => {
   244→        const config: StrategicPlannerConfig = {
   245→          ...DEFAULT_CONFIG,
   246→          horizonYears: request.horizonYears ?? DEFAULT_CONFIG.horizonYears,
   247→          populationSize: request.populationSize ?? DEFAULT_CONFIG.populationSize,
   248→          generations: request.generations ?? DEFAULT_CONFIG.generations,
   249→        };
   250→
   251→        const planId = crypto.randomUUID();
   252→        const startTime = new Date();
   253→        const horizonEnd = addYears(startTime, config.horizonYears);
   254→        const generationStats: GenerationStats[] = [];
   255→
   256→        // Report progress
   257→        const reportProgress = (stage: string, percent: number, extra?: Record<string, unknown>) => {
   258→          if (request.onProgress) {
   259→            request.onProgress({ stage, percent, ...extra });
   260→          }
   261→        };
   262→
   263→        reportProgress('INITIALIZING', 0, { message: 'Loading fleet state...' });
   264→
   265→        // 1. Load current fleet state
   266→        const fleetState = await this.getFleetState(request.fleetId);
   267→        this.log.info(
   268→          { trainCount: fleetState.trains.length },
   269→          'Loaded fleet state for strategic planning'
   270→        );
   271→
   272→        reportProgress('PROJECTING', 10, { message: 'Projecting PM requirements...' });
   273→
   274→        // 2. Project PM requirements for each train
   275→        const pmRequirements = await this.projectPMRequirements(
   276→          fleetState,
   277→          config.horizonYears,
   278→          request.fromBaselineId
   279→        );
   280→        this.log.info(
   281→          { pmCount: pmRequirements.length },
   282→          'Projected PM requirements'
   283→        );
   284→
   285→        reportProgress('GENERATING_ENVELOPES', 20, { message: 'Generating constraint envelopes...' });
   286→
   287→        // 3. Generate constraint envelopes for each quarter
   288→        const envelopes = await this.generateEnvelopes(startTime, horizonEnd);
   289→        this.log.info(
   290→          { envelopeCount: envelopes.length },
   291→          'Generated constraint envelopes'
   292→        );
   293→
   294→        reportProgress('OPTIMIZING', 30, { message: 'Running genetic algorithm...' });
   295→
   296→        // 4. Run genetic algorithm
   297→        const bestPlan = await this.runGeneticAlgorithm(
   298→          pmRequirements,
   299→          envelopes,
   300→          config,
   301→          (gen, fitness, avg) => {
   302→            // Report every 50 generations
   303→            if (gen % 50 === 0 || gen === config.generations - 1) {
   304→              const percent = 30 + Math.round((gen / config.generations) * 60);
   305→              reportProgress('OPTIMIZING', percent, {
   306→                generation: gen,
   307→                fitness,
   308→                message: `Generation ${gen}/${config.generations}, Fitness: ${fitness.toFixed(2)}`,
   309→              });
   310→            }
   311→            generationStats.push({
   312→              generation: gen,
   313→              bestFitness: fitness,
   314→              averageFitness: avg,
   315→              worstFitness: 0, // Calculated in GA
   316→            });
   317→          }
   318→        );
   319→
   320→        reportProgress('VALIDATING', 95, { message: 'Validating plan...' });
   321→
   322→        // 5. Validate against constraint envelopes
   323→        let validation = this.validatePlan(bestPlan, envelopes);
   324→
   325→        // 6. Attempt repair if invalid
   326→        let finalPlan = bestPlan;
   327→        if (!validation.valid) {
   328→          this.log.warn(
   329→            { violations: validation.violations.length },
   330→            'Plan has violations, attempting repair'
   331→          );
   332→          finalPlan = this.repairPlan(bestPlan, validation, envelopes);
   333→          validation = this.validatePlan(finalPlan, envelopes);
   334→        }
   335→
   336→        // 7. Calculate metrics
   337→        const metrics = this.calculateMetrics(finalPlan, envelopes);
   338→
   339→        reportProgress('COMPLETE', 100, { message: 'Strategic plan generated' });
   340→
   341→        return {
   342→          planId,
   343→          generatedAt: startTime,
   344→          horizonStart: startTime,
   345→          horizonEnd,
   346→          config,
   347→          plan: finalPlan,
   348→          validation,
   349→          metrics,
   350→          generationStats,
   351→        };
   352→      },
   353→      { 'planner.horizonYears': request.horizonYears ?? DEFAULT_CONFIG.horizonYears }
   354→    );
   355→  }

   368→  /**
   369→   * Project PM requirements for specific trains
   370→   */
   371→  async projectTrainPMs(
   372→    trainIds: string[],
   373→    horizonYears: number = 10
   374→  ): Promise<PMRequirement[]> {
   375→    const requirements: PMRequirement[] = [];
   376→
   377→    for (const trainId of trainIds) {
   378→      const train = await this.db.query.trains.findFirst({
   379→        where: eq(trains.id, trainId),
   380→      });
   381→
   382→      if (!train) continue;
   383→
   384→      // Convert to FleetState train format
   385→      const trainState = {
   386→        id: train.id,
   387→        trainNumber: train.trainNumber,
   388→        fleetId: train.fleetId,
   389→        currentMileageKm: train.currentMileageKm ?? 0,
   390→        commissioningDate: train.commissioningDate
   391→          ? new Date(train.commissioningDate)
   392→          : new Date(),
   393→        fitnessStatus: train.fitnessStatus,
   394→      };
   395→
   396→      const trainRules = await this.getPMRulesForFleet(train.fleetId);
   397→      const trainHistory = await this.getPMHistoryForTrain(trainId);
   398→      const dailyKm = await this.calculateDailyUtilization(trainState, trainHistory);
   399→
   400→      const trainProjections = await this.projectTrainPMsInternal(
   401→        trainState,
   402→        trainRules,
   403→        trainHistory,
   404→        addYears(new Date(), horizonYears),
   405→        dailyKm
   406→      );
   407→
   408→      requirements.push(...trainProjections);
   409→    }
   410→
   411→    return requirements;
   412→  }

   474→  /**
   475→   * Project all PM events for a single train over the horizon
   476→   */
   477→  private async projectTrainPMsInternal(
   478→    train: FleetState['trains'][0],
   479→    pmRules: PmType[],
   480→    pmHistory: PMHistory[],
   481→    horizonEnd: Date,
   482→    dailyKm: number
   483→  ): Promise<PMRequirement[]> {
   484→    const projections: PMRequirement[] = [];
   485→    const workingHistory = [...pmHistory];
   486→    let currentMileage = train.currentMileageKm;
   487→    let currentDate = new Date();
   488→
   489→    // Project until we exceed horizon
   490→    while (currentDate < horizonEnd) {
   491→      // Find next PM due across all PM types
   492→      let nextPM: {
   493→        rule: PmType;
   494→        date: Date;
   495→        mileage: number;
   496→        trigger: 'MILEAGE' | 'TIME';
   497→      } | null = null;
   498→
   499→      for (const rule of pmRules) {
   500→        if (!rule.active) continue;
   501→
   502→        const projection = this.calculateNextPMDue(
   503→          train,
   504→          rule,
   505→          pmRules,
   506→          workingHistory,
   507→          currentMileage,
   508→          currentDate,
   509→          dailyKm
   510→        );
   511→
   512→        if (!nextPM || projection.date < nextPM.date) {
   513→          nextPM = {
   514→            rule,
   515→            date: projection.date,
   516→            mileage: projection.mileage,
   517→            trigger: projection.trigger,
   518→          };
   519→        }
   520→      }
   521→
   522→      if (!nextPM || nextPM.date > horizonEnd) break;
   523→
   524→      // Calculate compliance window
   525→      const windowDays = this.calculateWindowDays(nextPM.rule, dailyKm);
   526→
   527→      // Create requirement
   528→      const requirement: PMRequirement = {
   529→        id: crypto.randomUUID(),
   530→        trainId: train.id,
   531→        trainNumber: train.trainNumber,
   532→        fleetId: train.fleetId,
   533→        pmTypeId: nextPM.rule.id,
   534→        pmCode: nextPM.rule.pmCode,
   535→        projectedDate: nextPM.date,
   536→        projectedMileageKm: nextPM.mileage,
   537→        triggerType: nextPM.trigger,
   538→        confidence: workingHistory.length > 0 ? 'HISTORICAL' : 'ESTIMATED',
   539→        estimatedHours: Number(nextPM.rule.estimatedHours) || 8,
   540→        trackType: nextPM.rule.trackType,
   541→        durationDays: nextPM.rule.minTrackDays ?? 1,
   542→        constraintStart: addDays(nextPM.date, -windowDays),
   543→        constraintEnd: addDays(nextPM.date, windowDays),
   544→      };
   545→
   546→      projections.push(requirement);
   547→
   548→      // Add to working history for cascading projection
   549→      workingHistory.push({
   550→        id: requirement.id,
   551→        trainId: train.id,
   552→        pmTypeId: nextPM.rule.id,
   553→        pmCode: nextPM.rule.pmCode,
   554→        completionDate: nextPM.date,
   555→        mileageAtCompletion: nextPM.mileage,
   556→      });
   557→
   558→      // Advance state past this PM
   559→      const trackDays = nextPM.rule.minTrackDays ?? 1;
   560→      currentDate = addDays(nextPM.date, trackDays);
   561→      currentMileage = nextPM.mileage + trackDays * dailyKm * 0.1; // Small mileage during PM
   562→    }
   563→
   564→    return projections;
   565→  }

   567→  /**
   568→   * Calculate when the next instance of a specific PM type is due
   569→   *
   570→   * Handles PM hierarchy (C4 satisfies C1/C2/C3) and cold start cases
   571→   */
   572→  private calculateNextPMDue(
   573→    train: FleetState['trains'][0],
   574→    targetRule: PmType,
   575→    allRules: PmType[],
   576→    history: PMHistory[],
   577→    currentMileage: number,
   578→    currentDate: Date,
   579→    dailyKm: number
   580→  ): { date: Date; mileage: number; trigger: 'MILEAGE' | 'TIME' } {
   581→    // 1. Find all PM types that SATISFY the target PM
   582→    const satisfyingCodes = this.getSatisfyingCodes(targetRule.pmCode, allRules);
   583→
   584→    // 2. Find the most recent PM that satisfies this requirement
   585→    const lastSatisfying = history
   586→      .filter((h) => satisfyingCodes.includes(h.pmCode))
   587→      .sort((a, b) => b.mileageAtCompletion - a.mileageAtCompletion)[0];
   588→
   589→    // 3. Handle "Cold Start" - no PM history exists
   590→    let lastMileage: number;
   591→    let lastDate: Date;
   592→
   593→    if (lastSatisfying) {
   594→      lastMileage = lastSatisfying.mileageAtCompletion;
   595→      lastDate = lastSatisfying.completionDate;
   596→    } else {
   597→      // COLD START FIX: Create "virtual" last PM
   598→      const intervalKm = targetRule.intervalKm ?? 180000;
   599→      const cyclesCompleted = Math.floor(currentMileage / intervalKm);
   600→
   601→      if (cyclesCompleted > 0) {
   602→        lastMileage = cyclesCompleted * intervalKm;
   603→        const daysAgo = (currentMileage - lastMileage) / Math.max(dailyKm, 1);
   604→        lastDate = addDays(currentDate, -Math.round(daysAgo));
   605→      } else {
   606→        lastMileage = 0;
   607→        lastDate = train.commissioningDate;
   608→      }
   609→    }
   610→
   611→    // 4. Calculate mileage-based due date
   612→    const intervalKm = targetRule.intervalKm ?? 180000;
   613→    const kmSinceLast = currentMileage - lastMileage;
   614→    const kmRemaining = Math.max(0, intervalKm - kmSinceLast);
   615→    const daysUntilMileageDue = dailyKm > 0 ? kmRemaining / dailyKm : 365;
   616→    const dateByMileage = addDays(currentDate, Math.round(daysUntilMileageDue));
   617→    const mileageAtDue = currentMileage + kmRemaining;
   618→
   619→    // 5. Calculate time-based due date (if interval defined)
   620→    let dateByTime: Date | null = null;
   621→    if (targetRule.intervalMonths) {
   622→      dateByTime = addMonths(lastDate, targetRule.intervalMonths);
   623→    }
   624→
   625→    // 6. Return whichever comes first (dual trigger)
   626→    if (dateByTime && dateByTime < dateByMileage) {
   627→      const daysUntilTime = Math.max(0, differenceInDays(dateByTime, currentDate));
   628→      const mileageAtTime = currentMileage + daysUntilTime * dailyKm;
   629→      return { date: dateByTime, mileage: mileageAtTime, trigger: 'TIME' };
   630→    } else {
   631→      return { date: dateByMileage, mileage: mileageAtDue, trigger: 'MILEAGE' };
   632→    }
   633→  }

   635→  /**
   636→   * Find all PM codes that satisfy a target PM code
   637→   */
   638→  private getSatisfyingCodes(targetCode: string, allRules: PmType[]): string[] {
   639→    const result = [targetCode];
   640→
   641→    for (const rule of allRules) {
   642→      if (rule.satisfiesCodes?.includes(targetCode)) {
   643→        result.push(rule.pmCode);
   644→      }
   645→    }
   646→
   647→    return result;
   648→  }

   650→  /**
   651→   * Calculate daily mileage rate for a train
   652→   */
   653→  private async calculateDailyUtilization(
   654→    train: FleetState['trains'][0],
   655→    history: PMHistory[]
   656→  ): Promise<number> {
   657→    // Try to get from route utilization service
   658→    try {
   659→      const routeService = getRouteUtilizationService();
   660→      const targetDate = format(new Date(), 'yyyy-MM-dd');
   661→      const rate = await routeService.calculateDailyRate(train.id, targetDate);
   662→      if (rate.dailyKm > 50 && rate.dailyKm < 500) {
   663→        return rate.dailyKm;
   664→      }
   665→    } catch {
   666→      // Fall through to calculation
   667→    }
   668→
   669→    // Filter valid history entries
   670→    const validHistory = history
   671→      .filter((h) => h.mileageAtCompletion > 0)
   672→      .sort((a, b) => b.completionDate.getTime() - a.completionDate.getTime());
   673→
   674→    // Try: Recent velocity between last 2 PMs
   675→    if (validHistory.length >= 2) {
   676→      const recent = validHistory[0];
   677→      const previous = validHistory[1];
   678→      const daysBetween = differenceInDays(recent.completionDate, previous.completionDate);
   679→      const kmBetween = recent.mileageAtCompletion - previous.mileageAtCompletion;
   680→
   681→      if (daysBetween >= 14 && kmBetween > 0) {
   682→        const rate = kmBetween / daysBetween;
   683→        if (rate > 50 && rate < 500) {
   684→          return rate;
   685→        }
   686→      }
   687→    }
   688→
   689→    // Fallback: Lifetime average from commissioning
   690→    const daysSinceCommission = differenceInDays(new Date(), train.commissioningDate);
   691→    if (daysSinceCommission > 0 && train.currentMileageKm > 0) {
   692→      return train.currentMileageKm / daysSinceCommission;
   693→    }
   694→
   695→    // Last resort: Fleet typical value
   696→    return 150; // Conservative metro average
   697→  }

   778→  private async generateEnvelopeForPeriod(
   779→    periodStart: Date,
   780→    periodEnd: Date,
   781→    depotIds?: string[]
   782→  ): Promise<ConstraintEnvelope> {
   783→    const quarterKey = this.getQuarterKey(periodStart);
   784→    const daysInPeriod = differenceInDays(periodEnd, periodStart);
   785→
   786→    // Get all depots if not specified
   787→    const depots = depotIds
   788→      ? depotIds
   789→      : await this.getAllDepotIds();
   790→
   791→    // Track capacity by depot and type
   792→    const trackCapacity: ConstraintEnvelope['trackCapacity'] = [];
   793→    for (const depotId of depots) {
   794→      const trackData = await this.db.query.tracks.findMany({
   795→        where: eq(tracks.depotId, depotId),
   796→      });
   797→
   798→      const tracksByType = new Map<string, number>();
   799→      for (const track of trackData) {
   800→        const current = tracksByType.get(track.trackType) ?? 0;
   801→        tracksByType.set(track.trackType, current + 1);
   802→      }
   803→
   804→      for (const [trackType, count] of tracksByType) {
   805→        trackCapacity.push({
   806→          trackType,
   807→          availableSlotDays: count * daysInPeriod * 0.85, // 85% utilization target
   808→          depotId,
   809→        });
   810→      }
   811→    }
   812→
   813→    // Labor capacity by depot
   814→    const laborCapacity: ConstraintEnvelope['laborCapacity'] = [];
   815→    for (const depotId of depots) {
   816→      const capacity = await this.db.query.manhourCapacity.findFirst({
   817→        where: eq(manhourCapacity.depotId, depotId),
   818→      });
   819→
   820→      if (capacity) {
   821→        const dailyHours =
   822→          Number(capacity.dayShiftHours ?? 0) +
   823→          Number(capacity.nightShiftHours ?? 0);
   824→        laborCapacity.push({
   825→          depotId,
   826→          availableHours: dailyHours * daysInPeriod,
   827→        });
   828→      }
   829→    }
   830→
   831→    const totalAvailableHours = laborCapacity.reduce((sum, l) => sum + l.availableHours, 0);
   832→    const totalTrackSlots = trackCapacity.reduce((sum, t) => sum + t.availableSlotDays, 0);
   833→
   834→    return {
   835→      period: periodStart,
   836→      quarterKey,
   837→      trackCapacity,
   838→      laborCapacity,
   839→      materialCapacity: {}, // Simplified - would need inventory integration
   840→      totalAvailableHours,
   841→      totalTrackSlots,
   842→    };
   843→  }

   860→  private async runGeneticAlgorithm(
   861→    requirements: PMRequirement[],
   862→    envelopes: ConstraintEnvelope[],
   863→    config: StrategicPlannerConfig,
   864→    onGeneration?: (gen: number, bestFitness: number, avgFitness: number) => void
   865→  ): Promise<StrategicAllocation[]> {
   866→    // Initialize population
   867→    let population = this.initializePopulation(requirements, envelopes, config.populationSize);
   868→
   869→    for (let gen = 0; gen < config.generations; gen++) {
   870→      // Evaluate fitness
   871→      const fitness = population.map((individual) =>
   872→        this.evaluateFitness(individual, envelopes)
   873→      );
   874→
   875→      const bestFitness = Math.max(...fitness);
   876→      const avgFitness = fitness.reduce((a, b) => a + b, 0) / fitness.length;
   877→
   878→      if (onGeneration) {
   879→        onGeneration(gen, bestFitness, avgFitness);
   880→      }
   881→
   882→      // Selection
   883→      const selected = this.tournamentSelection(population, fitness, config.populationSize);
   884→
   885→      // Crossover
   886→      const offspring = this.crossover(selected, config.populationSize - config.eliteCount);
   887→
   888→      // Mutation
   889→      const mutated = this.mutate(offspring, envelopes, config.mutationRate);
   890→
   891→      // Elitism - keep best individuals
   892→      const elite = this.selectElite(population, fitness, config.eliteCount);
   893→
   894→      // New population
   895→      population = [...elite, ...mutated];
   896→    }
   897→
   898→    // Return best individual
   899→    const finalFitness = population.map((i) => this.evaluateFitness(i, envelopes));
   900→    const bestIndex = finalFitness.indexOf(Math.max(...finalFitness));
   901→
   902→    return population[bestIndex];
   903→  }

   905→  private initializePopulation(
   906→    requirements: PMRequirement[],
   907→    envelopes: ConstraintEnvelope[],
   908→    populationSize: number
   909→  ): StrategicAllocation[][] {
   910→    const population: StrategicAllocation[][] = [];
   911→
   912→    for (let i = 0; i < populationSize; i++) {
   913→      const individual: StrategicAllocation[] = requirements.map((req) => {
   914→        // Find valid quarters for this requirement
   915→        const validQuarters = envelopes.filter(
   916→          (e) => e.period >= req.constraintStart && e.period <= req.constraintEnd
   917→        );
   918→
   919→        // Randomly select a quarter (or closest if none in window)
   920→        let targetEnvelope: ConstraintEnvelope;
   921→        if (validQuarters.length > 0) {
   922→          targetEnvelope = validQuarters[Math.floor(Math.random() * validQuarters.length)];
   923→        } else {
   924→          // Find closest envelope to constraint window
   925→          targetEnvelope = envelopes.reduce((closest, env) => {
   926→            const distCurrent = Math.abs(env.period.getTime() - req.projectedDate.getTime());
   927→            const distClosest = Math.abs(closest.period.getTime() - req.projectedDate.getTime());
   928→            return distCurrent < distClosest ? env : closest;
   929→          });
   930→        }
   931→
   932→        return {
   933→          requirementId: req.id,
   934→          trainId: req.trainId,
   935→          trainNumber: req.trainNumber,
   936→          pmTypeId: req.pmTypeId,
   937→          pmCode: req.pmCode,
   938→          targetQuarter: targetEnvelope.quarterKey,
   939→          targetDate: req.projectedDate,
   940→          estimatedHours: req.estimatedHours,
   941→          estimatedCost: req.estimatedHours * 75, // Rough cost estimate
   942→          depotId: null,
   943→        };
   944→      });
   945→
   946→      population.push(individual);
   947→    }
   948→
   949→    return population;
   950→  }

   952→  private evaluateFitness(
   953→    individual: StrategicAllocation[],
   954→    envelopes: ConstraintEnvelope[]
   955→  ): number {
   956→    let fitness = 1000; // Start with base score
   957→
   958→    // Penalty for constraint envelope violations
   959→    for (const envelope of envelopes) {
   960→      const quarterAllocations = individual.filter(
   961→        (a) => a.targetQuarter === envelope.quarterKey
   962→      );
   963→
   964→      // Labor capacity check
   965→      const totalHours = quarterAllocations.reduce((sum, a) => sum + a.estimatedHours, 0);
   966→      if (totalHours > envelope.totalAvailableHours) {
   967→        fitness -= (totalHours - envelope.totalAvailableHours) * 5;
   968→      }
   969→
   970→      // Track capacity check (simplified)
   971→      if (quarterAllocations.length > envelope.totalTrackSlots) {
   972→        fitness -= (quarterAllocations.length - envelope.totalTrackSlots) * 10;
   973→      }
   974→    }
   975→
   976→    // Bonus for workload leveling
   977→    const quarterHours = this.groupByQuarter(individual);
   978→    const variance = this.calculateVariance(Object.values(quarterHours));
   979→    fitness -= variance * 0.1; // Prefer even distribution
   980→
   981→    return Math.max(0, fitness);
   982→  }

   984→  private tournamentSelection(
   985→    population: StrategicAllocation[][],
   986→    fitness: number[],
   987→    targetSize: number
   988→  ): StrategicAllocation[][] {
   989→    const selected: StrategicAllocation[][] = [];
   990→    const tournamentSize = 3;
   991→
   992→    while (selected.length < targetSize) {
   993→      // Pick random tournament participants
   994→      const participants: number[] = [];
   995→      while (participants.length < tournamentSize) {
   996→        const idx = Math.floor(Math.random() * population.length);
   997→        if (!participants.includes(idx)) {
   998→          participants.push(idx);
   999→        }
  1000→      }
  1001→
  1002→      // Select winner (highest fitness)
  1003→      const winnerIdx = participants.reduce((best, current) =>
  1004→        fitness[current] > fitness[best] ? current : best
  1005→      );
  1006→
  1007→      selected.push([...population[winnerIdx]]);
  1008→    }
  1009→
  1010→    return selected;
  1011→  }

  1013→  private crossover(
  1014→    selected: StrategicAllocation[][],
  1015→    offspringCount: number
  1016→  ): StrategicAllocation[][] {
  1017→    const offspring: StrategicAllocation[][] = [];
  1018→
  1019→    while (offspring.length < offspringCount) {
  1020→      const parent1 = selected[Math.floor(Math.random() * selected.length)];
  1021→      const parent2 = selected[Math.floor(Math.random() * selected.length)];
  1022→
  1023→      // Single-point crossover
  1024→      const crossPoint = Math.floor(Math.random() * parent1.length);
  1025→
  1026→      const child = [
  1027→        ...parent1.slice(0, crossPoint),
  1028→        ...parent2.slice(crossPoint),
  1029→      ];
  1030→
  1031→      offspring.push(child);
  1032→    }
  1033→
  1034→    return offspring;
  1035→  }

  1067→  private groupByQuarter(individual: StrategicAllocation[]): Record<string, number> {
  1068→    const result: Record<string, number> = {};
  1069→    for (const alloc of individual) {

  1075→  private calculateVariance(values: number[]): number {
  1076→    if (values.length === 0) return 0;
  1077→    const mean = values.reduce((a, b) => a + b, 0) / values.length;
  1078→    const squaredDiffs = values.map((v) => Math.pow(v - mean, 2));
  1079→    return squaredDiffs.reduce((a, b) => a + b, 0) / values.length;
  1080→  }

  1086→  private validatePlan(
  1087→    plan: StrategicAllocation[],
  1088→    envelopes: ConstraintEnvelope[]
  1089→  ): PlanValidation {
  1090→    const violations: PlanViolation[] = [];
  1091→    const warnings: PlanWarning[] = [];
  1092→
  1093→    for (const envelope of envelopes) {
  1094→      const quarterAllocations = plan.filter(
  1095→        (a) => a.targetQuarter === envelope.quarterKey
  1096→      );
  1097→
  1098→      // Check labor capacity
  1099→      const totalHours = quarterAllocations.reduce((sum, a) => sum + a.estimatedHours, 0);
  1100→      if (totalHours > envelope.totalAvailableHours) {
  1101→        violations.push({
  1102→          type: 'LABOR_CAPACITY',
  1103→          quarter: envelope.quarterKey,
  1104→          description: `Labor hours (${totalHours.toFixed(0)}) exceed capacity (${envelope.totalAvailableHours.toFixed(0)})`,
  1105→          severity: (totalHours - envelope.totalAvailableHours) / envelope.totalAvailableHours,
  1106→        });
  1107→      } else if (totalHours > envelope.totalAvailableHours * 0.95) {
  1108→        warnings.push({
  1109→          type: 'HIGH_UTILIZATION',
  1110→          quarter: envelope.quarterKey,
  1111→          description: `High utilization (${((totalHours / envelope.totalAvailableHours) * 100).toFixed(1)}%)`,
  1112→        });
  1113→      }
  1114→
  1115→      // Check track capacity
  1116→      if (quarterAllocations.length > envelope.totalTrackSlots) {
  1117→        violations.push({
  1118→          type: 'TRACK_CAPACITY',
  1119→          quarter: envelope.quarterKey,
  1120→          description: `PMs (${quarterAllocations.length}) exceed track slots (${envelope.totalTrackSlots.toFixed(0)})`,
  1121→          severity: (quarterAllocations.length - envelope.totalTrackSlots) / envelope.totalTrackSlots,
  1122→        });
  1123→      }
  1124→    }
  1125→
  1126→    // Check for bow wave (accumulation in later quarters)
  1127→    const quarterLoads = this.groupByQuarter(plan);
  1128→    const quarters = Object.keys(quarterLoads).sort();
  1129→    if (quarters.length > 4) {
  1130→      const firstHalf = quarters.slice(0, Math.floor(quarters.length / 2));
  1131→      const secondHalf = quarters.slice(Math.floor(quarters.length / 2));
  1132→
  1133→      const firstHalfLoad = firstHalf.reduce((sum, q) => sum + (quarterLoads[q] ?? 0), 0);
  1134→      const secondHalfLoad = secondHalf.reduce((sum, q) => sum + (quarterLoads[q] ?? 0), 0);
  1135→
  1136→      if (secondHalfLoad > firstHalfLoad * 1.5) {
  1137→        violations.push({
  1138→          type: 'BOW_WAVE',
  1139→          quarter: 'OVERALL',
  1140→          description: 'Maintenance accumulating in later periods (bow wave risk)',
  1141→          severity: secondHalfLoad / firstHalfLoad - 1,
  1142→        });
  1143→      }
  1144→    }
  1145→
  1146→    const score = 100 - violations.reduce((sum, v) => sum + v.severity * 20, 0);
  1147→
  1148→    return {
  1149→      valid: violations.length === 0,
  1150→      violations,
  1151→      warnings,
  1152→      score: Math.max(0, score),
  1153→    };
  1154→  }

  1156→  private repairPlan(
  1157→    plan: StrategicAllocation[],
  1158→    validation: PlanValidation,
  1159→    envelopes: ConstraintEnvelope[]
  1160→  ): StrategicAllocation[] {
  1161→    const repairedPlan = [...plan];
  1162→
  1163→    // Try to move allocations from overloaded quarters
  1164→    for (const violation of validation.violations) {
  1165→      if (violation.type === 'LABOR_CAPACITY' || violation.type === 'TRACK_CAPACITY') {
  1166→        const overloadedQuarter = violation.quarter;
  1167→        const overloadedAllocations = repairedPlan.filter(
  1168→          (a) => a.targetQuarter === overloadedQuarter
  1169→        );
  1170→
  1171→        // Find underutilized quarters
  1172→        const quarterLoads = this.groupByQuarter(repairedPlan);
  1173→        const underutilized = envelopes
  1174→          .filter((e) => {
  1175→            const load = quarterLoads[e.quarterKey] ?? 0;
  1176→            return load < e.totalAvailableHours * 0.7;
  1177→          })
  1178→          .map((e) => e.quarterKey);
  1179→
  1180→        // Move some allocations
  1181→        let movedCount = 0;
  1182→        const targetMoves = Math.ceil(overloadedAllocations.length * 0.2);
  1183→
  1184→        for (let i = 0; i < repairedPlan.length && movedCount < targetMoves; i++) {
  1185→          if (repairedPlan[i].targetQuarter === overloadedQuarter && underutilized.length > 0) {
  1186→            repairedPlan[i] = {
  1187→              ...repairedPlan[i],
  1188→              targetQuarter: underutilized[Math.floor(Math.random() * underutilized.length)],
  1189→            };
  1190→            movedCount++;
  1191→          }
  1192→        }
  1193→      }
  1194→    }
  1195→
  1196→    return repairedPlan;
  1197→  }
