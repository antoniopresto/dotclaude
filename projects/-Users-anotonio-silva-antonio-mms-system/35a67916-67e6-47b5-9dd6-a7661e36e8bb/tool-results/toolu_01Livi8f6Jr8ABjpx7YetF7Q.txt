     1→/**
     2→ * MMS Infinite Canvas
     3→ *
     4→ * The main canvas component that renders PM schedule data at different zoom levels.
     5→ * Uses deck.gl for strategic view (WebGL) and Konva for tactical/operational views (Canvas 2D).
     6→ *
     7→ * Zoom Level Rendering:
     8→ * - STRATEGIC (0-4): deck.gl WebGL heatmap
     9→ * - TACTICAL_WIDE (4-8): deck.gl WebGL compressed blocks
    10→ * - TACTICAL (8-12): Konva HTML5 Canvas Gantt
    11→ * - OPERATIONAL (12-16): Konva HTML5 Canvas detailed cards
    12→ * - DETAIL (16+): React DOM overlay
    13→ *
    14→ * Based on specs/006-user-interface.md Section 6.3.
    15→ */
    16→

   214→  // ==========================================================================
   215→  // Drag-and-Drop Handlers

   218→  const handlePmDragStart = (pmId: string, sourceDate: string) => {
   219→    // Find PM in tactical data
   220→    const pm = tacticalData?.pmBlocks.find(p => p.pmId === pmId);
   221→    if (!pm) return;
   222→
   223→    // Store drag info
   224→    draggingPmRef.current = {
   225→      pmId,
   226→      pmCode: pm.pmCode,
   227→      trainId: pm.trainId,
   228→      trainNumber: pm.trainNumber,
   229→      sourceDate,
   230→      constraintStart: pm.constraintStart,
   231→      constraintEnd: pm.constraintEnd,
   232→      isLocked: pm.isLocked,
   233→    };
   234→
   235→    // Initialize drag state in PM slice
   236→    pmActions.startDrag({ pmId, sourceDate });
   237→
   238→    // Set initial validation (quick check)
   239→    const quickResult = quickValidatePmMove({
   240→      pmId,
   241→      currentDate: sourceDate,
   242→      targetDate: sourceDate,
   243→      trainId: pm.trainId,
   244→      constraintStart: pm.constraintStart,
   245→      constraintEnd: pm.constraintEnd,
   246→      isLocked: pm.isLocked,
   247→    });
   248→    dragValidationRef.current = quickResult;
   249→    pmActions.setDragValidation({
   250→      isValid: quickResult.isValid,
   251→      message: quickResult.message,
   252→      violations: quickResult.violations,
   253→    });
   254→  };
   255→
   256→  const handlePmDragMove = (pmId: string, targetDate: string) => {
   257→    const dragInfo = draggingPmRef.current;
   258→    if (!dragInfo || dragInfo.pmId !== pmId) return;
   259→
   260→    // Update target date in PM slice
   261→    pmActions.updateDragTarget(targetDate);
   262→
   263→    // Quick client-side validation
   264→    const quickResult = quickValidatePmMove({
   265→      pmId,
   266→      currentDate: dragInfo.sourceDate,
   267→      targetDate,
   268→      trainId: dragInfo.trainId,
   269→      constraintStart: dragInfo.constraintStart,
   270→      constraintEnd: dragInfo.constraintEnd,
   271→      isLocked: dragInfo.isLocked,
   272→    });
   273→    dragValidationRef.current = quickResult;
   274→    pmActions.setDragValidation({
   275→      isValid: quickResult.isValid,
   276→      message: quickResult.message,
   277→      violations: quickResult.violations,
   278→    });
   279→
   280→    // If quick validation passes, trigger debounced API validation
   281→    if (quickResult.isValid && targetDate !== dragInfo.sourceDate) {
   282→      debouncedValidationRef.current.validate({
   283→        pmId,
   284→        currentDate: dragInfo.sourceDate,
   285→        targetDate,
   286→        trainId: dragInfo.trainId,
   287→      });
   288→    }
   289→  };
   290→
   291→  const handlePmDragEnd = async (pmId: string, targetDate: string) => {
   292→    const dragInfo = draggingPmRef.current;
   293→    if (!dragInfo || dragInfo.pmId !== pmId) {
   294→      // Clear drag state
   295→      pmActions.cancelDrag();
   296→      draggingPmRef.current = null;
   297→      return;
   298→    }
   299→
   300→    // Cancel pending validation
   301→    debouncedValidationRef.current.cancel();
   302→
   303→    // If no change, just cancel
   304→    if (targetDate === dragInfo.sourceDate) {
   305→      pmActions.cancelDrag();
   306→      draggingPmRef.current = null;
   307→      return;
   308→    }
   309→
   310→    // Final quick validation
   311→    const finalValidation = quickValidatePmMove({
   312→      pmId,
   313→      currentDate: dragInfo.sourceDate,
   314→      targetDate,
   315→      trainId: dragInfo.trainId,
   316→      constraintStart: dragInfo.constraintStart,
   317→      constraintEnd: dragInfo.constraintEnd,
   318→      isLocked: dragInfo.isLocked,
   319→    });
   320→
   321→    // If invalid, snap back and show toast
   322→    if (!finalValidation.isValid) {
   323→      pmActions.cancelDrag();
   324→      draggingPmRef.current = null;
   325→      void showToast({
   326→        intent: Intent.DANGER,
   327→        icon: 'error',
   328→        message: finalValidation.message || 'Cannot reschedule: constraint violation',
   329→        timeout: 5000,
   330→      });
   331→      return;
   332→    }
   333→
   334→    // Open confirmation dialog
   335→    dragConfirmRef.current = {
   336→      isOpen: true,
   337→      pmId,
   338→      pmCode: dragInfo.pmCode,
   339→      trainNumber: dragInfo.trainNumber,
   340→      sourceDate: dragInfo.sourceDate,
   341→      targetDate,
   342→      impact: null,
   343→      isLoadingImpact: true,
   344→    };
   345→    // Force re-render by ending drag with commit=false (optimistic update)
   346→    pmActions.endDrag(false);
   347→
   348→    // Load impact assessment
   349→    try {
   350→      const impact = await getPmMoveImpact({
   351→        pmId,
   352→        currentDate: dragInfo.sourceDate,
   353→        targetDate,
   354→        trainId: dragInfo.trainId,
   355→      });
   356→      dragConfirmRef.current = {
   357→        ...dragConfirmRef.current,
   358→        impact,
   359→        isLoadingImpact: false,
   360→      };
   361→    } catch {
   362→      dragConfirmRef.current = {
   363→        ...dragConfirmRef.current,
   364→        isLoadingImpact: false,
   365→      };
   366→    }
   367→
   368→    // Keep draggingPmRef for the confirmation dialog
   369→  };
   370→
   371→  const handleDragConfirm = async () => {
   372→    const confirmState = dragConfirmRef.current;
   373→    const dragInfo = draggingPmRef.current;
   374→
   375→    if (!confirmState.pmId || !dragInfo) {
   376→      handleDragCancel();
   377→      return;
   378→    }
   379→
   380→    try {
   381→      // Commit the move to API
   382→      await commitPmMove({
   383→        pmId: confirmState.pmId,
   384→        currentDate: confirmState.sourceDate,
   385→        targetDate: confirmState.targetDate,
   386→        trainId: dragInfo.trainId,
   387→        reason: 'Manual reschedule via canvas drag-and-drop',
   388→      });
   389→
   390→      // Confirm move in PM slice (updates local state)
   391→      pmActions.confirmMove(confirmState.pmId);
   392→
   393→      // Show success toast
   394→      void showToast({
   395→        intent: Intent.SUCCESS,
   396→        icon: 'tick',
   397→        message: `${confirmState.pmCode} rescheduled to ${confirmState.targetDate}`,
   398→        timeout: 3000,
   399→      });
   400→
   401→      // Reload tactical data to reflect changes
   402→      handleLoadTacticalData();
   403→    } catch (error) {
   404→      // Reject move (revert to original date)
   405→      pmActions.rejectMove(confirmState.pmId);
   406→
   407→      // Show error toast
   408→      void showToast({
   409→        intent: Intent.DANGER,
   410→        icon: 'error',
   411→        message: error instanceof Error ? error.message : 'Failed to reschedule PM',
   412→        timeout: 5000,
   413→      });
   414→    } finally {
   415→      // Close dialog and clear state
   416→      dragConfirmRef.current = initialDragConfirmState;
   417→      draggingPmRef.current = null;
   418→    }
   419→  };
   420→
   421→  const handleDragCancel = () => {
   422→    const confirmState = dragConfirmRef.current;
