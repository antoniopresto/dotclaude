     1→/**
     2→ * UI Slice
     3→ *
     4→ * State management for global UI state: toasts, modals, loading, errors.
     5→ *
     6→ * @see specs/006-user-interface.md Section 6.8
     7→ */
     8→
     9→import { createSlice } from '@belt/tools';
    10→import type { Toast, ModalState, LoadingState, ErrorState, OptimizerJob } from '../types';
    11→
    12→// =============================================================================
    13→// UI Slice State
    14→// =============================================================================
    15→
    16→export interface UiSliceState {
    17→  // Toast notifications
    18→  toasts: Toast[];
    19→  maxToasts: number;
    20→
    21→  // Modal state
    22→  modal: ModalState;
    23→
    24→  // Loading indicators
    25→  loading: LoadingState;
    26→
    27→  // Global error state
    28→  error: ErrorState;
    29→
    30→  // Sidebar collapsed
    31→  sidebarCollapsed: boolean;
    32→
    33→  // Help panel open
    34→  helpPanelOpen: boolean;
    35→
    36→  // Keyboard shortcuts enabled
    37→  shortcutsEnabled: boolean;
    38→
    39→  // Theme
    40→  theme: 'light' | 'dark' | 'system';
    41→
    42→  // Optimizer jobs tracking
    43→  optimizerJobs: OptimizerJob[];
    44→  activeJobId: string | null;
    45→
    46→  // Connection status
    47→  isOnline: boolean;
    48→  websocketConnected: boolean;
    49→
    50→  // Sync status
    51→  lastSyncAt: number | null;
    52→  syncInProgress: boolean;
    53→
    54→  // Index signature for MachineState compatibility
    55→  [key: string]: unknown;
    56→}
    57→
    58→const initialUiState: UiSliceState = {
    59→  toasts: [],
    60→  maxToasts: 5,
    61→  modal: {
    62→    isOpen: false,
    63→    type: null,
    64→    data: null,
    65→  },
    66→  loading: {
    67→    trains: false,
    68→    pmSchedule: false,
    69→    canvas: false,
    70→    optimizer: false,
    71→  },
    72→  error: {
    73→    message: null,
    74→    code: null,
    75→    details: null,
    76→    timestamp: null,
    77→  },
    78→  sidebarCollapsed: false,
    79→  helpPanelOpen: false,
    80→  shortcutsEnabled: true,
    81→  theme: 'system',
    82→  optimizerJobs: [],
    83→  activeJobId: null,
    84→  isOnline: true,
    85→  websocketConnected: false,
    86→  lastSyncAt: null,
    87→  syncInProgress: false,
    88→};
    89→
    90→let toastIdCounter = 0;
    91→
    92→export const uiSlice = createSlice({
    93→  id: 'ui',
    94→  initial: initialUiState,
    95→  cache: { policy: 'session' },
    96→  reducers: {
    97→    // ==========================================================================
    98→    // Toast Notifications
    99→    // ==========================================================================
   100→
   101→    addToast: (
   102→      s,
   103→      toast: Omit<Toast, 'id' | 'createdAt'> & { id?: string }
   104→    ) => {
   105→      const state = s as UiSliceState;
   106→      const newToast: Toast = {
   107→        id: toast.id ?? `toast-${++toastIdCounter}`,
   108→        type: toast.type,
   109→        title: toast.title,
   110→        message: toast.message,
   111→        duration: toast.duration ?? 5000,
   112→        dismissible: toast.dismissible ?? true,
   113→        createdAt: Date.now(),
   114→      };
   115→
   116→      state.toasts.push(newToast);
   117→
   118→      // Trim to max toasts
   119→      if (state.toasts.length > state.maxToasts) {
   120→        state.toasts.shift();
   121→      }
   122→    },
   123→
   124→    removeToast: (s, toastId: string) => {
   125→      const state = s as UiSliceState;
   126→      state.toasts = state.toasts.filter((t) => t.id !== toastId);
   127→    },
   128→
   129→    clearToasts: (s) => {
   130→      const state = s as UiSliceState;
   131→      state.toasts = [];
   132→    },
   133→
   134→    // Convenience toast creators
   135→    showSuccess: (s, payload: { title: string; message: string }) => {
   136→      const state = s as UiSliceState;
   137→      state.toasts.push({
   138→        id: `toast-${++toastIdCounter}`,
   139→        type: 'success',
   140→        title: payload.title,
   141→        message: payload.message,
   142→        duration: 3000,
   143→        dismissible: true,
   144→        createdAt: Date.now(),
   145→      });
   146→    },
   147→
   148→    showError: (s, payload: { title: string; message: string }) => {
   149→      const state = s as UiSliceState;
   150→      state.toasts.push({
   151→        id: `toast-${++toastIdCounter}`,
   152→        type: 'error',
   153→        title: payload.title,
   154→        message: payload.message,
   155→        duration: 8000,
   156→        dismissible: true,
   157→        createdAt: Date.now(),
   158→      });
   159→    },
   160→
   161→    showWarning: (s, payload: { title: string; message: string }) => {
   162→      const state = s as UiSliceState;
   163→      state.toasts.push({
   164→        id: `toast-${++toastIdCounter}`,
   165→        type: 'warning',
   166→        title: payload.title,
   167→        message: payload.message,
   168→        duration: 6000,
   169→        dismissible: true,
   170→        createdAt: Date.now(),
   171→      });
   172→    },
   173→
   174→    showInfo: (s, payload: { title: string; message: string }) => {
   175→      const state = s as UiSliceState;
   176→      state.toasts.push({
   177→        id: `toast-${++toastIdCounter}`,
   178→        type: 'info',
   179→        title: payload.title,
   180→        message: payload.message,
   181→        duration: 5000,
   182→        dismissible: true,
   183→        createdAt: Date.now(),
   184→      });
   185→    },
   186→
   187→    // ==========================================================================
   188→    // Modal
   189→    // ==========================================================================
   190→
   191→    openModal: (s, payload: { type: string; data?: Record<string, unknown> }) => {
   192→      const state = s as UiSliceState;
   193→      state.modal = {
   194→        isOpen: true,
   195→        type: payload.type,
   196→        data: payload.data ?? null,
   197→      };
   198→    },
   199→
   200→    closeModal: (s) => {
   201→      const state = s as UiSliceState;
   202→      state.modal = {
   203→        isOpen: false,
   204→        type: null,
   205→        data: null,
   206→      };
   207→    },
   208→
   209→    updateModalData: (s, data: Record<string, unknown>) => {
   210→      const state = s as UiSliceState;
   211→      if (state.modal.isOpen) {
   212→        state.modal.data = { ...state.modal.data, ...data };
   213→      }
   214→    },
   215→
   216→    // ==========================================================================
   217→    // Loading
   218→    // ==========================================================================
   219→
   220→    setLoading: (s, payload: Partial<LoadingState>) => {
   221→      const state = s as UiSliceState;
   222→      state.loading = { ...state.loading, ...payload };
   223→    },
   224→
   225→    startLoading: (s, key: keyof LoadingState) => {
   226→      const state = s as UiSliceState;
   227→      state.loading[key] = true;
   228→    },
   229→
   230→    stopLoading: (s, key: keyof LoadingState) => {
   231→      const state = s as UiSliceState;
   232→      state.loading[key] = false;
   233→    },
   234→
   235→    clearLoading: (s) => {
   236→      const state = s as UiSliceState;
   237→      state.loading = {
   238→        trains: false,
   239→        pmSchedule: false,
   240→        canvas: false,
   241→        optimizer: false,
   242→      };
   243→    },
   244→
   245→    // ==========================================================================
   246→    // Error
   247→    // ==========================================================================
   248→
   249→    setError: (
   250→      s,
   251→      payload: { message: string; code?: string; details?: Record<string, unknown> }
   252→    ) => {
   253→      const state = s as UiSliceState;
   254→      state.error = {
   255→        message: payload.message,
   256→        code: payload.code ?? null,
   257→        details: payload.details ?? null,
   258→        timestamp: Date.now(),
   259→      };
   260→    },
   261→
   262→    clearError: (s) => {
   263→      const state = s as UiSliceState;
   264→      state.error = {
   265→        message: null,
   266→        code: null,
   267→        details: null,
   268→        timestamp: null,
   269→      };
   270→    },
   271→
   272→    // ==========================================================================
   273→    // Layout
   274→    // ==========================================================================
   275→
   276→    toggleSidebar: (s) => {
   277→      const state = s as UiSliceState;
   278→      state.sidebarCollapsed = !state.sidebarCollapsed;
   279→    },
   280→
   281→    setSidebarCollapsed: (s, collapsed: boolean) => {
   282→      const state = s as UiSliceState;
   283→      state.sidebarCollapsed = collapsed;
   284→    },
   285→
   286→    toggleHelpPanel: (s) => {
   287→      const state = s as UiSliceState;
   288→      state.helpPanelOpen = !state.helpPanelOpen;
   289→    },
   290→
   291→    setHelpPanelOpen: (s, open: boolean) => {
   292→      const state = s as UiSliceState;
   293→      state.helpPanelOpen = open;
   294→    },
   295→
   296→    // ==========================================================================
   297→    // Preferences
   298→    // ==========================================================================
   299→
   300→    setShortcutsEnabled: (s, enabled: boolean) => {
   301→      const state = s as UiSliceState;
   302→      state.shortcutsEnabled = enabled;
   303→    },
   304→
   305→    setTheme: (s, theme: 'light' | 'dark' | 'system') => {
   306→      const state = s as UiSliceState;
   307→      state.theme = theme;
   308→    },
   309→
   310→    // ==========================================================================
   311→    // Optimizer Jobs
   312→    // ==========================================================================
   313→
   314→    addOptimizerJob: (s, job: OptimizerJob) => {
   315→      const state = s as UiSliceState;
   316→      state.optimizerJobs.push(job);
   317→      state.activeJobId = job.id;
   318→    },
   319→
   320→    updateOptimizerJob: (s, job: Partial<OptimizerJob> & { id: string }) => {
   321→      const state = s as UiSliceState;
   322→      const index = state.optimizerJobs.findIndex((j) => j.id === job.id);
   323→      if (index !== -1) {
   324→        state.optimizerJobs[index] = { ...state.optimizerJobs[index], ...job };
   325→      }
   326→    },
   327→
   328→    removeOptimizerJob: (s, jobId: string) => {
   329→      const state = s as UiSliceState;
   330→      state.optimizerJobs = state.optimizerJobs.filter((j) => j.id !== jobId);
   331→      if (state.activeJobId === jobId) {
   332→        state.activeJobId = state.optimizerJobs[0]?.id ?? null;
   333→      }
   334→    },
   335→
   336→    setActiveJob: (s, jobId: string | null) => {
   337→      const state = s as UiSliceState;
   338→      state.activeJobId = jobId;
   339→    },
   340→
   341→    clearCompletedJobs: (s) => {
   342→      const state = s as UiSliceState;
   343→      state.optimizerJobs = state.optimizerJobs.filter(
   344→        (j) => j.status !== 'COMPLETED' && j.status !== 'FAILED'
   345→      );
   346→    },
   347→
   348→    // ==========================================================================
   349→    // Connection Status
   350→    // ==========================================================================
   351→
   352→    setOnlineStatus: (s, isOnline: boolean) => {
   353→      const state = s as UiSliceState;
   354→      state.isOnline = isOnline;
   355→    },
   356→
   357→    setWebsocketConnected: (s, connected: boolean) => {
   358→      const state = s as UiSliceState;
   359→      state.websocketConnected = connected;
   360→    },
   361→
   362→    // ==========================================================================
   363→    // Sync Status
   364→    // ==========================================================================
   365→
   366→    startSync: (s) => {
   367→      const state = s as UiSliceState;
   368→      state.syncInProgress = true;
   369→    },
   370→
   371→    finishSync: (s) => {
   372→      const state = s as UiSliceState;
   373→      state.syncInProgress = false;
   374→      state.lastSyncAt = Date.now();
   375→    },
   376→
   377→    // ==========================================================================
   378→    // Reset
   379→    // ==========================================================================
   380→
   381→    reset: (s) => {
   382→      const state = s as UiSliceState;
   383→      Object.assign(state, initialUiState);
   384→    },
   385→  },
   386→});
   387→
   388→// =============================================================================
   389→// Selectors
   390→// =============================================================================
   391→
   392→/** Check if any loading is in progress */
   393→export function isAnyLoading(state: UiSliceState): boolean {
   394→  return Object.values(state.loading).some(Boolean);
   395→}
   396→
   397→/** Get active optimizer job */
   398→export function selectActiveJob(state: UiSliceState): OptimizerJob | undefined {
   399→  return state.activeJobId
   400→    ? state.optimizerJobs.find((j) => j.id === state.activeJobId)
   401→    : undefined;
   402→}
   403→
   404→/** Get running optimizer jobs */
   405→export function selectRunningJobs(state: UiSliceState): OptimizerJob[] {
   406→  return state.optimizerJobs.filter((j) => j.status === 'RUNNING' || j.status === 'PENDING');
   407→}
   408→
   409→/** Check if there's an error */
   410→export function hasError(state: UiSliceState): boolean {
   411→  return state.error.message !== null;
   412→}
   413→
   414→/** Get visible toasts */
   415→export function selectVisibleToasts(state: UiSliceState): Toast[] {
   416→  const now = Date.now();
   417→  return state.toasts.filter((t) => now - t.createdAt < t.duration);
   418→}
   419→
   420→/** Check connection status */
   421→export function isFullyConnected(state: UiSliceState): boolean {
   422→  return state.isOnline && state.websocketConnected;
   423→}
