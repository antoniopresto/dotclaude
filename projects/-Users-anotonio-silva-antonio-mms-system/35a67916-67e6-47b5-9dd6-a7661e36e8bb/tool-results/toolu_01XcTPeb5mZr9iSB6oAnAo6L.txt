  3673→## 4.10 Costing Service
  3674→
  3675→### 4.10.1 Purpose
  3676→
  3677→Calculate TCO projections, track variance against budget, and provide cost impact analysis for adjustments.
  3678→
  3679→### 4.10.2 Implementation
  3680→
  3681→```typescript
  3682→interface CostBreakdown {
  3683→  laborCost: number;
  3684→  overtimeCost: number;
  3685→  materialCost: number;
  3686→  contractorCost: number;
  3687→  overheadCost: number;
  3688→  totalCost: number;
  3689→}

  3691→class CostingService {
  3692→  async calculatePMCost(
  3693→    pmScheduleId: string,
  3694→    includeProjected: boolean = false
  3695→  ): Promise<CostBreakdown> {
  3696→    const pm = await this.db.query.pmSchedule.findFirst({
  3697→      where: eq(pmSchedule.id, pmScheduleId),
  3698→      with: {
  3699→        pmType: true,
  3700→        assignedDepot: true
  3701→      }
  3702→    });
  3703→    
  3704→    if (!pm) {
  3705→      throw new NotFoundError('PM not found');
  3706→    }
  3707→    
  3708→    // Get labor rates
  3709→    const laborRate = await this.getLaborRate(pm.assignedDepotId, pm.targetDate);
  3710→    
  3711→    // Get material costs
  3712→    const materialCost = await this.calculateMaterialCost(pm.pmType.id);
  3713→    
  3714→    // Calculate labor cost
  3715→    const hours = pm.actualHours || pm.assignedHours || pm.pmType.estimatedHours;
  3716→    const laborCost = hours * laborRate.shopRateHourly;
  3717→    
  3718→    // Overtime (if applicable)
  3719→    let overtimeCost = 0;
  3720→    if (pm.assignedShift === 'NIGHT' || pm.assignedShift === 'WEEKEND') {
  3721→      const multiplier = pm.assignedShift === 'WEEKEND' 
  3722→        ? laborRate.weekendMultiplier 
  3723→        : laborRate.overtimeMultiplier;
  3724→      overtimeCost = hours * laborRate.regularHourlyRate * (multiplier - 1);
  3725→    }
  3726→    
  3727→    // Overhead (typically 15-20% of direct costs)
  3728→    const overheadRate = 0.18;
  3729→    const directCost = laborCost + materialCost;
  3730→    const overheadCost = directCost * overheadRate;
  3731→    
  3732→    return {
  3733→      laborCost,
  3734→      overtimeCost,
  3735→      materialCost,
  3736→      contractorCost: 0,
  3737→      overheadCost,
  3738→      totalCost: laborCost + overtimeCost + materialCost + overheadCost
  3739→    };
  3740→  }

  3742→  async projectTCO(
  3743→    horizonYears: number = 12
  3744→  ): Promise<TCOProjection> {
  3745→    const projections: YearlyCost[] = [];
  3746→    const startYear = new Date().getFullYear();
  3747→    
  3748→    for (let year = 0; year < horizonYears; year++) {
  3749→      const fiscalYear = startYear + year;
  3750→      
  3751→      // Get scheduled PMs for this year
  3752→      const yearStart = new Date(fiscalYear, 0, 1);
  3753→      const yearEnd = new Date(fiscalYear, 11, 31);
  3754→      
  3755→      const scheduledPMs = await this.db.query.pmSchedule.findMany({
  3756→        where: and(
  3757→          isNull(pmSchedule.scenarioId),
  3758→          gte(pmSchedule.targetDate, yearStart),
  3759→          lte(pmSchedule.targetDate, yearEnd)
  3760→        ),
  3761→        with: { pmType: true }
  3762→      });
  3763→      
  3764→      // Calculate costs
  3765→      let totalLabor = 0;
  3766→      let totalMaterial = 0;
  3767→      
  3768→      for (const pm of scheduledPMs) {
  3769→        const cost = await this.calculatePMCost(pm.id, true);
  3770→        totalLabor += cost.laborCost + cost.overtimeCost;
  3771→        totalMaterial += cost.materialCost;
  3772→      }
  3773→      
  3774→      // Apply inflation adjustment for future years
  3775→      const inflationRate = 0.03;
  3776→      const inflationFactor = Math.pow(1 + inflationRate, year);
  3777→      
  3778→      projections.push({
  3779→        fiscalYear,
  3780→        laborCost: totalLabor * inflationFactor,
  3781→        materialCost: totalMaterial * inflationFactor,
  3782→        pmCount: scheduledPMs.length,
  3783→        inflationFactor
  3784→      });
  3785→    }
  3786→    
  3787→    // Calculate NPV
  3788→    const discountRate = 0.05;
  3789→    let npv = 0;
  3790→    
  3791→    for (let i = 0; i < projections.length; i++) {
  3792→      const yearCost = projections[i].laborCost + projections[i].materialCost;
  3793→      npv += yearCost / Math.pow(1 + discountRate, i);
  3794→    }
  3795→    
  3796→    return {
  3797→      projections,
  3798→      summary: {
  3799→        totalLaborCost: projections.reduce((sum, p) => sum + p.laborCost, 0),
  3800→        totalMaterialCost: projections.reduce((sum, p) => sum + p.materialCost, 0),
  3801→        totalPMCount: projections.reduce((sum, p) => sum + p.pmCount, 0),
  3802→        npv,
  3803→        avgAnnualCost: npv / horizonYears
  3804→      }
  3805→    };
  3806→  }

  3808→  async calculateVariance(
  3809→    fiscalYear: number,
  3810→    depotId?: string
  3811→  ): Promise<CostVariance> {
  3812→    const baseline = await this.db.query.costBaselines.findFirst({

  3815→        depotId ? eq(costBaselines.depotId, depotId) : isNull(costBaselines.depotId)

  3818→    
  3819→    if (!baseline) {

  3822→    
  3823→    // Get actual costs from completed PMs

  3835→

  3845→    const actualTotal = actualLaborCost + actualMaterialCost;
  3846→    const budgetTotal = baseline.budgetTotal;
  3847→    
  3848→    return {
  3849→      fiscalYear,
  3850→      depotId,
  3851→      budget: {
  3852→        labor: baseline.budgetLaborCost,
  3853→        material: baseline.budgetMaterialCost,
  3854→        total: budgetTotal
  3855→      },
  3856→      actual: {
  3857→        labor: actualLaborCost,
  3858→        material: actualMaterialCost,
  3859→        total: actualTotal
  3860→      },
  3861→      variance: {
  3862→        labor: actualLaborCost - baseline.budgetLaborCost,
  3863→        material: actualMaterialCost - baseline.budgetMaterialCost,

  3865→        percentage: ((actualTotal - budgetTotal) / budgetTotal) * 100
  3866→      },
  3867→      status: actualTotal <= budgetTotal ? 'UNDER_BUDGET' : 
  3868→              actualTotal <= budgetTotal * 1.1 ? 'WITHIN_TOLERANCE' : 'OVER_BUDGET'
  3869→    };
  3870→  }
