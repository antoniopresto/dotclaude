    65→### 4.2.2 File Watcher
    66→
    67→```typescript
    68→interface FileWatcherConfig {
    69→  inboxPath: string;           // /data/inbox
    70→  processedPath: string;       // /data/processed
    71→  quarantinePath: string;      // /data/quarantine
    72→  expectedFilePattern: RegExp; // /^mileage_\d{8}\.csv$/
    73→  maxFileSizeBytes: number;    // 10MB
    74→  alertThresholdHours: number; // 2
    75→}
    76→
    77→class FileWatcherService {
    78→  private watcher: FSWatcher;
    79→  private lastFileReceivedAt: Date | null = null;
    80→  
    81→  async start(): Promise<void> {
    82→    this.watcher = chokidar.watch(this.config.inboxPath, {
    83→      persistent: true,
    84→      awaitWriteFinish: { stabilityThreshold: 2000, pollInterval: 500 }
    85→    });
    86→    
    87→    this.watcher.on('add', async (filePath) => {
    88→      this.lastFileReceivedAt = new Date();
    89→      await this.processFile(filePath);
    90→    });
    91→    
    92→    // Health check interval
    93→    setInterval(() => this.checkFileArrivalHealth(), 30 * 60 * 1000);
    94→  }
    95→  
    96→  private async processFile(filePath: string): Promise<void> {
    97→    const filename = path.basename(filePath);
    98→    const jobId = crypto.randomUUID();
    99→    
   100→    try {
   101→      // Validate file
   102→      await this.validateFile(filePath);
   103→      
   104→      // Queue for processing
   105→      await this.jobQueue.add('ingestion', {
   106→        jobId,
   107→        filePath,
   108→        filename,
   109→        receivedAt: new Date().toISOString()
   110→      }, {
   111→        attempts: 3,
   112→        backoff: { type: 'exponential', delay: 5000 }
   113→      });
   114→      
   115→      this.logger.info({ jobId, filename }, 'File queued for ingestion');
   116→      
   117→    } catch (error) {
   118→      // Move to quarantine
   119→      const quarantinePath = path.join(
   120→        this.config.quarantinePath, 
   121→        `${Date.now()}_${filename}`
   122→      );
   123→      await fs.rename(filePath, quarantinePath);
   124→      
   125→      this.logger.error({ filename, error }, 'File quarantined');
   126→      
   127→      await this.alertService.send({
   128→        type: 'FILE_VALIDATION_FAILED',
   129→        severity: 'WARNING',
   130→        message: `File ${filename} failed validation: ${error.message}`,
   131→        filePath: quarantinePath
   132→      });
   133→    }
   134→  }
   135→  
   136→  private async validateFile(filePath: string): Promise<void> {
   137→    const stats = await fs.stat(filePath);
   138→    
   139→    // Size check
   140→    if (stats.size > this.config.maxFileSizeBytes) {
   141→      throw new Error(`File exceeds maximum size: ${stats.size} bytes`);
   142→    }
   143→    
   144→    // Format check
   145→    const filename = path.basename(filePath);
   146→    if (!this.config.expectedFilePattern.test(filename)) {
   147→      throw new Error(`Filename does not match expected pattern: ${filename}`);
   148→    }
   149→    
   150→    // Content check (first 10 lines)
   151→    const sample = await this.readFirstLines(filePath, 10);
   152→    await this.validateCSVStructure(sample);
   153→  }
   154→  
   155→  private async checkFileArrivalHealth(): Promise<void> {
   156→    if (!this.lastFileReceivedAt) return;
   157→    
   158→    const hoursSinceLastFile = 
   159→      (Date.now() - this.lastFileReceivedAt.getTime()) / (1000 * 60 * 60);
   160→    
   161→    if (hoursSinceLastFile > this.config.alertThresholdHours) {
   162→      await this.alertService.send({
   163→        type: 'FILE_ARRIVAL_DELAYED',
   164→        severity: 'WARNING',
   165→        message: `No mileage files received in ${hoursSinceLastFile.toFixed(1)} hours`,
   166→        lastFileAt: this.lastFileReceivedAt.toISOString(),
   167→        expectedInterval: `${this.config.alertThresholdHours} hours`
   168→      });
   169→    }
   170→  }
   171→}
   172→```

   174→### 4.2.3 Ingestion Worker
   175→
   176→```typescript
   177→interface MileageRecord {
   178→  trainNumber: string;
   179→  readingDate: string;  // YYYY-MM-DD
   180→  odometerValue: number;
   181→  source?: string;
   182→}
   183→
   184→interface IngestionResult {
   185→  jobId: string;
   186→  filename: string;
   187→  totalRecords: number;
   188→  validRecords: number;
   189→  invalidRecords: number;
   190→  anomalies: AnomalyRecord[];
   191→  meterEvents: MeterEventRecord[];
   192→  processingTimeMs: number;
   193→}
   194→
   195→class IngestionWorker {
   196→  async process(job: Job<IngestionJobData>): Promise<IngestionResult> {
   197→    const startTime = Date.now();
   198→    const { jobId, filePath, filename } = job.data;
   199→    
   200→    const result: IngestionResult = {
   201→      jobId,
   202→      filename,
   203→      totalRecords: 0,
   204→      validRecords: 0,
   205→      invalidRecords: 0,
   206→      anomalies: [],
   207→      meterEvents: [],
   208→      processingTimeMs: 0
   209→    };
   210→    
   211→    // Parse CSV
   212→    const records = await this.parseCSV(filePath);
   213→    result.totalRecords = records.length;
   214→    
   215→    // Process in batches
   216→    const batches = this.chunk(records, 100);
   217→    
   218→    for (const batch of batches) {
   219→      await this.db.transaction(async (tx) => {
   220→        for (const record of batch) {
   221→          try {
   222→            await this.processRecord(tx, record, result);
   223→            result.validRecords++;
   224→          } catch (error) {
   225→            result.invalidRecords++;
   226→            this.logger.warn({ record, error }, 'Record processing failed');
   227→          }
   228→        }
   229→      });
   230→    }
   231→    
   232→    // Move file to processed
   233→    const processedPath = path.join(
   234→      this.config.processedPath,
   235→      `${Date.now()}_${filename}`
   236→    );
   237→    await fs.rename(filePath, processedPath);
   238→    
   239→    // Update train current mileage
   240→    await this.updateTrainMileage(records);
   241→    
   242→    // Trigger downstream processing
   243→    await this.triggerDriftCheck();
   244→    
   245→    result.processingTimeMs = Date.now() - startTime;
   246→    
   247→    return result;
   248→  }
   249→  
   250→  private async processRecord(
   251→    tx: Transaction,
   252→    record: MileageRecord,
   253→    result: IngestionResult
   254→  ): Promise<void> {
   255→    // Find train
   256→    const train = await tx.query.trains.findFirst({
   257→      where: eq(trains.trainNumber, record.trainNumber)
   258→    });
   259→    
   260→    if (!train) {
   261→      throw new Error(`Unknown train: ${record.trainNumber}`);
   262→    }
   263→    
   264→    // Get previous reading
   265→    const previousReading = await tx.query.mileageReadings.findFirst({
   266→      where: and(
   267→        eq(mileageReadings.trainId, train.id),
   268→        lt(mileageReadings.readingDate, record.readingDate)
   269→      ),
   270→      orderBy: desc(mileageReadings.readingDate)
   271→    });
   272→    
   273→    // Calculate adjusted value (handle meter events)
   274→    const adjustment = await this.getMeterAdjustment(tx, train.id, record.readingDate);
   275→    const adjustedReading = record.odometerValue + adjustment;
   276→    
   277→    // Calculate daily km
   278→    let dailyKm: number | null = null;
   279→    let anomalyType: string | null = null;
   280→    
   281→    if (previousReading) {
   282→      dailyKm = adjustedReading - previousReading.adjustedReading;
   283→      
   284→      // Anomaly detection
   285→      if (dailyKm < 0) {
   286→        anomalyType = 'NEGATIVE';
   287→        result.anomalies.push({
   288→          trainNumber: record.trainNumber,
   289→          date: record.readingDate,
   290→          type: 'NEGATIVE',
   291→          value: dailyKm,
   292→          message: `Negative daily mileage: ${dailyKm} km`
   293→        });
   294→      } else if (dailyKm > 1000) {  // Spike threshold
   295→        anomalyType = 'SPIKE';
   296→        result.anomalies.push({
   297→          trainNumber: record.trainNumber,
   298→          date: record.readingDate,
   299→          type: 'SPIKE',
   300→          value: dailyKm,
   301→          message: `Unusually high daily mileage: ${dailyKm} km`
   302→        });
   303→      }
   304→    }
   305→    
   306→    // Insert or update reading
   307→    await tx.insert(mileageReadings)
   308→      .values({
   309→        trainId: train.id,
   310→        readingDate: record.readingDate,
   311→        rawReading: record.odometerValue,
   312→        adjustedReading,
   313→        dailyKm,
   314→        validationStatus: anomalyType ? 'ANOMALY' : 'VALID',
   315→        anomalyType,
   316→        sourceFile: result.filename
   317→      })
   318→      .onConflictDoUpdate({
   319→        target: [mileageReadings.trainId, mileageReadings.readingDate],
   320→        set: {
   321→          rawReading: record.odometerValue,
   322→          adjustedReading,
   323→          dailyKm,
   324→          validationStatus: anomalyType ? 'ANOMALY' : 'VALID',
   325→          anomalyType,
   326→          sourceFile: result.filename
   327→        }
   328→      });
   329→  }
   330→  
   331→  private async getMeterAdjustment(
   332→    tx: Transaction,
   333→    trainId: string,
   334→    asOfDate: string
   335→  ): Promise<number> {
   336→    // Sum all meter event adjustments up to this date
   337→    const events = await tx.query.meterEvents.findMany({
   338→      where: and(
   339→        eq(meterEvents.trainId, trainId),
   340→        lte(meterEvents.eventDate, asOfDate)
   341→      )
   342→    });
   343→    
   344→    return events.reduce((sum, e) => sum + (e.adjustmentKm || 0), 0);
   345→  }
   346→  
   347→  private async updateTrainMileage(records: MileageRecord[]): Promise<void> {
   348→    // Group by train, find latest reading
   349→    const latestByTrain = new Map<string, MileageRecord>();
   350→    
   351→    for (const record of records) {
   352→      const existing = latestByTrain.get(record.trainNumber);
   353→      if (!existing || record.readingDate > existing.readingDate) {
   354→        latestByTrain.set(record.trainNumber, record);
   355→      }
   356→    }
   357→    
   358→    // Update train current_mileage
   359→    for (const [trainNumber, record] of latestByTrain) {
   360→      await this.db.update(trains)
   361→        .set({ 
   362→          currentMileageKm: record.odometerValue,
   363→          updatedAt: new Date()
   364→        })
   365→        .where(eq(trains.trainNumber, trainNumber));
   366→    }
   367→    
   368→    // Update fleet mileage targets
   369→    await this.db.execute(sql`SELECT update_fleet_mileage_targets()`);
   370→  }
   371→}
