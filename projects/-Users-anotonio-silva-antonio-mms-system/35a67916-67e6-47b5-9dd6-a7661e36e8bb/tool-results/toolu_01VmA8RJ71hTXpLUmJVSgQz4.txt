     1→/**
     2→ * MMS Error Classes
     3→ *
     4→ * Typed error classes for consistent error handling across the application.
     5→ * Each error includes a code for client-side handling and logging.
     6→ *
     7→ * Error code format: CATEGORY_NNN
     8→ * - FILE_001-099: File/CSV processing errors
     9→ * - ROW_001-099: Row/record validation errors
    10→ * - SYS_001-099: System/infrastructure errors
    11→ * - BUS_001-099: Business logic errors
    12→ * - AUTH_001-099: Authentication/authorization errors
    13→ * - EXT_001-099: External system errors (Maximo, etc.)
    14→ *
    15→ * @see specs/004-backend-specification.md Section 4.15
    16→ */
    17→
    18→/**
    19→ * Base error class for all MMS errors
    20→ */
    21→export abstract class MmsError extends Error {
    22→  abstract readonly code: string;
    23→  abstract readonly statusCode: number;
    24→  readonly isOperational: boolean = true;
    25→  readonly timestamp: Date = new Date();
    26→  readonly context?: Record<string, unknown>;
    27→
    28→  constructor(message: string, context?: Record<string, unknown>) {
    29→    super(message);
    30→    this.name = this.constructor.name;
    31→    this.context = context;
    32→    Error.captureStackTrace(this, this.constructor);
    33→  }
    34→
    35→  toJSON() {
    36→    return {
    37→      name: this.name,
    38→      code: this.code,
    39→      message: this.message,
    40→      statusCode: this.statusCode,
    41→      timestamp: this.timestamp.toISOString(),
    42→      ...(this.context && { context: this.context }),
    43→    };
    44→  }
    45→}

   583→
   584→/**
   585→ * Type guard to check if an error is an MmsError
   586→ */
   587→export function isMmsError(error: unknown): error is MmsError {
   588→  return error instanceof MmsError;
   589→}
   590→
   591→/**
   592→ * Convert unknown error to MmsError
   593→ */
   594→export function toMmsError(error: unknown): MmsError {
   595→  if (isMmsError(error)) {
   596→    return error;
   597→  }
   598→
   599→  if (error instanceof Error) {
   600→    return new SystemError(error.message, {
   601→      originalError: error.name,
   602→      stack: error.stack,
   603→    });
   604→  }
   605→
   606→  return new SystemError(String(error));
   607→}
