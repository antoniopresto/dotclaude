     1→/**
     2→ * MMS Sidebar - Train List Panel
     3→ *
     4→ * Left sidebar with train list, filters, and selection.
     5→ * Based on specs/006-user-interface.md Section 6.2.
     6→ */
     7→
     8→import {
     9→  Card,
    10→  Icon,
    11→  Tag,
    12→  InputGroup,
    13→  HTMLSelect,
    14→  Button,
    15→  Classes,
    16→} from '@blueprintjs/core';
    17→
    18→import { useFleet, useCanvas, usePm } from '../../store';
    19→import { FITNESS_STATUS_COLORS, FLEET_COLORS } from '../theme';
    20→import type { Train, FitnessStatus } from '../../store/types';
    21→import type { FleetSliceState } from '../../store/slices/fleet';
    22→import type { PmSliceState } from '../../store/slices/pm';
    23→
    24→interface TrainCardProps {
    25→  train: Train;
    26→  isSelected: boolean;
    27→  pmCount: number;
    28→  onSelect: (trainId: string) => void;
    29→}
    30→
    31→function TrainCard({ train, isSelected, pmCount, onSelect }: TrainCardProps) {
    32→  const fitnessColor = FITNESS_STATUS_COLORS[train.fitnessStatus];
    33→  const fleetColor = FLEET_COLORS[train.fleetCode as keyof typeof FLEET_COLORS] || '#6B7280';
    34→
    35→  const fitnessIcon = train.fitnessStatus === 'FIT' ? 'tick-circle' :
    36→                      train.fitnessStatus === 'RESTRICTED' ? 'warning-sign' :
    37→                      'error';
    38→
    39→  return (
    40→    <Card
    41→      interactive
    42→      elevation={isSelected ? 2 : 0}
    43→      onClick={() => onSelect(train.id)}
    44→      style={{
    45→        marginBottom: '4px',
    46→        padding: '8px 12px',
    47→        borderLeft: `4px solid ${fleetColor}`,
    48→        backgroundColor: isSelected ? 'var(--pt-intent-primary-active, #137cbd20)' : undefined,
    49→      }}
    50→    >
    51→      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'flex-start' }}>
    52→        <div>
    53→          <div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
    54→            <strong style={{ fontSize: '14px' }}>{train.trainNumber}</strong>
    55→            <Tag minimal style={{ backgroundColor: fleetColor, color: 'white', fontSize: '10px' }}>
    56→              {train.fleetCode}
    57→            </Tag>
    58→          </div>
    59→          <div style={{ display: 'flex', alignItems: 'center', gap: '4px', marginTop: '4px' }}>
    60→            <Icon icon={fitnessIcon} size={12} color={fitnessColor} />
    61→            <span style={{ fontSize: '11px', color: fitnessColor }}>
    62→              {train.fitnessStatus}
    63→            </span>
    64→          </div>
    65→        </div>
    66→        <div style={{ textAlign: 'right' }}>
    67→          <div style={{ fontSize: '12px', fontFamily: 'monospace' }}>
    68→            {train.currentMileageKm.toLocaleString()} km
    69→          </div>
    70→          {pmCount > 0 && (
    71→            <Tag minimal intent="primary" style={{ marginTop: '4px' }}>
    72→              {pmCount} PM{pmCount > 1 ? 's' : ''}
    73→            </Tag>
    74→          )}
    75→        </div>
    76→      </div>
    77→      {train.fitnessNotes && (
    78→        <div style={{ fontSize: '11px', color: 'var(--pt-text-muted, #5c7080)', marginTop: '4px' }}>
    79→          {train.fitnessNotes}
    80→        </div>
    81→      )}
    82→    </Card>
    83→  );
    84→}
    85→
    86→export function Sidebar() {
    87→  const { state: fleetStateRaw, actions: fleetActions } = useFleet();
    88→  const { actions: canvasActions } = useCanvas();
    89→  const { state: pmStateRaw } = usePm();
    90→
    91→  const fleetState = fleetStateRaw as unknown as FleetSliceState;
    92→  const pmState = pmStateRaw as unknown as PmSliceState;
    93→
    94→  const handleSelectTrain = (trainId: string) => {
    95→    fleetActions.selectTrain(trainId);
    96→    canvasActions.selectTrain(trainId);
    97→  };
    98→
    99→  const handleSearchChange = (event: React.ChangeEvent<HTMLInputElement>) => {
   100→    fleetActions.setSearchTerm(event.target.value);
   101→  };
   102→
   103→  const handleFitnessFilterChange = (event: React.ChangeEvent<HTMLSelectElement>) => {
   104→    const value = event.target.value as FitnessStatus | '';
   105→    fleetActions.setFitnessFilter(value ? [value] : []);
   106→  };
   107→
   108→  const handleClearFilters = () => {
   109→    fleetActions.clearFilters();
   110→  };
   111→
   112→  // Filter trains based on current filters
   113→  const filteredTrains = fleetState.trains.filter(train => {
   114→    const { fleetIds, fitnessStatuses, depotIds, searchTerm } = fleetState.filters;
   115→
   116→    if (searchTerm && !train.trainNumber.toLowerCase().includes(searchTerm.toLowerCase())) {
   117→      return false;
   118→    }
   119→    if (fleetIds.length > 0 && !fleetIds.includes(train.fleetId)) {
   120→      return false;
   121→    }
   122→    if (fitnessStatuses.length > 0 && !fitnessStatuses.includes(train.fitnessStatus)) {
   123→      return false;
   124→    }
   125→    if (depotIds.length > 0 && !depotIds.includes(train.homeDepotId || '') && !depotIds.includes(train.currentDepotId || '')) {
   126→      return false;
   127→    }
   128→    return true;
   129→  });
   130→
   131→  // Sort trains
   132→  const sortedTrains = [...filteredTrains].sort((a, b) => {
   133→    switch (fleetState.sort.field) {
   134→      case 'trainNumber':
   135→        return fleetState.sort.direction === 'asc'
   136→          ? a.trainNumber.localeCompare(b.trainNumber)
   137→          : b.trainNumber.localeCompare(a.trainNumber);
   138→      case 'mileage':
   139→        return fleetState.sort.direction === 'asc'
   140→          ? a.currentMileageKm - b.currentMileageKm
   141→          : b.currentMileageKm - a.currentMileageKm;
   142→      case 'fitness': {
   143→        const order = { FIT: 0, RESTRICTED: 1, GROUNDED: 2 };
   144→        return fleetState.sort.direction === 'asc'
   145→          ? order[a.fitnessStatus] - order[b.fitnessStatus]
   146→          : order[b.fitnessStatus] - order[a.fitnessStatus];
   147→      }
   148→      default:
   149→        return 0;
   150→    }
   151→  });
   152→
   153→  // Count PMs per train
   154→  const pmCountByTrain = pmState.pmSchedule.reduce((acc, pm) => {
   155→    acc[pm.trainId] = (acc[pm.trainId] || 0) + 1;
   156→    return acc;
   157→  }, {} as Record<string, number>);
   158→
   159→  const hasFilters = fleetState.filters.searchTerm ||
   160→                     fleetState.filters.fleetIds.length > 0 ||
   161→                     fleetState.filters.fitnessStatuses.length > 0 ||
   162→                     fleetState.filters.depotIds.length > 0;
   163→
   164→  const fitnessOptions = [
   165→    { value: '', label: 'All Status' },
   166→    { value: 'FIT', label: 'FIT' },
   167→    { value: 'RESTRICTED', label: 'RESTRICTED' },
   168→    { value: 'GROUNDED', label: 'GROUNDED' },
   169→  ];
   170→
   171→  const sortOptions = [
   172→    { value: 'trainNumber', label: 'Train Number' },
   173→    { value: 'mileage', label: 'Mileage' },
   174→    { value: 'fitness', label: 'Fitness Status' },
   175→  ];
   176→
   177→  const selectedFitness = fleetState.filters.fitnessStatuses[0] || '';
   178→
   179→  return (
   180→    <div className="mms-sidebar" style={{
   181→      width: '260px',
   182→      height: '100%',
   183→      display: 'flex',
   184→      flexDirection: 'column',
   185→      borderRight: '1px solid var(--pt-divider-black, #e1e8ed)',
   186→      backgroundColor: 'var(--pt-app-background-color, #f5f8fa)',
   187→    }}>
   188→      {/* Header */}
   189→      <div style={{ padding: '12px', borderBottom: '1px solid var(--pt-divider-black, #e1e8ed)' }}>
   190→        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '8px' }}>
   191→          <span className={Classes.TEXT_MUTED} style={{ fontSize: '12px', fontWeight: 600, textTransform: 'uppercase' }}>
   192→            Trains ({sortedTrains.length})
   193→          </span>
   194→          {hasFilters && (
   195→            <Button minimal icon="filter-remove" onClick={handleClearFilters} title="Clear filters" />
   196→          )}
   197→        </div>
   198→
   199→        <InputGroup
   200→          leftIcon="search"
   201→          placeholder="Search..."
   202→          value={fleetState.filters.searchTerm}
   203→          onChange={handleSearchChange}
   204→        />
   205→
   206→        <div style={{ display: 'flex', gap: '8px', marginTop: '8px' }}>
   207→          <HTMLSelect
   208→            minimal
   209→            options={fitnessOptions}
   210→            value={selectedFitness}
   211→            onChange={handleFitnessFilterChange}
   212→            style={{ flex: 1 }}
   213→          />
   214→          <HTMLSelect
   215→            minimal
   216→            options={sortOptions}
   217→            value={fleetState.sort.field}
   218→            onChange={(e) => fleetActions.setSort({ field: e.target.value, direction: fleetState.sort.direction })}
   219→            style={{ flex: 1 }}
   220→          />
   221→        </div>
   222→      </div>
   223→
   224→      {/* Train List */}
   225→      <div style={{ flex: 1, overflowY: 'auto', padding: '8px' }}>
   226→        {sortedTrains.length === 0 ? (
   227→          <div style={{ textAlign: 'center', padding: '24px', color: 'var(--pt-text-muted, #5c7080)' }}>
   228→            <Icon icon="search" size={32} style={{ opacity: 0.5 }} />
   229→            <p style={{ marginTop: '8px' }}>No trains match filters</p>
   230→          </div>
   231→        ) : (
   232→          sortedTrains.map(train => (
   233→            <TrainCard
   234→              key={train.id}
   235→              train={train}
   236→              isSelected={train.id === fleetState.selectedTrainId}
   237→              pmCount={pmCountByTrain[train.id] || 0}
   238→              onSelect={handleSelectTrain}
   239→            />
   240→          ))
   241→        )}
   242→      </div>
   243→
   244→      {/* Fleet Summary */}
   245→      <div style={{
   246→        padding: '12px',
   247→        borderTop: '1px solid var(--pt-divider-black, #e1e8ed)',
   248→        backgroundColor: 'var(--pt-app-background-color, #f5f8fa)',
   249→      }}>
   250→        <div style={{ display: 'flex', justifyContent: 'space-around', fontSize: '11px' }}>
   251→          <div style={{ textAlign: 'center' }}>
   252→            <div style={{ color: FITNESS_STATUS_COLORS.FIT, fontWeight: 600 }}>
   253→              {fleetState.trains.filter(t => t.fitnessStatus === 'FIT').length}
   254→            </div>
   255→            <div className={Classes.TEXT_MUTED}>FIT</div>
   256→          </div>
   257→          <div style={{ textAlign: 'center' }}>
   258→            <div style={{ color: FITNESS_STATUS_COLORS.RESTRICTED, fontWeight: 600 }}>
   259→              {fleetState.trains.filter(t => t.fitnessStatus === 'RESTRICTED').length}
   260→            </div>
   261→            <div className={Classes.TEXT_MUTED}>RESTR</div>
   262→          </div>
   263→          <div style={{ textAlign: 'center' }}>
   264→            <div style={{ color: FITNESS_STATUS_COLORS.GROUNDED, fontWeight: 600 }}>
   265→              {fleetState.trains.filter(t => t.fitnessStatus === 'GROUNDED').length}
   266→            </div>
   267→            <div className={Classes.TEXT_MUTED}>GRND</div>
   268→          </div>
   269→        </div>
   270→      </div>
   271→    </div>
   272→  );
   273→}
