     1→/**
     2→ * MMS Maximo Sync Service
     3→ *
     4→ * Implements bidirectional synchronization between MMS and IBM Maximo CMMS.
     5→ * Uses the shadow record pattern to enable offline operation and conflict resolution.
     6→ *
     7→ * Key responsibilities:
     8→ * - Create and manage shadow records for PM schedules
     9→ * - Push MMS changes to Maximo (create/update work orders)
    10→ * - Pull Maximo status changes to MMS
    11→ * - Detect and resolve conflicts
    12→ * - Validate edit operations against Maximo status locks
    13→ *
    14→ * Integration flow:
    15→ * 1. When a PM is scheduled in MMS, a shadow record is created (status: PENDING)
    16→ * 2. Sync job pushes shadow records to Maximo (creates work orders)
    17→ * 3. Maximo status changes (via polling or webhook) update shadow records
    18→ * 4. INPRG/COMP/CLOSE/CAN status locks PM in MMS (read-only)
    19→ *
    20→ * @see specs/007-integration.md Section 7.4
    21→ */

   212→/**
   213→ * Maximo Sync Service for bidirectional PM synchronization
   214→ *
   215→ * Implements the shadow record pattern from spec 007.4.2:
   216→ * - MMS is source of truth for scheduling
   217→ * - Shadow records track sync state with Maximo
   218→ * - Status-based locking prevents conflicts
   219→ *
   220→ * @example
   221→ * ```typescript
   222→ * const maximoSync = getMaximoSyncService();
   223→ *
   224→ * // Validate before editing a PM
   225→ * const validation = await maximoSync.validateEdit(pmId);
   226→ * if (!validation.valid) {
   227→ *   throw new MaximoLockError(validation.message);
   228→ * }
   229→ *
   230→ * // Create shadow record for new PM
   231→ * await maximoSync.createShadowRecord(pmId);
   232→ *
   233→ * // Push changes to Maximo
   234→ * const result = await maximoSync.pushToMaximo(pmId);
   235→ *
   236→ * // Pull status updates from Maximo
   237→ * await maximoSync.pullFromMaximo();
   238→ * ```
   239→ */
   240→export class MaximoSyncService extends BaseService {
   241→  private readonly circuitBreaker: CircuitBreaker;
   242→  private lastPullTimestamp: Date | null = null;
   243→
   244→  constructor() {
   245→    super({ name: 'maximo-sync-service' });
   246→
   247→    // Circuit breaker for Maximo API calls
   248→    this.circuitBreaker = new CircuitBreaker(
   249→      {
   250→        failureThreshold: 5,
   251→        successThreshold: 2,
   252→        resetTimeoutMs: 60000, // 60 seconds in open state
   253→        isFailure: (error) => {
   254→          // Only count network/availability errors as circuit breaker failures
   255→          if (error instanceof MaximoConnectionError) return true;
   256→          if (error instanceof Error) {
   257→            const msg = error.message.toLowerCase();
   258→            return (
   259→              msg.includes('econnrefused') ||
   260→              msg.includes('etimedout') ||
   261→              msg.includes('econnreset') ||
   262→              msg.includes('503') ||
   263→              msg.includes('504')
   264→            );
   265→          }
   266→          return false;
   267→        },
   268→      },
   269→      'maximo-sync'
   270→    );
   271→  }

   273→  // ===========================================================================
   274→  // Edit Validation
   275→  // ===========================================================================
   276→
   277→  /**
   278→   * Validate PM edit request against Maximo status
   279→   *
   280→   * MUST be called before any PM schedule modification.
   281→   * Prevents race conditions where a planner reschedules a PM that
   282→   * technicians are actively working on.
   283→   *
   284→   * @see specs/007-integration.md Section 7.4.6
   285→   */
   286→  async validateEdit(pmScheduleId: string): Promise<EditValidationResult> {
   287→    return this.traced('validateEdit', async (span) => {
   288→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   289→
   290→      // Get PM with shadow record
   291→      const pm = await this.getPmWithShadow(pmScheduleId);
   292→
   293→      if (!pm) {
   294→        return {
   295→          valid: false,
   296→          reason: 'NOT_FOUND',
   297→          message: 'PM schedule not found',
   298→        };
   299→      }
   300→
   301→      // Check 1: Maximo status lock
   302→      if (pm.shadowRecord?.maximoData) {
   303→        const maximoData = pm.shadowRecord.maximoData as { status?: string };
   304→        const maximoStatus = maximoData.status as MaximoWoStatus | undefined;
   305→
   306→        if (maximoStatus && LOCKED_MAXIMO_STATUSES.includes(maximoStatus)) {
   307→          span.setAttribute('maximo.status', maximoStatus);
   308→          span.setAttribute('maximo.locked', true);
   309→
   310→          return {
   311→            valid: false,
   312→            reason: 'MAXIMO_STATUS_LOCK',
   313→            message: `Cannot modify: Work order is ${MAXIMO_STATUS_LABELS[maximoStatus]} in Maximo`,
   314→            maximoStatus,
   315→            maximoWoNum: pm.shadowRecord.maximoWoNum ?? undefined,
   316→            suggestion:
   317→              maximoStatus === 'INPRG'
   318→                ? 'Wait for technicians to complete work, or contact supervisor to cancel in Maximo first'
   319→                : 'This PM has been completed/cancelled. Create a new PM if rescheduling is needed.',
   320→          };
   321→        }
   322→      }
   323→
   324→      // Check 2: Pending sync conflict
   325→      if (pm.shadowRecord?.syncStatus === 'CONFLICT') {
   326→        span.setAttribute('sync.conflict', true);
   327→
   328→        return {
   329→          valid: false,
   330→          reason: 'SYNC_CONFLICT',
   331→          message: 'Unresolved sync conflict exists. Resolve before editing.',
   332→          suggestion: 'Go to Integration > Conflicts to resolve',
   333→        };
   334→      }
   335→
   336→      // Check 3: Baseline lock (if PM is locked)
   337→      if (pm.pmSchedule.isLocked) {
   338→        span.setAttribute('baseline.locked', true);
   339→
   340→        return {
   341→          valid: false,
   342→          reason: 'BASELINE_LOCK',
   343→          message: 'PM is locked in published baseline',
   344→          suggestion: 'Unlock in baseline editor or create scenario to test changes',
   345→        };
   346→      }
   347→
   348→      span.setAttribute('edit.valid', true);
   349→      return { valid: true };
   350→    });
   351→  }

   365→  /**
   366→   * Create a shadow record for a PM schedule
   367→   *
   368→   * Called when a new PM is scheduled or when sync is first enabled.
   369→   */
   370→  async createShadowRecord(pmScheduleId: string): Promise<MaximoShadowRecord> {
   371→    return this.traced('createShadowRecord', async (span) => {
   372→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   373→
   374→      // Verify PM exists
   375→      const pm = await this.db.query.pmSchedule.findFirst({
   376→        where: eq(pmSchedule.id, pmScheduleId),
   377→        with: {
   378→          train: true,
   379→          pmType: true,
   380→        },
   381→      });
   382→
   383→      if (!pm) {
   384→        throw new NotFoundError(`PM schedule not found: ${pmScheduleId}`, {
   385→          entityType: 'PmSchedule',
   386→          entityId: pmScheduleId,
   387→        });
   388→      }
   389→
   390→      // Check if shadow record already exists
   391→      const existing = await this.db.query.maximoShadowRecords.findFirst({
   392→        where: eq(maximoShadowRecords.pmScheduleId, pmScheduleId),
   393→      });
   394→
   395→      if (existing) {
   396→        this.log.info(
   397→          { pmScheduleId, shadowRecordId: existing.id },
   398→          'Shadow record already exists'
   399→        );
   400→        return existing;
   401→      }
   402→
   403→      // Create MMS data snapshot
   404→      const mmsDataSnapshot = {
   405→        trainId: pm.trainId,
   406→        pmTypeCode: pm.pmType?.pmCode ?? 'UNKNOWN',
   407→        targetDate: pm.targetDate,
   408→        status: pm.status ?? 'SCHEDULED',
   409→        assignedDepotId: pm.assignedDepotId ?? undefined,
   410→        assignedHours: pm.assignedHours ?? undefined,
   411→      };
   412→
   413→      // Insert shadow record
   414→      const [shadowRecord] = await this.db
   415→        .insert(maximoShadowRecords)
   416→        .values({
   417→          pmScheduleId,
   418→          syncStatus: 'PENDING',
   419→          mmsDataSnapshot,
   420→          syncAttempts: 0,
   421→          nextSyncAt: new Date(), // Sync immediately
   422→        })
   423→        .returning();
   424→
   425→      span.setAttribute('shadow.record.id', shadowRecord.id);
   426→      this.log.info(
   427→        { pmScheduleId, shadowRecordId: shadowRecord.id },
   428→        'Created shadow record for PM schedule'
   429→      );
   430→
   431→      return shadowRecord;
   432→    });
   433→  }

   485→  /**
   486→   * List shadow records with filters
   487→   */
   488→  async listShadowRecords(
   489→    filters: ShadowRecordFilters = {},
   490→    pagination: PaginationOptions = {}
   491→  ): Promise<PaginatedShadowRecords> {
   492→    return this.traced('listShadowRecords', async (span) => {
   493→      const { page = 1, pageSize = 50 } = pagination;
   494→      const offset = (page - 1) * pageSize;
   495→
   496→      // Build where conditions
   497→      const conditions: SQL<unknown>[] = [];
   498→
   499→      if (filters.syncStatus) {
   500→        const statuses = Array.isArray(filters.syncStatus)
   501→          ? filters.syncStatus
   502→          : [filters.syncStatus];
   503→        conditions.push(inArray(maximoShadowRecords.syncStatus, statuses));
   504→      }
   505→
   506→      if (filters.pmScheduleId) {
   507→        conditions.push(eq(maximoShadowRecords.pmScheduleId, filters.pmScheduleId));
   508→      }
   509→
   510→      if (filters.maximoWoNum) {
   511→        conditions.push(eq(maximoShadowRecords.maximoWoNum, filters.maximoWoNum));
   512→      }
   513→
   514→      if (filters.hasConflict) {
   515→        conditions.push(eq(maximoShadowRecords.syncStatus, 'CONFLICT'));
   516→      }
   517→
   518→      if (filters.needsSync) {
   519→        const needsSyncCondition = or(
   520→          eq(maximoShadowRecords.syncStatus, 'PENDING'),
   521→          and(
   522→            eq(maximoShadowRecords.syncStatus, 'ERROR'),
   523→            lt(maximoShadowRecords.syncAttempts, 5)
   524→          )
   525→        );
   526→        if (needsSyncCondition) {
   527→          conditions.push(needsSyncCondition);
   528→        }
   529→      }
   530→
   531→      const whereClause = conditions.length > 0 ? and(...conditions) : undefined;
   532→
   533→      // Get total count
   534→      const [{ value: total }] = await this.db
   535→        .select({ value: count() })
   536→        .from(maximoShadowRecords)
   537→        .where(whereClause);
   538→
   539→      // Get records
   540→      const records = await this.db.query.maximoShadowRecords.findMany({
   541→        where: whereClause,
   542→        orderBy: [desc(maximoShadowRecords.updatedAt)],
   543→        limit: pageSize,
   544→        offset,
   545→        with: {
   546→          pmScheduleEntry: {
   547→            with: {
   548→              train: true,
   549→              pmType: true,
   550→              assignedDepot: true,
   551→            },
   552→          },
   553→        },
   554→      });
   555→
   556→      // Transform to ShadowRecordWithPm
   557→      const data: ShadowRecordWithPm[] = records.map((r) => ({
   558→        shadowRecord: r,
   559→        pmSchedule: {
   560→          ...r.pmScheduleEntry,
   561→          train: r.pmScheduleEntry.train
   562→            ? {
   563→                id: r.pmScheduleEntry.train.id,
   564→                trainNumber: r.pmScheduleEntry.train.trainNumber,
   565→              }
   566→            : undefined,
   567→          pmType: r.pmScheduleEntry.pmType
   568→            ? {
   569→                id: r.pmScheduleEntry.pmType.id,
   570→                pmCode: r.pmScheduleEntry.pmType.pmCode,
   571→                pmName: r.pmScheduleEntry.pmType.pmName,
   572→              }
   573→            : undefined,
   574→          depot: r.pmScheduleEntry.assignedDepot
   575→            ? {
   576→                id: r.pmScheduleEntry.assignedDepot.id,
   577→                code: r.pmScheduleEntry.assignedDepot.code,
   578→              }
   579→            : undefined,
   580→        },
   581→      }));
   582→
   583→      span.setAttribute('query.total', total);
   584→      span.setAttribute('query.returned', data.length);
   585→
   586→      return {
   587→        data,
   588→        total,
   589→        page,
   590→        pageSize,
   591→        totalPages: Math.ceil(total / pageSize),
   592→      };
   593→    });
   594→  }

   596→  // ===========================================================================
   597→  // Push Sync (MMS → Maximo)
   598→  // ===========================================================================
   599→
   600→  /**
   601→   * Push MMS changes to Maximo for a specific PM
   602→   *
   603→   * Creates or updates the Maximo work order based on shadow record state.
   604→   *
   605→   * @see specs/007-integration.md Section 7.4.7
   606→   */
   607→  async pushToMaximo(pmScheduleId: string): Promise<PushResult> {
   608→    return this.traced('pushToMaximo', async (span) => {
   609→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   610→
   611→      const pmWithShadow = await this.getPmWithShadow(pmScheduleId);
   612→
   613→      if (!pmWithShadow) {
   614→        return { action: 'ERROR', error: 'PM schedule not found' };
   615→      }
   616→
   617→      const { pmSchedule: pm, shadowRecord } = pmWithShadow;
   618→
   619→      // Create shadow record if doesn't exist
   620→      let shadow = shadowRecord;
   621→      if (!shadow?.id) {
   622→        shadow = await this.createShadowRecord(pmScheduleId);
   623→      }
   624→
   625→      // Check circuit breaker state
   626→      if (this.circuitBreaker.getState() === 'OPEN') {
   627→        this.log.warn({ pmScheduleId }, 'Circuit breaker is open, skipping push');
   628→        return { action: 'SKIPPED', reason: 'Circuit breaker open' };
   629→      }
   630→
   631→      const syncAttempts = shadow.syncAttempts ?? 0;
   632→
   633→      try {
   634→        // No Maximo WO yet - create new
   635→        if (!shadow.maximoWoNum) {
   636→          // Use circuit breaker for Maximo API call
   637→          const woNum = await this.circuitBreaker.execute(() =>
   638→            this.createMaximoWorkOrder(pm)
   639→          );
   640→
   641→          await this.updateShadowRecord(shadow.id, {
   642→            maximoWoNum: woNum,
   643→            syncStatus: 'SYNCED',
   644→            lastSyncAt: new Date(),
   645→            lastSyncError: null,
   646→            syncAttempts: syncAttempts + 1,
   647→            mmsDataSnapshot: this.createMmsSnapshot(pm),
   648→          });
   649→
   650→          await this.recordSyncHistory(shadow.id, 'PUSH', {
   651→            previousStatus: shadow.syncStatus,
   652→            newStatus: 'SYNCED',
   653→            success: true,
   654→          });
   655→
   656→          span.setAttribute('maximo.wo.created', woNum);
   657→
   658→          return { action: 'CREATED', woNum };
   659→        }
   660→
   661→        // Check if Maximo WO is in locked state
   662→        const maximoWo = await this.circuitBreaker.execute(() =>
   663→          this.fetchMaximoWorkOrder(shadow.maximoWoNum!)
   664→        );
   665→
   666→        if (maximoWo && LOCKED_MAXIMO_STATUSES.includes(maximoWo.status)) {
   667→          return {
   668→            action: 'SKIPPED',
   669→            reason: `Work order is ${MAXIMO_STATUS_LABELS[maximoWo.status]}`,
   670→            woNum: shadow.maximoWoNum,
   671→          };
   672→        }
   673→
   674→        // Check for conflicts
   675→        if (maximoWo?.changedate && shadow.lastSyncAt) {
   676→          const maximoChangeDate = new Date(maximoWo.changedate);
   677→          if (maximoChangeDate > shadow.lastSyncAt) {
   678→            await this.updateShadowRecord(shadow.id, {
   679→              syncStatus: 'CONFLICT',
   680→              maximoData: {
   681→                status: maximoWo.status,
   682→                schedStart: maximoWo.schedstart,
   683→                schedFinish: maximoWo.schedfinish,
   684→                actualStart: maximoWo.actstart,
   685→                actualFinish: maximoWo.actfinish,
   686→                actualLabor: maximoWo.actlabhrs,
   687→                changeDate: maximoWo.changedate,
   688→              },
   689→            });
   690→
   691→            await this.recordSyncHistory(shadow.id, 'PUSH', {
   692→              previousStatus: shadow.syncStatus,
   693→              newStatus: 'CONFLICT',
   694→              success: false,
   695→              errorMessage: 'Conflict detected - Maximo changed after last sync',
   696→            });
   697→
   698→            return { action: 'CONFLICT', woNum: shadow.maximoWoNum };
   699→          }
   700→        }
   701→
   702→        // Update existing work order
   703→        await this.circuitBreaker.execute(() =>
   704→          this.updateMaximoWorkOrder(shadow.maximoWoNum!, pm)
   705→        );
   706→
   707→        await this.updateShadowRecord(shadow.id, {
   708→          syncStatus: 'SYNCED',
   709→          lastSyncAt: new Date(),
   710→          lastSyncError: null,
   711→          syncAttempts: syncAttempts + 1,
   712→          mmsDataSnapshot: this.createMmsSnapshot(pm),
   713→          pendingChanges: null,
   714→        });
   715→
   716→        await this.recordSyncHistory(shadow.id, 'PUSH', {
   717→          previousStatus: shadow.syncStatus,
   718→          newStatus: 'SYNCED',
   719→          success: true,
   720→        });
   721→
   722→        span.setAttribute('maximo.wo.updated', shadow.maximoWoNum);
   723→
   724→        return { action: 'UPDATED', woNum: shadow.maximoWoNum };
   725→      } catch (error) {
   726→        const errorMessage = error instanceof Error ? error.message : String(error);
   727→
   728→        await this.updateShadowRecord(shadow.id, {
   729→          syncStatus: 'ERROR',
   730→          lastSyncError: errorMessage,
   731→          syncAttempts: syncAttempts + 1,
   732→          nextSyncAt: this.calculateNextRetry(syncAttempts + 1),
   733→        });
   734→
   735→        await this.recordSyncHistory(shadow.id, 'PUSH', {
   736→          previousStatus: shadow.syncStatus,
   737→          newStatus: 'ERROR',
   738→          success: false,
   739→          errorMessage,
   740→        });
   741→
   742→        span.setAttribute('error', true);
   743→        span.setAttribute('error.message', errorMessage);
   744→
   745→        return { action: 'ERROR', error: errorMessage };
   746→      }
   747→    });
   748→  }

   804→  /**
   805→   * Pull Maximo status changes to MMS
   806→   *
   807→   * Queries Maximo for work orders modified since last pull and updates
   808→   * MMS PM statuses accordingly.
   809→   *
   810→   * @see specs/007-integration.md Section 7.4.7
   811→   */
   812→  async pullFromMaximo(): Promise<PullResult> {
   813→    return this.traced('pullFromMaximo', async (span) => {
   814→      const result: PullResult = {
   815→        processed: 0,
   816→        updated: 0,
   817→        conflicts: 0,
   818→        errors: 0,
   819→        details: [],
   820→      };
   821→
   822→      // Check circuit breaker state
   823→      if (this.circuitBreaker.getState() === 'OPEN') {
   824→        this.log.warn('Circuit breaker is open, skipping pull');
   825→        return result;
   826→      }
   827→
   828→      // Get all synced shadow records
   829→      const syncedRecords = await this.db.query.maximoShadowRecords.findMany({
   830→        where: and(
   831→          eq(maximoShadowRecords.syncStatus, 'SYNCED'),
   832→          sql`${maximoShadowRecords.maximoWoNum} IS NOT NULL`
   833→        ),
   834→      });
   835→
   836→      span.setAttribute('synced.records', syncedRecords.length);
   837→
   838→      for (const record of syncedRecords) {
   839→        result.processed++;
   840→
   841→        try {
   842→          if (!record.maximoWoNum) continue;
   843→
   844→          // Use circuit breaker for Maximo API call
   845→          const maximoWo = await this.circuitBreaker.execute(() =>
   846→            this.fetchMaximoWorkOrder(record.maximoWoNum!)
   847→          );
   848→
   849→          if (!maximoWo) {
   850→            result.details.push({
   851→              pmScheduleId: record.pmScheduleId,
   852→              maximoWoNum: record.maximoWoNum,
   853→              action: 'SKIPPED',
   854→              error: 'Work order not found in Maximo',
   855→            });
   856→            continue;
   857→          }
   858→
   859→          const currentMaximoData = record.maximoData as { status?: string } | null;
   860→          const previousStatus = currentMaximoData?.status;
   861→
   862→          // Status changed
   863→          if (maximoWo.status !== previousStatus) {
   864→            await this.handleStatusChange(record, maximoWo);
   865→            result.updated++;
   866→            result.details.push({
   867→              pmScheduleId: record.pmScheduleId,
   868→              maximoWoNum: record.maximoWoNum,
   869→              previousStatus,
   870→              newStatus: maximoWo.status,
   871→              action: 'UPDATED',
   872→            });
   873→          }
   874→
   875→          // Record actuals if completed
   876→          if (
   877→            (maximoWo.status === 'COMP' || maximoWo.status === 'CLOSE') &&
   878→            maximoWo.actlabhrs
   879→          ) {
   880→            await this.recordActuals(record.pmScheduleId, maximoWo);
   881→          }
   882→        } catch (error) {
   883→          result.errors++;
   884→          result.details.push({
   885→            pmScheduleId: record.pmScheduleId,
   886→            maximoWoNum: record.maximoWoNum ?? '',
   887→            action: 'ERROR',
   888→            error: error instanceof Error ? error.message : String(error),
   889→          });
   890→        }
   891→      }
   892→
   893→      this.lastPullTimestamp = new Date();
   894→      span.setAttribute('pull.updated', result.updated);
   895→      span.setAttribute('pull.errors', result.errors);
   896→
   897→      return result;
   898→    });
   899→  }

   905→  /**
   906→   * Resolve sync conflict
   907→   *
   908→   * @param pmScheduleId - PM schedule ID with conflict
   909→   * @param resolution - Strategy: USE_MMS (force push) or USE_MAXIMO (apply Maximo data)
   910→   * @param userId - User resolving the conflict
   911→   */
   912→  async resolveConflict(
   913→    pmScheduleId: string,
   914→    resolution: ConflictResolution,
   915→    userId?: string
   916→  ): Promise<void> {
   917→    return this.traced('resolveConflict', async (span) => {
   918→      span.setAttribute(MmsAttributes.PM_SCHEDULE_ID, pmScheduleId);
   919→      span.setAttribute('resolution', resolution);
   920→
   921→      const shadow = await this.db.query.maximoShadowRecords.findFirst({
   922→        where: eq(maximoShadowRecords.pmScheduleId, pmScheduleId),
   923→      });
   924→
   925→      if (!shadow) {
   926→        throw new NotFoundError(`Shadow record not found for PM: ${pmScheduleId}`, {
   927→          entityType: 'MaximoShadowRecord',
   928→          entityId: pmScheduleId,
   929→        });
   930→      }
   931→
   932→      if (shadow.syncStatus !== 'CONFLICT') {
   933→        throw new ValidationError('Shadow record is not in conflict state');
   934→      }
   935→
   936→      if (!shadow.maximoWoNum) {
   937→        throw new ValidationError('Shadow record has no Maximo work order');
   938→      }
   939→
   940→      if (resolution === 'USE_MMS') {
   941→        // Force push MMS data to Maximo
   942→        const pm = await this.db.query.pmSchedule.findFirst({
   943→          where: eq(pmSchedule.id, pmScheduleId),
   944→          with: { train: true, pmType: true, assignedDepot: true },
   945→        });
   946→
   947→        if (!pm) {
   948→          throw new NotFoundError(`PM schedule not found: ${pmScheduleId}`, {
   949→            entityType: 'PmSchedule',
   950→            entityId: pmScheduleId,
   951→          });
   952→        }
   953→
   954→        await this.updateMaximoWorkOrder(shadow.maximoWoNum, pm, { force: true });
   955→      } else {
   956→        // Apply Maximo data to MMS
   957→        const maximoWo = await this.fetchMaximoWorkOrder(shadow.maximoWoNum);
   958→
   959→        if (!maximoWo) {
   960→          throw new MaximoConnectionError('Failed to fetch Maximo work order');
   961→        }
   962→
   963→        await this.applyMaximoDataToMms(pmScheduleId, maximoWo);
   964→      }
   965→
   966→      // Update shadow record
   967→      await this.updateShadowRecord(shadow.id, {
   968→        syncStatus: 'SYNCED',
   969→        lastSyncAt: new Date(),
   970→        conflictResolvedAt: new Date(),
   971→        conflictResolvedBy: userId ?? undefined,
   972→        conflictResolution: resolution,
   973→      });
   974→
   975→      await this.recordSyncHistory(shadow.id, 'CONFLICT_RESOLVE', {
   976→        previousStatus: 'CONFLICT',
   977→        newStatus: 'SYNCED',
   978→        success: true,
   979→      });
   980→
   981→      this.log.info(
   982→        { pmScheduleId, resolution, userId },
   983→        'Conflict resolved'
   984→      );
   985→    });
   986→  }

  1248→  private calculateNextRetry(attempts: number): Date {
  1249→    // Exponential backoff with max delay
  1250→    const initialDelayMs = 1000;
  1251→    const backoffMultiplier = 2;
  1252→    const maxDelayMs = 60000;
  1253→
  1254→    const delay = Math.min(
  1255→      initialDelayMs * Math.pow(backoffMultiplier, attempts - 1),
  1256→      maxDelayMs
  1257→    );
  1258→    return addSeconds(new Date(), delay / 1000);
  1259→  }
