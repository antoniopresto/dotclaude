     1→/**
     2→ * MMS Route Utilization Service
     3→ *
     4→ * Provides route-specific daily mileage rates for accurate long-term projections.
     5→ * Critical for PM scheduling as fleet-wide averages can result in 60,000+ km
     6→ * projection error over the 12-year contract period.
     7→ *
     8→ * Key responsibilities:
     9→ * - Retrieve route-specific utilization profiles
    10→ * - Calculate expected daily km for trains with fallback hierarchy
    11→ * - Project cumulative mileage over time horizons
    12→ * - Apply seasonal multipliers for peak/off-peak periods
    13→ *
    14→ * Fallback hierarchy (in priority order):
    15→ * 1. Route-specific profile from route_utilization_profiles table
    16→ * 2. Fleet-wide average from last 30 days of mileage readings
    17→ * 3. System default: 250 km/day (conservative estimate)
    18→ *
    19→ * @see specs/004-domain-services.md Section 4.2.7 (Route Utilization Service)
    20→ * @see specs/003-database-schema.md Section 3.2.2 (Route Utilization Profiles)
    21→ */

   222→  /**
   223→   * Get the utilization profile for a route at a specific date.
   224→   *
   225→   * Returns the most recent effective profile for the route. If no profile
   226→   * exists, returns null (callers should fall back to fleet average or default).
   227→   *
   228→   * @param routeId - Route UUID
   229→   * @param asOfDate - Date to find effective profile for (defaults to today)
   230→   * @returns Profile if found, null otherwise
   231→   */
   232→  async getProfile(
   233→    routeId: string,
   234→    asOfDate?: string
   235→  ): Promise<RouteProfileWithMeta | null> {
   236→    return this.traced('getProfile', async (span) => {
   237→      span.setAttribute('route.id', routeId);
   238→      const effectiveDate = asOfDate ?? new Date().toISOString().split('T')[0];
   239→      span.setAttribute('profile.effectiveDate', effectiveDate);
   240→
   241→      // Find most recent effective profile
   242→      const profile = await this.db.query.routeUtilizationProfiles.findFirst({
   243→        where: and(
   244→          eq(routeUtilizationProfiles.routeId, routeId),
   245→          lte(routeUtilizationProfiles.effectiveDate, effectiveDate)
   246→        ),
   247→        orderBy: desc(routeUtilizationProfiles.effectiveDate),
   248→        with: { route: true },
   249→      });
   250→
   251→      if (!profile) {
   252→        span.setAttribute('profile.found', false);
   253→        return null;
   254→      }
   255→
   256→      span.setAttribute('profile.found', true);
   257→      span.setAttribute('profile.id', profile.id);
   258→      span.setAttribute('profile.confidence', profile.confidence ?? 'MEDIUM');
   259→
   260→      return {
   261→        ...profile,
   262→        routeCode: profile.route?.routeCode,
   263→        routeName: profile.route?.routeName,
   264→      };
   265→    });
   266→  }

   328→  /**
   329→   * Calculate expected daily km for a train based on available data.
   330→   *
   331→   * This is the primary method for mileage projection. It follows the
   332→   * fallback hierarchy:
   333→   * 1. Route-specific profile (if train has assigned route or default route)
   334→   * 2. Fleet-wide average from recent mileage readings
   335→   * 3. System default (250 km/day)
   336→   *
   337→   * The result includes source and confidence metadata for transparency.
   338→   *
   339→   * @param trainId - Train UUID
   340→   * @param targetDate - Date to calculate rate for
   341→   * @param routeId - Optional specific route ID (if known)
   342→   * @param holidays - Optional array of holiday dates (YYYY-MM-DD)
   343→   * @returns Daily rate with source and confidence
   344→   */
   345→  async calculateDailyRate(
   346→    trainId: string,
   347→    targetDate: string,
   348→    routeId?: string,
   349→    holidays?: string[]
   350→  ): Promise<DailyRateResult> {
   351→    return this.traced('calculateDailyRate', async (span) => {
   352→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   353→      span.setAttribute('rate.targetDate', targetDate);
   354→
   355→      // Get train details
   356→      const train = await this.db.query.trains.findFirst({
   357→        where: eq(trains.id, trainId),
   358→        with: { fleet: true },
   359→      });
   360→
   361→      if (!train) {
   362→        throw new NotFoundError(`Train not found: ${trainId}`, {
   363→          entityType: 'train',
   364→          entityId: trainId,
   365→        });
   366→      }
   367→
   368→      span.setAttribute(MmsAttributes.FLEET_ID, train.fleetId);
   369→
   370→      // Determine day type
   371→      const dayType = this.getDayType(targetDate, holidays);
   372→      span.setAttribute('rate.dayType', dayType);
   373→
   374→      // Try route-specific profile first
   375→      if (routeId) {
   376→        const profile = await this.getProfile(routeId, targetDate);
   377→        if (profile) {
   378→          const result = this.calculateFromProfile(profile, targetDate, dayType);
   379→          span.setAttribute('rate.source', result.source);
   380→          span.setAttribute('rate.dailyKm', result.dailyKm);
   381→          span.setAttribute('rate.confidence', result.confidence);
   382→          return result;
   383→        }
   384→      }
   385→
   386→      // Try typical route assignment from recent history
   387→      const typicalRoute = await this.getTypicalRouteAssignment(trainId);
   388→      if (typicalRoute) {
   389→        const profile = await this.getProfile(typicalRoute.routeId, targetDate);
   390→        if (profile) {
   391→          const result = this.calculateFromProfile(profile, targetDate, dayType);
   392→          // Slightly lower confidence since it's inferred from history
   393→          result.confidence = result.confidence === 'HIGH' ? 'MEDIUM' : result.confidence;
   394→          result.details = {
   395→            ...result.details,
   396→            routeCode: typicalRoute.routeCode,
   397→          };
   398→          span.setAttribute('rate.source', result.source);
   399→          span.setAttribute('rate.sourceMethod', 'typical_route_assignment');
   400→          span.setAttribute('rate.dailyKm', result.dailyKm);
   401→          span.setAttribute('rate.confidence', result.confidence);
   402→
   403→          this.log.debug(
   404→            {
   405→              trainId,
   406→              trainNumber: train.trainNumber,
   407→              routeId: typicalRoute.routeId,
   408→              routeCode: typicalRoute.routeCode,
   409→              dailyKm: result.dailyKm,
   410→              assignmentCount: typicalRoute.assignmentCount,
   411→            },
   412→            `Using typical route ${typicalRoute.routeCode} for train ${train.trainNumber}: ${result.dailyKm} km/day`
   413→          );
   414→
   415→          return result;
   416→        }
   417→      }
   418→
   419→      // Fall back to fleet average
   420→      const fleetAverage = await this.getFleetAverage(train.fleetId, targetDate);
   421→      if (fleetAverage && fleetAverage.sampleSize >= MIN_SAMPLE_SIZE_FOR_MEDIUM_CONFIDENCE) {
   422→        const result: DailyRateResult = {
   423→          dailyKm: Math.round(fleetAverage.averageDailyKm),
   424→          source: 'fleet_average',
   425→          confidence: fleetAverage.confidence,
   426→          details: {
   427→            fleetId: train.fleetId,
   428→            fleetCode: train.fleet?.code,
   429→            dayType,
   430→          },
   431→        };
   432→
   433→        span.setAttribute('rate.source', result.source);
   434→        span.setAttribute('rate.dailyKm', result.dailyKm);
   435→        span.setAttribute('rate.confidence', result.confidence);
   436→
   437→        this.log.debug(
   438→          {
   439→            trainId,
   440→            fleetId: train.fleetId,
   441→            dailyKm: result.dailyKm,
   442→            sampleSize: fleetAverage.sampleSize,
   443→          },
   444→          `Using fleet average for train ${train.trainNumber}: ${result.dailyKm} km/day`
   445→        );
   446→
   447→        return result;
   448→      }
   449→
   450→      // Fall back to system default
   451→      const result: DailyRateResult = {
   452→        dailyKm: SYSTEM_DEFAULT_DAILY_KM,
   453→        source: 'system_default',
   454→        confidence: 'ESTIMATED',
   455→        details: {
   456→          fleetId: train.fleetId,
   457→          fleetCode: train.fleet?.code,
   458→          dayType,
   459→        },
   460→      };
   461→
   462→      span.setAttribute('rate.source', result.source);
   463→      span.setAttribute('rate.dailyKm', result.dailyKm);
   464→      span.setAttribute('rate.confidence', result.confidence);
   465→
   466→      this.log.warn(
   467→        {
   468→          trainId,
   469→          trainNumber: train.trainNumber,
   470→          fleetId: train.fleetId,
   471→          targetDate,
   472→        },
   473→        `No route profile or fleet average for train ${train.trainNumber}. Using system default ${SYSTEM_DEFAULT_DAILY_KM} km/day. This may cause significant projection errors over time.`
   474→      );
   475→
   476→      return result;
   477→    });
   478→  }

   491→      span.setAttribute('route.id', routeId);
   492→      span.setAttribute('rate.targetDate', targetDate);
   493→
   494→      // Verify route exists
   495→      const route = await this.db.query.routes.findFirst({
   496→        where: eq(routes.id, routeId),
   497→      });
   498→
   499→      if (!route) {
   500→        throw new NotFoundError(`Route not found: ${routeId}`, {
   501→          entityType: 'route',
   502→          entityId: routeId,
   503→        });
   504→      }
   505→
   506→      const dayType = this.getDayType(targetDate, holidays);
   507→      span.setAttribute('rate.dayType', dayType);
   508→
   509→      // Try route-specific profile
   510→      const profile = await this.getProfile(routeId, targetDate);
   511→      if (profile) {
   512→        const result = this.calculateFromProfile(profile, targetDate, dayType);
   513→        span.setAttribute('rate.source', result.source);
   514→        span.setAttribute('rate.dailyKm', result.dailyKm);
   515→        return result;
   516→      }
   517→
   518→      // Fall back to route's default daily km if defined
   519→      if (route.defaultDailyKm) {
   520→        const result: DailyRateResult = {
   521→          dailyKm: route.defaultDailyKm,
   522→          source: `route:${routeId}`,
   523→          confidence: 'LOW',
   524→          details: {
   525→            routeId,
   526→            routeCode: route.routeCode,
   527→            dayType,
   528→          },
   529→        };
   530→        span.setAttribute('rate.source', result.source);
   531→        span.setAttribute('rate.dailyKm', result.dailyKm);
   532→        return result;
   533→      }
   534→
   535→      // Fall back to expected km per day
   536→      if (route.expectedKmPerDay) {
   537→        const multiplier = DEFAULT_DAY_TYPE_MULTIPLIERS[dayType];
   538→        const result: DailyRateResult = {
   539→          dailyKm: Math.round(route.expectedKmPerDay * multiplier),
   540→          source: `route:${routeId}`,
   541→          confidence: 'LOW',
   542→          details: {
   543→            routeId,
   544→            routeCode: route.routeCode,
   545→            dayType,
   546→          },
   547→        };
   548→        span.setAttribute('rate.source', result.source);
   549→        span.setAttribute('rate.dailyKm', result.dailyKm);
   550→        return result;
   551→      }
   552→
   553→      // System default
   554→      const result: DailyRateResult = {
   555→        dailyKm: SYSTEM_DEFAULT_DAILY_KM,
   556→        source: 'system_default',
   557→        confidence: 'ESTIMATED',
   558→        details: {
   559→          routeId,
   560→          routeCode: route.routeCode,
   561→          dayType,
   562→        },
   563→      };
   564→      span.setAttribute('rate.source', result.source);
   565→      span.setAttribute('rate.dailyKm', result.dailyKm);
   566→      return result;
   567→    });

   574→  /**
   575→   * Project mileage for a train over a time horizon.
   576→   *
   577→   * Starting from the train's current mileage, calculates daily projections
   578→   * for the specified number of days. This is used by:
   579→   * - PM scheduling (determine when train will hit maintenance thresholds)
   580→   * - Drift monitoring (compare projected vs. actual)
   581→   * - Strategic planning (long-term fleet mileage distribution)
   582→   *
   583→   * @param trainId - Train UUID
   584→   * @param startDate - Projection start date (YYYY-MM-DD)
   585→   * @param horizonDays - Number of days to project
   586→   * @param routeId - Optional fixed route for all projections
   587→   * @param holidays - Optional array of holiday dates
   588→   * @returns Array of daily projections with cumulative mileage
   589→   */
   590→  async projectMileage(
   591→    trainId: string,
   592→    startDate: string,
   593→    horizonDays: number,
   594→    routeId?: string,
   595→    holidays?: string[]
   596→  ): Promise<MileageProjection[]> {
   597→    return this.traced('projectMileage', async (span) => {
   598→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   599→      span.setAttribute('projection.startDate', startDate);
   600→      span.setAttribute('projection.horizonDays', horizonDays);
   601→
   602→      if (horizonDays < 1) {
   603→        throw new ValidationError('Horizon days must be at least 1', {
   604→          field: 'horizonDays',
   605→          context: { value: horizonDays },
   606→        });
   607→      }
   608→
   609→      if (horizonDays > 3650) {
   610→        throw new ValidationError('Horizon days cannot exceed 3650 (10 years)', {
   611→          field: 'horizonDays',
   612→          context: { value: horizonDays },
   613→        });
   614→      }
   615→
   616→      // Get train's current mileage
   617→      const train = await this.db.query.trains.findFirst({
   618→        where: eq(trains.id, trainId),
   619→      });
   620→
   621→      if (!train) {
   622→        throw new NotFoundError(`Train not found: ${trainId}`, {
   623→          entityType: 'train',
   624→          entityId: trainId,
   625→        });
   626→      }
   627→
   628→      let cumulativeKm = train.currentMileageKm;
   629→      const projections: MileageProjection[] = [];
   630→      const start = new Date(startDate);
   631→
   632→      // Calculate daily projections
   633→      for (let day = 0; day < horizonDays; day++) {
   634→        const targetDate = addDays(start, day);
   635→        const dateStr = targetDate.toISOString().split('T')[0];
   636→
   637→        const rate = await this.calculateDailyRate(trainId, dateStr, routeId, holidays);
   638→        cumulativeKm += rate.dailyKm;
   639→
   640→        projections.push({
   641→          date: dateStr,
   642→          projectedDailyKm: rate.dailyKm,
   643→          cumulativeKm,
   644→          source: rate.source,
   645→          confidence: rate.confidence,
   646→        });
   647→      }
   648→
   649→      span.setAttribute('projection.count', projections.length);
   650→      span.setAttribute(
   651→        'projection.finalCumulativeKm',
   652→        projections[projections.length - 1]?.cumulativeKm ?? cumulativeKm
   653→      );
   654→
   655→      return projections;
   656→    });
   657→  }

   659→  /**
   660→   * Project when a train will reach a target mileage.
   661→   *
   662→   * Useful for PM scheduling - determines the date when a train will hit
   663→   * the next maintenance threshold.
   664→   *
   665→   * @param trainId - Train UUID
   666→   * @param targetMileageKm - Target cumulative mileage to reach
   667→   * @param maxHorizonDays - Maximum days to search (default: 365)
   668→   * @returns Projected date and details, or null if not reached within horizon
   669→   */
   670→  async projectDateForMileage(
   671→    trainId: string,
   672→    targetMileageKm: number,
   673→    maxHorizonDays: number = 365,
   674→    routeId?: string,
   675→    holidays?: string[]
   676→  ): Promise<{
   677→    projectedDate: string;
   678→    daysFromNow: number;
   679→    projectedMileage: number;
   680→    confidence: ConfidenceLevel;
   681→  } | null> {
   682→    return this.traced('projectDateForMileage', async (span) => {
   683→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   684→      span.setAttribute('target.mileageKm', targetMileageKm);
   685→      span.setAttribute('target.maxHorizonDays', maxHorizonDays);
   686→
   687→      // Get train's current mileage
   688→      const train = await this.db.query.trains.findFirst({
   689→        where: eq(trains.id, trainId),
   690→      });
   691→
   692→      if (!train) {
   693→        throw new NotFoundError(`Train not found: ${trainId}`, {
   694→          entityType: 'train',
   695→          entityId: trainId,
   696→        });
   697→      }
   698→
   699→      if (train.currentMileageKm >= targetMileageKm) {
   700→        span.setAttribute('result.alreadyReached', true);
   701→        return {
   702→          projectedDate: new Date().toISOString().split('T')[0],
   703→          daysFromNow: 0,
   704→          projectedMileage: train.currentMileageKm,
   705→          confidence: 'HIGH',
   706→        };
   707→      }
   708→
   709→      const startDate = new Date().toISOString().split('T')[0];
   710→      let cumulativeKm = train.currentMileageKm;
   711→      const start = new Date(startDate);
   712→      let worstConfidence: ConfidenceLevel = 'HIGH';
   713→
   714→      for (let day = 1; day <= maxHorizonDays; day++) {
   715→        const targetDate = addDays(start, day);
   716→        const dateStr = targetDate.toISOString().split('T')[0];
   717→
   718→        const rate = await this.calculateDailyRate(trainId, dateStr, routeId, holidays);
   719→        cumulativeKm += rate.dailyKm;
   720→
   721→        // Track worst confidence level
   722→        if (this.confidenceRank(rate.confidence) < this.confidenceRank(worstConfidence)) {
   723→          worstConfidence = rate.confidence;
   724→        }
   725→
   726→        if (cumulativeKm >= targetMileageKm) {
   727→          span.setAttribute('result.found', true);
   728→          span.setAttribute('result.projectedDate', dateStr);
   729→          span.setAttribute('result.daysFromNow', day);
   730→          return {
   731→            projectedDate: dateStr,
   732→            daysFromNow: day,
   733→            projectedMileage: cumulativeKm,
   734→            confidence: worstConfidence,
   735→          };
   736→        }
   737→      }
   738→
   739→      span.setAttribute('result.found', false);
   740→      span.setAttribute('result.reachedMileage', cumulativeKm);
   741→      return null;
   742→    });
   743→  }

   749→  /**
   750→   * Calculate fleet-wide average daily km from recent mileage readings.
   751→   *
   752→   * Uses the last 30 days of validated mileage readings to compute
   753→   * a reliable average. Returns confidence based on sample size.
   754→   *
   755→   * @param fleetId - Fleet UUID
   756→   * @param asOfDate - Calculate as of this date (defaults to today)
   757→   * @returns Fleet average with metadata, or null if no data
   758→   */
   759→  async getFleetAverage(
   760→    fleetId: string,
   761→    asOfDate?: string
   762→  ): Promise<FleetAverageResult | null> {
   763→    return this.traced('getFleetAverage', async (span) => {
   764→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   765→
   766→      const endDate = asOfDate ?? new Date().toISOString().split('T')[0];
   767→      const startDate = addDays(new Date(endDate), -FLEET_AVERAGE_LOOKBACK_DAYS)
   768→        .toISOString()
   769→        .split('T')[0];
   770→
   771→      span.setAttribute('average.startDate', startDate);
   772→      span.setAttribute('average.endDate', endDate);
   773→
   774→      // Get fleet for validation and metadata
   775→      const fleet = await this.db.query.fleets.findFirst({
   776→        where: eq(fleets.id, fleetId),
   777→      });
   778→
   779→      if (!fleet) {
   780→        throw new NotFoundError(`Fleet not found: ${fleetId}`, {
   781→          entityType: 'fleet',
   782→          entityId: fleetId,
   783→        });
   784→      }
   785→
   786→      // Calculate average from valid mileage readings
   787→      const result = await this.db
   788→        .select({
   789→          avgDaily: avg(mileageReadings.dailyKm),
   790→          sampleSize: sql<number>`COUNT(*)::int`,
   791→        })
   792→        .from(mileageReadings)
   793→        .innerJoin(trains, eq(mileageReadings.trainId, trains.id))
   794→        .where(
   795→          and(
   796→            eq(trains.fleetId, fleetId),
   797→            eq(trains.active, true),
   798→            gte(mileageReadings.readingDate, startDate),
   799→            lte(mileageReadings.readingDate, endDate),
   800→            eq(mileageReadings.validationStatus, 'VALID'),
   801→            sql`${mileageReadings.dailyKm} > 0` // Exclude zero or null readings
   802→          )
   803→        );
   804→
   805→      const { avgDaily, sampleSize } = result[0];
   806→
   807→      if (!avgDaily || sampleSize === 0) {
   808→        span.setAttribute('average.found', false);
   809→        this.log.debug(
   810→          { fleetId, fleetCode: fleet.code, startDate, endDate },
   811→          `No valid mileage readings found for fleet ${fleet.code}`
   812→        );
   813→        return null;
   814→      }
   815→
   816→      // Determine confidence based on sample size
   817→      let confidence: ConfidenceLevel;
   818→      if (sampleSize >= MIN_SAMPLE_SIZE_FOR_HIGH_CONFIDENCE) {
   819→        confidence = 'MEDIUM'; // Fleet average is never "HIGH" - route-specific is preferred
   820→      } else if (sampleSize >= MIN_SAMPLE_SIZE_FOR_MEDIUM_CONFIDENCE) {
   821→        confidence = 'LOW';
   822→      } else {
   823→        confidence = 'ESTIMATED';
   824→      }
   825→
   826→      span.setAttribute('average.found', true);
   827→      span.setAttribute('average.value', Number(avgDaily));
   828→      span.setAttribute('average.sampleSize', sampleSize);
   829→      span.setAttribute('average.confidence', confidence);
   830→
   831→      return {
   832→        fleetId,
   833→        fleetCode: fleet.code,
   834→        averageDailyKm: Math.round(Number(avgDaily)),
   835→        sampleSize,
   836→        periodStart: startDate,
   837→        periodEnd: endDate,
   838→        confidence,
   839→      };
   840→    });
   841→  }

   847→  /**

   849→   *
   850→   * Queries the run_assignments table to find the most frequently assigned
   851→   * route for the train in the last 30 days. This enables more accurate
   852→   * mileage projections when the train doesn't have a fixed route assignment.
   853→   *
   854→   * @param trainId - Train UUID
   855→   * @returns Most common route info if found, null otherwise
   856→   */
   857→  private async getTypicalRouteAssignment(
   858→    trainId: string
   859→  ): Promise<{ routeId: string; routeCode: string; assignmentCount: number } | null> {
   860→    return this.traced('getTypicalRouteAssignment', async (span) => {
   861→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   862→

   864→      span.setAttribute('lookback.startDate', thirtyDaysAgo);

   879→          )

   881→        .groupBy(runAssignments.routeId, routes.routeCode)

   899→          trainId,
   900→          routeId,
   901→          routeCode,
   902→          assignmentCount,
   903→        },
   904→        `Found typical route ${routeCode} for train with ${assignmentCount} assignments in last 30 days`
   905→      );
   906→
   907→      return { routeId, routeCode: routeCode ?? 'UNKNOWN', assignmentCount };
   908→    });

  1169→  /**
  1170→   * Analyze projection accuracy across the fleet.
  1171→   *
  1172→   * This method identifies trains that are using fallback calculations
  1173→   * (fleet average or system default) instead of route-specific profiles.
  1174→   * Fleet-wide averages can result in 60,000+ km projection error over
  1175→   * the 12-year contract period.
  1176→   *
  1177→   * @param fleetIds - Optional list of fleet IDs to analyze (all fleets if not specified)
  1178→   * @returns Comprehensive analysis of projection accuracy
  1179→   */
  1180→  async analyzeProjectionAccuracy(fleetIds?: string[]): Promise<ProjectionAccuracyAnalysis> {
  1181→    return this.traced('analyzeProjectionAccuracy', async (span) => {
  1182→      const analysisTime = new Date().toISOString();
  1183→      const targetDate = new Date().toISOString().split('T')[0];
  1184→
  1185→      // Get all active trains
  1186→      const allTrains = await this.db.query.trains.findMany({
  1187→        where: eq(trains.fitnessStatus, 'FIT'),
  1188→        with: { fleet: true },
  1189→      });
  1190→      const filteredTrains = fleetIds
  1191→        ? allTrains.filter((t) => fleetIds.includes(t.fleetId))
  1192→        : allTrains;
  1193→
  1194→      span.setAttribute('analysis.totalTrains', filteredTrains.length);
  1195→
  1196→      const fallbackTrains: TrainProjectionInfo[] = [];
  1197→      let routeProfileCount = 0;
  1198→      let fleetAverageCount = 0;
  1199→      let systemDefaultCount = 0;
  1200→      let totalImpactKm = 0;
  1201→
  1202→      // Analyze each train's projection source
  1203→      for (const train of filteredTrains) {
  1204→        const rate = await this.calculateDailyRate(train.id, targetDate);
  1205→
  1206→        if (rate.source.startsWith('route:')) {
  1207→          routeProfileCount++;
  1208→        } else if (rate.source === 'fleet_average') {
  1209→          fleetAverageCount++;
  1210→          // Estimate projection uncertainty: fleet averages have ~20% variance
  1211→          const dailyVariance = rate.dailyKm * 0.2;

  1465→      return { requiresWarning: true, warningSeverity: 'LOW' };
  1466→    }

  1500→  /**
  1501→   * Calculate daily rate from a route profile.
  1502→   */
  1503→  private calculateFromProfile(
  1504→    profile: RouteUtilizationProfile,
  1505→    targetDate: string,
  1506→    dayType: UtilizationDayType
  1507→  ): DailyRateResult {
  1508→    // Get base km for day type
  1509→    let baseKm: number;
  1510→    switch (dayType) {
  1511→      case 'SATURDAY':
  1512→        baseKm = profile.saturdayKm ?? Math.round(profile.weekdayKm * DEFAULT_DAY_TYPE_MULTIPLIERS.SATURDAY);
  1513→        break;
  1514→      case 'SUNDAY':
  1515→        baseKm = profile.sundayKm ?? Math.round(profile.weekdayKm * DEFAULT_DAY_TYPE_MULTIPLIERS.SUNDAY);
  1516→        break;
  1517→      case 'HOLIDAY':
  1518→        baseKm = profile.holidayKm ?? Math.round(profile.weekdayKm * DEFAULT_DAY_TYPE_MULTIPLIERS.HOLIDAY);
  1519→        break;
  1520→      default:
  1521→        baseKm = profile.weekdayKm;
  1522→    }
  1523→
  1524→    // Apply seasonal multiplier
  1525→    const isPeak = this.isPeakSeason(targetDate);
  1526→    const multiplier = isPeak
  1527→      ? parseFloat(profile.peakSeasonMultiplier ?? '1.0')
  1528→      : parseFloat(profile.offPeakMultiplier ?? '1.0');
  1529→
  1530→    const dailyKm = Math.round(baseKm * multiplier);
  1531→
  1532→    return {
  1533→      dailyKm,
  1534→      source: `route:${profile.routeId}`,
  1535→      confidence: (profile.confidence ?? 'MEDIUM') as ConfidenceLevel,
  1536→      details: {
  1537→        routeId: profile.routeId,
  1538→        dayType,
  1539→        seasonalMultiplier: multiplier,
  1540→        profileEffectiveDate: profile.effectiveDate,
  1541→      },
  1542→    };
  1543→  }
