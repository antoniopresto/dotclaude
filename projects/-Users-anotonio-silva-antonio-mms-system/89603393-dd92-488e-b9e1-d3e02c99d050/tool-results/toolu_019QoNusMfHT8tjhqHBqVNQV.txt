   369→```typescript
   370→// Schema validation (replaces Zod)
   371→import { createSchema, createType, InferType } from '@belt/tools';
   372→
   373→const TrainModel = createSchema({
   374→  id: 'string',
   375→  trainNumber: 'string',
   376→  mileageKm: 'number',
   377→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   378→  tags: 'string[]',
   379→  metadata: 'object?',
   380→});
   381→type Train = InferType<typeof TrainModel>;
   382→const train = TrainModel.parse(data);
   383→
   384→// State machine (replaces React hooks + zustand)
   385→import { createSlice, createMachine } from '@belt/tools';
   386→
   387→const fleetSlice = createSlice({
   388→  id: 'fleet',
   389→  initial: { trains: [], selected: null },
   390→  cache: { policy: 'session' },
   391→  reducers: {
   392→    setTrains: (state, trains) => { state.trains = trains; },
   393→    selectTrain: (state, id) => { state.selected = id; },
   394→  },
   395→});
   396→
   397→const machine = createMachine({
   398→  id: 'mms',
   399→  slices: { fleet: fleetSlice },
   400→  trackMetadata: false, // Critical for 60fps
   401→});
   402→
   403→// In React component (NO useState/useEffect!)
   404→const { state, actions } = machine.useMachine('fleet');
