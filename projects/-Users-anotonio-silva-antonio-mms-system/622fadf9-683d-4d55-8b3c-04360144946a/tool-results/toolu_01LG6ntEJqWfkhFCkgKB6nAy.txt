   381→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   382→  tags: 'string[]',
   383→  metadata: 'object?',
   384→});
   385→type Train = InferType<typeof TrainModel>;
   386→const train = TrainModel.parse(data);

   389→import { createSlice, createMachine } from '@belt/tools';
   390→

   392→  id: 'fleet',
   393→  initial: { trains: [], selected: null },
   394→  cache: { policy: 'session' },
   395→  reducers: {
   396→    setTrains: (state, trains) => { state.trains = trains; },
   397→    selectTrain: (state, id) => { state.selected = id; },
   398→  },
   399→});
   400→
   401→const machine = createMachine({
   402→  id: 'mms',
   403→  slices: { fleet: fleetSlice },
   404→  trackMetadata: false, // Critical for 60fps
   405→});
   406→
   407→// In React component (NO useState/useEffect!)
   408→const { state, actions } = machine.useMachine('fleet');
