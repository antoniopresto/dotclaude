    61→### 4.2.1 Purpose
    62→
    63→Process daily CSV files containing mileage data, validate records, handle meter events, and update the database.

    65→### 4.2.2 File Watcher
    66→
    67→```typescript
    68→interface FileWatcherConfig {
    69→  inboxPath: string;           // /data/inbox
    70→  processedPath: string;       // /data/processed
    71→  quarantinePath: string;      // /data/quarantine
    72→  expectedFilePattern: RegExp; // /^mileage_\d{8}\.csv$/
    73→  maxFileSizeBytes: number;    // 10MB
    74→  alertThresholdHours: number; // 2
    75→}
    76→
    77→class FileWatcherService {
    78→  private watcher: FSWatcher;
    79→  private lastFileReceivedAt: Date | null = null;
    80→  
    81→  async start(): Promise<void> {
    82→    this.watcher = chokidar.watch(this.config.inboxPath, {
    83→      persistent: true,
    84→      awaitWriteFinish: { stabilityThreshold: 2000, pollInterval: 500 }
    85→    });
    86→    
    87→    this.watcher.on('add', async (filePath) => {
    88→      this.lastFileReceivedAt = new Date();
    89→      await this.processFile(filePath);
    90→    });
    91→    
    92→    // Health check interval
    93→    setInterval(() => this.checkFileArrivalHealth(), 30 * 60 * 1000);
    94→  }
    95→  
    96→  private async processFile(filePath: string): Promise<void> {
    97→    const filename = path.basename(filePath);
    98→    const jobId = crypto.randomUUID();
    99→    
   100→    try {
   101→      // Validate file
   102→      await this.validateFile(filePath);
   103→      
   104→      // Queue for processing
   105→      await this.jobQueue.add('ingestion', {
   106→        jobId,
   107→        filePath,
   108→        filename,
   109→        receivedAt: new Date().toISOString()
   110→      }, {
   111→        attempts: 3,
   112→        backoff: { type: 'exponential', delay: 5000 }
   113→      });
   114→      
   115→      this.logger.info({ jobId, filename }, 'File queued for ingestion');
   116→      
   117→    } catch (error) {
   118→      // Move to quarantine
   119→      const quarantinePath = path.join(
   120→        this.config.quarantinePath, 
   121→        `${Date.now()}_${filename}`
   122→      );
   123→      await fs.rename(filePath, quarantinePath);
   124→      
   125→      this.logger.error({ filename, error }, 'File quarantined');
   126→      
   127→      await this.alertService.send({
   128→        type: 'FILE_VALIDATION_FAILED',
   129→        severity: 'WARNING',
   130→        message: `File ${filename} failed validation: ${error.message}`,
   131→        filePath: quarantinePath
   132→      });
   133→    }
   134→  }
   135→  
   136→  private async validateFile(filePath: string): Promise<void> {
   137→    const stats = await fs.stat(filePath);
   138→    
   139→    // Size check
   140→    if (stats.size > this.config.maxFileSizeBytes) {
   141→      throw new Error(`File exceeds maximum size: ${stats.size} bytes`);
   142→    }
   143→    
   144→    // Format check
   145→    const filename = path.basename(filePath);
   146→    if (!this.config.expectedFilePattern.test(filename)) {
   147→      throw new Error(`Filename does not match expected pattern: ${filename}`);
   148→    }
   149→    
   150→    // Content check (first 10 lines)
   151→    const sample = await this.readFirstLines(filePath, 10);
   152→    await this.validateCSVStructure(sample);
   153→  }
   154→  
   155→  private async checkFileArrivalHealth(): Promise<void> {
   156→    if (!this.lastFileReceivedAt) return;
   157→    
   158→    const hoursSinceLastFile = 
   159→      (Date.now() - this.lastFileReceivedAt.getTime()) / (1000 * 60 * 60);
   160→    
   161→    if (hoursSinceLastFile > this.config.alertThresholdHours) {
   162→      await this.alertService.send({
   163→        type: 'FILE_ARRIVAL_DELAYED',
   164→        severity: 'WARNING',
   165→        message: `No mileage files received in ${hoursSinceLastFile.toFixed(1)} hours`,
   166→        lastFileAt: this.lastFileReceivedAt.toISOString(),
   167→        expectedInterval: `${this.config.alertThresholdHours} hours`
   168→      });
   169→    }
   170→  }
   171→}

   174→### 4.2.3 Ingestion Worker
   175→
   176→```typescript
   177→interface MileageRecord {
   178→  trainNumber: string;
   179→  readingDate: string;  // YYYY-MM-DD
   180→  odometerValue: number;
   181→  source?: string;
   182→}
   183→
   184→interface IngestionResult {
   185→  jobId: string;
   186→  filename: string;
   187→  totalRecords: number;
   188→  validRecords: number;
   189→  invalidRecords: number;
   190→  anomalies: AnomalyRecord[];
   191→  meterEvents: MeterEventRecord[];
   192→  processingTimeMs: number;
   193→}
   194→
   195→class IngestionWorker {
   196→  async process(job: Job<IngestionJobData>): Promise<IngestionResult> {
   197→    const startTime = Date.now();
   198→    const { jobId, filePath, filename } = job.data;
   199→    
   200→    const result: IngestionResult = {
   201→      jobId,
   202→      filename,
   203→      totalRecords: 0,
   204→      validRecords: 0,
   205→      invalidRecords: 0,
   206→      anomalies: [],
   207→      meterEvents: [],
   208→      processingTimeMs: 0
   209→    };
   210→    
   211→    // Parse CSV
   212→    const records = await this.parseCSV(filePath);
   213→    result.totalRecords = records.length;
   214→    
   215→    // Process in batches
   216→    const batches = this.chunk(records, 100);
   217→    
   218→    for (const batch of batches) {
   219→      await this.db.transaction(async (tx) => {
   220→        for (const record of batch) {
   221→          try {
   222→            await this.processRecord(tx, record, result);
   223→            result.validRecords++;
   224→          } catch (error) {
   225→            result.invalidRecords++;
   226→            this.logger.warn({ record, error }, 'Record processing failed');
   227→          }
   228→        }
   229→      });
   230→    }
   231→    
   232→    // Move file to processed
   233→    const processedPath = path.join(
   234→      this.config.processedPath,
   235→      `${Date.now()}_${filename}`
   236→    );
   237→    await fs.rename(filePath, processedPath);
   238→    
   239→    // Update train current mileage
   240→    await this.updateTrainMileage(records);
   241→    
   242→    // Trigger downstream processing
   243→    await this.triggerDriftCheck();
   244→    
   245→    result.processingTimeMs = Date.now() - startTime;
   246→    
   247→    return result;
   248→  }
   249→  
   250→  private async processRecord(
   251→    tx: Transaction,
   252→    record: MileageRecord,
   253→    result: IngestionResult
   254→  ): Promise<void> {
   255→    // Find train
   256→    const train = await tx.query.trains.findFirst({
   257→      where: eq(trains.trainNumber, record.trainNumber)
   258→    });
   259→    
   260→    if (!train) {
   261→      throw new Error(`Unknown train: ${record.trainNumber}`);
   262→    }
   263→    
   264→    // Get previous reading
   265→    const previousReading = await tx.query.mileageReadings.findFirst({
   266→      where: and(
   267→        eq(mileageReadings.trainId, train.id),
   268→        lt(mileageReadings.readingDate, record.readingDate)
   269→      ),
   270→      orderBy: desc(mileageReadings.readingDate)
   271→    });
   272→    
   273→    // Calculate adjusted value (handle meter events)
   274→    const adjustment = await this.getMeterAdjustment(tx, train.id, record.readingDate);
   275→    const adjustedReading = record.odometerValue + adjustment;
   276→    
   277→    // Calculate daily km
   278→    let dailyKm: number | null = null;
   279→    let anomalyType: string | null = null;
   280→    
   281→    if (previousReading) {
   282→      dailyKm = adjustedReading - previousReading.adjustedReading;
   283→      
   284→      // Anomaly detection
   285→      if (dailyKm < 0) {
   286→        anomalyType = 'NEGATIVE';
   287→        result.anomalies.push({
   288→          trainNumber: record.trainNumber,
   289→          date: record.readingDate,
   290→          type: 'NEGATIVE',
   291→          value: dailyKm,
   292→          message: `Negative daily mileage: ${dailyKm} km`
   293→        });
   294→      } else if (dailyKm > 1000) {  // Spike threshold
   295→        anomalyType = 'SPIKE';
   296→        result.anomalies.push({
   297→          trainNumber: record.trainNumber,
   298→          date: record.readingDate,
   299→          type: 'SPIKE',
   300→          value: dailyKm,
   301→          message: `Unusually high daily mileage: ${dailyKm} km`
   302→        });
   303→      }
   304→    }
   305→    
   306→    // Insert or update reading
   307→    await tx.insert(mileageReadings)
   308→      .values({
   309→        trainId: train.id,
   310→        readingDate: record.readingDate,
   311→        rawReading: record.odometerValue,
   312→        adjustedReading,
   313→        dailyKm,
   314→        validationStatus: anomalyType ? 'ANOMALY' : 'VALID',
   315→        anomalyType,
   316→        sourceFile: result.filename
   317→      })
   318→      .onConflictDoUpdate({
   319→        target: [mileageReadings.trainId, mileageReadings.readingDate],
   320→        set: {
   321→          rawReading: record.odometerValue,
   322→          adjustedReading,
   323→          dailyKm,
   324→          validationStatus: anomalyType ? 'ANOMALY' : 'VALID',
   325→          anomalyType,
   326→          sourceFile: result.filename
   327→        }
   328→      });
   329→  }
   330→  
   331→  private async getMeterAdjustment(
   332→    tx: Transaction,
   333→    trainId: string,
   334→    asOfDate: string
   335→  ): Promise<number> {
   336→    // Sum all meter event adjustments up to this date
   337→    const events = await tx.query.meterEvents.findMany({
   338→      where: and(
   339→        eq(meterEvents.trainId, trainId),
   340→        lte(meterEvents.eventDate, asOfDate)
   341→      )
   342→    });
   343→    
   344→    return events.reduce((sum, e) => sum + (e.adjustmentKm || 0), 0);
   345→  }
   346→  
   347→  private async updateTrainMileage(records: MileageRecord[]): Promise<void> {
   348→    // Group by train, find latest reading
   349→    const latestByTrain = new Map<string, MileageRecord>();
   350→    
   351→    for (const record of records) {
   352→      const existing = latestByTrain.get(record.trainNumber);
   353→      if (!existing || record.readingDate > existing.readingDate) {
   354→        latestByTrain.set(record.trainNumber, record);
   355→      }
   356→    }
   357→    
   358→    // Update train current_mileage
   359→    for (const [trainNumber, record] of latestByTrain) {
   360→      await this.db.update(trains)
   361→        .set({ 
   362→          currentMileageKm: record.odometerValue,
   363→          updatedAt: new Date()
   364→        })
   365→        .where(eq(trains.trainNumber, trainNumber));
   366→    }
   367→    
   368→    // Update fleet mileage targets
   369→    await this.db.execute(sql`SELECT update_fleet_mileage_targets()`);
   370→  }
   371→}

   376→## 4.2.5 Unavailability Service (NEW v2.1)
   377→
   378→### 4.2.5.1 Purpose
   379→
   380→Track trains that are unavailable due to corrective maintenance, incidents, or other blocking factors. This service is **critical** for schedule validity - without it, the optimizer can schedule PMs for trains that are grounded for weeks.
   381→
   382→### 4.2.5.2 Implementation
   383→
   384→```typescript
   385→interface UnavailabilityEvent {
   386→  id: string;
   387→  trainId: string;
   388→  eventType: 'CORRECTIVE' | 'INCIDENT' | 'COLLISION' | 'VANDALISM' | 'EXTERNAL' | 'PLANNED_OUTAGE';
   389→  status: 'ACTIVE' | 'RESOLVED';
   390→  restrictionLevel: 'RESTRICTED' | 'GROUNDED';
   391→  
   392→  reportedAt: Date;
   393→  estimatedRepairHours?: number;
   394→  estimatedReturnDate?: Date;
   395→  actualReturnDate?: Date;
   396→  
   397→  sourceSystem: 'MAXIMO' | 'OCC' | 'MANUAL';
   398→  externalReference?: string;   // Maximo WO number
   399→  externalStatus?: string;      // Maximo status
   400→  
   401→  description?: string;
   402→  failureCode?: string;
   403→  affectedSystem?: string;
   404→  
   405→  blocksPmScheduling: boolean;
   406→  blocksRunAssignment: boolean;
   407→}
   408→
   409→interface CreateUnavailabilityInput {
   410→  trainId: string;
   411→  eventType: UnavailabilityEvent['eventType'];
   412→  restrictionLevel: UnavailabilityEvent['restrictionLevel'];
   413→  estimatedReturnDate?: Date;
   414→  estimatedRepairHours?: number;
   415→  sourceSystem: UnavailabilityEvent['sourceSystem'];
   416→  externalReference?: string;
   417→  description?: string;
   418→  affectedSystem?: string;
   419→}
   420→
   421→class UnavailabilityService {
   422→  constructor(
   423→    private db: Database,
   424→    private alertService: AlertService,
   425→    private auditService: AuditService
   426→  ) {}
   427→  
   428→  /**
   429→   * Create a new unavailability event (e.g., from Maximo webhook)
   430→   */
   431→  async createEvent(input: CreateUnavailabilityInput, userId?: string): Promise<UnavailabilityEvent> {
   432→    const event = await this.db.transaction(async (tx) => {
   433→      // 1. Create the unavailability event
   434→      const [newEvent] = await tx
   435→        .insert(unavailabilityEvents)
   436→        .values({
   437→          ...input,
   438→          status: 'ACTIVE',
   439→          reportedAt: new Date(),
   440→          blocksPmScheduling: true,
   441→          blocksRunAssignment: input.restrictionLevel === 'GROUNDED',
   442→        })
   443→        .returning();
   444→      
   445→      // 2. Update train fitness status (trigger handles this, but we can be explicit)
   446→      await tx
   447→        .update(trains)
   448→        .set({
   449→          fitnessStatus: input.restrictionLevel,
   450→          fitnessNotes: input.description || `Unavailability: ${input.eventType}`,
   451→          updatedAt: new Date()
   452→        })
   453→        .where(eq(trains.id, input.trainId));
   454→      
   455→      // 3. Check for conflicts with scheduled PMs
   456→      const conflicts = await this.checkPMConflicts(tx, input.trainId, input.estimatedReturnDate);
   457→      
   458→      return { event: newEvent, conflicts };
   459→    });
   460→    
   461→    // Alert if PM conflicts exist
   462→    if (event.conflicts.length > 0) {
   463→      await this.alertService.send({
   464→        type: 'UNAVAILABILITY_PM_CONFLICT',
   465→        severity: 'WARNING',
   466→        message: `Train ${event.event.trainId} unavailable - ${event.conflicts.length} scheduled PMs may be affected`,
   467→        data: {
   468→          trainId: event.event.trainId,
   469→          eventId: event.event.id,
   470→          conflictingPMs: event.conflicts.map(pm => ({
   471→            pmId: pm.id,
   472→            pmType: pm.pmTypeCode,
   473→            targetDate: pm.targetDate
   474→          }))
   475→        }
   476→      });
   477→    }
   478→    
   479→    // Audit log
   480→    await this.auditService.log({
   481→      entityType: 'unavailability_event',
   482→      entityId: event.event.id,
   483→      action: 'CREATE',
   484→      userId,
   485→      newValues: event.event
   486→    });
   487→    
   488→    return event.event;
   489→  }
   490→  
   491→  /**
   492→   * Resolve an unavailability event (train returns to service)
   493→   */
   494→  async resolveEvent(
   495→    eventId: string, 
   496→    actualReturnDate: Date, 
   497→    userId: string,
   498→    notes?: string
   499→  ): Promise<void> {
   500→    await this.db.transaction(async (tx) => {
   501→      const [event] = await tx
   502→        .select()
   503→        .from(unavailabilityEvents)
   504→        .where(eq(unavailabilityEvents.id, eventId));
   505→      
   506→      if (!event) {
   507→        throw new NotFoundError(`Unavailability event ${eventId} not found`);
   508→      }
   509→      
   510→      if (event.status === 'RESOLVED') {
   511→        throw new ConflictError('Event already resolved');
   512→      }
   513→      
   514→      // Update the event
   515→      await tx
   516→        .update(unavailabilityEvents)
   517→        .set({
   518→          status: 'RESOLVED',
   519→          actualReturnDate,
   520→          resolvedAt: new Date(),
   521→          resolvedBy: userId,
   522→          resolutionNotes: notes,
   523→          updatedAt: new Date()
   524→        })
   525→        .where(eq(unavailabilityEvents.id, eventId));
   526→      
   527→      // Check if there are other active unavailability events for this train
   528→      const otherActive = await tx
   529→        .select({ count: count() })
   530→        .from(unavailabilityEvents)
   531→        .where(and(
   532→          eq(unavailabilityEvents.trainId, event.trainId),
   533→          eq(unavailabilityEvents.status, 'ACTIVE'),
   534→          ne(unavailabilityEvents.id, eventId)
   535→        ));
   536→      
   537→      // If no other active events, restore train to FIT
   538→      if (otherActive[0].count === 0) {
   539→        await tx
   540→          .update(trains)
   541→          .set({
   542→            fitnessStatus: 'FIT',
   543→            fitnessNotes: null,
   544→            updatedAt: new Date()
   545→          })
   546→          .where(eq(trains.id, event.trainId));
   547→      }
   548→    });
   549→    
   550→    await this.auditService.log({
   551→      entityType: 'unavailability_event',
   552→      entityId: eventId,
   553→      action: 'RESOLVE',
   554→      userId,
   555→      newValues: { status: 'RESOLVED', actualReturnDate }
   556→    });
   557→  }
   558→  
   559→  /**
   560→   * Get all active unavailability events (for optimizer input)
   561→   */
   562→  async getActiveUnavailability(depotId?: string): Promise<UnavailabilityEvent[]> {
   563→    let query = this.db
   564→      .select({
   565→        event: unavailabilityEvents,
   566→        train: {
   567→          trainNumber: trains.trainNumber,
   568→          fleetId: trains.fleetId,
   569→          currentDepotId: trains.currentDepotId
   570→        }
   571→      })
   572→      .from(unavailabilityEvents)
   573→      .innerJoin(trains, eq(unavailabilityEvents.trainId, trains.id))
   574→      .where(eq(unavailabilityEvents.status, 'ACTIVE'));
   575→    
   576→    if (depotId) {
   577→      query = query.where(eq(trains.currentDepotId, depotId));
   578→    }
   579→    
   580→    const results = await query;
   581→    return results.map(r => ({
   582→      ...r.event,
   583→      trainNumber: r.train.trainNumber,
   584→      fleetId: r.train.fleetId
   585→    }));
   586→  }

   588→  /**

   615→  /**
   616→   * Process Maximo webhook for work order status change
   617→   */
   618→  async processMaximoWebhook(payload: MaximoWorkOrderWebhook): Promise<void> {
   619→    // Map Maximo work types to our event types
   620→    const eventTypeMap: Record<string, UnavailabilityEvent['eventType']> = {
   621→      'CM': 'CORRECTIVE',
   622→      'EM': 'CORRECTIVE',  // Emergency maintenance
   623→      'IN': 'INCIDENT'
   624→    };
   625→    
   626→    if (!eventTypeMap[payload.worktype]) {
   627→      // Not a corrective maintenance type, ignore
   628→      return;
   629→    }
   630→    
   631→    // Check if we already have an event for this WO
   632→    const existing = await this.db
   633→      .select()
   634→      .from(unavailabilityEvents)
   635→      .where(and(
   636→        eq(unavailabilityEvents.sourceSystem, 'MAXIMO'),
   637→        eq(unavailabilityEvents.externalReference, payload.wonum)
   638→      ))
   639→      .limit(1);
   640→    
   641→    if (existing.length > 0) {
   642→      // Update existing event based on Maximo status
   643→      await this.updateFromMaximo(existing[0], payload);
   644→    } else if (payload.status === 'WAPPR' || payload.status === 'INPRG') {
   645→      // Create new unavailability event
   646→      const train = await this.findTrainByAssetnum(payload.assetnum);
   647→      if (!train) {
   648→        this.logger.warn({ assetnum: payload.assetnum }, 'Unknown asset in Maximo webhook');
   649→        return;
   650→      }
   651→      
   652→      await this.createEvent({
   653→        trainId: train.id,
   654→        eventType: eventTypeMap[payload.worktype],
   655→        restrictionLevel: 'GROUNDED',  // Default to grounded for CM
   656→        estimatedReturnDate: payload.schedfinish ? new Date(payload.schedfinish) : undefined,
   657→        sourceSystem: 'MAXIMO',
   658→        externalReference: payload.wonum,
   659→        description: payload.description
   660→      });
   661→    }
   662→  }
   663→  
   664→  private async updateFromMaximo(
   665→    event: UnavailabilityEvent, 
   666→    payload: MaximoWorkOrderWebhook
   667→  ): Promise<void> {
   668→    const maximoStatusMap: Record<string, 'ACTIVE' | 'RESOLVED'> = {
   669→      'WAPPR': 'ACTIVE',
   670→      'APPR': 'ACTIVE',
   671→      'INPRG': 'ACTIVE',
   672→      'COMP': 'RESOLVED',
   673→      'CLOSE': 'RESOLVED',
   674→      'CAN': 'RESOLVED'
   675→    };
   676→    
   677→    const newStatus = maximoStatusMap[payload.status] || 'ACTIVE';
   678→    
   679→    if (newStatus === 'RESOLVED' && event.status === 'ACTIVE') {
   680→      await this.resolveEvent(
   681→        event.id, 
   682→        new Date(), 
   683→        'SYSTEM_MAXIMO',
   684→        `Resolved via Maximo status: ${payload.status}`
   685→      );
   686→    } else {
   687→      // Update estimated return date if changed
   688→      await this.db
   689→        .update(unavailabilityEvents)
   690→        .set({
   691→          externalStatus: payload.status,
   692→          estimatedReturnDate: payload.schedfinish ? new Date(payload.schedfinish) : undefined,
   693→          updatedAt: new Date()
   694→        })
   695→        .where(eq(unavailabilityEvents.id, event.id));
   696→    }
   697→  }

   712→        lte(pmSchedule.targetDate, endDate),
   713→        isNull(pmSchedule.scenarioId)
