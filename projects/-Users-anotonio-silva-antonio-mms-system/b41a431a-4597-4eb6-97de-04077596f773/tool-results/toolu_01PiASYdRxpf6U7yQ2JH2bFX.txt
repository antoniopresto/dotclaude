     1→/**
     2→ * UI Slice
     3→ *
     4→ * State management for global UI state: toasts, modals, loading, errors.
     5→ *
     6→ * @see specs/006-user-interface.md Section 6.8
     7→ */
     8→
     9→import { createSlice } from '@belt/tools';
    10→import type { Toast, ModalState, LoadingState, ErrorState, OptimizerJob } from '../types';
    11→
    12→// =============================================================================
    13→// UI Slice State
    14→// =============================================================================
    15→
    16→export interface UiSliceState {
    17→  // Toast notifications
    18→  toasts: Toast[];
    19→  maxToasts: number;
    20→
    21→  // Modal state
    22→  modal: ModalState;
    23→
    24→  // Loading indicators
    25→  loading: LoadingState;
    26→
    27→  // Global error state
    28→  error: ErrorState;
    29→
    30→  // Sidebar collapsed
    31→  sidebarCollapsed: boolean;
    32→
    33→  // Help panel open
    34→  helpPanelOpen: boolean;
    35→
    36→  // Keyboard shortcuts dialog visible
    37→  keyboardShortcutsDialogOpen: boolean;
    38→
    39→  // Keyboard shortcuts enabled
    40→  shortcutsEnabled: boolean;
    41→
    42→  // Theme
    43→  theme: 'light' | 'dark' | 'system';
    44→
    45→  // Optimizer jobs tracking
    46→  optimizerJobs: OptimizerJob[];
    47→  activeJobId: string | null;
    48→
    49→  // Connection status
    50→  isOnline: boolean;
    51→  websocketConnected: boolean;
    52→
    53→  // Sync status
    54→  lastSyncAt: number | null;
    55→  syncInProgress: boolean;
    56→
    57→  // Index signature for MachineState compatibility
    58→  [key: string]: unknown;
    59→}
    60→
    61→const initialUiState: UiSliceState = {
    62→  toasts: [],
    63→  maxToasts: 5,
    64→  modal: {
    65→    isOpen: false,
    66→    type: null,
    67→    data: null,
    68→  },
    69→  loading: {
    70→    trains: false,
    71→    pmSchedule: false,
    72→    canvas: false,
    73→    optimizer: false,
    74→  },
    75→  error: {
    76→    message: null,
    77→    code: null,
    78→    details: null,
    79→    timestamp: null,
    80→  },
    81→  sidebarCollapsed: false,
    82→  helpPanelOpen: false,
    83→  keyboardShortcutsDialogOpen: false,
    84→  shortcutsEnabled: true,
    85→  theme: 'system',
    86→  optimizerJobs: [],
    87→  activeJobId: null,
    88→  isOnline: true,
    89→  websocketConnected: false,
    90→  lastSyncAt: null,
    91→  syncInProgress: false,
    92→};
    93→
    94→let toastIdCounter = 0;
    95→
    96→export const uiSlice = createSlice({
    97→  id: 'ui',
    98→  initial: initialUiState,
    99→  cache: { policy: 'session' },
   100→  reducers: {
   101→    // ==========================================================================
   102→    // Toast Notifications
   103→    // ==========================================================================
   104→
   105→    addToast: (
   106→      s,
   107→      toast: Omit<Toast, 'id' | 'createdAt'> & { id?: string }
   108→    ) => {
   109→      const state = s as UiSliceState;
   110→      const newToast: Toast = {
   111→        id: toast.id ?? `toast-${++toastIdCounter}`,
   112→        type: toast.type,
   113→        title: toast.title,
   114→        message: toast.message,
   115→        duration: toast.duration ?? 5000,
   116→        dismissible: toast.dismissible ?? true,
   117→        createdAt: Date.now(),
   118→      };
   119→
   120→      state.toasts.push(newToast);
   121→
   122→      // Trim to max toasts
   123→      if (state.toasts.length > state.maxToasts) {
   124→        state.toasts.shift();
   125→      }
   126→    },
   127→
   128→    removeToast: (s, toastId: string) => {
   129→      const state = s as UiSliceState;
   130→      state.toasts = state.toasts.filter((t) => t.id !== toastId);
   131→    },
   132→
   133→    clearToasts: (s) => {
   134→      const state = s as UiSliceState;
   135→      state.toasts = [];
   136→    },
   137→
   138→    // Convenience toast creators
   139→    showSuccess: (s, payload: { title: string; message: string }) => {
   140→      const state = s as UiSliceState;
   141→      state.toasts.push({
   142→        id: `toast-${++toastIdCounter}`,
   143→        type: 'success',
   144→        title: payload.title,
   145→        message: payload.message,
   146→        duration: 3000,
   147→        dismissible: true,
   148→        createdAt: Date.now(),
   149→      });
   150→    },
   151→
   152→    showError: (s, payload: { title: string; message: string }) => {
   153→      const state = s as UiSliceState;
   154→      state.toasts.push({
   155→        id: `toast-${++toastIdCounter}`,
   156→        type: 'error',
   157→        title: payload.title,
   158→        message: payload.message,
   159→        duration: 8000,
   160→        dismissible: true,
   161→        createdAt: Date.now(),
   162→      });
   163→    },
   164→
   165→    showWarning: (s, payload: { title: string; message: string }) => {
   166→      const state = s as UiSliceState;
   167→      state.toasts.push({
   168→        id: `toast-${++toastIdCounter}`,
   169→        type: 'warning',
   170→        title: payload.title,
   171→        message: payload.message,
   172→        duration: 6000,
   173→        dismissible: true,
   174→        createdAt: Date.now(),
   175→      });
   176→    },
   177→
   178→    showInfo: (s, payload: { title: string; message: string }) => {
   179→      const state = s as UiSliceState;
   180→      state.toasts.push({
   181→        id: `toast-${++toastIdCounter}`,
   182→        type: 'info',
   183→        title: payload.title,
   184→        message: payload.message,
   185→        duration: 5000,
   186→        dismissible: true,
   187→        createdAt: Date.now(),
   188→      });
   189→    },
   190→
   191→    // ==========================================================================
   192→    // Modal
   193→    // ==========================================================================
   194→
   195→    openModal: (s, payload: { type: string; data?: Record<string, unknown> }) => {
   196→      const state = s as UiSliceState;
   197→      state.modal = {
   198→        isOpen: true,
   199→        type: payload.type,
   200→        data: payload.data ?? null,
   201→      };
   202→    },
   203→
   204→    closeModal: (s) => {
   205→      const state = s as UiSliceState;
   206→      state.modal = {
   207→        isOpen: false,
   208→        type: null,
   209→        data: null,
   210→      };
   211→    },
   212→
   213→    updateModalData: (s, data: Record<string, unknown>) => {
   214→      const state = s as UiSliceState;
   215→      if (state.modal.isOpen) {
   216→        state.modal.data = { ...state.modal.data, ...data };
   217→      }
   218→    },
   219→
   220→    // ==========================================================================
   221→    // Loading
   222→    // ==========================================================================
   223→
   224→    setLoading: (s, payload: Partial<LoadingState>) => {
   225→      const state = s as UiSliceState;
   226→      state.loading = { ...state.loading, ...payload };
   227→    },
   228→
   229→    startLoading: (s, key: keyof LoadingState) => {
   230→      const state = s as UiSliceState;
   231→      state.loading[key] = true;
   232→    },
   233→
   234→    stopLoading: (s, key: keyof LoadingState) => {
   235→      const state = s as UiSliceState;
   236→      state.loading[key] = false;
   237→    },
   238→
   239→    clearLoading: (s) => {
   240→      const state = s as UiSliceState;
   241→      state.loading = {
   242→        trains: false,
   243→        pmSchedule: false,
   244→        canvas: false,
   245→        optimizer: false,
   246→      };
   247→    },
   248→
   249→    // ==========================================================================
   250→    // Error
   251→    // ==========================================================================
   252→
   253→    setError: (
   254→      s,
   255→      payload: { message: string; code?: string; details?: Record<string, unknown> }
   256→    ) => {
   257→      const state = s as UiSliceState;
   258→      state.error = {
   259→        message: payload.message,
   260→        code: payload.code ?? null,
   261→        details: payload.details ?? null,
   262→        timestamp: Date.now(),
   263→      };
   264→    },
   265→
   266→    clearError: (s) => {
   267→      const state = s as UiSliceState;
   268→      state.error = {
   269→        message: null,
   270→        code: null,
   271→        details: null,
   272→        timestamp: null,
   273→      };
   274→    },
   275→
   276→    // ==========================================================================
   277→    // Layout
   278→    // ==========================================================================
   279→
   280→    toggleSidebar: (s) => {
   281→      const state = s as UiSliceState;
   282→      state.sidebarCollapsed = !state.sidebarCollapsed;
   283→    },
   284→
   285→    setSidebarCollapsed: (s, collapsed: boolean) => {
   286→      const state = s as UiSliceState;
   287→      state.sidebarCollapsed = collapsed;
   288→    },
   289→
   290→    toggleHelpPanel: (s) => {
   291→      const state = s as UiSliceState;
   292→      state.helpPanelOpen = !state.helpPanelOpen;
   293→    },
   294→
   295→    setHelpPanelOpen: (s, open: boolean) => {
   296→      const state = s as UiSliceState;
   297→      state.helpPanelOpen = open;
   298→    },
   299→
   300→    toggleKeyboardShortcutsDialog: (s) => {
   301→      const state = s as UiSliceState;
   302→      state.keyboardShortcutsDialogOpen = !state.keyboardShortcutsDialogOpen;
   303→    },
   304→
   305→    showKeyboardShortcutsDialog: (s) => {
   306→      const state = s as UiSliceState;
   307→      state.keyboardShortcutsDialogOpen = true;
   308→    },
   309→
   310→    hideKeyboardShortcutsDialog: (s) => {
   311→      const state = s as UiSliceState;
   312→      state.keyboardShortcutsDialogOpen = false;
   313→    },
   314→
   315→    // ==========================================================================
   316→    // Preferences
   317→    // ==========================================================================
   318→
   319→    setShortcutsEnabled: (s, enabled: boolean) => {
   320→      const state = s as UiSliceState;
   321→      state.shortcutsEnabled = enabled;
   322→    },
   323→
   324→    setTheme: (s, theme: 'light' | 'dark' | 'system') => {
   325→      const state = s as UiSliceState;
   326→      state.theme = theme;
   327→    },
   328→
   329→    // ==========================================================================
   330→    // Optimizer Jobs
   331→    // ==========================================================================
   332→
   333→    addOptimizerJob: (s, job: OptimizerJob) => {
   334→      const state = s as UiSliceState;
   335→      state.optimizerJobs.push(job);
   336→      state.activeJobId = job.id;
   337→    },
   338→
   339→    updateOptimizerJob: (s, job: Partial<OptimizerJob> & { id: string }) => {
   340→      const state = s as UiSliceState;
   341→      const index = state.optimizerJobs.findIndex((j) => j.id === job.id);
   342→      if (index !== -1) {
   343→        state.optimizerJobs[index] = { ...state.optimizerJobs[index], ...job };
   344→      }
   345→    },
   346→
   347→    removeOptimizerJob: (s, jobId: string) => {
   348→      const state = s as UiSliceState;
   349→      state.optimizerJobs = state.optimizerJobs.filter((j) => j.id !== jobId);
   350→      if (state.activeJobId === jobId) {
   351→        state.activeJobId = state.optimizerJobs[0]?.id ?? null;
   352→      }
   353→    },
   354→
   355→    setActiveJob: (s, jobId: string | null) => {
   356→      const state = s as UiSliceState;
   357→      state.activeJobId = jobId;
   358→    },
   359→
   360→    clearCompletedJobs: (s) => {
   361→      const state = s as UiSliceState;
   362→      state.optimizerJobs = state.optimizerJobs.filter(
   363→        (j) => j.status !== 'COMPLETED' && j.status !== 'FAILED'
   364→      );
   365→    },
   366→
   367→    // ==========================================================================
   368→    // Connection Status
   369→    // ==========================================================================
   370→
   371→    setOnlineStatus: (s, isOnline: boolean) => {
   372→      const state = s as UiSliceState;
   373→      state.isOnline = isOnline;
   374→    },
   375→
   376→    setWebsocketConnected: (s, connected: boolean) => {
   377→      const state = s as UiSliceState;
   378→      state.websocketConnected = connected;
   379→    },
   380→
   381→    // ==========================================================================
   382→    // Sync Status
   383→    // ==========================================================================
   384→
   385→    startSync: (s) => {
   386→      const state = s as UiSliceState;
   387→      state.syncInProgress = true;
   388→    },
   389→
   390→    finishSync: (s) => {
   391→      const state = s as UiSliceState;
   392→      state.syncInProgress = false;
   393→      state.lastSyncAt = Date.now();
   394→    },
   395→
   396→    // ==========================================================================
   397→    // Reset
   398→    // ==========================================================================
   399→
   400→    reset: (s) => {
   401→      const state = s as UiSliceState;
   402→      Object.assign(state, initialUiState);
   403→    },
   404→  },
   405→});
   406→
   407→// =============================================================================
   408→// Selectors
   409→// =============================================================================
   410→
   411→/** Check if any loading is in progress */
   412→export function isAnyLoading(state: UiSliceState): boolean {
   413→  return Object.values(state.loading).some(Boolean);
   414→}
   415→
   416→/** Get active optimizer job */
   417→export function selectActiveJob(state: UiSliceState): OptimizerJob | undefined {
   418→  return state.activeJobId
   419→    ? state.optimizerJobs.find((j) => j.id === state.activeJobId)
   420→    : undefined;
   421→}
   422→
   423→/** Get running optimizer jobs */
   424→export function selectRunningJobs(state: UiSliceState): OptimizerJob[] {
   425→  return state.optimizerJobs.filter((j) => j.status === 'RUNNING' || j.status === 'PENDING');
   426→}
   427→
   428→/** Check if there's an error */
   429→export function hasError(state: UiSliceState): boolean {
   430→  return state.error.message !== null;
   431→}
   432→
   433→/** Get visible toasts */
   434→export function selectVisibleToasts(state: UiSliceState): Toast[] {
   435→  const now = Date.now();
   436→  return state.toasts.filter((t) => now - t.createdAt < t.duration);
   437→}
   438→
   439→/** Check connection status */
   440→export function isFullyConnected(state: UiSliceState): boolean {
   441→  return state.isOnline && state.websocketConnected;
   442→}
