     1→/**
     2→ * MMS Status Bar
     3→ *
     4→ * Bottom status bar showing baseline info, drift alerts, sync status.
     5→ * Based on specs/006-user-interface.md Section 6.2.
     6→ */
     7→
     8→import {
     9→  Icon,
    10→  Tag,
    11→  Intent,
    12→  Popover,
    13→  Menu,
    14→  MenuItem,
    15→  MenuDivider,
    16→  Classes,
    17→} from '@blueprintjs/core';
    18→import { format, formatDistanceToNow } from 'date-fns';
    19→
    20→import { useScenario, useUi } from '../../store';
    21→import { DRIFT_COLORS } from '../theme';
    22→import type { ScenarioSliceState } from '../../store/slices/scenario';
    23→import type { UiSliceState } from '../../store/slices/ui';
    24→
    25→export function StatusBar() {
    26→  const { state: scenarioStateRaw } = useScenario();
    27→  const { state: uiStateRaw } = useUi();
    28→
    29→  const scenarioState = scenarioStateRaw as unknown as ScenarioSliceState;
    30→  const uiState = uiStateRaw as unknown as UiSliceState;
    31→
    32→  // Compute drift summary
    33→  const trainsRequiringReplan = scenarioState.driftAnalysis.filter(d => d.requiresReplan).length;
    34→  const driftSeverity = trainsRequiringReplan > 10 ? 'critical' :
    35→                        trainsRequiringReplan > 5 ? 'warning' : 'ok';
    36→
    37→  // Active baseline info
    38→  const activeBaseline = scenarioState.baselines.find(b => b.id === scenarioState.activeBaselineId);
    39→
    40→  // Connection status
    41→  const isConnected = uiState.isOnline && uiState.websocketConnected;
    42→  const connectionIntent: Intent = isConnected ? 'success' : 'danger';
    43→
    44→  // Sync status
    45→  const lastSyncText = uiState.lastSyncAt
    46→    ? formatDistanceToNow(uiState.lastSyncAt, { addSuffix: true })
    47→    : 'Never';
    48→
    49→  // Active jobs
    50→  const runningJobs = uiState.optimizerJobs.filter(j => j.status === 'RUNNING');
    51→
    52→  const baselineMenu = (
    53→    <Menu>
    54→      <MenuItem disabled text={`Version: ${activeBaseline?.version || 'N/A'}`} />
    55→      <MenuItem disabled text={`Published: ${activeBaseline?.publishedAt ? format(new Date(activeBaseline.publishedAt), 'MMM d, yyyy') : 'N/A'}`} />
    56→      <MenuDivider />
    57→      <MenuItem icon="history" text="View Baseline History" />
    58→      <MenuItem icon="comparison" text="Compare to Previous" />
    59→    </Menu>
    60→  );
    61→
    62→  const driftMenu = (
    63→    <Menu>
    64→      {scenarioState.driftAnalysis.filter(d => d.requiresReplan).slice(0, 5).map(drift => (
    65→        <MenuItem
    66→          key={drift.trainId}
    67→          icon="warning-sign"
    68→          text={`Train ${drift.trainNumber}: ${drift.mileageDriftPct.toFixed(1)}% drift`}
    69→          intent="warning"
    70→        />
    71→      ))}
    72→      {trainsRequiringReplan > 5 && (
    73→        <MenuItem disabled text={`... and ${trainsRequiringReplan - 5} more`} />
    74→      )}
    75→      <MenuDivider />
    76→      <MenuItem icon="refresh" text="Refresh Drift Analysis" />
    77→      <MenuItem icon="document" text="View Full Report" />
    78→    </Menu>
    79→  );
    80→
    81→  const jobsMenu = (
    82→    <Menu>
    83→      {runningJobs.map(job => (
    84→        <MenuItem
    85→          key={job.id}
    86→          icon="refresh"
    87→          text={`${job.type}: ${job.progress}%`}
    88→          labelElement={<Tag minimal>{job.status}</Tag>}
    89→        />
    90→      ))}
    91→      {runningJobs.length === 0 && (
    92→        <MenuItem disabled text="No active jobs" />
    93→      )}
    94→      <MenuDivider />
    95→      <MenuItem icon="history" text="View Job History" />
    96→    </Menu>
    97→  );
    98→
    99→  return (
   100→    <div className="mms-status-bar" style={{
   101→      display: 'flex',
   102→      justifyContent: 'space-between',
   103→      alignItems: 'center',
   104→      padding: '6px 16px',
   105→      borderTop: '1px solid var(--pt-divider-black, #e1e8ed)',
   106→      backgroundColor: 'var(--pt-app-background-color, #f5f8fa)',
   107→      fontSize: '12px',
   108→    }}>
   109→      {/* Left Section - Baseline & Drift */}
   110→      <div style={{ display: 'flex', gap: '24px', alignItems: 'center' }}>
   111→        <Popover content={baselineMenu} placement="top-start" interactionKind="hover">
   112→          <div style={{ display: 'flex', alignItems: 'center', gap: '6px', cursor: 'pointer' }}>
   113→            <Icon icon="git-commit" size={12} />
   114→            <span className={Classes.TEXT_MUTED}>Baseline:</span>
   115→            <span style={{ fontWeight: 500 }}>
   116→              {activeBaseline ? `${activeBaseline.name} v${activeBaseline.version}` : 'No baseline'}
   117→            </span>
   118→          </div>
   119→        </Popover>
   120→
   121→        {trainsRequiringReplan > 0 && (
   122→          <Popover content={driftMenu} placement="top" interactionKind="hover">
   123→            <Tag
   124→              intent={driftSeverity === 'critical' ? 'danger' : driftSeverity === 'warning' ? 'warning' : 'success'}
   125→              icon="warning-sign"
   126→              interactive
   127→              style={{ cursor: 'pointer' }}
   128→            >
   129→              Drift: {trainsRequiringReplan} train{trainsRequiringReplan !== 1 ? 's' : ''} need attention
   130→            </Tag>
   131→          </Popover>
   132→        )}
   133→
   134→        {trainsRequiringReplan === 0 && (
   135→          <Tag intent="success" icon="tick-circle">
   136→            All trains on track
   137→          </Tag>
   138→        )}
   139→      </div>
   140→
   141→      {/* Center Section - Active Scenario */}
   142→      {scenarioState.activeScenarioId && (
   143→        <div style={{ display: 'flex', alignItems: 'center', gap: '6px' }}>
   144→          <Tag intent="warning" icon="fork">
   145→            Scenario: {scenarioState.scenarios.find(s => s.id === scenarioState.activeScenarioId)?.name || 'Active'}
   146→          </Tag>
   147→          <span className={Classes.TEXT_MUTED}>
   148→            {scenarioState.activeDeltas.length} change{scenarioState.activeDeltas.length !== 1 ? 's' : ''}
   149→          </span>
   150→        </div>
   151→      )}
   152→
   153→      {/* Right Section - Sync & Connection Status */}
   154→      <div style={{ display: 'flex', gap: '16px', alignItems: 'center' }}>
   155→        {/* Running Jobs */}
   156→        {runningJobs.length > 0 && (
   157→          <Popover content={jobsMenu} placement="top-end" interactionKind="hover">
   158→            <Tag intent="primary" icon="refresh" interactive>
   159→              {runningJobs.length} job{runningJobs.length !== 1 ? 's' : ''} running
   160→            </Tag>
   161→          </Popover>
   162→        )}
   163→
   164→        {/* Last Sync */}
   165→        <div style={{ display: 'flex', alignItems: 'center', gap: '6px' }}>
   166→          <Icon icon="refresh" size={12} color={uiState.syncInProgress ? '#137cbd' : undefined} />
   167→          <span className={Classes.TEXT_MUTED}>
   168→            {uiState.syncInProgress ? 'Syncing...' : `Last sync: ${lastSyncText}`}
   169→          </span>
   170→        </div>
   171→
   172→        {/* Connection Status */}
   173→        <div style={{ display: 'flex', alignItems: 'center', gap: '4px' }}>
   174→          <Icon
   175→            icon={isConnected ? 'cell-tower' : 'disable'}
   176→            size={12}
   177→            intent={connectionIntent}
   178→          />
   179→          <span style={{ color: isConnected ? DRIFT_COLORS.ok : DRIFT_COLORS.critical }}>
   180→            {isConnected ? 'Connected' : 'Offline'}
   181→          </span>
   182→        </div>
   183→      </div>
   184→    </div>
   185→  );
   186→}
