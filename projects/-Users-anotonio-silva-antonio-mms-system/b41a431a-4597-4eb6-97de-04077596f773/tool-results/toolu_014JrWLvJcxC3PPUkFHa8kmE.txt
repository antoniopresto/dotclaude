    95→/**
    96→ * Validate ISO date string (YYYY-MM-DD)
    97→ */
    98→function isValidDateString(date: string): boolean {
    99→  if (!/^\d{4}-\d{2}-\d{2}$/.test(date)) {
   100→    return false;
   101→  }
   102→  const parsed = new Date(date);
   103→  return !isNaN(parsed.getTime());
   104→}

   106→/**
   107→ * Get detailed job information including result
   108→ */
   109→async function getDetailedJob(
   110→  queueName: typeof QUEUE_NAMES[keyof typeof QUEUE_NAMES],
   111→  jobId: string
   112→): Promise<{
   113→  id: string;
   114→  status: OptimizerJobStatus;
   115→  progress?: unknown;
   116→  result?: unknown;
   117→  failedReason?: string;
   118→  startedAt?: string;
   119→  completedAt?: string;
   120→  attemptsMade?: number;
   121→} | null> {
   122→  const queue = getQueue(queueName);
   123→  const job = await queue.getJob(jobId);
   124→
   125→  if (!job) return null;
   126→
   127→  const state = await job.getState();
   128→  const result = job.returnvalue;
   129→  const failedReason = job.failedReason;
   130→  const progress = job.progress;
   131→  const processedOn = job.processedOn;
   132→  const finishedOn = job.finishedOn;
   133→
   134→  return {
   135→    id: job.id!,
   136→    status: (state as OptimizerJobStatus) || 'unknown',
   137→    ...(progress !== undefined && { progress }),
   138→    ...(result !== undefined && { result }),
   139→    ...(failedReason && { failedReason }),
   140→    ...(processedOn && { startedAt: new Date(processedOn).toISOString() }),
   141→    ...(finishedOn && { completedAt: new Date(finishedOn).toISOString() }),
   142→    attemptsMade: job.attemptsMade,
   143→  };
   144→}
   145→
   146→/**
   147→ * Estimate completion time based on job complexity
   148→ */
   149→function estimateCompletionSeconds(trainCount: number, routeCount: number): number {
   150→  // Rough estimate: base time + scaling factor
   151→  const baseTime = 10;
   152→  const scaleFactor = 0.5;
   153→  return Math.ceil(baseTime + (trainCount * routeCount * scaleFactor) / 1000);
   154→}

   160→/**
   161→ * POST /optimizer/run-assignment
   162→ *
   163→ * Trigger run assignment optimization for a specific date and fleet.
   164→ * The optimization is performed asynchronously and returns a job ID
   165→ * that can be used to check status and retrieve results.
   166→ */
   167→optimizerRoutes.post('/optimizer/run-assignment', async (c) => {
   168→  const requestId = getRequestId(c);
   169→
   170→  // Parse request body
   171→  let body: RunAssignmentRequest;
   172→  try {
   173→    body = await c.req.json();
   174→  } catch {
   175→    return c.json(
   176→      error(ErrorCodes.VALIDATION_ERROR, 'Invalid JSON body', requestId),
   177→      400
   178→    );
   179→  }
   180→
   181→  // Validate required fields
   182→  if (!body.date) {
   183→    return c.json(
   184→      error(ErrorCodes.VALIDATION_ERROR, 'date is required', requestId, {
   185→        field: 'date',
   186→      }),
   187→      400
   188→    );
   189→  }
   190→
   191→  if (!isValidDateString(body.date)) {
   192→    return c.json(
   193→      error(
   194→        ErrorCodes.VALIDATION_ERROR,
   195→        'Invalid date format. Expected YYYY-MM-DD',
   196→        requestId,
   197→        { field: 'date', value: body.date }
   198→      ),
   199→      400
   200→    );
   201→  }
   202→
   203→  if (!body.fleetId) {
   204→    return c.json(
   205→      error(ErrorCodes.VALIDATION_ERROR, 'fleetId is required', requestId, {
   206→        field: 'fleetId',
   207→      }),
   208→      400
   209→    );
   210→  }
   211→
   212→  if (!isValidUuid(body.fleetId)) {
   213→    return c.json(
   214→      error(
   215→        ErrorCodes.VALIDATION_ERROR,
   216→        'Invalid fleetId format. Expected UUID',
   217→        requestId,
   218→        { field: 'fleetId', value: body.fleetId }
   219→      ),
   220→      400
   221→    );
   222→  }
   223→
   224→  // Validate options if provided
   225→  const maxSolveTimeSeconds = body.options?.maxSolveTimeSeconds ?? 30;
   226→  if (maxSolveTimeSeconds < 1 || maxSolveTimeSeconds > 300) {
   227→    return c.json(
   228→      error(
   229→        ErrorCodes.VALIDATION_ERROR,
   230→        'maxSolveTimeSeconds must be between 1 and 300',
   231→        requestId,
   232→        { field: 'options.maxSolveTimeSeconds', value: maxSolveTimeSeconds }
   233→      ),
   234→      400
   235→    );
   236→  }
   237→
   238→  const validVerbosityLevels = ['MINIMAL', 'STANDARD', 'DETAILED'];
   239→  const verbosity = body.options?.explanationVerbosity ?? 'STANDARD';
   240→  if (!validVerbosityLevels.includes(verbosity)) {
   241→    return c.json(
   242→      error(
   243→        ErrorCodes.VALIDATION_ERROR,
   244→        `explanationVerbosity must be one of: ${validVerbosityLevels.join(', ')}`,
   245→        requestId,
   246→        { field: 'options.explanationVerbosity', value: verbosity }
   247→      ),
   248→      400
   249→    );
   250→  }
   251→
   252→  // Build job data
   253→  const jobData: OptimizerJobData = {
   254→    date: body.date,
   255→    fleetId: body.fleetId,
   256→    timeoutMs: maxSolveTimeSeconds * 1000,
   257→    ...(body.constraints && { constraints: body.constraints }),
   258→  };
   259→
   260→  // Add job to queue
   261→  const { id: jobId } = await addJob(
   262→    QUEUE_NAMES.OPTIMIZER,
   263→    'run-assignment',
   264→    jobData,
   265→    {
   266→      priority: 1, // High priority for run assignments
   267→    }
   268→  );
   269→
   270→  // Estimate completion time (this would be more accurate with fleet size info)
   271→  const estimatedCompletionSeconds = estimateCompletionSeconds(50, 10);
   272→
   273→  return c.json(
   274→    success(
   275→      {
   276→        jobId,
   277→        status: 'QUEUED',
   278→        estimatedCompletionSeconds,
   279→      },
   280→      {
   281→        timing: { durationMs: getRequestDuration(c) },
   282→      }
   283→    ),
   284→    202
   285→  );
   286→});

   288→/**
   289→ * GET /optimizer/jobs/:jobId
   290→ *
   291→ * Get the status and results of an optimizer job.
   292→ */
   293→optimizerRoutes.get('/optimizer/jobs/:jobId', async (c) => {
   294→  const requestId = getRequestId(c);
   295→  const jobId = c.req.param('jobId');
   296→
   297→  // Try to find job in optimizer queue first
   298→  let job = await getDetailedJob(QUEUE_NAMES.OPTIMIZER, jobId);
   299→
   300→  // If not found, try tactical scheduler queue
   301→  if (!job) {
   302→    job = await getDetailedJob(QUEUE_NAMES.TACTICAL_SCHEDULER, jobId);
   303→  }
   304→
   305→  // If not found, try strategic planner queue
   306→  if (!job) {
   307→    job = await getDetailedJob(QUEUE_NAMES.STRATEGIC_PLANNER, jobId);
   308→  }
   309→
   310→  if (!job) {
   311→    return c.json(
   312→      error(ErrorCodes.NOT_FOUND, `Job ${jobId} not found`, requestId),
   313→      404
   314→    );
   315→  }
   316→
   317→  // Map internal status to API status
   318→  const statusMap: Record<string, string> = {
   319→    waiting: 'QUEUED',
   320→    delayed: 'QUEUED',
   321→    active: 'PROCESSING',
   322→    completed: 'COMPLETED',
   323→    failed: 'FAILED',
   324→  };
   325→
   326→  const apiStatus = statusMap[job.status] || 'UNKNOWN';
   327→
   328→  // Build response based on job status
   329→  interface JobResponse {
   330→    jobId: string;
   331→    status: string;
   332→    startedAt?: string;
   333→    completedAt?: string;
   334→    progress?: unknown;
   335→    result?: unknown;
   336→    error?: {
   337→      message: string;
   338→      attemptsMade?: number;
   339→    };
   340→  }
   341→
   342→  const response: JobResponse = {
   343→    jobId: job.id,
   344→    status: apiStatus,
   345→    ...(job.startedAt && { startedAt: job.startedAt }),
   346→    ...(job.completedAt && { completedAt: job.completedAt }),
   347→  };
   348→
   349→  // Include progress for active jobs
   350→  if (job.status === 'active' && job.progress) {
   351→    response.progress = job.progress;
   352→  }
   353→
   354→  // Include result for completed jobs
   355→  if (job.status === 'completed' && job.result) {
   356→    response.result = job.result;
   357→  }
   358→
   359→  // Include error for failed jobs
   360→  if (job.status === 'failed' && job.failedReason) {
   361→    response.error = {
   362→      message: job.failedReason,
   363→      attemptsMade: job.attemptsMade,
   364→    };
   365→  }
   366→
   367→  return c.json(
   368→    success(response, {
   369→      timing: { durationMs: getRequestDuration(c) },
   370→    })
   371→  );
   372→});
