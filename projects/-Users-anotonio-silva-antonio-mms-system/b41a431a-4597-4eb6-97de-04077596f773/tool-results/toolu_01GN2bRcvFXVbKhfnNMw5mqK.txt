   639→```typescript
   640→// Schema validation (replaces Zod)
   641→import { createSchema, createType, InferType } from '@belt/tools';
   642→
   643→const TrainModel = createSchema({
   644→  id: 'string',
   645→  trainNumber: 'string',
   646→  mileageKm: 'number',
   647→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   648→  tags: 'string[]',
   649→  metadata: 'object?',
   650→});
   651→type Train = InferType<typeof TrainModel>;
   652→const train = TrainModel.parse(data);
   653→
   654→// State machine (replaces React hooks + zustand)
   655→import { createSlice, createMachine } from '@belt/tools';
   656→
   657→const fleetSlice = createSlice({
   658→  id: 'fleet',
   659→  initial: { trains: [], selected: null },
   660→  cache: { policy: 'session' },
   661→  reducers: {
   662→    setTrains: (state, trains) => { state.trains = trains; },
   663→    selectTrain: (state, id) => { state.selected = id; },
   664→  },
   665→});
   666→
   667→const machine = createMachine({
   668→  id: 'mms',
   669→  slices: { fleet: fleetSlice },
   670→  trackMetadata: false, // Critical for 60fps
   671→});
   672→
   673→// In React component (NO useState/useEffect!)
   674→const { state, actions } = machine.useMachine('fleet');
