     1→/**
     2→ * MMS Service Demand Service
     3→ *
     4→ * Manages fleet availability requirements - a CRITICAL v2.1 service.
     5→ * Service demand is a HARD CONSTRAINT for the optimizer: schedules that
     6→ * violate minimum train availability must be rejected.
     7→ *
     8→ * Key responsibilities:
     9→ * - Define minimum train requirements by fleet, day type, and time period
    10→ * - Validate that proposed PM schedules don't violate availability
    11→ * - Calculate default demand (85% of fleet) when no explicit config exists
    12→ * - Support peak period requirements (AM, midday, PM, evening)
    13→ *
    14→ * Business rules:
    15→ * - Optimizer CANNOT schedule PM that would drop fleet below minTrainsRequired
    16→ * - Day types: WEEKDAY, SATURDAY, SUNDAY, HOLIDAY
    17→ * - Demand records are versioned by effectiveDate (most recent applies)
    18→ * - Peak period requirements are optional but enforced if defined
    19→ *
    20→ * @see specs/003-database-schema.md Section 3.2.1 (Service Demand)
    21→ * @see specs/004-domain-services.md Section 4.2.6 (Service Demand Service)
    22→ */

   174→  /**
   175→   * Get demand configuration for a fleet on a specific date.
   176→   *
   177→   * This is the primary method for the optimizer to determine minimum
   178→   * train requirements. It:
   179→   * 1. Determines the day type for the date
   180→   * 2. Finds the most recent effective demand config
   181→   * 3. Falls back to 85% of fleet size if no config exists
   182→   *
   183→   * @param fleetId - Fleet to get demand for
   184→   * @param targetDate - Date to check (YYYY-MM-DD)
   185→   * @param holidays - Optional array of holiday dates (YYYY-MM-DD)
   186→   * @returns Demand configuration for the date
   187→   */
   188→  async getDemand(
   189→    fleetId: string,
   190→    targetDate: string,
   191→    holidays?: string[]
   192→  ): Promise<DemandConfig> {
   193→    return this.traced('getDemand', async (span) => {
   194→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   195→      span.setAttribute('demand.targetDate', targetDate);
   196→
   197→      // Verify fleet exists
   198→      const fleet = await this.db.query.fleets.findFirst({
   199→        where: eq(fleets.id, fleetId),
   200→      });
   201→
   202→      if (!fleet) {
   203→        throw new NotFoundError(`Fleet not found: ${fleetId}`, {
   204→          entityType: 'fleet',
   205→          entityId: fleetId,
   206→        });
   207→      }
   208→
   209→      // Determine day type
   210→      const dayType = this.getDayType(targetDate, holidays);
   211→      span.setAttribute('demand.dayType', dayType);
   212→
   213→      // Find most recent effective demand config
   214→      const demandRecord = await this.db.query.serviceDemand.findFirst({
   215→        where: and(
   216→          eq(serviceDemand.fleetId, fleetId),
   217→          eq(serviceDemand.dayType, dayType),
   218→          lte(serviceDemand.effectiveDate, targetDate)
   219→        ),
   220→        orderBy: desc(serviceDemand.effectiveDate),
   221→      });
   222→
   223→      if (demandRecord) {
   224→        span.setAttribute('demand.source', 'CONFIG');
   225→        span.setAttribute('demand.minTrains', demandRecord.minTrainsRequired);
   226→
   227→        return {
   228→          fleetId,
   229→          fleetCode: fleet.code,
   230→          date: targetDate,
   231→          dayType,
   232→          minTrainsRequired: demandRecord.minTrainsRequired,
   233→          standbyBuffer: demandRecord.standbyBuffer ?? DEFAULT_STANDBY_BUFFER,
   234→          totalRequired:
   235→            demandRecord.minTrainsRequired + (demandRecord.standbyBuffer ?? DEFAULT_STANDBY_BUFFER),
   236→          peakAmRequirement: demandRecord.peakAmRequirement,
   237→          middayRequirement: demandRecord.middayRequirement,
   238→          peakPmRequirement: demandRecord.peakPmRequirement,
   239→          eveningRequirement: demandRecord.eveningRequirement,
   240→          source: 'CONFIG',
   241→          sourceRecord: demandRecord,
   242→        };
   243→      }
   244→
   245→      // No config found - calculate default (85% of fleet)
   246→      const defaultDemand = await this.calculateDefaultDemand(fleetId);
   247→      span.setAttribute('demand.source', 'DEFAULT');
   248→      span.setAttribute('demand.minTrains', defaultDemand.minTrainsRequired);
   249→
   250→      this.log.warn(
   251→        {
   252→          fleetId,
   253→          fleetCode: fleet.code,
   254→          targetDate,
   255→          dayType,
   256→          calculatedDemand: defaultDemand.minTrainsRequired,
   257→        },
   258→        `No service demand config found for ${fleet.code} on ${targetDate} (${dayType}). Using default 85% of fleet.`
   259→      );
   260→
   261→      return {
   262→        fleetId,
   263→        fleetCode: fleet.code,
   264→        date: targetDate,
   265→        dayType,
   266→        minTrainsRequired: defaultDemand.minTrainsRequired,
   267→        standbyBuffer: DEFAULT_STANDBY_BUFFER,
   268→        totalRequired: defaultDemand.minTrainsRequired + DEFAULT_STANDBY_BUFFER,
   269→        peakAmRequirement: null,
   270→        middayRequirement: null,
   271→        peakPmRequirement: null,
   272→        eveningRequirement: null,
   273→        source: 'DEFAULT',
   274→        sourceRecord: null,
   275→      };
   276→    });
   277→  }

   279→  /**
   280→   * Get demand for a specific time period within a day.
   281→   *
   282→   * Used for more granular availability checking during peak hours.
   283→   */
   284→  async getDemandForPeriod(
   285→    fleetId: string,
   286→    targetDate: string,
   287→    period: TimePeriod,
   288→    holidays?: string[]
   289→  ): Promise<number> {
   290→    return this.traced('getDemandForPeriod', async (span) => {
   291→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   292→      span.setAttribute('demand.period', period);
   293→
   294→      const demandConfig = await this.getDemand(fleetId, targetDate, holidays);
   295→
   296→      if (period === 'ALL_DAY') {
   297→        return demandConfig.totalRequired;
   298→      }
   299→
   300→      // Check period-specific requirement, fall back to base requirement
   301→      const periodRequirement = {
   302→        PEAK_AM: demandConfig.peakAmRequirement,
   303→        MIDDAY: demandConfig.middayRequirement,
   304→        PEAK_PM: demandConfig.peakPmRequirement,
   305→        EVENING: demandConfig.eveningRequirement,
   306→      }[period];
   307→
   308→      return periodRequirement ?? demandConfig.minTrainsRequired;
   309→    });
   310→  }

   316→  /**
   317→   * Validate fleet availability for a date.
   318→   *
   319→   * This is a CRITICAL method for the optimizer. It checks whether the
   320→   * fleet has enough available trains to meet service demand, considering:
   321→   * - Current train fitness status
   322→   * - Active unavailability events
   323→   * - Scheduled PM on that date
   324→   * - Any additional proposed unavailability
   325→   *
   326→   * @param fleetId - Fleet to validate
   327→   * @param targetDate - Date to check
   328→   * @param additionalUnavailable - Number of additional trains that would be unavailable (e.g., proposed PM)
   329→   * @param holidays - Optional array of holiday dates
   330→   * @returns Validation result with detailed breakdown
   331→   */
   332→  async validateAvailability(
   333→    fleetId: string,
   334→    targetDate: string,
   335→    additionalUnavailable: number = 0,
   336→    holidays?: string[]
   337→  ): Promise<AvailabilityValidation> {
   338→    return this.traced('validateAvailability', async (span) => {
   339→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   340→      span.setAttribute('availability.targetDate', targetDate);
   341→      span.setAttribute('availability.additionalUnavailable', additionalUnavailable);
   342→
   343→      // Get demand config
   344→      const demandConfig = await this.getDemand(fleetId, targetDate, holidays);
   345→
   346→      // Count total and FIT trains in fleet
   347→      const trainCounts = await this.db
   348→        .select({
   349→          total: count(),
   350→          fit: sql<number>`COUNT(*) FILTER (WHERE ${trains.fitnessStatus} = 'FIT')`,
   351→          restricted: sql<number>`COUNT(*) FILTER (WHERE ${trains.fitnessStatus} = 'RESTRICTED')`,
   352→          grounded: sql<number>`COUNT(*) FILTER (WHERE ${trains.fitnessStatus} = 'GROUNDED')`,
   353→        })
   354→        .from(trains)
   355→        .where(and(eq(trains.fleetId, fleetId), eq(trains.active, true)));
   356→
   357→      const { total, fit, restricted, grounded } = trainCounts[0];
   358→      const unavailableTrains = restricted + grounded;
   359→
   360→      // Count trains in maintenance on that date
   361→      const pmCounts = await this.db
   362→        .select({ inMaintenance: count() })
   363→        .from(pmSchedule)
   364→        .innerJoin(trains, eq(pmSchedule.trainId, trains.id))
   365→        .where(
   366→          and(
   367→            eq(trains.fleetId, fleetId),
   368→            eq(pmSchedule.targetDate, targetDate),
   369→            inArray(pmSchedule.status, ['SCHEDULED', 'IN_PROGRESS']),
   370→            sql`${pmSchedule.scenarioId} IS NULL` // Production only
   371→          )
   372→        );
   373→
   374→      const inMaintenanceTrains = pmCounts[0]?.inMaintenance ?? 0;
   375→
   376→      // Calculate available
   377→      const availableTrains = fit - inMaintenanceTrains - additionalUnavailable;
   378→      const required = demandConfig.totalRequired;
   379→      const surplus = availableTrains - required;
   380→
   381→      // Determine severity
   382→      let severity: 'OK' | 'WARNING' | 'VIOLATION';
   383→      let message: string;
   384→
   385→      if (surplus < 0) {
   386→        severity = 'VIOLATION';
   387→        message = `HARD CONSTRAINT VIOLATED: ${demandConfig.fleetCode} needs ${required} trains but only ${availableTrains} available (shortfall: ${Math.abs(surplus)})`;
   388→      } else if (surplus < total * WARNING_THRESHOLD_PCT) {
   389→        severity = 'WARNING';
   390→        message = `Low availability margin: ${demandConfig.fleetCode} has ${surplus} surplus trains (${((availableTrains / required) * 100).toFixed(1)}% of requirement)`;
   391→      } else {
   392→        severity = 'OK';
   393→        message = `${demandConfig.fleetCode} availability OK: ${availableTrains} available, ${required} required, ${surplus} surplus`;
   394→      }
   395→
   396→      const result: AvailabilityValidation = {
   397→        isValid: surplus >= 0,
   398→        fleetId,
   399→        date: targetDate,
   400→        dayType: demandConfig.dayType,
   401→        totalTrains: total,
   402→        fitTrains: fit,
   403→        unavailableTrains,
   404→        inMaintenanceTrains,
   405→        additionalUnavailable,
   406→        availableTrains,
   407→        required,
   408→        surplus,
   409→        severity,
   410→        message,
   411→      };
   412→
   413→      span.setAttribute('availability.isValid', result.isValid);
   414→      span.setAttribute('availability.surplus', surplus);
   415→      span.setAttribute('availability.severity', severity);
   416→
   417→      if (severity === 'VIOLATION') {
   418→        this.log.error(
   419→          {
   420→            fleetId,
   421→            fleetCode: demandConfig.fleetCode,
   422→            targetDate,
   423→            required,
   424→            availableTrains,
   425→            surplus,
   426→          },
   427→          message
   428→        );
   429→      } else if (severity === 'WARNING') {
   430→        this.log.warn(
   431→          {
   432→            fleetId,
   433→            fleetCode: demandConfig.fleetCode,
   434→            targetDate,
   435→            required,
   436→            availableTrains,
   437→            surplus,
   438→          },
   439→          message
   440→        );
   441→      }
   442→
   443→      return result;
   444→    });
   445→  }

   496→  /**
   497→   * Find all service demand violations for a set of proposed PMs.
   498→   *
   499→   * Used by the optimizer to validate a proposed schedule before committing.
   500→   * Returns detailed violation information for each date/train combination.
   501→   *
   502→   * @param fleetId - Fleet to check
   503→   * @param proposedPMs - Array of proposed PM schedules
   504→   * @param holidays - Optional array of holiday dates
   505→   * @returns Array of violations (empty if no violations)
   506→   */
   507→  async findViolations(
   508→    fleetId: string,
   509→    proposedPMs: ProposedPM[],
   510→    holidays?: string[]
   511→  ): Promise<DemandViolation[]> {
   512→    return this.traced('findViolations', async (span) => {
   513→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   514→      span.setAttribute('violations.proposedCount', proposedPMs.length);
   515→
   516→      if (proposedPMs.length === 0) {
   517→        return [];
   518→      }
   519→
   520→      // Group proposed PMs by date
   521→      const pmsByDate = new Map<string, ProposedPM[]>();
   522→      for (const pm of proposedPMs) {
   523→        const existing = pmsByDate.get(pm.targetDate) ?? [];
   524→        existing.push(pm);
   525→        pmsByDate.set(pm.targetDate, existing);
   526→      }
   527→
   528→      // Get train details for error messages
   529→      const trainIds = [...new Set(proposedPMs.map((p) => p.trainId))];
   530→      const trainMap = new Map<string, { trainNumber: string }>();
   531→
   532→      if (trainIds.length > 0) {
   533→        const trainRecords = await this.db.query.trains.findMany({
   534→          where: inArray(trains.id, trainIds),
   535→          columns: { id: true, trainNumber: true },
   536→        });
   537→        for (const t of trainRecords) {
   538→          trainMap.set(t.id, { trainNumber: t.trainNumber });
   539→        }
   540→      }
   541→
   542→      // Check each date
   543→      const violations: DemandViolation[] = [];
   544→
   545→      for (const [date, pms] of pmsByDate) {
   546→        const validation = await this.validateAvailability(
   547→          fleetId,
   548→          date,
   549→          pms.length, // All proposed PMs on this date
   550→          holidays
   551→        );
   552→
   553→        if (!validation.isValid) {
   554→          // Create violation for each PM on this date
   555→          for (const pm of pms) {
   556→            const trainInfo = trainMap.get(pm.trainId);
   557→            violations.push({
   558→              date,
   559→              dayType: validation.dayType,
   560→              trainId: pm.trainId,
   561→              trainNumber: trainInfo?.trainNumber ?? 'unknown',
   562→              pmCode: pm.pmCode ?? 'unknown',
   563→              required: validation.required,
   564→              available: validation.availableTrains + pms.length, // What would be available without these PMs
   565→              shortfall: Math.abs(validation.surplus),
   566→              severity: 'HARD',
   567→              message: `Scheduling ${pm.pmCode ?? 'PM'} for train ${trainInfo?.trainNumber ?? pm.trainId} on ${date} would violate service demand (need ${validation.required}, would have ${validation.availableTrains})`,
   568→            });
   569→          }
   570→        }
   571→      }
   572→
   573→      span.setAttribute('violations.count', violations.length);
   574→
   575→      if (violations.length > 0) {
   576→        this.log.warn(
   577→          {
   578→            fleetId,
   579→            proposedPMs: proposedPMs.length,
   580→            violations: violations.length,
   581→            violatedDates: [...new Set(violations.map((v) => v.date))],
   582→          },
   583→          `Found ${violations.length} service demand violations for proposed PM schedule`
   584→        );
   585→      }
   586→
   587→      return violations;
   588→    });
   589→  }

   591→  /**
   592→   * Find violations for a date range (horizon planning).
   593→   *
   594→   * Checks existing scheduled PMs against service demand for an entire
   595→   * horizon period. Used for drift monitoring and planning validation.
   596→   *
   597→   * @param fleetId - Fleet to check
   598→   * @param startDate - Start of range (YYYY-MM-DD)
   599→   * @param horizonDays - Number of days to check
   600→   * @param holidays - Optional array of holiday dates
   601→   * @returns Array of dates with violations
   602→   */
   603→  async findViolationsInHorizon(
   604→    fleetId: string,
   605→    startDate: string,
   606→    horizonDays: number,
   607→    holidays?: string[]
   608→  ): Promise<{ date: string; validation: AvailabilityValidation }[]> {
   609→    return this.traced('findViolationsInHorizon', async (span) => {
   610→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   611→      span.setAttribute('horizon.startDate', startDate);
   612→      span.setAttribute('horizon.days', horizonDays);
   613→
   614→      const violations: { date: string; validation: AvailabilityValidation }[] = [];
   615→      const start = new Date(startDate);
   616→
   617→      for (let i = 0; i < horizonDays; i++) {
   618→        const checkDate = new Date(start);
   619→        checkDate.setDate(checkDate.getDate() + i);
   620→        const dateStr = checkDate.toISOString().split('T')[0];
   621→
   622→        const validation = await this.validateAvailability(
   623→          fleetId,
   624→          dateStr,
   625→          0, // Check existing schedule only
   626→          holidays

   898→  async getFleetAvailabilitySummary(
   899→    fleetId: string,
   900→    holidays?: string[]
   901→  ): Promise<FleetAvailabilitySummary> {
   902→    return this.traced('getFleetAvailabilitySummary', async (span) => {
   903→      span.setAttribute(MmsAttributes.FLEET_ID, fleetId);
   904→
   905→      const today = new Date().toISOString().split('T')[0];
   906→      const demandConfig = await this.getDemand(fleetId, today, holidays);
   907→      const validation = await this.validateAvailability(fleetId, today, 0, holidays);
   908→
   909→      // Get detailed counts
   910→      const trainCounts = await this.db
   911→        .select({
   912→          fit: sql<number>`COUNT(*) FILTER (WHERE ${trains.fitnessStatus} = 'FIT')`,
   913→          restricted: sql<number>`COUNT(*) FILTER (WHERE ${trains.fitnessStatus} = 'RESTRICTED')`,
   914→          grounded: sql<number>`COUNT(*) FILTER (WHERE ${trains.fitnessStatus} = 'GROUNDED')`,
   915→        })
   916→        .from(trains)
   917→        .where(and(eq(trains.fleetId, fleetId), eq(trains.active, true)));
   918→
   919→      const { fit, restricted, grounded } = trainCounts[0];
   920→
   921→      // Determine status
   922→      let status: 'OK' | 'WARNING' | 'CRITICAL';
   923→      if (!validation.isValid) {
   924→        status = 'CRITICAL';
   925→      } else if (validation.severity === 'WARNING') {
   926→        status = 'WARNING';
   927→      } else {
   928→        status = 'OK';
   929→      }
   930→
   931→      return {
   932→        fleetId,
   933→        fleetCode: demandConfig.fleetCode,
   934→        totalTrains: validation.totalTrains,
   935→        fitTrains: fit,
   936→        restrictedTrains: restricted,
   937→        groundedTrains: grounded,
   938→        inMaintenanceTrains: validation.inMaintenanceTrains,
   939→        availableTrains: validation.availableTrains,
   940→        demandConfig,
   941→        utilizationPct:
   942→          validation.availableTrains > 0
   943→            ? (demandConfig.totalRequired / validation.availableTrains) * 100
   944→            : 0,
   945→        status,
   946→      };
   947→    });
   948→  }

   979→  /**
   980→   * Determine day type for a given date.
   981→   *
   982→   * @param dateStr - Date string (YYYY-MM-DD)
   983→   * @param holidays - Optional array of holiday dates
   984→   * @returns Day type enum value
   985→   */
   986→  getDayType(dateStr: string, holidays?: string[]): DayType {
   987→    // Check if it's a holiday first
   988→    if (holidays?.includes(dateStr)) {
   989→      return 'HOLIDAY';
   990→    }
   991→
   992→    const date = new Date(dateStr);
   993→    const dayOfWeek = date.getDay();
   994→
   995→    if (dayOfWeek === 0) return 'SUNDAY';
   996→    if (dayOfWeek === 6) return 'SATURDAY';
   997→    return 'WEEKDAY';
   998→  }
