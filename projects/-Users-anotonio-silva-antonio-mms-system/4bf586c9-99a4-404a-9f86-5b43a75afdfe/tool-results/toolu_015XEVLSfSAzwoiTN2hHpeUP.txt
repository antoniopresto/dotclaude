   349→#### Reference: @belt/tools Patterns
   350→```typescript
   351→// Schema validation (replaces Zod)
   352→import { createSchema, createType, InferType } from '@belt/tools';
   353→
   354→const TrainModel = createSchema({
   355→  id: 'string',
   356→  trainNumber: 'string',
   357→  mileageKm: 'number',
   358→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   359→  tags: 'string[]',
   360→  metadata: 'object?',
   361→});
   362→type Train = InferType<typeof TrainModel>;
   363→const train = TrainModel.parse(data);
   364→
   365→// State machine (replaces React hooks + zustand)
   366→import { createSlice, createMachine } from '@belt/tools';
   367→
   368→const fleetSlice = createSlice({
   369→  id: 'fleet',
   370→  initial: { trains: [], selected: null },
   371→  cache: { policy: 'session' },
   372→  reducers: {
   373→    setTrains: (state, trains) => { state.trains = trains; },
   374→    selectTrain: (state, id) => { state.selected = id; },
   375→  },
   376→});
   377→
