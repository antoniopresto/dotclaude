    63→/**
    64→ * LOD Transition Configuration
    65→ *
    66→ * Defines blend zones around LOD thresholds for smooth transitions.
    67→ * During blend zones, both adjacent layers render with interpolated opacity.
    68→ */
    69→const LOD_THRESHOLDS = {
    70→  macroToMeso: 0.5,  // Below 0.5 = MACRO
    71→  mesoToMicro: 2.0,  // Above 2.0 = MICRO
    72→};
    73→
    74→// Size of blend zone around each threshold (±this value)
    75→const BLEND_ZONE_SIZE = 0.15;
    76→
    77→/**
    78→ * Calculate layer opacities for smooth LOD transitions
    79→ *
    80→ * Returns opacity values for each layer (0-1).
    81→ * During blend zones, adjacent layers overlap with interpolated opacity.
    82→ */
    83→function calculateLayerOpacities(zoom: number): {
    84→  macro: number;
    85→  meso: number;
    86→  micro: number;
    87→} {
    88→  const { macroToMeso, mesoToMicro } = LOD_THRESHOLDS;
    89→
    90→  // Define blend zone boundaries
    91→  const macroEnd = macroToMeso - BLEND_ZONE_SIZE; // 0.35
    92→  const mesoStart = macroToMeso + BLEND_ZONE_SIZE; // 0.65
    93→  const mesoEnd = mesoToMicro - BLEND_ZONE_SIZE; // 1.85
    94→  const microStart = mesoToMicro + BLEND_ZONE_SIZE; // 2.15
    95→
    96→  let macro = 0;
    97→  let meso = 0;
    98→  let micro = 0;
    99→
   100→  // Calculate MACRO opacity
   101→  if (zoom <= macroEnd) {
   102→    macro = 1;
   103→  } else if (zoom < mesoStart) {
   104→    // Blend zone MACRO → MESO
   105→    const t = (zoom - macroEnd) / (mesoStart - macroEnd);
   106→    macro = 1 - t;
   107→  }
   108→
   109→  // Calculate MESO opacity
   110→  if (zoom < macroEnd) {
   111→    meso = 0;
   112→  } else if (zoom < mesoStart) {
   113→    // Blend zone MACRO → MESO
   114→    const t = (zoom - macroEnd) / (mesoStart - macroEnd);
   115→    meso = t;
   116→  } else if (zoom <= mesoEnd) {
   117→    meso = 1;
   118→  } else if (zoom < microStart) {
   119→    // Blend zone MESO → MICRO
   120→    const t = (zoom - mesoEnd) / (microStart - mesoEnd);
   121→    meso = 1 - t;
   122→  }
   123→
   124→  // Calculate MICRO opacity
   125→  if (zoom <= mesoEnd) {
   126→    micro = 0;
   127→  } else if (zoom < microStart) {
   128→    // Blend zone MESO → MICRO
   129→    const t = (zoom - mesoEnd) / (microStart - mesoEnd);
   130→    micro = t;
   131→  } else {
   132→    micro = 1;
   133→  }
   134→
   135→  return { macro, meso, micro };
   136→}

   138→/**
   139→ * LODLayers - Orchestrates layer rendering with smooth transitions
   140→ *
   141→ * Renders multiple layers simultaneously during blend zones,
   142→ * with opacity interpolation for smooth visual transitions.
   143→ */
   144→function LODLayers({
   145→  tasks,
   146→  trainCount,
   147→  zoom,
   148→  viewportBounds,
   149→  hoveredTaskId,
   150→  selectedTaskId,
   151→  focusedTaskId,
   152→}: {
   153→  tasks: ReturnType<typeof getRenderData>['tasks'];
   154→  trainCount: number;
   155→  zoom: number;
   156→  viewportBounds: ViewportBounds;
   157→  hoveredTaskId: string | null;
   158→  selectedTaskId: string | null;
   159→  focusedTaskId: string | null;
   160→}) {
   161→  const opacities = useMemo(() => calculateLayerOpacities(zoom), [zoom]);
   162→
   163→  return (
   164→    <group>
   165→      {/* Render each layer if opacity > 0 */}
   166→      {opacities.macro > 0 && (
   167→        <MacroLayer
   168→          tasks={tasks}
   169→          trainCount={trainCount}
   170→          opacity={opacities.macro}
   171→        />
   172→      )}
   173→      {opacities.meso > 0 && (
   174→        <MesoLayer
   175→          tasks={tasks}
   176→          trainCount={trainCount}
   177→          opacity={opacities.meso}
   178→          hoveredTaskId={hoveredTaskId}
   179→          selectedTaskId={selectedTaskId}
   180→          focusedTaskId={focusedTaskId}
   181→        />
   182→      )}
   183→      {opacities.micro > 0 && (
   184→        <MicroLayer
   185→          tasks={tasks}
   186→          trainCount={trainCount}
   187→          opacity={opacities.micro}
   188→          viewportBounds={viewportBounds}
   189→          hoveredTaskId={hoveredTaskId}
   190→          selectedTaskId={selectedTaskId}
   191→          focusedTaskId={focusedTaskId}
   192→        />
   193→      )}
   194→    </group>
   195→  );
   196→}

   342→/**
   343→ * Get tasks visible within the current viewport
   344→ * Sorted by train index (Y) then start day (X) for logical tab order
   345→ */
   346→function getVisibleTasks(
   347→  tasks: ReturnType<typeof getRenderData>['tasks'],
   348→  viewportBounds: ViewportBounds
   349→): ReturnType<typeof getRenderData>['tasks'] {
   350→  const visibleTasks = tasks.filter((task) => {
   351→    const taskStartX = task.startDay * pxPerDay;
   352→    const taskEndX = taskStartX + task.duration * pxPerDay;
   353→    const taskY = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
   354→
   355→    // Check if task overlaps with viewport
   356→    const xOverlap = taskEndX >= viewportBounds.minX && taskStartX <= viewportBounds.maxX;
   357→    const yOverlap = taskY >= viewportBounds.minY - pxPerTrain && taskY <= viewportBounds.maxY + pxPerTrain;
   358→
   359→    return xOverlap && yOverlap;
   360→  });
   361→
   362→  // Sort by train index (row), then by start day (column) for natural reading order
   363→  return visibleTasks.sort((a, b) => {
   364→    if (a.trainIndex !== b.trainIndex) {
   365→      return a.trainIndex - b.trainIndex;
   366→    }
   367→    return a.startDay - b.startDay;
   368→  });
   369→}

   371→/**
   372→ * Find next task in a specific direction from the current focused task
   373→ * Direction: 'left' | 'right' | 'up' | 'down'
   374→ */
   375→function findNextTaskInDirection(
   376→  tasks: ReturnType<typeof getRenderData>['tasks'],
   377→  currentTaskId: string,
   378→  direction: 'left' | 'right' | 'up' | 'down'
   379→): string | null {
   380→  const currentTask = tasks.find((t) => t.id === currentTaskId);
   381→  if (!currentTask) return null;
   382→
   383→  const currentX = currentTask.startDay * pxPerDay + (currentTask.duration * pxPerDay) / 2;
   384→  const currentY = -currentTask.trainIndex;
   385→
   386→  let bestMatch: typeof currentTask | null = null;
   387→  let bestScore = Infinity;
   388→
   389→  for (const task of tasks) {
   390→    if (task.id === currentTaskId) continue;
   391→
   392→    const taskX = task.startDay * pxPerDay + (task.duration * pxPerDay) / 2;
   393→    const taskY = -task.trainIndex;
   394→
   395→    let isInDirection = false;
   396→    let score = 0;
   397→
   398→    switch (direction) {
   399→      case 'left':
   400→        isInDirection = taskX < currentX;
   401→        if (isInDirection) {
   402→          // Prefer same row, then closest to the left
   403→          const rowPenalty = Math.abs(taskY - currentY) * 10000;
   404→          score = (currentX - taskX) + rowPenalty;
   405→        }
   406→        break;
   407→      case 'right':
   408→        isInDirection = taskX > currentX;
   409→        if (isInDirection) {
   410→          const rowPenalty = Math.abs(taskY - currentY) * 10000;
   411→          score = (taskX - currentX) + rowPenalty;
   412→        }
   413→        break;
   414→      case 'up':
   415→        isInDirection = taskY > currentY;
   416→        if (isInDirection) {
   417→          // Prefer closest row above with similar X
   418→          const xPenalty = Math.abs(taskX - currentX) / 100;
   419→          score = (taskY - currentY) + xPenalty;
   420→        }
   421→        break;
   422→      case 'down':
   423→        isInDirection = taskY < currentY;
   424→        if (isInDirection) {
   425→          const xPenalty = Math.abs(taskX - currentX) / 100;
   426→          score = (currentY - taskY) + xPenalty;
   427→        }
   428→        break;
   429→    }
   430→
   431→    if (isInDirection && score < bestScore) {
   432→      bestScore = score;
   433→      bestMatch = task;
   434→    }
   435→  }
   436→
   437→  return bestMatch?.id || null;
   438→}

   440→/**
   441→ * Find task at screen position (hit testing for hover/click)
   442→ *
   443→ * Converts screen coordinates to world coordinates and finds
   444→ * which task (if any) is under the cursor.
   445→ *
   446→ * @param screenX - Screen X coordinate (e.g., clientX)
   447→ * @param screenY - Screen Y coordinate (e.g., clientY)
   448→ * @param containerRect - Container element bounding rect
   449→ * @param cameraX - Camera X position in world coordinates
   450→ * @param cameraY - Camera Y position in world coordinates
   451→ * @param zoom - Current zoom level
   452→ * @param canvasWidth - Canvas width in pixels
   453→ * @param canvasHeight - Canvas height in pixels
   454→ * @param tasks - Array of tasks to test against
   455→ * @returns Task ID if found, null otherwise
   456→ */
   457→function findTaskAtPosition(
   458→  screenX: number,
   459→  screenY: number,
   460→  containerRect: DOMRect,
   461→  cameraX: number,
   462→  cameraY: number,
   463→  zoom: number,
   464→  canvasWidth: number,
   465→  canvasHeight: number,
   466→  tasks: ReturnType<typeof getRenderData>['tasks']
   467→): string | null {
   468→  // Convert screen coords to canvas coords (relative to container)
   469→  const canvasX = screenX - containerRect.left;
   470→  const canvasY = screenY - containerRect.top;
   471→
   472→  // Calculate half viewport in world units
   473→  const halfViewX = canvasWidth / 2 / zoom;
   474→  const halfViewY = canvasHeight / 2 / zoom;
   475→
   476→  // Convert canvas coords to world coords
   477→  // canvasX=0 is left edge of viewport, which is cameraX - halfViewX in world
   478→  // canvasY=0 is top edge of viewport, which is cameraY + halfViewY in world
   479→  const worldX = cameraX - halfViewX + (canvasX / zoom);
   480→  const worldY = cameraY + halfViewY - (canvasY / zoom);
   481→
   482→  // Find task at world position
   483→  for (const task of tasks) {
   484→    const taskStartX = task.startDay * pxPerDay;
   485→    const taskEndX = taskStartX + task.duration * pxPerDay;
   486→    const taskCenterY = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
   487→    const taskTopY = taskCenterY + TASK_HEIGHT / 2;
   488→    const taskBottomY = taskCenterY - TASK_HEIGHT / 2;
   489→
   490→    if (
   491→      worldX >= taskStartX &&
   492→      worldX <= taskEndX &&
   493→      worldY <= taskTopY &&
   494→      worldY >= taskBottomY
   495→    ) {
   496→      return task.id;
   497→    }
   498→  }
   499→
   500→  return null;
   501→}

   503→/**
   504→ * Camera controller component (inside R3F context)
   505→ * Handles initial positioning and syncs camera with external state
   506→ *
   507→ * IMPORTANT: Uses R3F's `size` from useThree() which is the actual canvas size,
   508→ * not the container div size (which can differ due to flexbox/layout).
   509→ */
   510→function CameraController({
   511→  cameraX,
   512→  cameraY,
   513→  zoom,
   514→  worldBounds,
   515→  isInitialized,
   516→  onInitialized,
   517→  onViewportChange,
   518→}: {
   519→  cameraX: number;
   520→  cameraY: number;
   521→  zoom: number;
   522→  worldBounds: WorldBounds;
   523→  isInitialized: boolean;
   524→  onInitialized: (pos: { x: number; y: number }, size: { width: number; height: number }) => void;
   525→  onViewportChange?: (viewport: ViewportInfo) => void;
   526→}) {
   527→  const { camera, size } = useThree();
   528→
   529→  // Initialize camera position on first render using actual R3F size
   530→  useEffect(() => {
   531→    if (isInitialized || size.width === 0 || size.height === 0) return;
   532→
   533→    const halfViewX = size.width / 2 / zoom;
   534→    const halfViewY = size.height / 2 / zoom;
   535→
   536→    // Position camera to show top-left corner (X=0, Y=0)
   537→    const initialCameraX = halfViewX;
   538→    const initialCameraY = -halfViewY;
   539→
   540→    // Clamp to world bounds
   541→    const minCameraX = worldBounds.minX + halfViewX;
   542→    const maxCameraX = worldBounds.maxX - halfViewX;
   543→    const minCameraY = worldBounds.minY + halfViewY;
   544→    const maxCameraY = worldBounds.maxY - halfViewY;
   545→
   546→    const clampedX = Math.max(minCameraX, Math.min(maxCameraX, initialCameraX));
   547→    const clampedY = Math.max(minCameraY, Math.min(maxCameraY, initialCameraY));
   548→
   549→    // Update camera
   550→    camera.position.x = clampedX;
   551→    camera.position.y = clampedY;
   552→    (camera as any).zoom = zoom;
   553→    camera.updateProjectionMatrix();
   554→
   555→    // Notify parent of initial position and actual canvas size
   556→    onInitialized({ x: clampedX, y: clampedY }, { width: size.width, height: size.height });
   557→
   558→    // Notify viewport change with full viewport info
   559→    const topY = clampedY + halfViewY;
   560→    const leftX = clampedX - halfViewX;
   561→    const rightX = clampedX + halfViewX;
   562→    onViewportChange?.({ topY, leftX, rightX });
   563→  }, [size, zoom, worldBounds, camera, isInitialized, onInitialized, onViewportChange]);
   564→
   565→  // Sync camera with external state changes (after initialization)
   566→  useEffect(() => {
   567→    if (!isInitialized) return;
   568→
   569→    camera.position.x = cameraX;
   570→    camera.position.y = cameraY;
   571→    (camera as any).zoom = zoom;
   572→    camera.updateProjectionMatrix();
   573→  }, [camera, cameraX, cameraY, zoom, isInitialized]);
   574→
   575→  return null;
   576→}

   982→    const vx = totalDx / dt;

  1098→    const zoomFactor = Math.exp(-accumulator.delta);
  1099→    const newZoom = Math.max(effectiveMinZoom, Math.min(MAX_ZOOM, currentZoom * zoomFactor));
  1100→

  1102→      accumulator.delta = 0;
  1103→      return;
  1104→    }
  1105→

  1107→    const cursorX = accumulator.cursorX - rect.left - rect.width / 2;
  1108→    const cursorY = accumulator.cursorY - rect.top - rect.height / 2;
  1109→

  1112→    const currentCameraY = anim.isAnimating ? anim.targetY : state.viewport.camera.y;
  1113→
  1114→    // Calculate target camera position
  1115→    const worldX = cursorX / currentZoom + currentCameraX;
  1116→    const worldY = -cursorY / currentZoom + currentCameraY;
  1117→    const newCameraX = worldX - cursorX / newZoom;
  1118→    const newCameraY = worldY + cursorY / newZoom;
  1119→    const clamped = clampCameraToWorld(newCameraX, newCameraY, newZoom);
  1120→
  1121→    // Reset accumulator
  1122→    accumulator.delta = 0;
  1123→    accumulator.timeout = null;
  1124→
  1125→    // Animate to new zoom
  1126→    animateZoomTo(newZoom, clamped.x, clamped.y, ZOOM_ANIMATION_DURATION);
  1127→  }, [zoom, effectiveMinZoom, clampCameraToWorld, animateZoomTo]);
  1128→
  1129→  /**
  1130→   * Normalize wheel delta for cross-platform consistency (F121)
  1131→   * Handles deltaMode (LINE/PAGE), detects trackpad vs mouse
  1132→   * Returns normalized pixel delta
  1133→   */
  1134→  const normalizeWheelDelta = useCallback((e: WheelEvent): number => {
  1135→    let delta = e.deltaY;
  1136→
  1137→    // Normalize deltaMode to pixels (F121: cross-platform normalization)
  1138→    // deltaMode 0 = PIXEL, 1 = LINE, 2 = PAGE
  1139→    if (e.deltaMode === 1) {
  1140→      // LINE mode (Firefox on Windows)
  1141→      delta *= LINE_HEIGHT;
  1142→    } else if (e.deltaMode === 2) {
  1143→      // PAGE mode (rare, but handle it)
  1144→      delta *= PAGE_HEIGHT;
  1145→    }
  1146→
  1147→    // Detect trackpad vs mouse by event frequency
  1148→    const accumulator = wheelAccumulatorRef.current;
  1149→    const now = performance.now();
  1150→    const timeSinceLastEvent = now - accumulator.lastEventTime;
  1151→
  1152→    if (timeSinceLastEvent < TRACKPAD_DETECTION_THRESHOLD) {
  1153→      accumulator.rapidEventCount++;
  1154→      if (accumulator.rapidEventCount >= TRACKPAD_EVENT_COUNT_THRESHOLD) {
  1155→        accumulator.isTrackpad = true;
  1156→      }
  1157→    } else {
  1158→      // Reset detection if events are spaced out (mouse behavior)
  1159→      accumulator.rapidEventCount = 1;
  1160→      // Also check delta magnitude - trackpads produce small deltas

  1169→      ? TRACKPAD_ZOOM_SENSITIVITY

  1173→    const normalizedDelta = delta * sensitivity;
  1174→    return Math.max(-MAX_ZOOM_DELTA_PER_EVENT, Math.min(MAX_ZOOM_DELTA_PER_EVENT, normalizedDelta));

  1177→/**
  1178→   * Handle wheel events - Google Maps-like behavior (BUG-003)
  1179→   *
  1180→   * Behavior:
  1181→   * - No modifier (2-finger scroll): Pan in both directions (deltaX → horizontal, deltaY → vertical)
  1182→   * - Ctrl+scroll (pinch gesture): Zoom toward cursor position
  1183→   *
  1184→   * This follows Google Maps UX pattern where:
  1185→   * - Trackpad 2-finger drag pans the map
  1186→   * - Trackpad pinch (sends ctrlKey=true) zooms
  1187→   * - Mouse Ctrl+scroll zooms
  1188→   *
  1189→   * With cross-platform normalization (F121)
  1190→   */
  1191→  const handleWheel = useCallback(
  1192→    (e: WheelEvent) => {
  1193→      e.preventDefault();
  1194→      if (!isInitialized) return;
  1195→
  1196→      // Stop momentum on any wheel interaction
  1197→      stopMomentum();
  1198→
  1199→      const rect = containerRef.current?.getBoundingClientRect();
  1200→      if (!rect) return;
  1201→
  1202→      // Normalize deltas for cross-platform consistency
  1203→      const getNormalizedDeltas = () => {
  1204→        let deltaX = e.deltaX;
  1205→        let deltaY = e.deltaY;
  1206→        if (e.deltaMode === 1) {
  1207→          deltaX *= LINE_HEIGHT;
  1208→          deltaY *= LINE_HEIGHT;
  1209→        } else if (e.deltaMode === 2) {
  1210→          deltaX *= PAGE_HEIGHT;
  1211→          deltaY *= PAGE_HEIGHT;
  1212→        }
  1213→        return { deltaX, deltaY };
  1214→      };
  1215→
  1216→      // Ctrl+scroll = ZOOM (pinch gesture on trackpad sends ctrlKey=true)
  1217→      if (e.ctrlKey) {
  1218→        const accumulator = wheelAccumulatorRef.current;
  1219→
  1220→        // Normalize delta for cross-platform consistency (F121)
  1221→        const normalizedDelta = normalizeWheelDelta(e);
  1222→
  1223→        // Accumulate normalized delta (positive = zoom out, negative = zoom in)
  1224→        accumulator.delta += normalizedDelta;
  1225→        accumulator.cursorX = e.clientX;
  1226→        accumulator.cursorY = e.clientY;
  1227→
  1228→        // Clear existing timeout
  1229→        if (accumulator.timeout) {
  1230→          clearTimeout(accumulator.timeout);
  1231→        }
  1232→
  1233→        // Set timeout to execute accumulated zoom
  1234→        accumulator.timeout = setTimeout(() => {
  1235→          executeAccumulatedZoom();
  1236→        }, ZOOM_ACCUMULATION_TIMEOUT);
  1237→        return;
  1238→      }
  1239→
  1240→      // Default behavior: PAN in both directions (Google Maps style)
  1241→      // 2-finger trackpad scroll sends both deltaX and deltaY
  1242→      const { deltaX, deltaY } = getNormalizedDeltas();
  1243→
  1244→      // Apply pan sensitivity (scaled by zoom for consistent world movement)
  1245→      const panDeltaX = (deltaX * SCROLL_PAN_SENSITIVITY) / zoom;
  1246→      const panDeltaY = (deltaY * SCROLL_PAN_SENSITIVITY) / zoom;
  1247→
  1248→      const currentCamera = canvasStore.getState().viewport.camera;
  1249→      const newX = currentCamera.x + panDeltaX;
  1250→      const newY = currentCamera.y - panDeltaY; // Negative because Y is inverted in canvas coords
  1251→      const clamped = clampCameraToWorld(newX, newY, zoom);
  1252→
  1253→      updateCameraPos(clamped);
  1254→      notifyViewportChange(clamped.x, clamped.y, zoom);
  1255→    },
  1256→    [isInitialized, stopMomentum, normalizeWheelDelta, executeAccumulatedZoom, zoom, clampCameraToWorld, updateCameraPos, notifyViewportChange]
  1257→  );
