   316→#### Reference: @belt/tools Patterns
   317→```typescript
   318→// Schema validation (replaces Zod)
   319→import { createSchema, createType, InferType } from '@belt/tools';
   320→
   321→const TrainModel = createSchema({
   322→  id: 'string',
   323→  trainNumber: 'string',
   324→  mileageKm: 'number',
   325→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   326→  tags: 'string[]',
   327→  metadata: 'object?',
   328→});
   329→type Train = InferType<typeof TrainModel>;
   330→const train = TrainModel.parse(data);
   331→
   332→// State machine (replaces React hooks + zustand)
   333→import { createSlice, createMachine } from '@belt/tools';
   334→
   335→const fleetSlice = createSlice({
   336→  id: 'fleet',
   337→  initial: { trains: [], selected: null },
   338→  cache: { policy: 'session' },
   339→  reducers: {
   340→    setTrains: (state, trains) => { state.trains = trains; },
   341→    selectTrain: (state, id) => { state.selected = id; },
   342→  },
   343→});
   344→
   345→const machine = createMachine({
   346→  id: 'mms',
   347→  slices: { fleet: fleetSlice },
   348→  trackMetadata: false, // Critical for 60fps
   349→});
   350→
   351→// In React component (NO useState/useEffect!)
   352→const { state, actions } = machine.useMachine('fleet');
   353→```
