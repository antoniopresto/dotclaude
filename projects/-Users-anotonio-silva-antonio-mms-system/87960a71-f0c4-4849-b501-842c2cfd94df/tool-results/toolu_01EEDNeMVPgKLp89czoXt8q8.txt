    24→const { pxPerDay, pxPerTrain } = DEFAULT_VIEW_CONFIG;

    26→/**
    27→ * Hook that creates and manages a spatial index for tasks.
    28→ *
    29→ * The index is automatically rebuilt when tasks change.
    30→ * Uses RBushSpatialIndex for O(log n) queries (STATE-ARCH-001.Phase2).
    31→ *
    32→ * @param tasks - Array of tasks to index
    33→ * @returns Spatial index instance
    34→ */
    35→export function useSpatialIndex(tasks: TaskRender[]): ISpatialIndex {
    36→  // Keep a stable reference to the index
    37→  const indexRef = useRef<ISpatialIndex | null>(null);
    38→
    39→  // Create index if not exists - now using RBush for O(log n) queries
    40→  if (!indexRef.current) {
    41→    indexRef.current = new RBushSpatialIndex();
    42→  }
    43→
    44→  // Rebuild index when tasks change
    45→  useEffect(() => {
    46→    if (indexRef.current && tasks.length > 0) {
    47→      const items = tasksToSpatialItems(tasks, pxPerDay, pxPerTrain);
    48→      indexRef.current.load(items);
    49→
    50→      // Log stats in development
    51→      if (import.meta.env.DEV) {
    52→        const stats = indexRef.current.getStats();
    53→        console.log(`[SpatialIndex] Loaded ${stats.itemCount} items in ${stats.lastBuildTimeMs?.toFixed(2)}ms`);
    54→      }
    55→    }
    56→  }, [tasks]);
    57→
    58→  return indexRef.current;
    59→}
    60→
    61→/**
    62→ * Hook that queries visible task IDs from a spatial index.
    63→ *
    64→ * @param spatialIndex - Spatial index to query
    65→ * @param viewportBounds - Current viewport bounds
    66→ * @param padding - Extra padding around viewport
    67→ * @returns Set of visible task IDs
    68→ */
    69→export function useVisibleTaskIds(
    70→  spatialIndex: ISpatialIndex | null,
    71→  viewportBounds: { minX: number; maxX: number; minY: number; maxY: number } | null,
    72→  padding: number = 50
    73→): Set<string> {
    74→  return useMemo(() => {
    75→    if (!spatialIndex || !viewportBounds) {
    76→      return new Set<string>();
    77→    }
    78→
    79→    const paddedBounds = {
    80→      minX: viewportBounds.minX - padding,
    81→      maxX: viewportBounds.maxX + padding,
    82→      minY: viewportBounds.minY - padding,
    83→      maxY: viewportBounds.maxY + padding,
    84→    };
    85→
    86→    const visibleIds = spatialIndex.queryViewport(paddedBounds);
    87→
    88→    // Log query stats in development
    89→    if (import.meta.env.DEV) {
    90→      const stats = spatialIndex.getStats();
    91→      if (stats.lastQueryTimeMs !== undefined) {
    92→        console.log(
    93→          `[SpatialIndex] Query returned ${stats.lastQueryResultCount} items in ${stats.lastQueryTimeMs.toFixed(2)}ms`
    94→        );
    95→      }
    96→    }
    97→
    98→    return new Set(visibleIds);
    99→  }, [spatialIndex, viewportBounds, padding]);
   100→}
