    41→    async def get_available_trains(self, allocation_date: date) -> list[dict]:
    42→        """
    43→        Get all trains available for allocation on a given date.
    44→
    45→        Returns trains that are:
    46→        - Fitness status = 'fit'
    47→        - Not already allocated to another service on that date
    48→        - Not under maintenance on that date
    49→        """
    50→        query = """
    51→            SELECT
    52→                t.id,
    53→                t.train_id,
    54→                t.fleet_id,
    55→                t.base_depot_id,
    56→                t.current_km,
    57→                t.target_km,
    58→                t.fitness_status,
    59→                t.next_maintenance_km,
    60→                t.latitude,
    61→                t.longitude,
    62→                d.name as depot_name,
    63→                d.code as depot_code,
    64→                d.latitude as depot_lat,
    65→                d.longitude as depot_lon,
    66→                f.name as fleet_name,
    67→                f.pm1_interval_km,
    68→                f.pm2_interval_km,
    69→                f.hm_interval_km
    70→            FROM trains t
    71→            LEFT JOIN depots d ON t.base_depot_id = d.id
    72→            LEFT JOIN fleets f ON t.fleet_id = f.id
    73→            WHERE t.fitness_status = 'fit'
    74→            AND NOT EXISTS (
    75→                SELECT 1 FROM allocations a
    76→                WHERE a.train_id = t.id
    77→                AND a.allocation_date = $1
    78→                AND a.status != 'cancelled'
    79→            )
    80→            AND NOT EXISTS (
    81→                SELECT 1 FROM maintenance_schedules m
    82→                WHERE m.train_id = t.id
    83→                AND $1 BETWEEN m.scheduled_date
    84→                    AND m.scheduled_date + (m.duration_days - 1) * interval '1 day'
    85→                AND m.status = 'scheduled'
    86→            )
    87→            ORDER BY t.train_id
    88→        """
    89→        async with self.pool.acquire() as conn:
    90→            rows = await conn.fetch(query, allocation_date)
    91→            return [dict(row) for row in rows]

   136→    async def get_upcoming_maintenance(self, train_id: str, days_ahead: int = 7) -> list[dict]:
   137→        """Get scheduled maintenance for a train in the next N days."""
   138→        query = """
   139→            SELECT
   140→                m.id,
   141→                m.maintenance_type,
   142→                m.scheduled_date,
   143→                m.duration_days,
   144→                d.name as depot_name,

   147→            JOIN depots d ON m.depot_id = d.id
   148→            WHERE m.train_id = $1
   149→            AND m.scheduled_date BETWEEN CURRENT_DATE AND CURRENT_DATE + $2 * interval '1 day'
   150→            AND m.status = 'scheduled'
   151→            ORDER BY m.scheduled_date
   152→        """
   153→        async with self.pool.acquire() as conn:
