     1‚Üí/**
     2‚Üí * MicroLayer - Detailed task card visualization
     3‚Üí *
     4‚Üí * Renders detailed task cards with labels at zoom > 2.0x (MICRO LOD level).
     5‚Üí *
     6‚Üí * PERFORMANCE OPTIMIZATIONS:
     7‚Üí * - Frustum culling: Only renders tasks visible in the viewport
     8‚Üí * - InstancedMesh for grid lines: Single draw call instead of ~251 meshes
     9‚Üí * - Lazy label rendering: Only labels for visible tasks
    10‚Üí * - InstancedMesh for row backgrounds: Single draw call instead of 50 meshes
    11‚Üí */

    27‚Üí  viewportBounds?: ViewportBounds;
    28‚Üí  /** Currently hovered task ID for highlight effect */
    29‚Üí  hoveredTaskId?: string | null;
    30‚Üí  /** Currently selected task ID for selection highlight (F019) */

    43‚Üíconst COLORS: Record<string, string> = {

    45‚Üí  orange: '#f59e0b',

    52‚Üíconst GRID_COLOR_MINOR = '#e5e7eb'; // Gray-200

    58‚Üí/**
    59‚Üí * Filter tasks that are visible within the viewport bounds
    60‚Üí * This is the core of frustum culling for performance
    61‚Üí */
    62‚Üífunction filterVisibleTasks(tasks: TaskRender[], viewportBounds?: ViewportBounds): TaskRender[] {
    63‚Üí  if (!viewportBounds) return tasks;
    64‚Üí
    65‚Üí  // Add padding to viewport bounds (render slightly outside visible area for smooth panning)
    66‚Üí  const padding = 100; // world units
    67‚Üí  const minX = viewportBounds.minX - padding;
    68‚Üí  const maxX = viewportBounds.maxX + padding;
    69‚Üí  const minY = viewportBounds.minY - padding;
    70‚Üí  const maxY = viewportBounds.maxY + padding;
    71‚Üí
    72‚Üí  return tasks.filter((task) => {
    73‚Üí    const taskStartX = task.startDay * pxPerDay;
    74‚Üí    const taskEndX = taskStartX + task.duration * pxPerDay;
    75‚Üí    const taskY = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
    76‚Üí
    77‚Üí    // Check if task overlaps with viewport
    78‚Üí    const xOverlap = taskEndX >= minX && taskStartX <= maxX;
    79‚Üí    const yOverlap = taskY + pxPerTrain / 2 >= minY && taskY - pxPerTrain / 2 <= maxY;
    80‚Üí
    81‚Üí    return xOverlap && yOverlap;
    82‚Üí  });
    83‚Üí}
    84‚Üí
    85‚Üí// Highlight color multiplier for hovered tasks
    86‚Üíconst HOVER_BRIGHTNESS = 1.3;
    87‚Üí
    88‚Üí// Selection indicator color (distinct cyan/teal)
    89‚Üíconst SELECTION_COLOR = '#06b6d4'; // cyan-500
    90‚Üí
    91‚Üí// Focus indicator color (F021 - keyboard focus)
    92‚Üíconst FOCUS_COLOR = '#8b5cf6'; // violet-500

    94‚Üí/**
    95‚Üí * Brighten a color for hover effect
    96‚Üí */
    97‚Üífunction brightenColor(hexColor: string, factor: number): Color {
    98‚Üí  const color = new Color(hexColor);
    99‚Üí  color.r = Math.min(1, color.r * factor);
   100‚Üí  color.g = Math.min(1, color.g * factor);
   101‚Üí  color.b = Math.min(1, color.b * factor);
   102‚Üí  return color;
   103‚Üí}

   105‚Üí/**
   106‚Üí * Task cards using InstancedMesh (same pattern as MesoLayer)
   107‚Üí */
   108‚Üífunction TaskCards({
   109‚Üí  tasks,
   110‚Üí  opacity = 1,
   111‚Üí  hoveredTaskId,
   112‚Üí  selectedTaskId,
   113‚Üí  focusedTaskId,
   114‚Üí}: {
   115‚Üí  tasks: TaskRender[];
   116‚Üí  opacity?: number;
   117‚Üí  hoveredTaskId?: string | null;
   118‚Üí  selectedTaskId?: string | null;
   119‚Üí  focusedTaskId?: string | null;
   120‚Üí}) {
   121‚Üí  const meshRef = useRef<InstancedMesh>(null);
   122‚Üí  const tempObject = useMemo(() => new Object3D(), []);
   123‚Üí  const tempColor = useMemo(() => new Color(), []);
   124‚Üí
   125‚Üí  useEffect(() => {
   126‚Üí    if (!meshRef.current || tasks.length === 0) return;
   127‚Üí
   128‚Üí    const mesh = meshRef.current;
   129‚Üí
   130‚Üí    tasks.forEach((task, i) => {
   131‚Üí      const width = task.duration * pxPerDay - CARD_PADDING * 2;
   132‚Üí      const height = CARD_HEIGHT;
   133‚Üí      const x = task.startDay * pxPerDay + width / 2 + CARD_PADDING;
   134‚Üí      const y = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
   135‚Üí      // Z ordering: selected > focused > conflict > hovered > normal
   136‚Üí      const isSelected = task.id === selectedTaskId;
   137‚Üí      const isFocused = task.id === focusedTaskId;
   138‚Üí      const isHovered = task.id === hoveredTaskId;
   139‚Üí      const z = isSelected ? 0.2 : isFocused ? 0.15 : task.hasConflict ? 0.1 : isHovered ? 0.05 : 0;
   140‚Üí
   141‚Üí      tempObject.position.set(x, y, z);
   142‚Üí      tempObject.scale.set(width, height, 1);
   143‚Üí      tempObject.updateMatrix();
   144‚Üí      mesh.setMatrixAt(i, tempObject.matrix);
   145‚Üí
   146‚Üí      // Set color: selected gets distinct color, focused gets focus color, hovered gets brightened, normal is base
   147‚Üí      if (isSelected) {
   148‚Üí        tempColor.set(SELECTION_COLOR);
   149‚Üí        mesh.setColorAt(i, tempColor);
   150‚Üí      } else if (isFocused) {
   151‚Üí        tempColor.set(FOCUS_COLOR);
   152‚Üí        mesh.setColorAt(i, tempColor);
   153‚Üí      } else if (isHovered) {
   154‚Üí        const baseColor = task.hasConflict ? CONFLICT_COLOR : COLORS[task.color];
   155‚Üí        const brightened = brightenColor(baseColor, HOVER_BRIGHTNESS);
   156‚Üí        mesh.setColorAt(i, brightened);
   157‚Üí      } else {
   158‚Üí        const baseColor = task.hasConflict ? CONFLICT_COLOR : COLORS[task.color];
   159‚Üí        tempColor.set(baseColor);
   160‚Üí        mesh.setColorAt(i, tempColor);
   161‚Üí      }
   162‚Üí    });
   163‚Üí
   164‚Üí    mesh.instanceMatrix.needsUpdate = true;
   165‚Üí    if (mesh.instanceColor) {
   166‚Üí      mesh.instanceColor.needsUpdate = true;
   167‚Üí    }
   168‚Üí  }, [tasks, tempObject, tempColor, hoveredTaskId, selectedTaskId, focusedTaskId]);
   169‚Üí
   170‚Üí  if (tasks.length === 0) return null;
   171‚Üí
   172‚Üí  return (
   173‚Üí    <instancedMesh ref={meshRef} args={[undefined, undefined, tasks.length]}>
   174‚Üí      <planeGeometry args={[1, 1]} />
   175‚Üí      <meshBasicMaterial toneMapped={false} transparent={opacity < 1} opacity={opacity} />
   176‚Üí    </instancedMesh>
   177‚Üí  );
   178‚Üí}

   219‚Üí/**
   220‚Üí * HTML labels for tasks - ONLY for visible tasks (lazy rendering)
   221‚Üí * Maximum of 50 labels rendered at once for performance
   222‚Üí */
   223‚Üífunction TaskLabels({ tasks, opacity = 1 }: { tasks: TaskRender[]; opacity?: number }) {
   224‚Üí  const { camera } = useThree();
   225‚Üí  const zoom = (camera as any).zoom || 1;
   226‚Üí
   227‚Üí  // Only show labels on tasks wide enough (> 40 screen pixels)
   228‚Üí  // And limit to max 50 labels for performance
   229‚Üí  const labeledTasks = useMemo(() => {
   230‚Üí    const minWidthForLabel = 40 / zoom; // Convert screen pixels to world pixels
   231‚Üí    const filtered = tasks.filter((t) => t.duration * pxPerDay > minWidthForLabel);
   232‚Üí    // Sort by width (largest first) and take top 50
   233‚Üí    return filtered
   234‚Üí      .sort((a, b) => b.duration - a.duration)
   235‚Üí      .slice(0, 50);
   236‚Üí  }, [tasks, zoom]);
   237‚Üí
   238‚Üí  // Don't render labels during low opacity transitions (performance optimization)
   239‚Üí  if (opacity < 0.3) return null;
   240‚Üí
   241‚Üí  return (
   242‚Üí    <>
   243‚Üí      {labeledTasks.map((task) => {
   244‚Üí        const width = task.duration * pxPerDay;
   245‚Üí        const x = task.startDay * pxPerDay + width / 2;
   246‚Üí        const y = -task.trainIndex * pxPerTrain - pxPerTrain / 2;
   247‚Üí        const label = task.type.charAt(0).toUpperCase() + task.type.slice(1);
   248‚Üí        // F022: Generate descriptive aria-label for screen readers
   249‚Üí        const ariaLabel = `${label} task, Train ${task.trainIndex + 1}, Day ${task.startDay}, Duration ${task.duration} days${task.hasConflict ? ', has conflict' : ''}`;
   250‚Üí
   251‚Üí        return (
   252‚Üí          <Html
   253‚Üí            key={task.id}
   254‚Üí            position={[x, y, 0.2]}
   255‚Üí            center
   256‚Üí            style={{ pointerEvents: 'none', opacity }}
   257‚Üí          >
   258‚Üí            <div
   259‚Üí              role="img"
   260‚Üí              aria-label={ariaLabel}
   261‚Üí              style={{
   262‚Üí                color: 'white',
   263‚Üí                fontSize: '10px',
   264‚Üí                fontFamily: 'system-ui, sans-serif',
   265‚Üí                fontWeight: 600,
   266‚Üí                textShadow: '0 1px 2px rgba(0,0,0,0.3)',
   267‚Üí                whiteSpace: 'nowrap',
   268‚Üí              }}
   269‚Üí            >
   270‚Üí              {label}
   271‚Üí            </div>
   272‚Üí          </Html>

   274‚Üí      })}
   275‚Üí    </>
   276‚Üí  );
   277‚Üí}

   279‚Üí/**
   280‚Üí * Grid lines using InstancedMesh (OPTIMIZED - single draw call)
   281‚Üí * Previously: ~251 individual meshes = 251 draw calls
   282‚Üí * Now: 2 InstancedMesh = 2 draw calls
   283‚Üí */
   284‚Üífunction GridLines({
   285‚Üí  totalDays,
   286‚Üí  trainCount,
   287‚Üí  opacity = 1,
   288‚Üí  viewportBounds,
   289‚Üí}: {
   290‚Üí  totalDays: number;
   291‚Üí  trainCount: number;
   292‚Üí  opacity?: number;
   293‚Üí  viewportBounds?: ViewportBounds;
   294‚Üí}) {
   295‚Üí  const totalWidth = totalDays * pxPerDay;
   296‚Üí  const totalHeight = trainCount * pxPerTrain;
   297‚Üí
   298‚Üí  // Horizontal grid lines (train boundaries) - using InstancedMesh
   299‚Üí  const hMeshRef = useRef<InstancedMesh>(null);
   300‚Üí  const hTempObject = useMemo(() => new Object3D(), []);
   301‚Üí
   302‚Üí  // Vertical grid lines (week boundaries) - using InstancedMesh
   303‚Üí  const vMeshRef = useRef<InstancedMesh>(null);
   304‚Üí  const vTempObject = useMemo(() => new Object3D(), []);
   305‚Üí
   306‚Üí  // Calculate visible horizontal lines based on viewport
   307‚Üí  const visibleHLines = useMemo(() => {
   308‚Üí    if (!viewportBounds) {
   309‚Üí      return Array.from({ length: trainCount + 1 }, (_, i) => i);
   310‚Üí    }
   311‚Üí    const minTrainIndex = Math.max(0, Math.floor(-viewportBounds.maxY / pxPerTrain) - 1);
   312‚Üí    const maxTrainIndex = Math.min(trainCount, Math.ceil(-viewportBounds.minY / pxPerTrain) + 1);
   313‚Üí    return Array.from({ length: maxTrainIndex - minTrainIndex + 1 }, (_, i) => minTrainIndex + i);
   314‚Üí  }, [trainCount, viewportBounds]);
   315‚Üí
   316‚Üí  // Calculate visible vertical lines (weeks) based on viewport
   317‚Üí  const visibleVLines = useMemo(() => {
   318‚Üí    if (!viewportBounds) {
   319‚Üí      const lines = [];
   320‚Üí      for (let day = 0; day <= totalDays; day += 7) {
   321‚Üí        lines.push(day);
   322‚Üí      }
   323‚Üí      return lines.slice(0, 200);
   324‚Üí    }
   325‚Üí    const minDay = Math.max(0, Math.floor(viewportBounds.minX / pxPerDay / 7) * 7 - 7);
   326‚Üí    const maxDay = Math.min(totalDays, Math.ceil(viewportBounds.maxX / pxPerDay / 7) * 7 + 7);
   327‚Üí    const lines = [];
   328‚Üí    for (let day = minDay; day <= maxDay; day += 7) {
   329‚Üí      lines.push(day);
   330‚Üí    }
   331‚Üí    return lines;
   332‚Üí  }, [totalDays, viewportBounds]);
   333‚Üí
   334‚Üí  // Update horizontal lines InstancedMesh
   335‚Üí  useEffect(() => {
   336‚Üí    if (!hMeshRef.current || visibleHLines.length === 0) return;
   337‚Üí
   338‚Üí    const mesh = hMeshRef.current;
   339‚Üí    visibleHLines.forEach((trainIndex, i) => {
   340‚Üí      hTempObject.position.set(totalWidth / 2, -trainIndex * pxPerTrain, -0.05);
   341‚Üí      hTempObject.scale.set(totalWidth, 1, 1);
   342‚Üí      hTempObject.updateMatrix();
   343‚Üí      mesh.setMatrixAt(i, hTempObject.matrix);
   344‚Üí    });
   345‚Üí    mesh.instanceMatrix.needsUpdate = true;
   346‚Üí  }, [visibleHLines, totalWidth, hTempObject]);
   347‚Üí
   348‚Üí  // Update vertical lines InstancedMesh
   349‚Üí  useEffect(() => {
   350‚Üí    if (!vMeshRef.current || visibleVLines.length === 0) return;
   351‚Üí
   352‚Üí    const mesh = vMeshRef.current;
   353‚Üí    visibleVLines.forEach((day, i) => {
   354‚Üí      const isMajor = day % 30 === 0;
   355‚Üí      vTempObject.position.set(day * pxPerDay, -totalHeight / 2, -0.05);
   356‚Üí      vTempObject.scale.set(isMajor ? 1.5 : 1, totalHeight, 1);
   357‚Üí      vTempObject.updateMatrix();
   358‚Üí      mesh.setMatrixAt(i, vTempObject.matrix);
   359‚Üí      mesh.setColorAt(i, new Color(isMajor ? GRID_COLOR_MAJOR : GRID_COLOR_MINOR));
   360‚Üí    });
   361‚Üí    mesh.instanceMatrix.needsUpdate = true;
   362‚Üí    if (mesh.instanceColor) {
   363‚Üí      mesh.instanceColor.needsUpdate = true;
   364‚Üí    }
   365‚Üí  }, [visibleVLines, totalHeight, vTempObject]);
   366‚Üí
   367‚Üí  return (
   368‚Üí    <group>
   369‚Üí      {/* Horizontal lines - InstancedMesh */}
   370‚Üí      {visibleHLines.length > 0 && (
   371‚Üí        <instancedMesh ref={hMeshRef} args={[undefined, undefined, visibleHLines.length]}>
   372‚Üí          <planeGeometry args={[1, 1]} />
   373‚Üí          <meshBasicMaterial color={GRID_COLOR_MAJOR} transparent opacity={0.5 * opacity} />
   374‚Üí        </instancedMesh>
   375‚Üí      )}
   376‚Üí
   377‚Üí      {/* Vertical lines - InstancedMesh */}
   378‚Üí      {visibleVLines.length > 0 && (
   379‚Üí        <instancedMesh ref={vMeshRef} args={[undefined, undefined, visibleVLines.length]}>
   380‚Üí          <planeGeometry args={[1, 1]} />
   381‚Üí          <meshBasicMaterial transparent opacity={0.4 * opacity} />
   382‚Üí        </instancedMesh>
   383‚Üí      )}
   384‚Üí    </group>
   385‚Üí  );
   386‚Üí}

   388‚Üí/**
   389‚Üí * Row backgrounds using InstancedMesh (OPTIMIZED - single draw call)
   390‚Üí * Previously: 50 individual meshes = 50 draw calls
   391‚Üí * Now: 1 InstancedMesh = 1 draw call
   392‚Üí */
   393‚Üífunction RowBackgrounds({
   394‚Üí  trainCount,
   395‚Üí  totalDays,
   396‚Üí  opacity = 1,
   397‚Üí  viewportBounds,
   398‚Üí}: {
   399‚Üí  trainCount: number;
   400‚Üí  totalDays: number;
   401‚Üí  opacity?: number;
   402‚Üí  viewportBounds?: ViewportBounds;
   403‚Üí}) {
   404‚Üí  const totalWidth = totalDays * pxPerDay;
   405‚Üí  const meshRef = useRef<InstancedMesh>(null);
   406‚Üí  const tempObject = useMemo(() => new Object3D(), []);
   407‚Üí
   408‚Üí  // Calculate visible rows based on viewport
   409‚Üí  const visibleRows = useMemo(() => {
   410‚Üí    if (!viewportBounds) {
   411‚Üí      return Array.from({ length: trainCount }, (_, i) => i);
   412‚Üí    }
   413‚Üí    const minTrainIndex = Math.max(0, Math.floor(-viewportBounds.maxY / pxPerTrain) - 1);
   414‚Üí    const maxTrainIndex = Math.min(trainCount - 1, Math.ceil(-viewportBounds.minY / pxPerTrain) + 1);
   415‚Üí    return Array.from({ length: maxTrainIndex - minTrainIndex + 1 }, (_, i) => minTrainIndex + i);
   416‚Üí  }, [trainCount, viewportBounds]);
   417‚Üí
   418‚Üí  useEffect(() => {
   419‚Üí    if (!meshRef.current || visibleRows.length === 0) return;
   420‚Üí
   421‚Üí    const mesh = meshRef.current;
   422‚Üí    visibleRows.forEach((trainIndex, i) => {
   423‚Üí      tempObject.position.set(totalWidth / 2, -trainIndex * pxPerTrain - pxPerTrain / 2, -0.1);
   424‚Üí      tempObject.scale.set(totalWidth, pxPerTrain, 1);
   425‚Üí      tempObject.updateMatrix();
   426‚Üí      mesh.setMatrixAt(i, tempObject.matrix);
   427‚Üí      mesh.setColorAt(i, new Color(trainIndex % 2 === 0 ? ROW_BG_EVEN : ROW_BG_ODD));
   428‚Üí    });
   429‚Üí    mesh.instanceMatrix.needsUpdate = true;
   430‚Üí    if (mesh.instanceColor) {
   431‚Üí      mesh.instanceColor.needsUpdate = true;
   432‚Üí    }

   457‚Üí  opacity = 1,
   458‚Üí  viewportBounds,
   459‚Üí}: {

   462‚Üí  viewportBounds?: ViewportBounds;
   463‚Üí}) {
   464‚Üí  // Build a map of task ID to task for quick lookup

   468‚Üí

   493‚Üí        const fromX = (from.startDay + from.duration) * pxPerDay;

   513‚Üí          ];

   535‚Üí        const arrowTip = lastPoint.clone();

   540‚Üí          arrowTip,
   541‚Üí          arrowBase1,
   542‚Üí          arrowBase2,
   543‚Üí          arrowTip, // Close the triangle
   544‚Üí        ];

   546‚Üí        const key = `dep-${from.id}-${to.id}`;
   547‚Üí
   548‚Üí        return (
