    90→/**
    91→ * Defines the navigable world boundaries.
    92→ * Camera will be clamped to these bounds.

   101→  minY: number;  // -trainCount * pxPerTrain (bottom)

   152→/**
   153→ * Calculate world bounds from dataset metadata
   154→ */
   155→export function calculateWorldBounds(
   156→  trainCount: number,
   157→  totalDays: number,
   158→  config: ViewConfig = DEFAULT_VIEW_CONFIG
   159→): WorldBounds {
   160→  return {
   161→    minX: 0,
   162→    maxX: totalDays * config.pxPerDay,
   163→    minY: -trainCount * config.pxPerTrain,
   164→    maxY: 0,
   165→  };
   166→}

   182→/**
   183→ * Calculate actual data bounds from tasks (not theoretical world bounds)
   184→ * This is used to fit the initial view to the actual data range.
   185→ *
   186→ * @param tasks - Array of rendered tasks with position data
   187→ * @param trainCount - Total number of trains
   188→ * @param config - View configuration
   189→ * @param padding - Padding in pixels around the data bounds
   190→ */
   191→export function calculateDataBounds(
   192→  tasks: TaskRender[],
   193→  trainCount: number,
   194→  config: ViewConfig = DEFAULT_VIEW_CONFIG,
   195→  padding: number = 50
   196→): DataBounds {
   197→  if (tasks.length === 0) {
   198→    // No tasks, return default bounds
   199→    return {
   200→      minX: 0,
   201→      maxX: config.pxPerDay * 365,
   202→      minY: -trainCount * config.pxPerTrain,
   203→      maxY: 0,
   204→      width: config.pxPerDay * 365,
   205→      height: trainCount * config.pxPerTrain,
   206→      centerX: (config.pxPerDay * 365) / 2,
   207→      centerY: (-trainCount * config.pxPerTrain) / 2,
   208→    };
   209→  }
   210→
   211→  // Find min/max from actual task positions using d3-array (D3-MIGRATE-001)
   212→  const [minStartDay, maxStartDay] = extent(tasks, (d) => d.startDay) as [number, number];
   213→  const maxEndDay = max(tasks, (d) => d.startDay + d.duration) ?? maxStartDay;
   214→  const [minTrainIndex, maxTrainIndex] = extent(tasks, (d) => d.trainIndex) as [number, number];
   215→
   216→  // Convert to pixel coordinates
   217→  const minX = minStartDay * config.pxPerDay - padding;
   218→  const maxX = maxEndDay * config.pxPerDay + padding;
   219→  // Y is negative (trains go downward)
   220→  const maxY = -minTrainIndex * config.pxPerTrain + padding;
   221→  const minY = -(maxTrainIndex + 1) * config.pxPerTrain - padding;
   222→
   223→  const width = maxX - minX;
   224→  const height = maxY - minY;
   225→
   226→  return {
   227→    minX,
   228→    maxX,
   229→    minY,
   230→    maxY,
   231→    width,
   232→    height,
   233→    centerX: minX + width / 2,
   234→    centerY: minY + height / 2,
   235→  };
   236→}
   237→
   238→/**
   239→ * Calculate initial camera position and zoom to fit data bounds in viewport
   240→ *
   241→ * Follows the pattern used by D3.js fitExtent, Gantt chart zoomToFit, and pixi-viewport fitWorld
   242→ *
   243→ * Strategy:
   244→ * - If data fits comfortably at reasonable zoom (0.5x - 1.5x), fit all data
   245→ * - If data is too large (would require < 0.5x zoom), show starting portion at 0.5x zoom
   246→ * - This provides a good initial UX while allowing users to zoom out for full overview
   247→ *
   248→ * @param dataBounds - The data bounds to fit
   249→ * @param viewportWidth - Viewport width in pixels
   250→ * @param viewportHeight - Viewport height in pixels
   251→ * @param maxZoom - Maximum zoom level (for performance/UX)
   252→ * @param minZoom - Minimum zoom level for initial view
   253→ */
   254→export function calculateFitToView(
   255→  dataBounds: DataBounds,
   256→  viewportWidth: number,
   257→  viewportHeight: number,
   258→  maxZoom: number = 1.5,
   259→  minZoom: number = 0.5
   260→): { cameraX: number; cameraY: number; zoom: number } {
   261→  // Calculate zoom needed to fit all data in viewport
   262→  const zoomX = viewportWidth / dataBounds.width;
   263→  const zoomY = viewportHeight / dataBounds.height;
   264→  const fitAllZoom = Math.min(zoomX, zoomY);
   265→
   266→  // Clamp zoom to reasonable initial range
   267→  const zoom = Math.max(minZoom, Math.min(maxZoom, fitAllZoom));
   268→
   269→  // If we can fit all data at the calculated zoom, center on data
   270→  if (fitAllZoom >= minZoom) {
   271→    return {
   272→      cameraX: dataBounds.centerX,
   273→      cameraY: dataBounds.centerY,
   274→      zoom,
   275→    };
   276→  }
   277→
   278→  // Data is too large - show starting portion (top-left of data bounds)
   279→  // Calculate camera position to show left edge and top of data
   280→  const halfViewX = viewportWidth / 2 / zoom;
   281→  const halfViewY = viewportHeight / 2 / zoom;
   282→
   283→  return {
   284→    cameraX: dataBounds.minX + halfViewX,
   285→    cameraY: dataBounds.maxY - halfViewY, // maxY is the top (less negative)
   286→    zoom,
   287→  };
   288→}
