    74→/**
    75→ * TypedArray storage for task entities
    76→ * Structure of Arrays (SoA) pattern for optimal cache utilization
    77→ *
    78→ * Memory layout for N tasks:
    79→ * - startDay: Float32Array(N) - 4 bytes per task
    80→ * - duration: Float32Array(N) - 4 bytes per task
    81→ * - trainIndex: Uint16Array(N) - 2 bytes per task
    82→ * - type: Uint8Array(N) - 1 byte per task
    83→ * - color: Uint8Array(N) - 1 byte per task
    84→ * - hasConflict: Uint8Array(N) - 1 byte per task (0 or 1)
    85→ * - predecessorIndex: Int32Array(N) - 4 bytes per task (-1 = no predecessor)
    86→ *
    87→ * Total: 17 bytes per task = ~1.7MB for 100k tasks
    88→ * Compare to ~50MB+ for object-based storage with same data
    89→ */
    90→export interface TaskArrays {
    91→  /** Start day offset from dataset start date */
    92→  startDay: Float32Array;
    93→  /** Duration in days */
    94→  duration: Float32Array;
    95→  /** Train index (Y position) */
    96→  trainIndex: Uint16Array;
    97→  /** Task type (0=maintenance, 1=overhaul, 2=inspection) */
    98→  type: Uint8Array;
    99→  /** Task color (0=blue, 1=orange, 2=gray) */
   100→  color: Uint8Array;
   101→  /** Conflict flag (0=false, 1=true) */
   102→  hasConflict: Uint8Array;
   103→  /** Predecessor task index (-1 = no predecessor) */
   104→  predecessorIndex: Int32Array;
   105→}
