     1→/**
     2→ * Interaction State Slice
     3→ *
     4→ * Manages hover, selection, and focus state for task interactions.
     5→ * Uses @zazcart/tools external store pattern for synchronous access.
     6→ *
     7→ * @see F018 - Hover highlight on tasks
     8→ * @see F019 - Click to select task
     9→ * @see F021 - Keyboard navigation for tasks
    10→ */
    11→
    12→import { createSyncSlice } from '@zazcart/tools';
    13→
    14→// ─────────────────────────────────────────────────────────────
    15→// STATE TYPES
    16→// ─────────────────────────────────────────────────────────────
    17→
    18→export interface InteractionSliceState {
    19→  /** Currently hovered task ID (null if none) */
    20→  hoveredTaskId: string | null;
    21→
    22→  /** Mouse position in world coordinates (for hover detection) */
    23→  mouseWorldPos: { x: number; y: number } | null;
    24→
    25→  /** Currently selected task ID (null if none) - for F019 */
    26→  selectedTaskId: string | null;
    27→
    28→  /** Currently keyboard-focused task ID (null if none) - for F021 */
    29→  focusedTaskId: string | null;
    30→
    31→  /** Whether cursor is over a hoverable area */
    32→  isHoverable: boolean;
    33→}
    34→
    35→// ─────────────────────────────────────────────────────────────
    36→// INITIAL STATE
    37→// ─────────────────────────────────────────────────────────────
    38→
    39→const initialState: InteractionSliceState = {
    40→  hoveredTaskId: null,
    41→  mouseWorldPos: null,
    42→  selectedTaskId: null,
    43→  focusedTaskId: null,
    44→  isHoverable: false,
    45→};
    46→
    47→// ─────────────────────────────────────────────────────────────
    48→// INTERACTION SLICE
    49→// ─────────────────────────────────────────────────────────────
    50→
    51→export const interactionSlice = createSyncSlice({
    52→  id: 'interaction',
    53→  initial: initialState,
    54→  cache: { policy: 'session' },
    55→  reducers: {
    56→    // Hover state
    57→    setHoveredTask(state: InteractionSliceState, payload: string | null) {
    58→      state.hoveredTaskId = payload;
    59→      state.isHoverable = payload !== null;
    60→    },
    61→
    62→    setMouseWorldPos(
    63→      state: InteractionSliceState,
    64→      payload: { x: number; y: number } | null
    65→    ) {
    66→      state.mouseWorldPos = payload;
    67→    },
    68→
    69→    clearHover(state: InteractionSliceState) {
    70→      state.hoveredTaskId = null;
    71→      state.mouseWorldPos = null;
    72→      state.isHoverable = false;
    73→    },
    74→
    75→    // Selection state (F019)
    76→    setSelectedTask(state: InteractionSliceState, payload: string | null) {
    77→      state.selectedTaskId = payload;
    78→    },
    79→
    80→    clearSelection(state: InteractionSliceState) {
    81→      state.selectedTaskId = null;
    82→    },
    83→
    84→    // Focus state (F021 - keyboard navigation)
    85→    setFocusedTask(state: InteractionSliceState, payload: string | null) {
    86→      state.focusedTaskId = payload;
    87→    },
    88→
    89→    clearFocus(state: InteractionSliceState) {
    90→      state.focusedTaskId = null;
    91→    },
    92→
    93→    // Reset all
    94→    reset() {
    95→      return initialState;
    96→    },
    97→  },
    98→});
    99→
   100→export type InteractionSlice = typeof interactionSlice;
