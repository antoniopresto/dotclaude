     1→/**
     2→ * TaskDetailsPanel Component (F026)
     3→ *
     4→ * Shows detailed task information in a side panel when a task is selected.
     5→ * Uses BlueprintJS Drawer for the panel.
     6→ *
     7→ * Features:
     8→ * - Opens when a task is selected
     9→ * - Displays comprehensive task information
    10→ * - Closes on deselect or close button click
    11→ * - Shows conflict status with visual indicator
    12→ */
    13→
    14→import { useMemo, useCallback } from 'react';
    15→import {
    16→  Drawer,
    17→  DrawerSize,
    18→  Button,
    19→  Tag,
    20→  Intent,
    21→  Icon,
    22→  Divider,
    23→  H4,
    24→  H5,
    25→  Classes,
    26→} from '@blueprintjs/core';
    27→import { getData } from '../data/generator';
    28→import { useCanvasSelector, canvasStore, canvasActions } from '../state';
    29→import type { TaskType, AssetType } from '../data/types';
    30→
    31→// ─────────────────────────────────────────────────────────────
    32→// HELPERS
    33→// ─────────────────────────────────────────────────────────────
    34→
    35→/**
    36→ * Get intent color for task type
    37→ */
    38→function getTypeIntent(type: TaskType): Intent {
    39→  switch (type) {
    40→    case 'maintenance':
    41→      return Intent.PRIMARY;
    42→    case 'overhaul':
    43→      return Intent.WARNING;
    44→    case 'inspection':
    45→      return Intent.NONE;
    46→  }
    47→}
    48→
    49→/**
    50→ * Get icon for task type
    51→ */
    52→function getTypeIcon(type: TaskType): string {
    53→  switch (type) {
    54→    case 'maintenance':
    55→      return 'wrench';
    56→    case 'overhaul':
    57→      return 'build';
    58→    case 'inspection':
    59→      return 'eye-open';
    60→  }
    61→}
    62→
    63→/**
    64→ * Get icon for asset type (train fleet)
    65→ */
    66→function getAssetIcon(assetType: AssetType): string {
    67→  return assetType === 'ALSTOM' ? 'train' : 'tram';
    68→}
    69→
    70→/**
    71→ * Format date for display (e.g., "Mar 15, 2024")
    72→ */
    73→function formatDate(isoDate: string): string {
    74→  const date = new Date(isoDate);
    75→  return date.toLocaleDateString('en-US', {
    76→    weekday: 'short',
    77→    month: 'short',
    78→    day: 'numeric',
    79→    year: 'numeric',
    80→  });
    81→}
    82→
    83→/**
    84→ * Calculate duration in days from ISO dates
    85→ */
    86→function calculateDuration(startDate: string, endDate: string): number {
    87→  const start = new Date(startDate);
    88→  const end = new Date(endDate);
    89→  const diffTime = end.getTime() - start.getTime();
    90→  return Math.round(diffTime / (1000 * 60 * 60 * 24));
    91→}
    92→
    93→/**
    94→ * Capitalize first letter
    95→ */
    96→function capitalize(str: string): string {
    97→  return str.charAt(0).toUpperCase() + str.slice(1);
    98→}
    99→
   100→// ─────────────────────────────────────────────────────────────
   101→// COMPONENT
   102→// ─────────────────────────────────────────────────────────────
   103→
   104→export function TaskDetailsPanel() {
   105→  // Get selected task ID from canvas state
   106→  const selectedTaskId = useCanvasSelector((s) => s.interaction.selectedTaskId);
   107→
   108→  // Close handler - clears selection
   109→  const handleClose = useCallback(() => {
   110→    canvasStore.dispatch(canvasActions.interaction.clearSelection());
   111→  }, []);
   112→
   113→  // Get task details
   114→  const taskData = useMemo(() => {
   115→    if (!selectedTaskId) return null;
   116→
   117→    const data = getData();
   118→    const task = data.tasks.find((t) => t.id === selectedTaskId);
   119→    if (!task) return null;
   120→
   121→    const train = data.trains.find((t) => t.id === task.trainId);
   122→    const duration = calculateDuration(task.startDate, task.endDate);
   123→
   124→    // Find predecessor task if exists
   125→    let predecessorTask = null;
   126→    if (task.predecessorId) {
   127→      predecessorTask = data.tasks.find((t) => t.id === task.predecessorId);
   128→    }
   129→
   130→    // Find successor tasks (tasks that depend on this one)
   131→    const successorTasks = data.tasks.filter(
   132→      (t) => t.predecessorId === task.id
   133→    );
   134→
   135→    return {
   136→      id: task.id,
   137→      trainId: task.trainId,
   138→      trainName: train?.name || task.trainId,
   139→      assetType: train?.assetType || 'ALSTOM',
   140→      type: task.type,
   141→      startDate: task.startDate,
   142→      endDate: task.endDate,
   143→      duration,
   144→      hasConflict: task.hasConflict,
   145→      predecessorId: task.predecessorId,
   146→      predecessorTask,
   147→      successorTasks,
   148→    };
   149→  }, [selectedTaskId]);
   150→
   151→  // Render nothing if no task selected
   152→  const isOpen = taskData !== null;
   153→
   154→  return (
   155→    <Drawer
   156→      isOpen={isOpen}
   157→      onClose={handleClose}
   158→      size={DrawerSize.SMALL}
   159→      title={
   160→        <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
   161→          <Icon icon="info-sign" />
   162→          <span>Task Details</span>
   163→        </div>
   164→      }
   165→      canOutsideClickClose={false}
   166→      canEscapeKeyClose={true}
   167→      hasBackdrop={false}
   168→      position="right"
   169→      portalClassName="task-details-portal"
   170→      className="task-details-drawer"
   171→    >
   172→      {taskData && (
   173→        <div className={Classes.DRAWER_BODY}>
   174→          <div className={Classes.DIALOG_BODY} style={{ padding: 20 }}>
   175→            {/* Task Type Header */}
   176→            <div
   177→              style={{
   178→                display: 'flex',
   179→                alignItems: 'center',
   180→                gap: 12,
   181→                marginBottom: 16,
   182→              }}
   183→            >
   184→              <Icon icon={getTypeIcon(taskData.type)} size={24} />
   185→              <div>
   186→                <H4 style={{ margin: 0 }}>{capitalize(taskData.type)}</H4>
   187→                <Tag
   188→                  intent={getTypeIntent(taskData.type)}
   189→                  minimal
   190→                  style={{ marginTop: 4 }}
   191→                >
   192→                  {taskData.type.toUpperCase()}
   193→                </Tag>
   194→              </div>
   195→              {taskData.hasConflict && (
   196→                <Tag intent={Intent.DANGER} icon="warning-sign" large>
   197→                  CONFLICT
   198→                </Tag>
   199→              )}
   200→            </div>
   201→
   202→            <Divider />
   203→
   204→            {/* Train Information */}
   205→            <div style={{ marginTop: 16 }}>
   206→              <H5 style={{ marginBottom: 12, color: '#5c7080' }}>
   207→                <Icon icon="train" style={{ marginRight: 8 }} />
   208→                Train Information
   209→              </H5>
   210→              <div
   211→                style={{
   212→                  display: 'grid',
   213→                  gridTemplateColumns: '100px 1fr',
   214→                  gap: '8px 12px',
   215→                  fontSize: 14,
   216→                }}
   217→              >
   218→                <span style={{ color: '#5c7080' }}>Train ID:</span>
   219→                <span style={{ fontFamily: 'monospace', fontWeight: 600 }}>
   220→                  {taskData.trainId}
   221→                </span>
   222→
   223→                <span style={{ color: '#5c7080' }}>Fleet:</span>
   224→                <span style={{ display: 'flex', alignItems: 'center', gap: 6 }}>
   225→                  <Icon icon={getAssetIcon(taskData.assetType)} size={14} />
   226→                  {taskData.assetType}
   227→                </span>
   228→              </div>
   229→            </div>
   230→
   231→            <Divider style={{ margin: '16px 0' }} />
   232→
   233→            {/* Schedule Information */}
   234→            <div>
   235→              <H5 style={{ marginBottom: 12, color: '#5c7080' }}>
   236→                <Icon icon="calendar" style={{ marginRight: 8 }} />
   237→                Schedule
   238→              </H5>
   239→              <div
   240→                style={{
   241→                  display: 'grid',
   242→                  gridTemplateColumns: '100px 1fr',
   243→                  gap: '8px 12px',
   244→                  fontSize: 14,
   245→                }}
   246→              >
   247→                <span style={{ color: '#5c7080' }}>Start Date:</span>
   248→                <span>{formatDate(taskData.startDate)}</span>
   249→
   250→                <span style={{ color: '#5c7080' }}>End Date:</span>
   251→                <span>{formatDate(taskData.endDate)}</span>
   252→
   253→                <span style={{ color: '#5c7080' }}>Duration:</span>
   254→                <span style={{ fontWeight: 600 }}>
   255→                  {taskData.duration} day{taskData.duration !== 1 ? 's' : ''}
   256→                </span>
   257→              </div>
   258→            </div>
   259→
   260→            <Divider style={{ margin: '16px 0' }} />
   261→
   262→            {/* Status */}
   263→            <div>
   264→              <H5 style={{ marginBottom: 12, color: '#5c7080' }}>
   265→                <Icon icon="tick-circle" style={{ marginRight: 8 }} />
   266→                Status
   267→              </H5>
   268→              <div
   269→                style={{
   270→                  display: 'flex',
   271→                  flexWrap: 'wrap',
   272→                  gap: 8,
   273→                }}
   274→              >
   275→                <Tag
   276→                  intent={taskData.hasConflict ? Intent.DANGER : Intent.SUCCESS}
   277→                  icon={taskData.hasConflict ? 'warning-sign' : 'tick'}
   278→                >
   279→                  {taskData.hasConflict
   280→                    ? 'Has Scheduling Conflict'
   281→                    : 'No Conflicts'}
   282→                </Tag>
   283→                <Tag intent={Intent.NONE} icon="time" minimal>
   284→                  Scheduled
   285→                </Tag>
   286→              </div>
   287→            </div>
   288→
   289→            {/* Dependencies (if any) */}
   290→            {(taskData.predecessorId || taskData.successorTasks.length > 0) && (
   291→              <>
   292→                <Divider style={{ margin: '16px 0' }} />
   293→                <div>
   294→                  <H5 style={{ marginBottom: 12, color: '#5c7080' }}>
   295→                    <Icon icon="link" style={{ marginRight: 8 }} />
   296→                    Dependencies
   297→                  </H5>
   298→
   299→                  {taskData.predecessorTask && (
   300→                    <div style={{ marginBottom: 8 }}>
   301→                      <span
   302→                        style={{
   303→                          fontSize: 12,
   304→                          color: '#5c7080',
   305→                          display: 'block',
   306→                          marginBottom: 4,
   307→                        }}
   308→                      >
   309→                        Depends on:
   310→                      </span>
   311→                      <Tag
   312→                        icon={getTypeIcon(taskData.predecessorTask.type)}
   313→                        intent={getTypeIntent(taskData.predecessorTask.type)}
   314→                        minimal
   315→                      >
   316→                        {capitalize(taskData.predecessorTask.type)} (
   317→                        {taskData.predecessorTask.trainId})
   318→                      </Tag>
   319→                    </div>
   320→                  )}
   321→
   322→                  {taskData.successorTasks.length > 0 && (
   323→                    <div>
   324→                      <span
   325→                        style={{
   326→                          fontSize: 12,
   327→                          color: '#5c7080',
   328→                          display: 'block',
   329→                          marginBottom: 4,
   330→                        }}
   331→                      >
   332→                        Required by:
   333→                      </span>
   334→                      <div style={{ display: 'flex', flexWrap: 'wrap', gap: 4 }}>
   335→                        {taskData.successorTasks.map((t) => (
   336→                          <Tag
   337→                            key={t.id}
   338→                            icon={getTypeIcon(t.type)}
   339→                            intent={getTypeIntent(t.type)}
   340→                            minimal
   341→                          >
   342→                            {capitalize(t.type)} ({t.trainId})
   343→                          </Tag>
   344→                        ))}
   345→                      </div>
   346→                    </div>
   347→                  )}
   348→                </div>
   349→              </>
   350→            )}
   351→
   352→            {/* Technical Info (collapsed by default) */}
   353→            <Divider style={{ margin: '16px 0' }} />
   354→            <div>
   355→              <H5 style={{ marginBottom: 12, color: '#5c7080' }}>
   356→                <Icon icon="code" style={{ marginRight: 8 }} />
   357→                Technical Info
   358→              </H5>
   359→              <div
   360→                style={{
   361→                  backgroundColor: '#f5f8fa',
   362→                  padding: 12,
   363→                  borderRadius: 4,
   364→                  fontSize: 12,
   365→                  fontFamily: 'monospace',
   366→                }}
   367→              >
   368→                <div>Task ID: {taskData.id}</div>
   369→              </div>
   370→            </div>
   371→          </div>
   372→        </div>
   373→      )}
   374→
   375→      {/* Footer with close button */}
   376→      <div className={Classes.DRAWER_FOOTER} style={{ padding: 16 }}>
   377→        <Button intent={Intent.PRIMARY} onClick={handleClose} fill>
   378→          Close
   379→        </Button>
   380→      </div>
   381→    </Drawer>
   382→  );
   383→}
