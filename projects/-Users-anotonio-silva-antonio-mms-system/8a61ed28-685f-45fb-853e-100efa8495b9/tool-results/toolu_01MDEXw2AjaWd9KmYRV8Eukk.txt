   383→// Schema validation (replaces Zod)
   384→import { createSchema, createType, InferType } from '@belt/tools';
   385→
   386→const TrainModel = createSchema({

   388→  trainNumber: 'string',
   389→  mileageKm: 'number',
   390→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   391→  tags: 'string[]',
   392→  metadata: 'object?',
   393→});
   394→type Train = InferType<typeof TrainModel>;
   395→const train = TrainModel.parse(data);
   396→
   397→// State machine (replaces React hooks + zustand)
   398→import { createSlice, createMachine } from '@belt/tools';
   399→
   400→const fleetSlice = createSlice({
   401→  id: 'fleet',
   402→  initial: { trains: [], selected: null },
   403→  cache: { policy: 'session' },
   404→  reducers: {
   405→    setTrains: (state, trains) => { state.trains = trains; },
   406→    selectTrain: (state, id) => { state.selected = id; },
   407→  },
   408→});
   409→
   410→const machine = createMachine({
   411→  id: 'mms',
   412→  slices: { fleet: fleetSlice },
   413→  trackMetadata: false, // Critical for 60fps
   414→});
   415→
   416→// In React component (NO useState/useEffect!)
   417→const { state, actions } = machine.useMachine('fleet');
