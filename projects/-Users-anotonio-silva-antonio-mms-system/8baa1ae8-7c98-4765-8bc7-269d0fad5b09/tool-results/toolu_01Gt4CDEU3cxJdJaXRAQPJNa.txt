   291→/**
   292→ * HTTP middleware helper - measure request duration
   293→ */
   294→export function measureHttpRequest(
   295→  method: string,
   296→  path: string,
   297→  status: number,
   298→  durationMs: number
   299→): void {
   300→  const labels = { method, path, status: String(status) };
   301→  httpRequestsTotal.inc(labels);
   302→  httpRequestDuration.observe(labels, durationMs / 1000);
   303→}
   304→
   305→/**
   306→ * Queue job helper - record job completion
   307→ */
   308→export function recordQueueJob(
   309→  queue: string,
   310→  jobName: string,
   311→  status: 'completed' | 'failed',
   312→  durationMs: number
   313→): void {
   314→  queueJobsTotal.inc({ queue, status });
   315→  queueJobDuration.observe({ queue, job_name: jobName }, durationMs / 1000);
   316→}
   317→
   318→/**
   319→ * Optimizer helper - record solve completion
   320→ */
   321→export function recordOptimizerRun(
   322→  solverType: 'cpsat' | 'fallback',
   323→  status: 'optimal' | 'feasible' | 'infeasible' | 'timeout' | 'error',
   324→  durationMs: number,
   325→  objective?: number
   326→): void {
   327→  optimizerRunsTotal.inc({ solver_type: solverType, status });
   328→  optimizerSolveSeconds.observe({ solver_type: solverType, status }, durationMs / 1000);
   329→  if (objective !== undefined) {
   330→    optimizerObjectiveValue.set({ solver_type: solverType }, objective);
   331→  }
   332→}
