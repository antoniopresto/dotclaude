    32→/**
    33→ * Meter Events - Odometer replacements, resets, calibrations
    34→ *
    35→ * Tracks events that affect mileage continuity:
    36→ * - REPLACEMENT: New meter installed
    37→ * - RESET: Meter zeroed
    38→ * - CALIBRATION: Meter adjusted

    44→export const meterEvents = pgTable(
    45→  'meter_events',
    46→  {

    70→/**
