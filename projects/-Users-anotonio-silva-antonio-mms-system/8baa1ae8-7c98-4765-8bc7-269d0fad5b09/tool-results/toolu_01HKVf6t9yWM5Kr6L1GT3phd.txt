   394→// =============================================================================
   395→// Validation Helpers
   396→// =============================================================================
   397→
   398→/**
   399→ * Parse and validate data, returning typed result
   400→ */
   401→export function parseSchema<T extends z.ZodType>(
   402→  schema: T,
   403→  data: unknown
   404→): { success: true; data: z.infer<T> } | { success: false; errors: z.ZodError } {
   405→  const result = schema.safeParse(data);
   406→  if (result.success) {
   407→    return { success: true, data: result.data };
   408→  }
   409→  return { success: false, errors: result.error };
   410→}
   411→
   412→/**
   413→ * Format Zod errors for API response
   414→ */
   415→export function formatZodErrors(error: z.ZodError): Array<{ field: string; message: string }> {
   416→  return error.errors.map((err) => ({
   417→    field: err.path.join('.'),
   418→    message: err.message,
   419→  }));
   420→}
