   811→  /**

   814→   * Material constraints:

   817→   */
   818→  async checkMaterialAvailability(
   819→    pmTypeId: string,
   820→    depotId: string,
   821→    targetDate: string
   822→  ): Promise<MaterialConstraintResult> {
   823→    return this.traced('checkMaterialAvailability', async (span) => {
   824→      span.setAttribute('pm.typeId', pmTypeId);
   825→      span.setAttribute('depot.id', depotId);

   829→      const requirements = await this.db.query.pmMaterialRequirements.findMany({
   830→        where: eq(pmMaterialRequirements.pmTypeId, pmTypeId),
   831→      });
   832→
   833→      if (requirements.length === 0) {
   834→        return {
   835→          isBlocked: false,
   836→          hardConstraintViolations: [],
   837→          softConstraintWarnings: [],
   838→          totalMissingParts: 0,
   839→          reservations: [],
   840→        };
   841→      }
   842→
   843→      const today = new Date();
   844→      const pmDate = new Date(targetDate);
   845→      const daysUntilPm = Math.ceil((pmDate.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   846→
   847→      const hardViolations: MaterialAvailability[] = [];
   848→      const softWarnings: MaterialAvailability[] = [];
   849→      const reservations: Array<{ partNumber: string; quantity: number; type: ReservationType }> = [];
   850→
   851→      for (const req of requirements) {
   852→        // Get inventory at the depot
   853→        const inventory = await this.db.query.inventoryPositions.findFirst({
   854→          where: and(

   858→        });

   860→        const availableQty = inventory
   861→          ? (inventory.quantityOnHand ?? 0) - (inventory.quantityReserved ?? 0)
   862→          : 0;
   863→
   864→        const availability: MaterialAvailability = {
   865→          partNumber: req.partNumber,
   866→          partDescription: req.partDescription,
   867→          criticality: req.criticality as MaterialCriticality,

   872→          shortfall: Math.max(0, req.quantityPerPm - availableQty),
   873→          leadTimeDays: req.leadTimeDays ?? 14,

   875→        };

   884→        }
   885→
   886→        // Create reservation record

   888→          reservations.push({
   889→            partNumber: req.partNumber,
   890→            quantity: req.quantityPerPm,
   891→            type: daysUntilPm <= HARD_MATERIAL_WINDOW_DAYS ? 'HARD' : 'SOFT',
   892→          });
   893→        }
   894→      }
