   308→#### Reference: @belt/tools Patterns
   309→```typescript
   310→// Schema validation (replaces Zod)
   311→import { createSchema, createType, InferType } from '@belt/tools';
   312→
   313→const TrainModel = createSchema({
   314→  id: 'string',
   315→  trainNumber: 'string',
   316→  mileageKm: 'number',
   317→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   318→  tags: 'string[]',
   319→  metadata: 'object?',
   320→});
   321→type Train = InferType<typeof TrainModel>;
   322→const train = TrainModel.parse(data);
   323→
   324→// State machine (replaces React hooks + zustand)
   325→import { createSlice, createMachine } from '@belt/tools';
   326→
   327→const fleetSlice = createSlice({
   328→  id: 'fleet',
   329→  initial: { trains: [], selected: null },
   330→  cache: { policy: 'session' },
   331→  reducers: {
   332→    setTrains: (state, trains) => { state.trains = trains; },
   333→    selectTrain: (state, id) => { state.selected = id; },
   334→  },
   335→});
   336→
   337→const machine = createMachine({
   338→  id: 'mms',
   339→  slices: { fleet: fleetSlice },
   340→  trackMetadata: false, // Critical for 60fps
   341→});
   342→
   343→// In React component (NO useState/useEffect!)
   344→const { state, actions } = machine.useMachine('fleet');
   345→```
