   288→  // ===========================================================================
   289→  // Compliance Window Calculation
   290→  // ===========================================================================
   291→
   292→  /**
   293→   * Calculate compliance window for a PM type and train.
   294→   *
   295→   * The compliance window defines the valid date range for scheduling a PM.
   296→   * It is calculated based on:
   297→   * - Mileage trigger: intervalKm ± tolerance%
   298→   * - Time trigger: intervalMonths ± tolerance%
   299→   * - Whichever comes first determines the due date
   300→   *
   301→   * @param trainId - Train to calculate for
   302→   * @param pmTypeId - PM type to calculate for
   303→   * @returns Compliance window with start/end dates and mileage
   304→   */
   305→  async calculateComplianceWindow(
   306→    trainId: string,
   307→    pmTypeId: string
   308→  ): Promise<PmDueResult> {
   309→    return this.traced('calculateComplianceWindow', async (span) => {
   310→      span.setAttribute(MmsAttributes.TRAIN_ID, trainId);
   311→      span.setAttribute('pm.typeId', pmTypeId);
   312→
   313→      // Get train details
   314→      const train = await this.db.query.trains.findFirst({
   315→        where: eq(trains.id, trainId),
   316→        with: { fleet: true },
   317→      });
   318→
   319→      if (!train) {
   320→        throw new TrainNotFoundError(trainId);
   321→      }
   322→
   323→      span.setAttribute(MmsAttributes.TRAIN_NUMBER, train.trainNumber);
   324→      span.setAttribute(MmsAttributes.FLEET_ID, train.fleetId);
   325→
   326→      // Get PM type details
   327→      const pmType = await this.db.query.pmTypes.findFirst({
   328→        where: and(
   329→          eq(pmTypes.id, pmTypeId),
   330→          eq(pmTypes.active, true)
   331→        ),
   332→      });
   333→
   334→      if (!pmType) {
   335→        throw new NotFoundError(`PM type not found: ${pmTypeId}`, {
   336→          entityType: 'pm_type',
   337→          entityId: pmTypeId,
   338→        });
   339→      }
   340→
   341→      // Get last completed PM of this type for this train
   342→      const lastPm = await this.getLastCompletedPm(trainId, pmTypeId);
   343→
   344→      // Check if a higher-level PM has satisfied this requirement
   345→      const hierarchySatisfied = await this.checkHierarchySatisfaction(
   346→        trainId,
   347→        pmType,
   348→        lastPm?.completedAt ?? null
   349→      );
   350→
   351→      // Use hierarchy satisfaction if more recent
   352→      const effectiveLastPm = hierarchySatisfied?.satisfiedByPm &&
   353→        (!lastPm || new Date(hierarchySatisfied.satisfiedByPm.completedAt ?? 0) > new Date(lastPm.completedAt ?? 0))
   354→        ? hierarchySatisfied.satisfiedByPm
   355→        : lastPm;
   356→
   357→      const lastPmDate = effectiveLastPm?.completedAt
   358→        ? new Date(effectiveLastPm.completedAt).toISOString().split('T')[0]
   359→        : train.commissioningDate ?? null;
   360→      const lastPmMileageKm = effectiveLastPm?.completedMileageKm ?? 0;
   361→
   362→      // Calculate due date and mileage
   363→      const currentMileageKm = train.currentMileageKm ?? 0;
   364→      const today = new Date();
   365→      const todayStr = today.toISOString().split('T')[0];
   366→
   367→      // Get early/late tolerance
   368→      const earlyTolerancePct = Number(pmType.earlyTolerancePct) || DEFAULT_TOLERANCE_PCT;
   369→      const lateTolerancePct = Number(pmType.lateTolerancePct) || DEFAULT_TOLERANCE_PCT;
   370→
   371→      // Calculate mileage trigger
   372→      let dueMileageKm: number | null = null;
   373→      let kmUntilDue: number | null = null;
   374→      let mileageWindowStart: number | null = null;
   375→      let mileageWindowEnd: number | null = null;
   376→
   377→      if (pmType.intervalKm) {
   378→        dueMileageKm = lastPmMileageKm + pmType.intervalKm;
   379→        kmUntilDue = dueMileageKm - currentMileageKm;
   380→
   381→        const toleranceKm = pmType.intervalKm * (earlyTolerancePct / 100);
   382→        mileageWindowStart = dueMileageKm - toleranceKm;
   383→        mileageWindowEnd = dueMileageKm + (pmType.intervalKm * (lateTolerancePct / 100));
   384→      }
   385→
   386→      // Calculate time trigger
   387→      let dueDate: string | null = null;
   388→      let daysUntilDue: number | null = null;
   389→      let dateWindowStart: string | null = null;
   390→      let dateWindowEnd: string | null = null;
   391→
   392→      if (pmType.intervalMonths && lastPmDate) {
   393→        const lastDate = new Date(lastPmDate);
   394→        const dueDateTime = new Date(lastDate);
   395→        dueDateTime.setMonth(dueDateTime.getMonth() + pmType.intervalMonths);
   396→        dueDate = dueDateTime.toISOString().split('T')[0];
   397→        daysUntilDue = Math.ceil((dueDateTime.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   398→
   399→        const toleranceDays = Math.round((pmType.intervalMonths * 30) * (earlyTolerancePct / 100));
   400→        const lateDays = Math.round((pmType.intervalMonths * 30) * (lateTolerancePct / 100));
   401→
   402→        const windowStartDate = new Date(dueDateTime);
   403→        windowStartDate.setDate(windowStartDate.getDate() - toleranceDays);
   404→        dateWindowStart = windowStartDate.toISOString().split('T')[0];
   405→
   406→        const windowEndDate = new Date(dueDateTime);
   407→        windowEndDate.setDate(windowEndDate.getDate() + lateDays);
   408→        dateWindowEnd = windowEndDate.toISOString().split('T')[0];
   409→      }
   410→
   411→      // Determine trigger type (which comes first)
   412→      let triggerType: 'MILEAGE' | 'TIME' | 'BOTH' = 'BOTH';
   413→      let idealDate = todayStr;
   414→      let idealMileageKm = currentMileageKm;
   415→
   416→      if (dueMileageKm !== null && dueDate !== null) {
   417→        // Convert mileage due to date using daily rate
   418→        const dailyKm = await this.getAverageDailyKm(trainId);
   419→        const mileageDueDateMs = today.getTime() + ((kmUntilDue ?? 0) / dailyKm) * 24 * 60 * 60 * 1000;
   420→        const mileageDueDate = new Date(mileageDueDateMs);
   421→
   422→        const timeDueDate = new Date(dueDate);
   423→
   424→        if (mileageDueDate < timeDueDate) {
   425→          triggerType = 'MILEAGE';
   426→          idealDate = mileageDueDate.toISOString().split('T')[0];
   427→          idealMileageKm = dueMileageKm;
   428→        } else if (timeDueDate < mileageDueDate) {
   429→          triggerType = 'TIME';
   430→          idealDate = dueDate;
   431→          // Estimate mileage at time due date
   432→          const daysToTimeDue = Math.ceil((timeDueDate.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   433→          idealMileageKm = currentMileageKm + (daysToTimeDue * dailyKm);
   434→        } else {
   435→          triggerType = 'BOTH';
   436→          idealDate = dueDate;
   437→          idealMileageKm = dueMileageKm;
   438→        }
   439→      } else if (dueMileageKm !== null) {
   440→        triggerType = 'MILEAGE';
   441→        const dailyKm = await this.getAverageDailyKm(trainId);
   442→        const mileageDueDateMs = today.getTime() + ((kmUntilDue ?? 0) / dailyKm) * 24 * 60 * 60 * 1000;
   443→        idealDate = new Date(mileageDueDateMs).toISOString().split('T')[0];
   444→        idealMileageKm = dueMileageKm;
   445→      } else if (dueDate !== null) {
   446→        triggerType = 'TIME';
   447→        idealDate = dueDate;
   448→        const dailyKm = await this.getAverageDailyKm(trainId);
   449→        const daysToTimeDue = Math.ceil((new Date(dueDate).getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
   450→        idealMileageKm = currentMileageKm + (daysToTimeDue * dailyKm);
   451→      }
   452→
   453→      // Calculate compliance window dates
   454→      const dailyKm = await this.getAverageDailyKm(trainId);
   455→      let complianceStartDate = dateWindowStart ?? todayStr;
   456→      let complianceEndDate = dateWindowEnd ?? todayStr;
   457→
   458→      if (mileageWindowStart !== null) {
   459→        const kmToStart = mileageWindowStart - currentMileageKm;
   460→        const daysToStart = Math.max(0, Math.ceil(kmToStart / dailyKm));
   461→        const startDateFromMileage = new Date(today);
   462→        startDateFromMileage.setDate(startDateFromMileage.getDate() + daysToStart);
   463→
   464→        if (!dateWindowStart || startDateFromMileage < new Date(dateWindowStart)) {
   465→          complianceStartDate = startDateFromMileage.toISOString().split('T')[0];
   466→        }
   467→      }
   468→
   469→      if (mileageWindowEnd !== null) {
   470→        const kmToEnd = mileageWindowEnd - currentMileageKm;
   471→        const daysToEnd = Math.ceil(kmToEnd / dailyKm);
   472→        const endDateFromMileage = new Date(today);
   473→        endDateFromMileage.setDate(endDateFromMileage.getDate() + daysToEnd);
   474→
   475→        if (!dateWindowEnd || endDateFromMileage < new Date(dateWindowEnd)) {
   476→          complianceEndDate = endDateFromMileage.toISOString().split('T')[0];
   477→        }
   478→      }
   479→
   480→      // Ensure start date is not in the past
   481→      if (complianceStartDate < todayStr) {
   482→        complianceStartDate = todayStr;
   483→      }
   484→
   485→      // Check if overdue
   486→      const isOverdue = complianceEndDate < todayStr;
   487→
   488→      const result: PmDueResult = {
   489→        pmTypeId,
   490→        pmCode: pmType.pmCode,
   491→        pmName: pmType.pmName,
   492→        trainId,
   493→        trainNumber: train.trainNumber,
   494→        fleetId: train.fleetId,
   495→        dueMileageKm,
   496→        dueDate,
   497→        currentMileageKm,
   498→        kmUntilDue,
   499→        daysUntilDue,
   500→        triggerType,
