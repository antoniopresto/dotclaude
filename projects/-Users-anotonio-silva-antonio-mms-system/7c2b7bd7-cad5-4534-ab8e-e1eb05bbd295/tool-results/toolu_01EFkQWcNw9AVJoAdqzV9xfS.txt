   648→```typescript
   649→// Schema validation (replaces Zod)
   650→import { createSchema, createType, InferType } from '@belt/tools';
   651→
   652→const TrainModel = createSchema({
   653→  id: 'string',
   654→  trainNumber: 'string',
   655→  mileageKm: 'number',
   656→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   657→  tags: 'string[]',
   658→  metadata: 'object?',
   659→});
   660→type Train = InferType<typeof TrainModel>;
   661→const train = TrainModel.parse(data);
   662→
   663→// State machine (replaces React hooks + zustand)
   664→import { createSlice, createMachine } from '@belt/tools';
   665→
   666→const fleetSlice = createSlice({
   667→  id: 'fleet',
   668→  initial: { trains: [], selected: null },
   669→  cache: { policy: 'session' },
   670→  reducers: {
   671→    setTrains: (state, trains) => { state.trains = trains; },
   672→    selectTrain: (state, id) => { state.selected = id; },
   673→  },
   674→});
   675→
   676→const machine = createMachine({
   677→  id: 'mms',
   678→  slices: { fleet: fleetSlice },
   679→  trackMetadata: false, // Critical for 60fps
   680→});
   681→
   682→// In React component (NO useState/useEffect!)
   683→const { state, actions } = machine.useMachine('fleet');
