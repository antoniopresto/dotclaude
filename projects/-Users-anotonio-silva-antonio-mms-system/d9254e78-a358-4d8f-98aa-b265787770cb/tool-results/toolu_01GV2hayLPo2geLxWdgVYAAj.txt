   597→  trainNumber: 'string',
   598→  mileageKm: 'number',
   599→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   600→  tags: 'string[]',

   610→  id: 'fleet',
   611→  initial: { trains: [], selected: null },
   612→  cache: { policy: 'session' },
   613→  reducers: {
   614→    setTrains: (state, trains) => { state.trains = trains; },
   615→    selectTrain: (state, id) => { state.selected = id; },
   616→  },
   617→});
   618→

   620→  id: 'mms',
   621→  slices: { fleet: fleetSlice },
   622→  trackMetadata: false, // Critical for 60fps
   623→});
   624→
