    30→/**
    31→ * Main application state machine.
    32→ *
    33→ * Usage in React components:
    34→ * ```tsx
    35→ * function TrainList() {
    36→ *   const { state, actions } = mmsMachine.useMachine('fleet');
    37→ *   return (
    38→ *     <ul>
    39→ *       {state.trains.map(train => (
    40→ *         <li key={train.id} onClick={() => actions.selectTrain(train.id)}>
    41→ *           {train.trainNumber}
    42→ *         </li>
    43→ *       ))}
    44→ *     </ul>
    45→ *   );
    46→ * }
    47→ * ```
    48→ */
    49→export const mmsMachine = createMachine({
    50→  id: 'mms',
    51→  slices: {
    52→    fleet: fleetSlice,
    53→    fleetApi: fleetAsyncSlice,
    54→    canvas: canvasSlice,
    55→    canvasData: canvasDataSlice,
    56→    pm: pmSlice,
    57→    pmApi: pmAsyncSlice,
    58→    scenario: scenarioSlice,
    59→    scenarioApi: scenarioAsyncSlice,
    60→    ui: uiSlice,
    61→    overlay: overlaySlice,
    62→  },
    63→  context: {
    64→    // User ID for private cache policy
    65→    userID: null as string | null,
    66→    // Context properties for async slices
    67→    baseUrl: '/api/v1',
    68→    startDate: new Date().toISOString().split('T')[0],
    69→    endDate: new Date(Date.now() + 84 * 24 * 60 * 60 * 1000).toISOString().split('T')[0],
    70→  },
    71→});
