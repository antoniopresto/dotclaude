     1→/**
     2→ * Fleet Slice
     3→ *
     4→ * State management for fleet, train, depot, and route data.
     5→ * Uses @belt/tools createSlice for reactive state.
     6→ *
     7→ * @see specs/006-user-interface.md Section 6.4

    10→import { createSlice, createAsyncSlice } from '@belt/tools';
    11→import type {
    12→  Train,
    13→  Fleet,
    14→  Depot,
    15→  Route,
    16→  TrainFilters,
    17→  SortOptions,
    18→  FitnessStatus,
    19→} from '../types';
    20→
    21→// =============================================================================
    22→// Fleet Slice State
    23→// =============================================================================
    24→
    25→export interface FleetSliceState {
    26→  fleets: Fleet[];
    27→  trains: Train[];
    28→  depots: Depot[];
    29→  routes: Route[];
    30→  selectedTrainId: string | null;
    31→  selectedFleetId: string | null;
    32→  filters: TrainFilters;
    33→  sort: SortOptions;
    34→  lastUpdated: number | null;
    35→  [key: string]: unknown;
    36→}
    37→
    38→const initialFleetState: FleetSliceState = {
    39→  fleets: [],
    40→  trains: [],
    41→  depots: [],
    42→  routes: [],
    43→  selectedTrainId: null,
    44→  selectedFleetId: null,
    45→  filters: {
    46→    fleetIds: [],
    47→    fitnessStatuses: [],
    48→    depotIds: [],
    49→    searchTerm: '',
    50→  },
    51→  sort: {
    52→    field: 'trainNumber',
    53→    direction: 'asc',
    54→  },
    55→  lastUpdated: null,
    56→};
    57→
    58→export const fleetSlice = createSlice({
    59→  id: 'fleet',
    60→  initial: initialFleetState,
    61→  cache: { policy: 'session' },
    62→  reducers: {
    63→    // Fleet operations
    64→    setFleets: (s, fleets: Fleet[]) => {
    65→      const state = s as FleetSliceState;
    66→      state.fleets = fleets;
    67→      state.lastUpdated = Date.now();
    68→    },
    69→
    70→    // Train operations
    71→    setTrains: (s, trains: Train[]) => {
    72→      const state = s as FleetSliceState;
    73→      state.trains = trains;
    74→      state.lastUpdated = Date.now();
    75→    },
    76→
    77→    updateTrain: (s, train: Partial<Train> & { id: string }) => {
    78→      const state = s as FleetSliceState;
    79→      const index = state.trains.findIndex((t) => t.id === train.id);
    80→      if (index !== -1) {
    81→        state.trains[index] = { ...state.trains[index], ...train };
    82→        state.lastUpdated = Date.now();
    83→      }
    84→    },
    85→
    86→    updateTrainFitness: (
    87→      s,
    88→      payload: { trainId: string; status: FitnessStatus; notes?: string }
    89→    ) => {
    90→      const state = s as FleetSliceState;
    91→      const train = state.trains.find((t) => t.id === payload.trainId);
    92→      if (train) {
    93→        train.fitnessStatus = payload.status;
    94→        if (payload.notes !== undefined) {
    95→          train.fitnessNotes = payload.notes;
    96→        }
    97→        state.lastUpdated = Date.now();
    98→      }
    99→    },
   100→
   101→    updateTrainMileage: (s, payload: { trainId: string; mileageKm: number }) => {
   102→      const state = s as FleetSliceState;
   103→      const train = state.trains.find((t) => t.id === payload.trainId);
   104→      if (train) {
   105→        train.currentMileageKm = payload.mileageKm;
   106→        state.lastUpdated = Date.now();
   107→      }
   108→    },
   109→
   110→    // Depot operations
   111→    setDepots: (s, depots: Depot[]) => {
   112→      const state = s as FleetSliceState;
   113→      state.depots = depots;
   114→      state.lastUpdated = Date.now();
   115→    },
   116→
   117→    // Route operations
   118→    setRoutes: (s, routes: Route[]) => {
   119→      const state = s as FleetSliceState;
   120→      state.routes = routes;
   121→      state.lastUpdated = Date.now();
   122→    },
   123→
   124→    // Selection
   125→    selectTrain: (s, trainId: string | null) => {
   126→      const state = s as FleetSliceState;
   127→      state.selectedTrainId = trainId;
   128→    },
   129→
   130→    selectFleet: (s, fleetId: string | null) => {
   131→      const state = s as FleetSliceState;
   132→      state.selectedFleetId = fleetId;
   133→    },
   134→
   135→    clearSelection: (s) => {
   136→      const state = s as FleetSliceState;
   137→      state.selectedTrainId = null;
   138→      state.selectedFleetId = null;
   139→    },
   140→
   141→    // Filtering
   142→    setFilters: (s, filters: Partial<TrainFilters>) => {
   143→      const state = s as FleetSliceState;
   144→      state.filters = { ...state.filters, ...filters };
   145→    },
   146→
   147→    setFleetFilter: (s, fleetIds: string[]) => {
   148→      const state = s as FleetSliceState;
   149→      state.filters.fleetIds = fleetIds;
   150→    },
   151→
   152→    setFitnessFilter: (s, statuses: FitnessStatus[]) => {
   153→      const state = s as FleetSliceState;
   154→      state.filters.fitnessStatuses = statuses;
   155→    },
   156→
   157→    setDepotFilter: (s, depotIds: string[]) => {
   158→      const state = s as FleetSliceState;
   159→      state.filters.depotIds = depotIds;
   160→    },
   161→
   162→    setSearchTerm: (s, term: string) => {
   163→      const state = s as FleetSliceState;
   164→      state.filters.searchTerm = term;
   165→    },
   166→
   167→    clearFilters: (s) => {
   168→      const state = s as FleetSliceState;
   169→      state.filters = {
   170→        fleetIds: [],
   171→        fitnessStatuses: [],
   172→        depotIds: [],
   173→        searchTerm: '',
   174→      };
   175→    },
   176→
   177→    // Sorting
   178→    setSort: (s, sort: SortOptions) => {
   179→      const state = s as FleetSliceState;
   180→      state.sort = sort;
   181→    },
   182→
   183→    toggleSortDirection: (s) => {
   184→      const state = s as FleetSliceState;
   185→      state.sort.direction = state.sort.direction === 'asc' ? 'desc' : 'asc';
   186→    },
   187→
   188→    // Bulk operations
   189→    importFleetData: (
   190→      s,
   191→      payload: { fleets: Fleet[]; trains: Train[]; depots: Depot[]; routes: Route[] }
   192→    ) => {
   193→      const state = s as FleetSliceState;
   194→      state.fleets = payload.fleets;
   195→      state.trains = payload.trains;
   196→      state.depots = payload.depots;
   197→      state.routes = payload.routes;
   198→      state.lastUpdated = Date.now();
   199→    },
   200→
   201→    // Reset
   202→    reset: (s) => {
   203→      const state = s as FleetSliceState;
   204→      Object.assign(state, initialFleetState);
   205→    },
   206→  },
   207→});

   209→// =============================================================================
   210→// Async Fleet Slice
   211→// =============================================================================
   212→
   213→export interface FleetApiResponse {
   214→  fleets: Fleet[];
   215→  trains: Train[];
   216→  depots: Depot[];
   217→  routes: Route[];
   218→}
   219→
   220→export const fleetAsyncSlice = createAsyncSlice({
   221→  id: 'fleetApi',
   222→  cache: {
   223→    policy: 'public',
   224→    ttl: '5m',
   225→    swr: true,
   226→  },
   227→  context: { baseUrl: '/api/v1' },
   228→  load: async (ctx) => {
   229→    const baseUrl = (ctx as { baseUrl: string }).baseUrl;
   230→
   231→    const [fleetsRes, trainsRes, depotsRes, routesRes] = await Promise.all([
   232→      fetch(`${baseUrl}/fleets`),
   233→      fetch(`${baseUrl}/trains?pageSize=500`),
   234→      fetch(`${baseUrl}/depots`),
   235→      fetch(`${baseUrl}/routes`),
   236→    ]);
   237→
   238→    if (!fleetsRes.ok || !trainsRes.ok || !depotsRes.ok || !routesRes.ok) {
   239→      throw new Error('Failed to fetch fleet data');
   240→    }
   241→
   242→    const [fleetsData, trainsData, depotsData, routesData] = await Promise.all([
   243→      fleetsRes.json() as Promise<{ data: Fleet[] }>,
   244→      trainsRes.json() as Promise<{ data: Train[] }>,
   245→      depotsRes.json() as Promise<{ data: Depot[] }>,
   246→      routesRes.json() as Promise<{ data: Route[] }>,
   247→    ]);
   248→
   249→    return {
   250→      fleets: fleetsData.data,
   251→      trains: trainsData.data,
   252→      depots: depotsData.data,
   253→      routes: routesData.data,
   254→    };
   255→  },
   256→});

   262→export function selectFilteredTrains(state: FleetSliceState): Train[] {
   263→  let filtered = [...state.trains];
   264→
   265→  if (state.filters.fleetIds.length > 0) {
   266→    filtered = filtered.filter((t) => state.filters.fleetIds.includes(t.fleetId));
   267→  }
   268→
   269→  if (state.filters.fitnessStatuses.length > 0) {
   270→    filtered = filtered.filter((t) => state.filters.fitnessStatuses.includes(t.fitnessStatus));
   271→  }
   272→
   273→  if (state.filters.depotIds.length > 0) {
   274→    filtered = filtered.filter(
   275→      (t) => t.currentDepotId && state.filters.depotIds.includes(t.currentDepotId)
   276→    );
   277→  }
   278→
   279→  if (state.filters.searchTerm) {
   280→    const term = state.filters.searchTerm.toLowerCase();
   281→    filtered = filtered.filter(
   282→      (t) =>
   283→        t.trainNumber.toLowerCase().includes(term) ||
   284→        t.fleetCode.toLowerCase().includes(term) ||
   285→        (t.fitnessNotes && t.fitnessNotes.toLowerCase().includes(term))
   286→    );
   287→  }
   288→
   289→  filtered.sort((a, b) => {
   290→    const field = state.sort.field as keyof Train;
   291→    const aVal = a[field];
   292→    const bVal = b[field];
   293→
   294→    if (aVal === null || aVal === undefined) return 1;
   295→    if (bVal === null || bVal === undefined) return -1;
   296→
   297→    const comparison = aVal < bVal ? -1 : aVal > bVal ? 1 : 0;
   298→    return state.sort.direction === 'asc' ? comparison : -comparison;
   299→  });
   300→
   301→  return filtered;
   302→}

   304→export function selectTrainById(state: FleetSliceState, trainId: string): Train | undefined {
   305→  return state.trains.find((t) => t.id === trainId);
   306→}
   307→
   308→export function selectFleetById(state: FleetSliceState, fleetId: string): Fleet | undefined {
   309→  return state.fleets.find((f) => f.id === fleetId);
   310→}
   311→
   312→export function selectDepotById(state: FleetSliceState, depotId: string): Depot | undefined {
   313→  return state.depots.find((d) => d.id === depotId);
   314→}
   315→
   316→export function selectSelectedTrain(state: FleetSliceState): Train | undefined {
   317→  return state.selectedTrainId
   318→    ? state.trains.find((t) => t.id === state.selectedTrainId)
   319→    : undefined;
   320→}
   321→
   322→export function selectTrainsByFleet(state: FleetSliceState, fleetId: string): Train[] {
   323→  return state.trains.filter((t) => t.fleetId === fleetId);
   324→}
   325→
   326→export function selectFleetStats(state: FleetSliceState): {
   327→  totalTrains: number;
   328→  fitTrains: number;
   329→  restrictedTrains: number;
   330→  groundedTrains: number;
   331→  totalMileage: number;
   332→  avgMileage: number;
   333→} {
   334→  const trains = state.trains;
   335→  const fitTrains = trains.filter((t) => t.fitnessStatus === 'FIT').length;
   336→  const restrictedTrains = trains.filter((t) => t.fitnessStatus === 'RESTRICTED').length;
   337→  const groundedTrains = trains.filter((t) => t.fitnessStatus === 'GROUNDED').length;
   338→  const totalMileage = trains.reduce((sum, t) => sum + t.currentMileageKm, 0);
   339→
   340→  return {
   341→    totalTrains: trains.length,
   342→    fitTrains,
   343→    restrictedTrains,
   344→    groundedTrains,
   345→    totalMileage,
   346→    avgMileage: trains.length > 0 ? Math.round(totalMileage / trains.length) : 0,
   347→  };
   348→}
