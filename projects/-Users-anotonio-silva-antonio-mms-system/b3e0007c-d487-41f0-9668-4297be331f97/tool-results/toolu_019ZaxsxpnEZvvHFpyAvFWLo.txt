    20→function AppContent() {

    24→
    25→  const uiState = uiStateRaw as unknown as UiSliceState;

    28→
    29→  // Apply theme class to body
    30→  const themeClass = uiState.theme === 'dark' ? Classes.DARK : '';
    31→
    32→  // Determine if any overlay panels should be visible
    33→  const showManhourPanel = canvasState.layers?.manhourOverlay ?? false;
    34→  const showDriftPanel = canvasState.layers?.driftOverlay ?? false;
    35→  const showScenarioComparison = scenarioState.activeScenarioId && scenarioState.comparisonBaselineId;
    36→  const hasRightPanel = showManhourPanel || showDriftPanel || showScenarioComparison;
    37→
    38→  return (
    39→    <div className={`mms-app ${themeClass}`} data-testid="mms-app">

    71→            }}>
    72→              {showManhourPanel && <ManHourOverlayPanel />}
    73→              {showDriftPanel && <DriftOverlayPanel />}

    79→        {/* Status Bar */}

    83→      {/* Modal Container - renders active modals */}

    86→      {/* Toast Notifications - positioned top-right */}
