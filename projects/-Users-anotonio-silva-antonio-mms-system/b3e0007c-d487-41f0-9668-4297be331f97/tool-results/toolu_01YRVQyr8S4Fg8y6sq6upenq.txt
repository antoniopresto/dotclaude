   395→  const handlePmDragStart = (pmId: string, sourceDate: string) => {
   396→    // Find PM in tactical data
   397→    const pm = tacticalData?.pmBlocks.find(p => p.pmId === pmId);
   398→    if (!pm) return;
   399→
   400→    // Store drag info
   401→    draggingPmRef.current = {
   402→      pmId,
   403→      pmCode: pm.pmCode,
   404→      trainId: pm.trainId,
   405→      trainNumber: pm.trainNumber,
   406→      sourceDate,
   407→      constraintStart: pm.constraintStart,
   408→      constraintEnd: pm.constraintEnd,
   409→      isLocked: pm.isLocked,
   410→    };
   411→
   412→    // Initialize drag state in PM slice
   413→    pmActions.startDrag({ pmId, sourceDate });
   414→
   415→    // Set initial validation (quick check)
   416→    const quickResult = quickValidatePmMove({
   417→      pmId,
   418→      currentDate: sourceDate,
   419→      targetDate: sourceDate,
   420→      trainId: pm.trainId,
   421→      constraintStart: pm.constraintStart,
   422→      constraintEnd: pm.constraintEnd,
   423→      isLocked: pm.isLocked,
   424→    });

   426→    pmActions.setDragValidation({
   427→      isValid: quickResult.isValid,
   428→      message: quickResult.message,
   429→      violations: quickResult.violations,
   430→    });
   431→  };
   432→
   433→  const handlePmDragMove = (pmId: string, targetDate: string) => {
   434→    const dragInfo = draggingPmRef.current;
   435→    if (!dragInfo || dragInfo.pmId !== pmId) return;
   436→
   437→    // Update target date in PM slice
   438→    pmActions.updateDragTarget(targetDate);
   439→
   440→    // Quick client-side validation
   441→    const quickResult = quickValidatePmMove({
   442→      pmId,
   443→      currentDate: dragInfo.sourceDate,
   444→      targetDate,
   445→      trainId: dragInfo.trainId,
   446→      constraintStart: dragInfo.constraintStart,
   447→      constraintEnd: dragInfo.constraintEnd,
   448→      isLocked: dragInfo.isLocked,
   449→    });
   450→    dragValidationRef.current = quickResult;
   451→    pmActions.setDragValidation({
   452→      isValid: quickResult.isValid,
   453→      message: quickResult.message,
   454→      violations: quickResult.violations,
   455→    });
   456→
   457→    // If quick validation passes, trigger debounced API validation
   458→    if (quickResult.isValid && targetDate !== dragInfo.sourceDate) {
   459→      debouncedValidationRef.current.validate({
   460→        pmId,
   461→        currentDate: dragInfo.sourceDate,
   462→        targetDate,
   463→        trainId: dragInfo.trainId,
   464→      });
   465→    }
   466→  };
   467→
   468→  const handlePmDragEnd = async (pmId: string, targetDate: string) => {
   469→    const dragInfo = draggingPmRef.current;
   470→    if (!dragInfo || dragInfo.pmId !== pmId) {
   471→      // Clear drag state
   472→      pmActions.cancelDrag();
   473→      draggingPmRef.current = null;
   474→      return;
   475→    }
   476→
   477→    // Cancel pending validation
   478→    debouncedValidationRef.current.cancel();
   479→
   480→    // If no change, just cancel
   481→    if (targetDate === dragInfo.sourceDate) {
   482→      pmActions.cancelDrag();
   483→      draggingPmRef.current = null;
   484→      return;
   485→    }
   486→
   487→    // Final quick validation
   488→    const finalValidation = quickValidatePmMove({
   489→      pmId,
   490→      currentDate: dragInfo.sourceDate,
   491→      targetDate,
   492→      trainId: dragInfo.trainId,
   493→      constraintStart: dragInfo.constraintStart,
   494→      constraintEnd: dragInfo.constraintEnd,
   495→      isLocked: dragInfo.isLocked,
   496→    });
   497→
   498→    // If invalid, snap back and show toast
   499→    if (!finalValidation.isValid) {
   500→      pmActions.cancelDrag();
   501→      draggingPmRef.current = null;
   502→      void showToast({
   503→        intent: Intent.DANGER,
   504→        icon: 'error',
   505→        message: finalValidation.message || 'Cannot reschedule: constraint violation',
   506→        timeout: 5000,
   507→      });
   508→      return;
   509→    }
   510→
   511→    // Open confirmation dialog
   512→    dragConfirmRef.current = {
   513→      isOpen: true,
   514→      pmId,
   515→      pmCode: dragInfo.pmCode,
   516→      trainNumber: dragInfo.trainNumber,
   517→      sourceDate: dragInfo.sourceDate,
   518→      targetDate,
   519→      impact: null,
   520→      isLoadingImpact: true,
   521→    };
   522→    // Force re-render by ending drag with commit=false (optimistic update)
   523→    pmActions.endDrag(false);
   524→
   525→    // Load impact assessment
   526→    try {
   527→      const impact = await getPmMoveImpact({
   528→        pmId,
   529→        currentDate: dragInfo.sourceDate,
   530→        targetDate,
   531→        trainId: dragInfo.trainId,
   532→      });
   533→      dragConfirmRef.current = {
   534→        ...dragConfirmRef.current,
   535→        impact,
   536→        isLoadingImpact: false,
   537→      };
   538→    } catch {
   539→      dragConfirmRef.current = {
   540→        ...dragConfirmRef.current,
   541→        isLoadingImpact: false,
   542→      };
   543→    }
   544→
   545→    // Keep draggingPmRef for the confirmation dialog
   546→  };
   547→
   548→  const handleDragConfirm = async () => {
   549→    const confirmState = dragConfirmRef.current;
   550→    const dragInfo = draggingPmRef.current;
   551→
   552→    if (!confirmState.pmId || !dragInfo) {
   553→      handleDragCancel();
   554→      return;
   555→    }
   556→
   557→    try {
   558→      // Commit the move to API
   559→      await commitPmMove({
   560→        pmId: confirmState.pmId,
   561→        currentDate: confirmState.sourceDate,
   562→        targetDate: confirmState.targetDate,
   563→        trainId: dragInfo.trainId,
   564→        reason: 'Manual reschedule via canvas drag-and-drop',
   565→      });
   566→
   567→      // Confirm move in PM slice (updates local state)
   568→      pmActions.confirmMove(confirmState.pmId);
   569→
   570→      // Show success toast
   571→      void showToast({
   572→        intent: Intent.SUCCESS,
   573→        icon: 'tick',
   574→        message: `${confirmState.pmCode} rescheduled to ${confirmState.targetDate}`,
   575→        timeout: 3000,
   576→      });
   577→
   578→      // Reload tactical data to reflect changes
   579→      handleLoadTacticalData();
   580→    } catch (error) {
   581→      // Reject move (revert to original date)
   582→      pmActions.rejectMove(confirmState.pmId);
   583→
   584→      // Show error toast
   585→      void showToast({
   586→        intent: Intent.DANGER,
   587→        icon: 'error',
   588→        message: error instanceof Error ? error.message : 'Failed to reschedule PM',
   589→        timeout: 5000,
   590→      });
   591→    } finally {
   592→      // Close dialog and clear state
   593→      dragConfirmRef.current = initialDragConfirmState;
   594→      draggingPmRef.current = null;
   595→    }
   596→  };
   597→
   598→  const handleDragCancel = () => {
   599→    const confirmState = dragConfirmRef.current;
   600→
   601→    // Revert the move if pending
   602→    if (confirmState.pmId) {
   603→      pmActions.rejectMove(confirmState.pmId);
   604→    }
   605→
   606→    // Clear state
   607→    dragConfirmRef.current = initialDragConfirmState;
   608→    draggingPmRef.current = null;
   609→  };
   610→
   611→  const handlePmBlockHover = (pm: PmBlock | null) => {
