   121→# Cross-platform date from timestamp
   122→date_from_timestamp() {
   123→    local timestamp="$1"
   124→    if date --version >/dev/null 2>&1; then
   125→        # GNU date (Linux)
   126→        date -d "@$timestamp" '+%Y-%m-%d %H:%M:%S' 2>/dev/null || echo "N/A"
   127→    else
   128→        # BSD date (macOS)
   129→        date -r "$timestamp" '+%Y-%m-%d %H:%M:%S' 2>/dev/null || echo "N/A"
   130→    fi
   131→}

   207→        grep -E "FATAL|CRITICAL|Exception|Traceback" "$output_file" | head -1

   220→push_changes() {
   221→    local branch="$1"
   222→    local max_retries=3
   223→    local retry=0
   224→    
   225→    while [[ $retry -lt $max_retries ]]; do
   226→        if git push origin "$branch" 2>&1; then
   227→            return 0
   228→        fi
   229→        
   230→        retry=$((retry + 1))
   231→        
   232→        if [[ $retry -eq 1 ]]; then
   233→            log_warn "Push failed. Trying to create remote branch..."
   234→            if git push -u origin "$branch" 2>&1; then
   235→                return 0
   236→            fi
   237→        fi
   238→        
   239→        log_warn "Push attempt $retry/$max_retries failed. Retrying in 5s..."
   240→        sleep 5
   241→    done
   242→    
   243→    return 1
   244→}

   246→# Run Claude CLI with retry logic for transient failures
   247→run_claude_with_retry() {
   248→    local prompt_file="$1"
   249→    local output_file="$2"
   250→    local mode="$3"
   251→    local retry=0
   252→    local exit_code
   253→    
   254→    while [[ $retry -lt $MAX_RETRIES ]]; do
   255→        if [[ "$mode" == "plan-work" ]]; then
   256→            envsubst < "$prompt_file" | claude -p \
   257→                --dangerously-skip-permissions \
   258→                --output-format=stream-json \
   259→                --model "$MODEL" \
   260→                $VERBOSE 2>&1 | tee "$output_file"
   261→            exit_code=${PIPESTATUS[0]}
   262→        else
   263→            cat "$prompt_file" | claude -p \
   264→                --dangerously-skip-permissions \
   265→                --output-format=stream-json \
   266→                --model "$MODEL" \
   267→                $VERBOSE 2>&1 | tee "$output_file"
   268→            exit_code=${PIPESTATUS[0]}
   269→        fi
   270→        
   271→        # Check for transient errors that should be retried
   272→        if [[ $exit_code -eq 0 ]]; then
   273→            return 0
   274→        fi
   275→        
   276→        # Check if it's a retryable error
   277→        if grep -qE "ETIMEDOUT|ECONNRESET|ECONNREFUSED|503|504|rate.?limit" "$output_file" 2>/dev/null; then
   278→            retry=$((retry + 1))
   279→            if [[ $retry -lt $MAX_RETRIES ]]; then
   280→                log_warn "Transient error detected. Retrying in ${RETRY_DELAY}s... (attempt $((retry + 1))/$MAX_RETRIES)"
   281→                sleep "$RETRY_DELAY"
   282→                continue
   283→            fi
   284→        fi
   285→        
   286→        # Non-retryable error or max retries reached
   287→        return $exit_code
   288→    done
   289→    
   290→    return 1
   291→}
