     1→/**
     2→ * MMS Infinite Canvas
     3→ *
     4→ * The main canvas component that renders PM schedule data at different zoom levels.
     5→ * Uses deck.gl for strategic view (WebGL) and Konva for tactical/operational views (Canvas 2D).
     6→ *
     7→ * Zoom Level Rendering:
     8→ * - STRATEGIC (0-4): deck.gl WebGL heatmap
     9→ * - TACTICAL_WIDE (4-8): deck.gl WebGL compressed blocks
    10→ * - TACTICAL (8-12): Konva HTML5 Canvas Gantt
    11→ * - OPERATIONAL (12-16): Konva HTML5 Canvas detailed cards
    12→ * - DETAIL (16+): React DOM overlay
    13→ *
    14→ * Based on specs/006-user-interface.md Section 6.3.
    15→ */

   332→  const handlePmDragStart = (pmId: string, sourceDate: string) => {
   333→    // Find PM in tactical data
   334→    const pm = tacticalData?.pmBlocks.find(p => p.pmId === pmId);
   335→    if (!pm) return;
   336→
   337→    // Store drag info
   338→    draggingPmRef.current = {
   339→      pmId,
   340→      pmCode: pm.pmCode,
   341→      trainId: pm.trainId,
   342→      trainNumber: pm.trainNumber,
   343→      sourceDate,
   344→      constraintStart: pm.constraintStart,
   345→      constraintEnd: pm.constraintEnd,
   346→      isLocked: pm.isLocked,
   347→    };
   348→
   349→    // Initialize drag state in PM slice
   350→    pmActions.startDrag({ pmId, sourceDate });
   351→
   352→    // Set initial validation (quick check)
   353→    const quickResult = quickValidatePmMove({
   354→      pmId,
   355→      currentDate: sourceDate,
   356→      targetDate: sourceDate,
   357→      trainId: pm.trainId,
   358→      constraintStart: pm.constraintStart,
   359→      constraintEnd: pm.constraintEnd,
   360→      isLocked: pm.isLocked,
   361→    });
   362→    dragValidationRef.current = quickResult;
   363→    pmActions.setDragValidation({
   364→      isValid: quickResult.isValid,
   365→      message: quickResult.message,
   366→      violations: quickResult.violations,
   367→    });
   368→  };
   369→
   370→  const handlePmDragMove = (pmId: string, targetDate: string) => {
   371→    const dragInfo = draggingPmRef.current;
   372→    if (!dragInfo || dragInfo.pmId !== pmId) return;
   373→
   374→    // Update target date in PM slice
   375→    pmActions.updateDragTarget(targetDate);
   376→
   377→    // Quick client-side validation
   378→    const quickResult = quickValidatePmMove({
   379→      pmId,
   380→      currentDate: dragInfo.sourceDate,
   381→      targetDate,
   382→      trainId: dragInfo.trainId,
   383→      constraintStart: dragInfo.constraintStart,
   384→      constraintEnd: dragInfo.constraintEnd,
   385→      isLocked: dragInfo.isLocked,
   386→    });
   387→    dragValidationRef.current = quickResult;
   388→    pmActions.setDragValidation({
   389→      isValid: quickResult.isValid,
   390→      message: quickResult.message,
   391→      violations: quickResult.violations,
   392→    });
   393→
   394→    // If quick validation passes, trigger debounced API validation
   395→    if (quickResult.isValid && targetDate !== dragInfo.sourceDate) {
   396→      debouncedValidationRef.current.validate({
   397→        pmId,
   398→        currentDate: dragInfo.sourceDate,
   399→        targetDate,
   400→        trainId: dragInfo.trainId,
   401→      });
   402→    }
   403→  };
   404→
   405→  const handlePmDragEnd = async (pmId: string, targetDate: string) => {
   406→    const dragInfo = draggingPmRef.current;
   407→    if (!dragInfo || dragInfo.pmId !== pmId) {
   408→      // Clear drag state
   409→      pmActions.cancelDrag();
   410→      draggingPmRef.current = null;
   411→      return;
   412→    }
   413→
   414→    // Cancel pending validation
   415→    debouncedValidationRef.current.cancel();
   416→
   417→    // If no change, just cancel
   418→    if (targetDate === dragInfo.sourceDate) {
   419→      pmActions.cancelDrag();
   420→      draggingPmRef.current = null;
   421→      return;
   422→    }
   423→
   424→    // Final quick validation
   425→    const finalValidation = quickValidatePmMove({
   426→      pmId,
   427→      currentDate: dragInfo.sourceDate,
   428→      targetDate,
   429→      trainId: dragInfo.trainId,
   430→      constraintStart: dragInfo.constraintStart,
   431→      constraintEnd: dragInfo.constraintEnd,
   432→      isLocked: dragInfo.isLocked,
   433→    });
   434→
   435→    // If invalid, snap back and show toast
   436→    if (!finalValidation.isValid) {
   437→      pmActions.cancelDrag();
   438→      draggingPmRef.current = null;
   439→      void showToast({
   440→        intent: Intent.DANGER,
   441→        icon: 'error',
   442→        message: finalValidation.message || 'Cannot reschedule: constraint violation',
   443→        timeout: 5000,
   444→      });
   445→      return;
   446→    }
   447→
   448→    // Open confirmation dialog
   449→    dragConfirmRef.current = {
   450→      isOpen: true,
   451→      pmId,
   452→      pmCode: dragInfo.pmCode,
   453→      trainNumber: dragInfo.trainNumber,
   454→      sourceDate: dragInfo.sourceDate,
   455→      targetDate,
   456→      impact: null,
   457→      isLoadingImpact: true,
   458→    };
   459→    // Force re-render by ending drag with commit=false (optimistic update)
   460→    pmActions.endDrag(false);
   461→
   462→    // Load impact assessment
   463→    try {
   464→      const impact = await getPmMoveImpact({
   465→        pmId,
   466→        currentDate: dragInfo.sourceDate,
   467→        targetDate,
   468→        trainId: dragInfo.trainId,
   469→      });
   470→      dragConfirmRef.current = {
   471→        ...dragConfirmRef.current,
   472→        impact,
   473→        isLoadingImpact: false,
   474→      };
   475→    } catch {
   476→      dragConfirmRef.current = {
   477→        ...dragConfirmRef.current,
   478→        isLoadingImpact: false,
   479→      };
   480→    }
   481→
   482→    // Keep draggingPmRef for the confirmation dialog
   483→  };
   484→
   485→  const handleDragConfirm = async () => {
   486→    const confirmState = dragConfirmRef.current;
   487→    const dragInfo = draggingPmRef.current;
   488→
   489→    if (!confirmState.pmId || !dragInfo) {
   490→      handleDragCancel();
   491→      return;
   492→    }
   493→
   494→    try {
   495→      // Commit the move to API
   496→      await commitPmMove({
   497→        pmId: confirmState.pmId,
   498→        currentDate: confirmState.sourceDate,
   499→        targetDate: confirmState.targetDate,
   500→        trainId: dragInfo.trainId,
   501→        reason: 'Manual reschedule via canvas drag-and-drop',
   502→      });
   503→
   504→      // Confirm move in PM slice (updates local state)
   505→      pmActions.confirmMove(confirmState.pmId);
   506→
   507→      // Show success toast
   508→      void showToast({
   509→        intent: Intent.SUCCESS,
   510→        icon: 'tick',
   511→        message: `${confirmState.pmCode} rescheduled to ${confirmState.targetDate}`,
   512→        timeout: 3000,
   513→      });
   514→
   515→      // Reload tactical data to reflect changes
   516→      handleLoadTacticalData();
   517→    } catch (error) {
   518→      // Reject move (revert to original date)
   519→      pmActions.rejectMove(confirmState.pmId);
   520→
   521→      // Show error toast
   522→      void showToast({
   523→        intent: Intent.DANGER,
   524→        icon: 'error',
   525→        message: error instanceof Error ? error.message : 'Failed to reschedule PM',
   526→        timeout: 5000,
   527→      });
   528→    } finally {
   529→      // Close dialog and clear state
   530→      dragConfirmRef.current = initialDragConfirmState;
   531→      draggingPmRef.current = null;
   532→    }
   533→  };
   534→
   535→  const handleDragCancel = () => {
   536→    const confirmState = dragConfirmRef.current;
   537→
   538→    // Revert the move if pending
   539→    if (confirmState.pmId) {
   540→      pmActions.rejectMove(confirmState.pmId);
   541→    }
   542→
   543→    // Clear state
   544→    dragConfirmRef.current = initialDragConfirmState;
   545→    draggingPmRef.current = null;
   546→  };

   580→    if (tooltipRef.current) {

   584→        tooltipRef.current.style.top = `${tooltipPosRef.current.y + 10}px`;
   585→        tooltipRef.current.innerHTML = `
