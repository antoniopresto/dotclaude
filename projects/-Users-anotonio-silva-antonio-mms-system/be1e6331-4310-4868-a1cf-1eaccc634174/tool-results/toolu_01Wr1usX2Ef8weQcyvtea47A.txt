   637→```typescript
   638→// Schema validation (replaces Zod)
   639→import { createSchema, createType, InferType } from '@belt/tools';
   640→
   641→const TrainModel = createSchema({
   642→  id: 'string',
   643→  trainNumber: 'string',
   644→  mileageKm: 'number',
   645→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   646→  tags: 'string[]',
   647→  metadata: 'object?',
   648→});
   649→type Train = InferType<typeof TrainModel>;
   650→const train = TrainModel.parse(data);
   651→
   652→// State machine (replaces React hooks + zustand)
   653→import { createSlice, createMachine } from '@belt/tools';
   654→
   655→const fleetSlice = createSlice({
   656→  id: 'fleet',
   657→  initial: { trains: [], selected: null },
   658→  cache: { policy: 'session' },
   659→  reducers: {
   660→    setTrains: (state, trains) => { state.trains = trains; },
   661→    selectTrain: (state, id) => { state.selected = id; },
   662→  },
   663→});
   664→
   665→const machine = createMachine({
   666→  id: 'mms',
   667→  slices: { fleet: fleetSlice },
   668→  trackMetadata: false, // Critical for 60fps
   669→});
   670→
   671→// In React component (NO useState/useEffect!)
   672→const { state, actions } = machine.useMachine('fleet');
