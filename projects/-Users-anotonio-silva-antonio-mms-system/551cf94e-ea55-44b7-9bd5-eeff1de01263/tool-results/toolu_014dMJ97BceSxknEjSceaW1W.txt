   617→```typescript
   618→// Schema validation (replaces Zod)
   619→import { createSchema, createType, InferType } from '@belt/tools';
   620→
   621→const TrainModel = createSchema({
   622→  id: 'string',
   623→  trainNumber: 'string',
   624→  mileageKm: 'number',
   625→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   626→  tags: 'string[]',
   627→  metadata: 'object?',
   628→});
   629→type Train = InferType<typeof TrainModel>;
   630→const train = TrainModel.parse(data);
   631→

   633→import { createSlice, createMachine } from '@belt/tools';
   634→

   636→  id: 'fleet',
   637→  initial: { trains: [], selected: null },
   638→  cache: { policy: 'session' },
   639→  reducers: {
   640→    setTrains: (state, trains) => { state.trains = trains; },
   641→    selectTrain: (state, id) => { state.selected = id; },
   642→  },
   643→});
   644→
   645→const machine = createMachine({
   646→  id: 'mms',
   647→  slices: { fleet: fleetSlice },
   648→  trackMetadata: false, // Critical for 60fps
   649→});
   650→
   651→// In React component (NO useState/useEffect!)
   652→const { state, actions } = machine.useMachine('fleet');
