   398→// Schema validation (replaces Zod)
   399→import { createSchema, createType, InferType } from '@belt/tools';

   403→  trainNumber: 'string',
   404→  mileageKm: 'number',
   405→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   406→  tags: 'string[]',
   407→  metadata: 'object?',
   408→});
   409→type Train = InferType<typeof TrainModel>;
   410→const train = TrainModel.parse(data);

   413→import { createSlice, createMachine } from '@belt/tools';
   414→

   416→  id: 'fleet',
   417→  initial: { trains: [], selected: null },
   418→  cache: { policy: 'session' },
   419→  reducers: {
   420→    setTrains: (state, trains) => { state.trains = trains; },
   421→    selectTrain: (state, id) => { state.selected = id; },
   422→  },
   423→});
   424→
   425→const machine = createMachine({
   426→  id: 'mms',
   427→  slices: { fleet: fleetSlice },
   428→  trackMetadata: false, // Critical for 60fps
   429→});
   430→
   431→// In React component (NO useState/useEffect!)
   432→const { state, actions } = machine.useMachine('fleet');
