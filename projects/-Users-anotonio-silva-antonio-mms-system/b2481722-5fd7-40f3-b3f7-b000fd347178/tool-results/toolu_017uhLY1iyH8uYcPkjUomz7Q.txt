     1→/**
     2→ * Baseline Assumption Routes
     3→ *
     4→ * REST endpoints for managing PM baseline assumptions during cold start
     5→ * data migration. Includes manual review queue for LOW/ESTIMATED confidence
     6→ * assumptions.
     7→ *
     8→ * Key features:
     9→ * - CRUD operations for baseline assumptions
    10→ * - Review queue for manual verification
    11→ * - Inference rules (from commissioning date, higher PM)
    12→ * - Review/approval workflow
    13→ *
    14→ * @see specs/003-database-schema.md Section 3.2.6
    15→ * @see ADR-014: PM History Assumptions for Cold Start
    16→ */

    53→/**
    54→ * GET /baseline-assumptions/review-queue
    55→ *
    56→ * Get assumptions requiring manual review.
    57→ * Returns assumptions with LOW or ESTIMATED confidence that have not been reviewed.
    58→ *
    59→ * Query Parameters:
    60→ * - page: Page number (default: 1)
    61→ * - pageSize: Items per page (default: 20, max: 100)
    62→ * - sortBy: Sort field (createdAt, confidenceLevel, trainNumber)
    63→ * - sortOrder: Sort direction (asc, desc) - default: asc (oldest first)
    64→ */
    65→baselineAssumptionRoutes.get('/baseline-assumptions/review-queue', async (c) => {
    66→  const service = getService();
    67→
    68→  const pagination = parsePaginationQuery({
    69→    page: c.req.query('page') ? Number(c.req.query('page')) : undefined,
    70→    pageSize: c.req.query('pageSize') ? Number(c.req.query('pageSize')) : undefined,
    71→    sort: c.req.query('sortBy')
    72→      ? `${c.req.query('sortBy')}:${c.req.query('sortOrder') ?? 'asc'}`
    73→      : undefined,
    74→  });
    75→
    76→  const result = await service.getReviewQueue({
    77→    page: pagination.page,
    78→    pageSize: Math.min(pagination.pageSize, 100),
    79→    sortBy: (c.req.query('sortBy') as 'createdAt' | 'confidenceLevel' | 'trainNumber') ?? 'createdAt',
    80→    sortOrder: (c.req.query('sortOrder') as 'asc' | 'desc') ?? 'asc',
    81→  });
    82→
    83→  return c.json(
    84→    paginatedSuccess(
    85→      result.items.map(mapAssumptionToResponse),
    86→      {
    87→        page: result.page,
    88→        pageSize: result.pageSize,
    89→        totalItems: result.total,
    90→      },
    91→      { durationMs: getRequestDuration(c) }
    92→    )
    93→  );
    94→});

   124→/**
   125→ * GET /baseline-assumptions
   126→ *
   127→ * List baseline assumptions with filtering and pagination.
   128→ *
   129→ * Query Parameters:
   130→ * - trainId: Filter by train ID
   131→ * - pmTypeId: Filter by PM type ID
   132→ * - confidenceLevel: Filter by confidence level (can be comma-separated)
   133→ * - assumptionType: Filter by assumption type (can be comma-separated)
   134→ * - requiresReview: Filter by review requirement (true/false)
   135→ * - reviewed: Filter by reviewed status (true/false)
   136→ * - applied: Filter by applied status (true/false)
   137→ * - page, pageSize, sortBy, sortOrder: Pagination
   138→ */
   139→baselineAssumptionRoutes.get('/baseline-assumptions', async (c) => {
   140→  const service = getService();
   141→
   142→  // Parse query filters
   143→  const filters: AssumptionFilters = {};
   144→
   145→  const trainId = c.req.query('trainId');
   146→  if (trainId && isValidUuid(trainId)) {
   147→    filters.trainId = trainId;
   148→  }
   149→
   150→  const pmTypeId = c.req.query('pmTypeId');
   151→  if (pmTypeId && isValidUuid(pmTypeId)) {
   152→    filters.pmTypeId = pmTypeId;
   153→  }
   154→
   155→  const confidenceLevel = c.req.query('confidenceLevel');
   156→  if (confidenceLevel) {
   157→    const validLevels: ConfidenceLevel[] = ['HIGH', 'MEDIUM', 'LOW', 'ESTIMATED'];
   158→    const levels = confidenceLevel
   159→      .split(',')
   160→      .filter((l) => validLevels.includes(l as ConfidenceLevel)) as ConfidenceLevel[];
   161→    if (levels.length === 1) {
   162→      filters.confidenceLevel = levels[0];
   163→    } else if (levels.length > 1) {
   164→      filters.confidenceLevel = levels;
   165→    }
   166→  }
   167→
   168→  const assumptionType = c.req.query('assumptionType');
   169→  if (assumptionType) {
   170→    const validTypes: AssumptionType[] = [
   171→      'INFERRED_FROM_COMMISSIONING',
   172→      'INFERRED_FROM_HIGHER_PM',
   173→      'MANUAL_OVERRIDE',
   174→      'IMPORTED',
   175→      'NO_HISTORY',
   176→    ];
   177→    const types = assumptionType
   178→      .split(',')
   179→      .filter((t) => validTypes.includes(t as AssumptionType)) as AssumptionType[];
   180→    if (types.length === 1) {
   181→      filters.assumptionType = types[0];
   182→    } else if (types.length > 1) {
   183→      filters.assumptionType = types;
   184→    }
   185→  }
   186→
   187→  const requiresReview = c.req.query('requiresReview');
   188→  if (requiresReview === 'true' || requiresReview === 'false') {
   189→    filters.requiresReview = requiresReview === 'true';
   190→  }
   191→
   192→  const reviewed = c.req.query('reviewed');
   193→  if (reviewed === 'true' || reviewed === 'false') {
   194→    filters.reviewed = reviewed === 'true';
   195→  }
   196→
   197→  const applied = c.req.query('applied');
   198→  if (applied === 'true' || applied === 'false') {
   199→    filters.applied = applied === 'true';
   200→  }
   201→
   202→  const pagination = parsePaginationQuery({
   203→    page: c.req.query('page') ? Number(c.req.query('page')) : undefined,
   204→    pageSize: c.req.query('pageSize') ? Number(c.req.query('pageSize')) : undefined,
   205→    sort: c.req.query('sortBy')
   206→      ? `${c.req.query('sortBy')}:${c.req.query('sortOrder') ?? 'desc'}`
   207→      : undefined,
   208→  });
   209→
   210→  const result = await service.listAssumptions(filters, {
   211→    page: pagination.page,
   212→    pageSize: Math.min(pagination.pageSize, 100),
   213→    sortBy: c.req.query('sortBy') as 'createdAt' | 'confidenceLevel' | 'trainNumber' | undefined,
   214→    sortOrder: (c.req.query('sortOrder') as 'asc' | 'desc') ?? 'desc',
   215→  });
   216→
   217→  return c.json(
   218→    paginatedSuccess(
   219→      result.items.map(mapAssumptionToResponse),
   220→      {
   221→        page: result.page,
   222→        pageSize: result.pageSize,
   223→        totalItems: result.total,
   224→      },
   225→      { durationMs: getRequestDuration(c) }
   226→    )
   227→  );
   228→});

   254→/**
   255→ * POST /baseline-assumptions
   256→ *
   257→ * Create a new baseline assumption.
   258→ *
   259→ * Request Body:
   260→ * - trainId: Train ID (required)
   261→ * - pmTypeId: PM type ID (required)
   262→ * - assumptionType: How the assumption was derived (required)
   263→ * - assumedLastPmDate: Assumed date of last PM (optional)
   264→ * - assumedLastPmMileage: Assumed mileage at last PM (optional)
   265→ * - confidenceLevel: Data quality indicator (required)
   266→ * - inferenceSource: Description of how assumption was derived (optional)
   267→ * - inferenceData: Supporting data as JSON (optional)
   268→ * - requiresReview: Override review requirement (optional)
   269→ */
   270→baselineAssumptionRoutes.post('/baseline-assumptions', async (c) => {
   271→  const service = getService();
   272→  const body = await c.req.json();
   273→
   274→  // Validate required fields
   275→  if (!body.trainId || !isValidUuid(body.trainId)) {
   276→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'Valid trainId is required'), 400);
   277→  }
   278→  if (!body.pmTypeId || !isValidUuid(body.pmTypeId)) {
   279→    return c.json(error(ErrorCodes.VALIDATION_ERROR, 'Valid pmTypeId is required'), 400);
   280→  }
   281→
   282→  const validAssumptionTypes: AssumptionType[] = [
   283→    'INFERRED_FROM_COMMISSIONING',
   284→    'INFERRED_FROM_HIGHER_PM',
   285→    'MANUAL_OVERRIDE',
   286→    'IMPORTED',
   287→    'NO_HISTORY',
   288→  ];
   289→  if (!body.assumptionType || !validAssumptionTypes.includes(body.assumptionType)) {
   290→    return c.json(
   291→      error(ErrorCodes.VALIDATION_ERROR, `assumptionType must be one of: ${validAssumptionTypes.join(', ')}`),
   292→      400
   293→    );
   294→  }
   295→
   296→  const validConfidenceLevels: ConfidenceLevel[] = ['HIGH', 'MEDIUM', 'LOW', 'ESTIMATED'];
   297→  if (!body.confidenceLevel || !validConfidenceLevels.includes(body.confidenceLevel)) {
   298→    return c.json(
   299→      error(ErrorCodes.VALIDATION_ERROR, `confidenceLevel must be one of: ${validConfidenceLevels.join(', ')}`),
   300→      400
   301→    );
   302→  }
   303→
   304→  // Get user ID from context (if authenticated)
   305→  const userId = c.get('user')?.id;
   306→
   307→  try {
   308→    const assumption = await service.createAssumption(
   309→      {
   310→        trainId: body.trainId,
   311→        pmTypeId: body.pmTypeId,
   312→        assumptionType: body.assumptionType,
   313→        assumedLastPmDate: body.assumedLastPmDate,
   314→        assumedLastPmMileage: body.assumedLastPmMileage,
   315→        confidenceLevel: body.confidenceLevel,
   316→        inferenceSource: body.inferenceSource,
   317→        inferenceData: body.inferenceData,
   318→        requiresReview: body.requiresReview,
   319→      },
   320→      userId
   321→    );
   322→
   323→    return c.json(success(mapAssumptionToResponse(assumption), { durationMs: getRequestDuration(c) }), 201);
   324→  } catch (err) {
   325→    if (err instanceof NotFoundError) {
   326→      return c.json(error(ErrorCodes.NOT_FOUND, err.message), 404);
   327→    }
   328→    if (err instanceof ConflictError) {
   329→      return c.json(error(ErrorCodes.CONFLICT, err.message), 409);
   330→    }
   331→    throw err;
   332→  }
   333→});

   578→ * - knownPmTypeCode: PM type code that was completed (required, e.g., "C4")
   579→ * - knownPmDate: Date of the PM completion (required)

   602→    const result = await service.inferFromHigherPm(
