   421→    describe('Review Status Determination', () => {
   422→      interface MockAssumption {
   423→        appliedAt: Date | null;
   424→        reviewedAt: Date | null;
   425→        reviewNotes: string | null;
   426→      }
   427→
   428→      const determineReviewStatus = (assumption: MockAssumption): ReviewStatus => {
   429→        if (assumption.appliedAt) {
   430→          return 'APPLIED';
   431→        } else if (assumption.reviewedAt) {
   432→          if (assumption.reviewNotes?.startsWith('REJECTED:')) {
   433→            return 'REJECTED';
   434→          } else {
   435→            return 'REVIEWED';
   436→          }
   437→        } else {
   438→          return 'PENDING_REVIEW';
   439→        }
   440→      };
   441→
   442→      test('should return APPLIED when appliedAt is set', () => {
   443→        const assumption: MockAssumption = {
   444→          appliedAt: new Date(),
   445→          reviewedAt: new Date(),
   446→          reviewNotes: 'All good',
   447→        };
   448→        expect(determineReviewStatus(assumption)).toBe('APPLIED');
   449→      });

   451→      test('should return REVIEWED when reviewedAt is set but not applied', () => {

   453→          appliedAt: null,

   456→        };
