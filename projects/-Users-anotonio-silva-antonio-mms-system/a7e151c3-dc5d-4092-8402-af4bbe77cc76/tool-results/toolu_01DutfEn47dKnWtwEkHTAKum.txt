   315→#### Reference: @belt/tools Patterns
   316→```typescript
   317→// Schema validation (replaces Zod)
   318→import { createSchema, createType, InferType } from '@belt/tools';
   319→
   320→const TrainModel = createSchema({
   321→  id: 'string',
   322→  trainNumber: 'string',
   323→  mileageKm: 'number',
   324→  fitnessStatus: { enum: ['FIT', 'RESTRICTED', 'GROUNDED'] },
   325→  tags: 'string[]',
   326→  metadata: 'object?',
   327→});
   328→type Train = InferType<typeof TrainModel>;
   329→const train = TrainModel.parse(data);
   330→
   331→// State machine (replaces React hooks + zustand)
   332→import { createSlice, createMachine } from '@belt/tools';
   333→
   334→const fleetSlice = createSlice({
   335→  id: 'fleet',
   336→  initial: { trains: [], selected: null },
   337→  cache: { policy: 'session' },
   338→  reducers: {
   339→    setTrains: (state, trains) => { state.trains = trains; },
   340→    selectTrain: (state, id) => { state.selected = id; },
   341→  },
   342→});
   343→
   344→const machine = createMachine({
   345→  id: 'mms',
   346→  slices: { fleet: fleetSlice },
   347→  trackMetadata: false, // Critical for 60fps
   348→});
   349→
   350→// In React component (NO useState/useEffect!)
   351→const { state, actions } = machine.useMachine('fleet');
   352→```
